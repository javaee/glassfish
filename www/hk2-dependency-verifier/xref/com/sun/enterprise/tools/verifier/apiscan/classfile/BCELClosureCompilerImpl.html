82      //See corresponding method of ClosureCompiler for javadocs
83      public boolean buildClosure(String className) {
84          logger.entering(myClassName, "buildClosure", className); // NOI18N
85          ClassFile cf;
86          if (!needToBuildClosure(className))
87              return true;
88          try {
89              cf = loader.load(className);
90          } catch (IOException e) {
91              handleFailure(className);
92              return false;
93          }
94          return buildClosure(cf);
95      }
96  
97      /***
98       * @param cf class file whose closure needs to be computed. The behavior is
99       *           same as the other buildClosure() method.
100      */
101     private boolean buildClosure(ClassFile cf) {
102         boolean result = true;
103         callStack.push(cf);
104         if (needToBuildClosure(cf.getName())) {
105             visitedClasses.add(cf.getName());
106             Collection<String> names = cf.getAllReferencedClassNames();
107             closure.addAll(names);
108             // TODO: We should not be doing this here. Its just a quick &
109             // dirty solution.
110             for(Method m : cf.getMethods()) {
111                 if(m.isNative()) {
112                     final String methodDesc =
113                             m.getOwningClass().getName()+ "." + m.getName(); // NOI18N
114                     nativeMethods.add(methodDesc);
115                 }
116             }
117             for (Iterator i = names.iterator(); i.hasNext();) {
118                 String nextExternalName = (String) i.next();
119                 if (!needToBuildClosure(nextExternalName)) continue;
120                 ClassFile next;
121                 try {
122                     next = loader.load(nextExternalName);
123                 } catch (IOException e) {
124                     result = false;
125                     handleFailure(nextExternalName);
126                     continue;
127                 }
128                 boolean newresult = buildClosure(next);//recurssive call
129                 result = newresult && result;
130             }
131         }
132         callStack.pop();
133         return result;
134     }
135 
136     private void handleFailure(String referencedClass) {
137         String referencingPath = "";
138         try {
139             StringBuilder referencingPathBuffer = new StringBuilder();
140             for (Iterator i = callStack.iterator(); i.hasNext();) {
141                 if (referencingPathBuffer.length() != 0)
142                     referencingPathBuffer.append(File.separator);
143                 referencingPathBuffer.append(((ClassFile) i.next()).getName());
144             }
145             referencingPath = referencingPathBuffer.toString();
146         } catch (EmptyStackException e) {
147         }
148         logger.finer(
149                 "Could not locate " + referencingPath + File.separator + // NOI18N
150                 referencedClass);
151         List<String> failedList = failed.get(referencingPath);
152         if (failedList == null) {
153             failedList = new ArrayList<String>();
154             failed.put(referencingPath, failedList);
155         }
156         failedList.add(referencedClass);
157     }

169     /***
170      * Reset the closure for next closure computation.
171      * Clear the internal cache. It includes the result it has collected since
172      * last reset(). But it does not clear the excludedd list. If you want to
173      * reset the excluded list, create a new ClosureCompiler.

176         closure.clear();
177         visitedClasses.clear();
178         failed.clear();
179         nativeMethods.clear();

186     public String toString() {
187         StringBuilder sb=new StringBuilder();
188         if(logger.isLoggable(Level.FINER)){
189             sb.append("\n<Closure>"); // NOI18N
190 
191             sb.append("\n\t<ExcludedClasses>"); // NOI18N
192             for(Iterator i=excludedClasses.iterator(); i.hasNext();) {

196             sb.append("\n\t</ExcludedClasses>"); // NOI18N
197 
198             sb.append("\n\t<ExcludedPackages>"); // NOI18N
199             for(Iterator i=excludedPackages.iterator(); i.hasNext();){
200                 sb.append("\n\t\t"); // NOI18N
201                 sb.append((String)i.next());
202             }
203             sb.append("\n\t</ExcludedPackages>"); // NOI18N
204             
205             sb.append("\n\t<ExcludedPatterns>"); // NOI18N
206             for(Iterator i=excludedPatterns.iterator(); i.hasNext();){
207                 sb.append("\n\t\t"); // NOI18N
208                 sb.append((String)i.next());
209             }
210             sb.append("\n\t</ExcludedPatterns>"); // NOI18N
211             
212             sb.append("\n\t<Classes>"); // NOI18N
213             for(Iterator i=closure.iterator(); i.hasNext();){
214                 sb.append("\n\t\t"); // NOI18N
215                 sb.append((String)i.next());
216             }
217             sb.append("\n\t</Classes>"); // NOI18N
218         }
219         sb.append("\n\t<Failed>"); // NOI18N
220         for(Iterator i=failed.entrySet().iterator(); i.hasNext();) {
221             Map.Entry referencingPathToFailedList=(Map.Entry)i.next();
222             sb.append("\n\t\t"); // NOI18N
223             sb.append("<ReferencingPath>"); // NOI18N
224             sb.append("\n\t\t\t"); // NOI18N
225             sb.append(referencingPathToFailedList.getKey());
226             sb.append("\n\t\t"); // NOI18N
227             sb.append("</ReferencingPath>"); // NOI18N
228             sb.append("\n\t\t"); // NOI18N
229             sb.append("<Classes>"); // NOI18N
230             for(Iterator iii=((List)referencingPathToFailedList.getValue()).iterator(); iii.hasNext();){
231                 sb.append("\n\t\t\t"); // NOI18N
232                 sb.append((String)iii.next());
233             }
234             sb.append("\n\t\t"); // NOI18N
235             sb.append("</Classes>"); // NOI18N
236         }
237         sb.append("\n\t</Failed>"); // NOI18N
238 
239         sb.append("\n\t<NativeMethods>"); // NOI18N
240         for(String s : nativeMethods) {
241             sb.append("\n\t\t"); // NOI18N
242             sb.append(s);
243         }
244         sb.append("\n\t</NativeMethods>"); // NOI18N
245 
246         if(logger.isLoggable(Level.FINER)){
247             sb.append("\n</Closure>"); // NOI18N
248         }
249         return sb.toString();        
250     }
