90      /***
91       * @param is        is could be a Zip or Jar InputStream or a regular
92       *                  InputStream.
93       * @param file_name is the name of the .class file. If is a Zip or Jar Input
94       *                  stream, then this name should just be the entry path,
95       *                  because it is used internally in the is.getEntry(file_name)
96       *                  to locate the .class file. If is is a regular stream,
97       *                  then file_name should not be really needed, I don't have
98       *                  a choice as BCEL ClassFile does not have a constructor
99       *                  that just takes an InputStream. Any way, if is is a
100      *                  regular stream, then it should either be the internal
101      *                  name of the class or the path to the actual .class file
102      *                  that the input stream represents. This constructor does
103      *                  not check if the ClassFile created here indeed
104      *                  represents the class that is being requested. That check
105      *                  should be done by ClassFileLoader.
106      */
107     public BCELClassFile(InputStream is, String file_name) throws IOException {
108         logger.entering(myClassName, "<init>(InputStream, String)", file_name); // NOI18N
109         jc = new ClassParser(is, file_name).parse();

139         if (classNames == null) {
140             getAllReferencedClassNamesInInternalForm();

142         HashSet<String> extClassNames = new HashSet<String>(classNames.size());
143         for (Iterator i = classNames.iterator(); i.hasNext();) {

218     //returns a list of all the classnames embedded in the given signature string.
219     //This method knows about basic data types, so for them classname is not returned.
220     //e.g. given below are the signature string and class name that would be
221     // returned by this call...
222     // Ljava/lang/Integer;    {java/lang/Integer}
223     // [Ljava/lang/Integer;   {java/lang/Integer}
224     // [[I                    {}
225     // I                      {I}
226     // (F[La/b/P;La/b/Q;[[La/b/P;)I    {a/b/P, a/b/Q, a/b/P}
227     // a method like "int foo(float f, a/b/P[] ps, a/b/Q q, a/b/P[][] pss)" will
228     // have above signature.
229     private static List<String> signatureToClassNames(String signature) {
230         logger.entering(myClassName, "signatureToClassNames", signature); // NOI18N
231         List<String> result = new ArrayList<String>();
232         int i = 0;
233         while ((i = signature.indexOf('L', i)) != -1) {
234             int j = signature.indexOf(';', i);
235             if (j > i) {
236                 // get name, minus leading 'L' and trailing ';'
237                 String className = signature.substring(i + 1, j);
238                 if (!Util.isPrimitive(className)) result.add(className);
239                 i = j + 1;
240             } else
241                 break;
242         }
243         if (logger.isLoggable(Level.FINE)) {
244             StringBuffer sb = new StringBuffer("Class Names are {"); // NOI18N
245             int size = result.size();
246             for (int k = 0; k < size; k++) {
247                 sb.append((String) result.get(k));
248                 if (k != size - 1) sb.append(", "); // NOI18N
249             }
250             sb.append("}"); // NOI18N
251             logger.finer(sb.toString());
252         }
253         return result;
254     }

263         /* Now override the visitor methods of our interest from EmptyVisitor class. */
264         /*** 
265          */
266         public void visitConstantClass(ConstantClass obj) {
267             logger.entering(myClassName, "visitConstantClass", obj); // NOI18N
268             String className = obj.getBytes(jc.getConstantPool());
269             logger.finer("Class name is " + className); // NOI18N
270             //sometimes we get names like Ljava.lang.Integer; or [I. So we need
271             // to decode the names.
272             //A good test case is java/io/ObjectInputStream.class
273             if (className.indexOf(';') != -1 || className.indexOf('[') != -1) {
274                 classNames.addAll(signatureToClassNames(className));
275             } else {
276                 classNames.add(className);
277             }
278         }
279 
280         public void visitField(Field field) {
281             logger.entering(myClassName, "visitField", field); // NOI18N
282             String signature = field.getSignature();
283             logger.finer("Signature is " + signature); // NOI18N
284             //for BCEL 5.1, use field.getType().getSignature() if the above does
285             //not work
286             classNames.addAll(signatureToClassNames(signature));
287         }
288 
289         public synchronized void visitMethod(Method method) {
290             logger.entering(myClassName, "visitMethod", method); // NOI18N
291             String signature = method.getSignature();
292             logger.finer("Signature is " + signature); // NOI18N
293             methods.add(new BCELMethod(cf, method));
294             classNames.addAll(signatureToClassNames(signature));
295         }
296     }//class Visitor
