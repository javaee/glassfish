41  package com.sun.enterprise.tools.verifier.apiscan.classfile;
42  
43  import java.io.File;
44  import java.io.IOException;
45  import java.io.InputStream;
46  import java.net.MalformedURLException;
47  import java.net.URL;
48  import java.net.URLClassLoader;
49  import java.net.URLConnection;
50  import java.util.ArrayList;
51  import java.util.StringTokenizer;
52  import java.util.logging.Level;
53  import java.util.logging.Logger;
54  
55  /**
56   * *This is a factory for {@link BCELClassFile}. This is not a public class, as
57   * I expect users to use {@link ClassFileLoaderFactory} interface. This class
58   * internally uses the the standard Java ClassLoader to load the resource and
59   * construct BCELClassFile object out of it.
60   *
61   * @author Sanjeeb.Sahoo@Sun.COM
62   */
63  class BCELClassFileLoader implements ClassFileLoader {
64  
65      private ClassLoader cl;
66      private static String resourceBundleName = "com.sun.enterprise.tools.verifier.apiscan.LocalStrings";
67      private static Logger logger = Logger.getLogger("apiscan.classfile", resourceBundleName); // NOI18N
68      private final static String myClassName = "BCELClassFileLoader"; // NOI18N
69  
70      /**
71       * Creates a new instance of BCELClassFileLoader.
72       *
73       * @param cp that will be used to create a new java.net.URLClassLoader. In
74       *           subsequent load operations, this classloader will be used.
75       */
76      public BCELClassFileLoader(String cp) {
77          ArrayList<URL> urls = new ArrayList<URL>();
78          for (StringTokenizer st = new StringTokenizer(cp, File.pathSeparator);
79               st.hasMoreTokens();) {
80              String entry = st.nextToken();
81              try {
82                  // toURI().toURL() takes care of all the escape characters
83                  // in the absolutePath. The toURI() method encodes all escape
84                  // characters. Later URLClassLoader decodes and regenerates 
85                  // correct escape characters.
86                  urls.add(new File(entry).toURI().toURL());
87              } catch (MalformedURLException e) {
88                  logger.logp(Level.WARNING, myClassName, "init<>", getClass().getName() + ".exception1", new Object[]{entry});
89                  logger.log(Level.WARNING, "", e);
90              }
91          }
92          //We do not want system class loader or even extension class loadera s our parent.
93          //We want only boot class loader as our parent. Boot class loader is represented as null.
94          cl = new URLClassLoader((URL[]) urls.toArray(new URL[0]), null);
95      }
96  
97      /**
98       * Creates a new instance of BCELClassFileLoader.
99       *
100      * @param cl is the classloader that will be used in subsequent load
101      *           operations.
102      */
103     public BCELClassFileLoader(ClassLoader cl) {
104         this.cl = cl;
105     }
106 
107     //See corresponding method of ClassFileLoader
108     public ClassFile load(String externalClassName) throws IOException {
109         logger.entering("BCELClassFileLoader", "load", externalClassName); // NOI18N
110         //URLClassLoader library expects me to pass in internal form.
111         String internalClassName = externalClassName.replace('.', '/');
112         URL resource = cl.getResource(internalClassName + ".class");
113         //URLClassLoader returns null if resource is not found.
114         if(resource == null)
115             throw new IOException("Not able to load " + internalClassName + ".class");
116         URLConnection urlcon = resource.openConnection();
117         urlcon.setUseCaches(false); // bug 6328564
118         InputStream is = urlcon.getInputStream();
119         try {
120             ClassFile cf = new BCELClassFile(is, internalClassName + ".class"); // NOI18N
121             matchClassSignature(cf, externalClassName);
122             return cf;
123         } finally {
124             is.close();
125         }
126     }
127 
128     //This method is neede to be protected against users who are passing us
129     //internal class names instead of external class names or
130     //when the file actually represents some other class, but it isnot 
131     //available in in proper package hierarchy.
132     private void matchClassSignature(ClassFile cf, String externalClassName)
133             throws IOException {
134         String nameOfLoadedClass = cf.getName();
135         if (!nameOfLoadedClass.equals(externalClassName)) {
136             throw new IOException(externalClassName + ".class represents " +
137                     cf.getName() +
138                     ". Perhaps your package name is incorrect or you passed the " +
139                     "name using internal form instead of using external form.");
140         }
141     }
