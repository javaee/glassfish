43  package com.sun.enterprise.tools.verifier.apiscan.classfile;
44  
45  import java.util.Collection;
46  import java.util.Map;
47  
48  /***
49   * This is single most important interface of the apiscan package. This class is
50   * used to compute the complete closure of a set of classes.
51   *
52   * @author Sanjeeb.Sahoo@Sun.COM
53   * @see ClosureCompilerImpl
54   */
55  public interface ClosureCompiler {
56      /***
57       * @param externalClsName class name (in external format) whose closure will
58       *                        be computed.
59       * @return true if it can compute the closure for all the classes that are
60       *         new to it, else false. In other words, this operation is not
61       *         idempotent. e.g. ClosureCompiler cc; cc.reset(); boolean
62       *         first=cc.buildClosure("a.B"); boolean second=cc.buildClosure("a.B");
63       *         second will always be true irrespective of value of first. This
64       *         is because it uses list of classes that it has already visited.
65       *         That list gets cleared when {@link #reset()} is called).
66       */
67      boolean buildClosure(String externalClsName);
68  
69      /***
70       * @return unmodifiable collection of class names which it visited during
71       *         closure computation. e.g. Let's say a.class references b1.class
72       *         and b2.class. b1.class references c1.class, c2.class and c3.class
73       *         b2.class references d1.class, d2.class and d3.class.
74       *         c1/c2/d1/d2.class are all not loadable where as c3 and d3.class
75       *         are loadable. When we build the closure of a.class, closure will
76       *         contain the following... {"a", "b1", "b2", "c3", "d3"}
77       */
78      Collection getClosure();
79  
80      /***
81       * @return unmodifiable collection of class names whose closure could not be
82       *         computed. The typical reason for not able to build closure for a
83       *         class is that class not being found in loader's search path. See
84       *         it returns a map which is keyed by the access path and the value
85       *         is a list of class names which could not be loaded. e.g. Let's
86       *         say a.class references b1.class and b2.class. b1.class references
87       *         c1.class, c2.class and c3.class b2.class references d1.class,
88       *         d2.class and d3.class. c1/c2/d1/d2.class are all not loadable
89       *         where as c3 and d3.class are loadable. When we build the closure
90       *         of a.class, failed map will contain the following... {("a:b1",
91       *         {"c1","c2"}), ("a.b2", {"d1","d2"})}
92       */
93      Map getFailed();
94  
95      /***
96       * Clear the internal cache. It includes the result it has collected since
97       * last reset(). But it does not clear the excludedd list. If you want to
98       * reset the excluded list, create a new ClosureCompiler.
99       */
100     void reset();
101 
102 }
