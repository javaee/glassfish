41  package com.sun.enterprise.tools.verifier.apiscan.classfile;
42  
43  import java.util.Collection;
44  import java.util.Map;
45  
46  /**
47   * This is single most important interface of the apiscan package. This class is
48   * used to compute the complete closure of a set of classes.
49   *
50   * @author Sanjeeb.Sahoo@Sun.COM
51   * @see ClosureCompilerImpl
52   */
53  public interface ClosureCompiler {
54      /**
55       * @param externalClsName class name (in external format) whose closure will
56       *                        be computed.
57       * @return true if it can compute the closure for all the classes that are
58       *         new to it, else false. In other words, this operation is not
59       *         idempotent. e.g. ClosureCompiler cc; cc.reset(); boolean
60       *         first=cc.buildClosure("a.B"); boolean second=cc.buildClosure("a.B");
61       *         second will always be true irrespective of value of first. This
62       *         is because it uses list of classes that it has already visited.
63       *         That list gets cleared when {@link #reset()} is called).
64       */
65      boolean buildClosure(String externalClsName);
66  
67      /**
68       * @return unmodifiable collection of class names which it visited during
69       *         closure computation. e.g. Let's say a.class references b1.class
70       *         and b2.class. b1.class references c1.class, c2.class and c3.class
71       *         b2.class references d1.class, d2.class and d3.class.
72       *         c1/c2/d1/d2.class are all not loadable where as c3 and d3.class
73       *         are loadable. When we build the closure of a.class, closure will
74       *         contain the following... {"a", "b1", "b2", "c3", "d3"}
75       */
76      Collection getClosure();
77  
78      /**
79       * @return unmodifiable collection of class names whose closure could not be
80       *         computed. The typical reason for not able to build closure for a
81       *         class is that class not being found in loader's search path. See
82       *         it returns a map which is keyed by the access path and the value
83       *         is a list of class names which could not be loaded. e.g. Let's
84       *         say a.class references b1.class and b2.class. b1.class references
85       *         c1.class, c2.class and c3.class b2.class references d1.class,
86       *         d2.class and d3.class. c1/c2/d1/d2.class are all not loadable
87       *         where as c3 and d3.class are loadable. When we build the closure
88       *         of a.class, failed map will contain the following... {("a:b1",
89       *         {"c1","c2"}), ("a.b2", {"d1","d2"})}
90       */
91      Map getFailed();
92  
93      /**
94       * Clear the internal cache. It includes the result it has collected since
95       * last reset(). But it does not clear the excludedd list. If you want to
96       * reset the excluded list, create a new ClosureCompiler.
97       */
98      void reset();
99  
100 }
