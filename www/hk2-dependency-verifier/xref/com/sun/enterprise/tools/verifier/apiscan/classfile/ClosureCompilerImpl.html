48  /**
49   * This is single most important class of the apiscan package. This class is
50   * used to compute the complete closure of a set of classes. It uses {@link
51   * ClassFile} to find out dependencies of a class. For each references class
52   * name, it loads the corresponding ClassFile using the ClassFileLoader passed
53   * to it in constructor. Then it recurssively computes the dependencies until it
54   * visits either Java primitive classes or the class name matches the exclude
55   * list. Example of using this class is given below...
56   * <blockquote><pre>
57   * String classpath="your own classpath";
58   * ClassFileLoader cfl=ClassFileLoaderFactory.newInstance(new
59   * Object[]{classpath});
60   * ClosureCompilerImpl cc=new ClosureCompilerImpl(cfl);
61   * cc.addExcludePattern("java.");//exclude all classes that start with java.
62   * Most of the J2SE classes will be excluded thus.
63    * cc.addExcludePattern("javax.");//Most of the J2EE classes can be excluded
64   * like this.
65   * cc.addExcludePackage("org.omg.CORBA");//Exclude classes whose package name is
66   * org.omg.CORBA
67   * cc.addExcludeClass("mypackage.Foo");//Exclude class whose name is
68   * myPackage.Foo
69   * boolean successful=cc.buildClosure("a.b.MyEjb");
70   * successful=cc.buildClosure("a.b.YourEjb") && successful; //note the order of
71   * &&.
72   * Collection closure=cc.getClosure();//now closure contains the union of
73   * closure for the two classes.
74   * Map failed=cc.getFailed();//now failure contains the union of failures for
75   * the two classes.
76   * cc.reset();//clear the results collected so far so that we can start afresh.
77   * //Now you can again start computing closure of another set of classes.
78   * //The excluded stuff are still valid.
79   * </pre></blockquote>
80   *
81   * @author Sanjeeb.Sahoo@Sun.COM
82   */
83  public class ClosureCompilerImpl implements ClosureCompiler {

100     /**
101      * @param loader the ClassFileLoader that is used to load the referenced
102      *               classes.
103      */
104     public ClosureCompilerImpl(ClassFileLoader loader) {
105         /*
106          * See how it binds to a runtime implementation
107          * TODO: Ideally we should introduce an AbstractFactory for
108          * both ClassFileLoader & ClassClosureCompiler product types.
109          */
110         if(loader instanceof BCELClassFileLoader ||
111                 loader instanceof BCELClassFileLoader1) {
112             imp = new BCELClosureCompilerImpl(loader);
113         } else {
114             throw new RuntimeException("Unknown loader type [" + loader + "]");
115         }
116     }

118     /**
119      * I don't expect this constructor to be used. Only defined for
120      * testing purpose.
121      * @param imp the implementation in the bridge design pattern.
122      */
123     public ClosureCompilerImpl(ClosureCompilerImplBase imp) {
124         this.imp = imp;
125     }

127     /**
128      * @param className the class name to be excluded from closure
129      *                     computation. It is in the external class name format
130      *                     (i.e. java.util.Map$Entry instead of java.util.Map.Entry).
131      *                     When the closure compiler sees a class matches this
132      *                     name, it does not try to compute its closure any
133      *                     more. It merely adds this name to the closure. So the
134      *                     final closure will contain this class name, but not
135      *                     its dependencies.
136      */
137     public void addExcludedClass(String className) {
138         imp.addExcludedClass(className);
139     }
140 
141     //takes in external format, i.e. java.util
142     /**
143      * @param pkgName the package name of classes to be excluded from
144      *                     closure computation. It is in the external format
145      *                     (i.e. java.lang (See no trailing '.'). When the
146      *                     closure compiler sees a class whose package name
147      *                     matches this name, it does not try to compute the
148      *                     closure of that class any more. It merely adds that
149      *                     class name to the closure. So the final closure will
150      *                     contain that class name, but not its dependencies.
151      */
152     public void addExcludedPackage(String pkgName) {
153         imp.addExcludedPackage(pkgName);
154     }
155 
156     /**
157      * @param pattern the pattern for the names of classes to be excluded from
158      *                closure computation. It is in the external format (i.e.
159      *                org.apache.). When the closure compiler sees a class whose
160      *                name begins with this pattern, it does not try to compute
161      *                the closure of that class any more. It merely adds that
162      *                class name to the closure. So the final closure will
163      *                contain that class name, but not its dependencies. Among
164      *                all the excluded list, it is given the lowest priority in
165      *                search order.
166      */
167     public void addExcludedPattern(String pattern) {
168         imp.addExcludedPattern(pattern);
169     }

172     public boolean buildClosure(String className) {
173         logger.entering(myClassName, "buildClosure", className); // NOI18N
174         return imp.buildClosure(className);
175     }

177     /**
178      * @param jar whose classes it will try to build closure of. This is a
179      *            convenience method which iterates over all the entries in a
180      *            jar file and computes their closure.

182     public boolean buildClosure(java.util.jar.JarFile jar) throws IOException {
183         return imp.buildClosure(jar);
184     }

214     public static void main(String[] args) {
215         if (args.length < 2) {
216             System.out.println(
217                     "Usage : java " + com.sun.enterprise.tools.verifier.apiscan.classfile.ClosureCompilerImpl.class.getName() + // NOI18N
218                     " <classpath> <external class name(s)>"); // NOI18N
219             System.out.println("Example: to find the closure of " + // NOI18N
220                     "mypkg.MySessionBean which is packaged in myejb.jar run\n" + // NOI18N
221                     " java " + com.sun.enterprise.tools.verifier.apiscan.classfile.ClosureCompilerImpl.class.getName() + // NOI18N
222                     " path_to_j2ee.jar"+File.pathSeparator+"path_to_myejb.jar"+ // NOI18N
223                     " mypkg.MySessionBean"); // NOI18N
224             System.exit(1);
225         }
226 
227         String cp=args[0];
228         System.out.println("Using classpath " + cp); // NOI18N
229         ClassFileLoader cfl = ClassFileLoaderFactory.newInstance(
230                 new Object[]{cp});
231         ClosureCompilerImpl closure = new ClosureCompilerImpl(cfl);
232         closure.addExcludedPattern("java."); // NOI18N
233         for (int i = 1; i < args.length; i++) {
234             String clsName = args[i];
235             System.out.println("Building closure for " + clsName); // NOI18N
236             closure.reset();
237             closure.buildClosure(clsName);
238             System.out.println("The closure is [" + closure+"\n]"); // NOI18N
239         }
240     }
