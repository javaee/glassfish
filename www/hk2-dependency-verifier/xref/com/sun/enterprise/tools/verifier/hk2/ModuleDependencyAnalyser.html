83              // add all HK2 package patterns, because we don't handle reexport yet.
84              , "org.jvnet.hk2."
85              , "com.sun.hk2."
86              , "com.sun.enterprise.module."

90      /**
91       * Create a new analyser.
92       * @param moduleDef module whose dependency needs to be analysed
93       * @param moduleRepository repository used to satisfy dependencies
94       * @throws IOException
95       */
96      public ModuleDependencyAnalyser(ModuleDefinition moduleDef,
97                                      Repository moduleRepository) throws IOException {
98          this.moduleDef = moduleDef;
99          this.moduleRepository = moduleRepository;
100         moduleJar = new File(moduleDef.getLocations()[0]);
101         // Make a classpath consisting of only module jar file.
102         String classpath = moduleJar.getAbsolutePath();
103         ClassFileLoader cfl = ClassFileLoaderFactory.newInstance(new Object[]{classpath});
104         closure = new ClosureCompilerImpl(cfl);
105         for (String pattern : excludedPatterns) {
106             closure.addExcludedPattern(pattern);
107         }
108         excludeImportedPackages();
109         excludeExportedClasses();
110     }

112     /**
113      * Analyse dependency of a module. It uses the repository to look up
114      * modules that this module depends on.
115      * @return true if all the dependencies are OK, false if something is missing
116      * @throws IOException if there is any failure in reading module information
117      */
118     public synchronized boolean analyse()
119             throws IOException {
120         closure.buildClosure(new JarFile(moduleJar));
121         if (System.getProperty("debugOutput") != null) {
122             logger.setLevel(Level.FINER);
123             PrintStream out = new PrintStream(new FileOutputStream(System.getProperty("debugOutput")));
124             out.println(closure);
125         }
126         return closure.getFailed().isEmpty();
127     }

156     /**
157      * This method adds packages imported by this bundle to
158      * to the list of excluded package names.
159      */
160     private void excludeImportedPackages() {
161         Attributes attributes = moduleDef.getManifest().getMainAttributes();
162         String exportedPkgsAttr = attributes.getValue("Import-Package");
163         if (exportedPkgsAttr==null) return;
164         StringTokenizer st = new StringTokenizer(exportedPkgsAttr, ",", false);
165         Set<String> importedPkgs = new HashSet<String>();
166         while (st.hasMoreTokens()) {
167             String token = st.nextToken().trim();
168             int idx = token.indexOf(';');
169             String pkg = (idx == -1) ? token : token.substring(0, idx);
170 //            System.out.println("pkg = " + pkg);
171             importedPkgs.add(pkg);
172         }
173         for (String pkg : importedPkgs) {
174             closure.addExcludedPackage(pkg);
175         }
176     }
177 
178     /**
179      * This method adds classes exported by all the bundles that this bundle imports
180      * to the list of excluded class names.
181      */
182     private void excludeExportedClasses() throws IOException {
183         ModuleDependency[] dependencies = moduleDef.getDependencies();
184         for (ModuleDependency dependency : dependencies) {
185             ModuleDefinition requiredModuleDef =
186                     moduleRepository.find(dependency.getName(), dependency.getVersion());
187             for (String cls : getExportedClasses(requiredModuleDef)) {
188                 closure.addExcludedClass(cls);
189             }
190         }
191     }

193     private Set<String> getExportedClasses(ModuleDefinition moduleDef) throws IOException {
194         Set<String> exportedPkgs = new HashSet<String>();
195         java.util.jar.Manifest m = moduleDef.getManifest();
196         if (m==null) return exportedPkgs;
197         Attributes attributes = m.getMainAttributes();
198         String exportedPkgsAttr = attributes.getValue("Export-Package");
199         StringTokenizer st = new StringTokenizer(exportedPkgsAttr, ",", false);
200         while (st.hasMoreTokens()) {
201             String token = st.nextToken().trim();
202             int idx = token.indexOf(';');
203             String pkg = (idx == -1) ? token : token.substring(0, idx);
204 //            System.out.println("pkg = " + pkg);
205             exportedPkgs.add(pkg);
206         }
207         Set<String> exportedClasses = new HashSet<String>();
208         JarFile moduleJar = new JarFile(new File(moduleDef.getLocations()[0]));
209         Enumeration<JarEntry> moduleEntries = moduleJar.entries();
210         while (moduleEntries.hasMoreElements()) {
211             String entry = moduleEntries.nextElement().getName();
212             if (entry.endsWith(".class")) {
213                 String clsName = Util.convertToExternalClassName(
214                         entry.substring(0, entry.length() - ".class".length()));
215                 String pkgName = Util.getPackageName(clsName);
216                 if (exportedPkgs.contains(pkgName)) {
217                     exportedClasses.add(clsName);
218                 }
219             }
220         }
221         return exportedClasses;
222     }

224     public String getResultAsString() {
225         StringBuilder sb = new StringBuilder();
226         Map<String, Collection<String>> failed = closure.getFailed();
227         if (failed.isEmpty()) return "";
228         for (String referencingPath : failed.keySet()) {
229             String referencingClass;
230             int idx = referencingPath.lastIndexOf('/');
231             if (idx == -1) {
232                 referencingClass = referencingPath;
233             } else {
234                 referencingClass = referencingPath.substring(idx + 1);
235             }
236             sb.append("\t").append(referencingClass);
237             sb.append(" -> {");
238             String[] notFoundClasses = failed.get(referencingPath).toArray(new String[0]);
239             for (int i = 0; i < notFoundClasses.length; ++i) {
240                 sb.append(notFoundClasses[i]);
241                 if (i != (notFoundClasses.length - 1)) { // skip appending ',' at the end
242                     sb.append(", ");
243                 }
244             }
245             sb.append("}\n");
246         }
247         return sb.toString();
248     }

254     public static void main(String[] args) throws Exception {
255         if (args.length == 0) {
256             System.out.println("Usage: java " + ModuleDependencyAnalyser.class.getName() +
257                     " <Repository Dir Path> [Path to module]...");
258 
259             System.out.println("Examples:\n" +
260                     "1. Following command verifies all modules in the specified repository:\n" +
261                     " java " + ModuleDependencyAnalyser.class.getName() +
262                     " /tmp/glassfish/modules/\n\n" +
263                     "2. Following command verifies only the specified modules in the specified repository:\n" +
264                     " java " + ModuleDependencyAnalyser.class.getName() +
265                     " /tmp/glassfish/modules/ kernel-10.0-SNAPSHOT.jar amx-impl-10.0SNAPSHOT.jar\n\n");
266             return;
267         }
268         String repoPath = args[0];
269         File f = new File(repoPath) {
270             @Override public File[] listFiles() {
271                 List<File> files = new ArrayList<File>();
272                 for (File f : super.listFiles()) {
273                     if (f.isDirectory()) {
274                         for (File f2 : f.listFiles()) {
275                             if (f2.isFile() && f2.getName().endsWith(".jar")) {
276                                 files.add(f2);
277                             }
278                         }
279                     } else if (f.isFile() && f.getName().endsWith(".jar")) {
280                         files.add(f);
281                     }
282                 }
283                 return files.toArray(new File[files.size()]);
284             }
285         };
286         Repository moduleRepository = new DirectoryBasedRepository("repo", f);
287         moduleRepository.initialize();
288         List<ModuleDefinition> moduleDefs = new ArrayList<ModuleDefinition>();
289         if (args.length > 1) {
290             for (int i = 1; i < args.length; ++i) {
291                 File moduleFile = new File(args[i]);
292                 if (!moduleFile.isAbsolute()) {
293                     moduleFile = new File(repoPath, args[i]);
294                 }
295                 ModuleDefinition moduleDef = new DefaultModuleDefinition(moduleFile);
296                 moduleDefs.add(moduleDef);
297             }
298         } else {
299             moduleDefs = moduleRepository.findAll();
300         }
301         List<URI> badModules = new ArrayList<URI>();
302         for (ModuleDefinition moduleDef : moduleDefs) {
303             ModuleDependencyAnalyser analyser =
304                     new ModuleDependencyAnalyser(moduleDef, moduleRepository);
305             if (System.getProperty("ExcludedPatterns")!=null) {
306                 StringTokenizer st = new StringTokenizer(
307                         System.getProperty("ExcludedPatterns"), ",", false);
308                 Set<String> patterns = new HashSet<String>();
309                 while (st.hasMoreTokens()) {
310                     patterns.add(st.nextToken());
311                 }
312                 analyser.excludePatterns(patterns);
313             }
314             if (!analyser.analyse()) {
315                 URI badModule = moduleRepository.getLocation().relativize(moduleDef.getLocations()[0]);
316                 badModules.add(badModule);
317                 System.out.println("<Module name = " + badModule + ">");
318                 System.out.println(analyser.getResultAsString());
319                 System.out.println("</Module>");
320             }
321         }
322         if (badModules.isEmpty()) {
323             System.out.println("All modules are OK");
324         } else {
325             System.out.println("Dependencies are not correctly set up for following modules:");
326             for (URI badModule : badModules) {
327                 System.out.print(badModule + " ");
328             }
329             System.out.println("");
330         }
331     }
