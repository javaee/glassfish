150     /**
151      * Returns a ClassLoader capable of loading classes from a set of modules identified
152      * by their module definition and also load new urls.
153      *
154      * @param parent the parent class loader for the returned class loader instance
155      * @param mds module definitions for all modules this classloader should be
156      *        capable of loading
157      * @param urls urls to be added to the module classloader
158      * @return class loader instance
159      * @throws com.sun.enterprise.module.ResolveError if one of the provided module
160      *         definition cannot be resolved
161      */
162     public ClassLoader getModulesClassLoader(final ClassLoader parent,
163                                              Collection<ModuleDefinition> mds,
164                                              URL[] urls) throws ResolveError {
165         final List<ClassLoader> delegateCLs = new ArrayList<ClassLoader>();
166         final List<Module> delegateModules = new ArrayList<Module>();
167         for (ModuleDefinition md : mds) {
168             Module m = makeModuleFor(md.getName(), md.getVersion());
169             delegateModules.add(m);
170             delegateCLs.add(m.getClassLoader());
171         }
172         return new URLClassLoader(urls!=null?urls:new URL[0], parent) {
173             /*
174              * This is a delegating class loader.
175              * This extends URLClassLoader, because web layer (Jasper to be specific)
176              * expects it to be a URLClassLoader so that it can extract Classpath information
177              * used for javac.
178              * It always delegates to a chain of OSGi bundle's class loader.
179              * ClassLoader.defineClass() is never called in the context of this class.
180              * There will never be a class for which getClassLoader()
181              * would return this class loader.
182              */
183             @Override
184             public URL[] getURLs() {
185                 List<URL> result = new ArrayList<URL>();
186                 if (parent instanceof URLClassLoader) {
187                     URL[] parentURLs = URLClassLoader.class.cast(parent).getURLs();
188                     result.addAll(Arrays.asList(parentURLs));
189                 }
190                 for (Module m : delegateModules) {
191                     ModuleDefinition md = m.getModuleDefinition();
192                     URI[] uris = md.getLocations();
193                     URL[] urls = new URL[uris.length];
194                     for (int i = 0; i < uris.length; ++i) {
195                         try {
196                             urls[i] = uris[i].toURL();
197                         } catch (MalformedURLException e) {
198                             logger.warning("Exception " + e + " while converting " + uris[i] + " to URL");
199                         }
200                     }
201                     result.addAll(Arrays.asList(urls));
202                 }
203                 return result.toArray(new URL[0]);
204             }
205 
206             @Override
207             protected Class<?> findClass(String name) throws ClassNotFoundException {
208                 for (ClassLoader delegate : delegateCLs) {
209                     try {
210                         return delegate.loadClass(name);
211                     } catch(ClassNotFoundException e) {
212                         // This is expected, so ignore
213                     }
214                 }
215                 throw new ClassNotFoundException(name);
216             }
217 
218             @Override
219             public URL findResource(String name) {
220                 URL resource = null;
221                 for (ClassLoader delegate : delegateCLs) {
222                     resource = delegate.getResource(name);
223                     if (resource != null) {
224                         return resource;
225                     }
226                 }
227                 return resource;
228             }
229 
230             @Override
231             public Enumeration<URL> findResources(String name) throws IOException {
232                 List<Enumeration<URL>> enumerators = new ArrayList<Enumeration<URL>>();
233                 for (ClassLoader delegate : delegateCLs) {
234                     Enumeration<URL> enumerator = delegate.getResources(name);
235                     enumerators.add(enumerator);
236                 }
237                 return new CompositeEnumeration(enumerators);
238             }
239 
240         };
241     }

272     public void addModuleChangeListener(final ModuleChangeListener listener, final OSGiModuleImpl module) {
273         BundleListener bundleListener = new BundleListener() {
274             public void bundleChanged(BundleEvent event) {
275                 if ((event.getBundle() == module.getBundle()) &&
276                         ((event.getType() & BundleEvent.UPDATED) == BundleEvent.UPDATED)) {
277                     listener.changed(module);
278                 }
279             }
280         };
281         bctx.addBundleListener(bundleListener);
282         moduleChangeListeners.put(listener, bundleListener);
283     }
284 
285     public boolean removeModuleChangeListener(ModuleChangeListener listener) {
286         BundleListener bundleListener = moduleChangeListeners.remove(listener);
287         if (bundleListener!= null) {
288             bctx.removeBundleListener(bundleListener);
289             return true;
290         }
291         return false;
292     }

294     public void register(final ModuleLifecycleListener listener) {
295         // This is purposefully made an asynchronous bundle listener
296         BundleListener bundleListener = new BundleListener() {
297             public void bundleChanged(BundleEvent event) {
298                 switch (event.getType()) {
299                     case BundleEvent.INSTALLED:
300                         listener.moduleInstalled(getModule(event.getBundle()));
301                         break;
302                     case BundleEvent.UPDATED:
303                         listener.moduleUpdated(getModule(event.getBundle()));
304                         break;
305                     case BundleEvent.RESOLVED:
306                         listener.moduleResolved(getModule(event.getBundle()));
307                         break;
308                     case BundleEvent.STARTED:
309                         listener.moduleStarted(getModule(event.getBundle()));
310                         break;
311                     case BundleEvent.STOPPED:
312                         listener.moduleStopped(getModule(event.getBundle()));
313                         break;
314                 }
315             }
316         };
317         bctx.addBundleListener(bundleListener);
318         moduleLifecycleListeners.put(listener,  bundleListener);
319     }
320 
321     public void unregister(ModuleLifecycleListener listener) {
322         BundleListener bundleListener = moduleLifecycleListeners.remove(listener);
323         if (bundleListener!=null) {
324             bctx.removeBundleListener(bundleListener);
325         }
326     }

332     public void remove(Module module) {
333         super.remove(module);
334         
335         if (!(module instanceof OSGiModuleImpl)) {
336             return;
337         }
338         
339         OSGiModuleImpl oModule = (OSGiModuleImpl) module;
340         Bundle bundle = oModule.getBundle();
341         
342         String bsn = bundle.getSymbolicName();
343         String version = bundle.getVersion().toString();
344         
345         Set<ServiceLocator> locators = getAllServiceLocators();
346         
347         for (ServiceLocator locator : locators) {
348             if (!ServiceLocatorState.RUNNING.equals(locator.getState())) continue;
349             
350             ServiceLocatorUtilities.removeFilter(locator, new RemoveFilter(bsn, version));
351         }
352     }

354     private static class RemoveFilter implements Filter {
355         private final String bsn;
356         private final String version;
357         
358         private RemoveFilter(String bsn, String version) {
359             this.bsn = bsn;
360             this.version = version;
361         }
362         
363         private static String getMetadataValue(Descriptor d, String key) {
364             Map<String, List<String>> metadata = d.getMetadata();
365             
366             List<String> values = metadata.get(key);
367             if (values == null || values.size() <= 0) {
368                 return null;
369             }
370             
371             return values.get(0);
372         }
373 
374         @Override
375         public boolean matches(Descriptor d) {
376             String dBSN = getMetadataValue(d, OsgiPopulatorPostProcessor.BUNDLE_SYMBOLIC_NAME);
377             if (dBSN == null || !dBSN.equals(bsn)) return false;
378             
379             String dVersion = getMetadataValue(d, OsgiPopulatorPostProcessor.BUNDLE_VERSION);
380             if (dVersion == null) return false;
381             
382             return dVersion.equals(version);
383         }
