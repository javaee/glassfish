122     public synchronized void addRepository(URI obrUri) throws Exception {
123         if (isDirectory(obrUri)) {
124             setupRepository(new File(obrUri), isSynchronous());
125         } else {
126             repositories.add(getRepositoryAdmin().getHelper().repository(obrUri.toURL()));
127         }

155     private boolean isSynchronous() {
156         String property = context.getProperty(Constants.INITIALIZE_OBR_SYNCHRONOUSLY);
157         // default is synchronous as we are not sure if we have covered every race condition in asynchronous path
158         return property == null || Boolean.TRUE.toString().equalsIgnoreCase(property);
159     }

190     private File getRepositoryFile(File repoDir) {
191         String extn = ".xml";
192         String cacheDir = context.getProperty(Constants.HK2_CACHE_DIR);
193         if (cacheDir == null) {
194             return null; // caching is disabled, so don't do it.
195         }
196         return new File(cacheDir, Constants.OBR_FILE_NAME_PREFIX + repoDir.getName() + extn);
197     }

199     /**
200      * Create a new Repository from a directory by recurssively traversing all the jar files found there.
201      *
202      * @param repoFile
203      * @param repoDir
204      * @return
205      * @throws IOException
206      */
207     private Repository createRepository(File repoFile, File repoDir) throws IOException {
208         DataModelHelper dmh = getRepositoryAdmin().getHelper();
209         List<Resource> resources = new ArrayList<Resource>();
210         for (File jar : findAllJars(repoDir)) {
211             Resource r = dmh.createResource(jar.toURI().toURL());
212             
213             if (r == null) {
214             	logger.logp(Level.WARNING, "ObrHandler", "createRepository", "{0} not an OSGi bundle", jar.toURI().toURL());
215             } else {
216                 resources.add(r);
217             }
218         }
219         Repository repository = dmh.repository(resources.toArray(new Resource[resources.size()]));
220         logger.logp(Level.INFO, "ObrHandler", "createRepository", "Created {0} containing {1} resources.",
221                 new Object[]{repoFile, resources.size()});
222         if (repoFile != null) {
223             saveRepository(repoFile, repository);
224         }
225         return repository;
226     }

228     private void saveRepository(File repoFile, Repository repository) throws IOException {
229         assert (repoFile != null);
230         final FileWriter writer = new FileWriter(repoFile);
231         
232         getRepositoryAdmin().getHelper().writeRepository(repository, writer);
233         writer.flush();
234     }

241     private Repository updateRepository(File repoFile, File repoDir) throws Exception {
242         Repository repository = loadRepository(repoFile);
243         if (isObsoleteRepo(repository, repoFile, repoDir)) {
244             if (!repoFile.delete()) {
245                 throw new IOException("Failed to delete " + repoFile.getAbsolutePath());
246             }
247             logger.logp(Level.INFO, "ObrHandler", "updateRepository", "Recreating {0}", new Object[]{repoFile});
248             repository = createRepository(repoFile, repoDir);
249         }
250         return repository;
251     }

253     private boolean isObsoleteRepo(Repository repository, File repoFile, File repoDir) {
254         // TODO(Sahoo): Revisit this...
255         // This method assumes that the cached repoFile has been created before a newer jar is created.
256         // So, this method does not always detect stale repoFile. Imagine the following situation:
257         // time t1: v1 version of jar is released.
258         // time t2: v2 version of jar is released.
259         // time t3: repo.xml is populated using v1 version of jar, so repo.xml records a timestamp of t3 > t2.
260         // time t4: v2 version of jar is unzipped on modules/ and unzip maintains the timestamp of jar as t2.
261         // Next time when we compare timestamp, we will see that repo.xml is newer than this jar, when it is not.
262         // So, we include a size check. We go for the total size check...
263 
264         long lastModifiedTime = repoFile.lastModified();
265         // optimistic: see if the repoDir has been touched. dir timestamp changes when files are added or removed.
266         if (repoDir.lastModified() > lastModifiedTime) {
267             return true;
268         }
269         long totalSize = 0;
270         // now compare timestamp of each jar and take a sum of size of all jars.
271         for (File jar : findAllJars(repoDir)) {
272             if (jar.lastModified() > lastModifiedTime) {
273                 logger.logp(Level.INFO, "ObrHandler", "isObsoleteRepo", "{0} is newer than {1}", new Object[]{jar, repoFile});
274                 return true;
275             }
276             totalSize += jar.length();
277         }
278         // time stamps didn't identify any difference, so check sizes. The probabibility of sizes of all jars being same
279         // when some jars have changed is very very low.
280         for (Resource r : repository.getResources()) {
281             totalSize -= r.getSize();
282         }
283         if (totalSize != 0) {
284             logger.logp(Level.INFO, "ObrHandler", "isObsoleteRepo", "Change in size detected by {0} bytes", new Object[]{totalSize});
285             return true;
286         }
287         return false;
288     }

290     private List<File> findAllJars(File repo) {
291         final List<File> files = new ArrayList<File>();
292         repo.listFiles(new FileFilter() {
293             @Override
294             public boolean accept(File pathname) {
295                 if (pathname.isDirectory()) {
296                     pathname.listFiles(this);
297                 } else if (pathname.getName().endsWith("jar")) {
298                     files.add(pathname);
299                 }
300                 return true;
301             }
302         });
303         return files;
304     }

307     synchronized Bundle deploy(Resource resource) {
308         final Resolver resolver = getRepositoryAdmin().resolver(getRepositories());
309         boolean resolved = resolve(resolver, resource);
310         if (resolved) {
311             final int flags = !deployOptionalRequirements ? Resolver.NO_OPTIONAL_RESOURCES : 0;
312             resolver.deploy(flags);
313             return getBundle(resource);
314         } else {
315             Reason[] reqs = resolver.getUnsatisfiedRequirements();
316             logger.logp(Level.WARNING, "ObrHandler", "deploy",
317                     "Unable to satisfy the requirements: {0}", new Object[]{Arrays.toString(reqs)});
318             return null;
319         }
320     }

349     synchronized Bundle deploy(String name, String version) {
350         Resource resource = findResource(name, version);
351         if (resource == null) {
352             logger.logp(Level.INFO, "ObrHandler", "deploy",
353                     "No resource matching name = {0} and version = {1} ", new Object[]{name, version});
354             return null;
355         }
356         if (resource.isLocal()) {
357             return getBundle(resource);
358         }
359         return deploy(resource);
360     }

362     private Bundle getBundle(Resource resource) {
363         for (Bundle b : context.getBundles()) {
364             final String bsn = b.getSymbolicName();
365             final Version bv = b.getVersion();
366             final String rsn = resource.getSymbolicName();
367             final Version rv = resource.getVersion();
368             boolean versionMatching = (rv == bv) || (rv != null && rv.equals(bv));
369             boolean nameMatching = (bsn == rsn) || (bsn != null && bsn.equals(rsn));
370             if (nameMatching && versionMatching) return b;
371         }
372         return null;
373     }

375     private Resource findResource(String name, String version) {
376         final RepositoryAdmin repositoryAdmin = getRepositoryAdmin();
377         if (repositoryAdmin == null) {
378             logger.logp(Level.WARNING, "ObrHandler", "findResource",
379                     "OBR is not yet available, so can't find resource with name = {0} and version = {1} from repository",
380                     new Object[]{name, version});
381             return null;
382         }
383         String s1 = "(symbolicname=" + name + ")";
384         String s2 = "(version=" + version + ")";
385         String query = (version != null) ? "(&" + s1 + s2 + ")" : s1;
386         try {
387             Resource[] resources = discoverResources(query);
388             logger.logp(Level.INFO, "ObrHandler", "findResource",
389                     "Using the first one from the list of {0} discovered bundles shown below: {1}",
390                     new Object[]{resources.length, Arrays.toString(resources)});
391             return resources.length > 0 ? resources[0] : null;
392         } catch (InvalidSyntaxException e) {
393             throw new RuntimeException(e); // TODO(Sahoo): Proper Exception Handling
394         }
395     }

397     private Resource[] discoverResources(String filterExpr) throws InvalidSyntaxException {
398         // TODO(Sahoo): File a bug against Obr to add a suitable method to Repository interface.
399         // We can't use the following method, because we can't rely on the RepositoryAdmin to have the correct
400         // list of repositories. So, we do the discovery ourselves.
401         // return getRepositoryAdmin().discoverResources(query);
402         Filter filter = filterExpr != null ? getRepositoryAdmin().getHelper().filter(filterExpr) : null;
403         Resource[] resources;
404         Repository[] repos = getRepositories();
405         List<Resource> matchList = new ArrayList<Resource>();
406         for (int repoIdx = 0; (repos != null) && (repoIdx < repos.length); repoIdx++) {
407             resources = repos[repoIdx].getResources();
408             for (int resIdx = 0; (resources != null) && (resIdx < resources.length); resIdx++) {
409                 Properties dict = new Properties();
410                 dict.putAll(resources[resIdx].getProperties());
411                 if (filter == null || filter.match(dict)) {
412                     matchList.add(resources[resIdx]);
413                 }
414             }
415         }
416 
417         return matchList.toArray(new Resource[matchList.size()]);
418     }

425     private StringBuffer getResolverOutput(Resolver resolver) {
426         Resource[] addedResources = resolver.getAddedResources();
427         Resource[] requiredResources = resolver.getRequiredResources();
428         Resource[] optionalResources = resolver.getOptionalResources();
429         Reason[] unsatisfiedRequirements = resolver.getUnsatisfiedRequirements();
430         StringBuffer sb = new StringBuffer("Added resources: [");
431         for (Resource r : addedResources) {
432             sb.append("\n").append(r.getSymbolicName()).append(", ").append(r.getVersion()).append(", ").append(r.getURI());
433         }
434         sb.append("]\nRequired Resources: [");
435         for (Resource r : requiredResources) {
436             sb.append("\n").append(r.getURI());
437         }
438         String optionalRequirementsDeployed = deployOptionalRequirements ? "deployed" : "not deployed";
439         sb.append("]\nOptional resources (" + optionalRequirementsDeployed + "): [");
440         for (Resource r : optionalResources) {
441             sb.append("\n").append(r.getURI());
442         }
443         sb.append("]\nUnsatisfied requirements: [");
444         for (Reason r : unsatisfiedRequirements) {
445             sb.append("\n").append(r.getRequirement());
446         }
447         sb.append("]");
448         return sb;
449     }
