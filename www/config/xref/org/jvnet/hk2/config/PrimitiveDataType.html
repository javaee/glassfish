6   package org.jvnet.hk2.config;
7   
8   import org.jvnet.hk2.annotations.Service;
9   
10  /*** Represents a Java primitive (and its wrapper) data type. Not all Java primitives
11   *  are relevant from a configuration standpoint.
12   * @see DataType
13   * @see DomDocument#PRIMS
14   * @author &#2325;&#2375;&#2342;&#2366;&#2352 (km@dev.java.net)
15   * @since hk2 0.3.10
16   */
17  @Service
18  public final class PrimitiveDataType implements DataType {
19  
20      private final String realType;
21      PrimitiveDataType(String realType) {
22          assert DomDocument.PRIMS.contains(realType) : "This class can't validate: " + realType;
23          this.realType = realType;
24      }
25      
26      public void validate(String value) throws ValidationException {
27          if (value.startsWith("${") && value.endsWith("}")) //it's a token
28            return;
29          boolean match = false;
30          if ("int".equals(realType) || "java.lang.Integer".equals(realType))
31              match = representsInteger(value);
32          else if ("boolean".equals(realType) || "java.lang.Boolean".endsWith(realType))
33              match = representsBoolean(value);
34          else if ("char".equals(realType) || "java.lang.Character".equals(realType))
35              representsChar(value);
36          //no need for else as we are asserting it in the constructor
37          if (!match) {
38              String msg = "This value: " + value + " is not of type: " + realType + ", validation failed";
39              throw new ValidationException(msg);            
40          }
41      }
42      
43      private static boolean representsBoolean(String value) {
44          boolean isBoolean = "true".equalsIgnoreCase(value) || "false".equalsIgnoreCase(value);
45          return (isBoolean);
46      }
47      private static boolean representsChar(String value) {
48              if (value.length() == 1)
49                  return true;
50              return false;
51      }
52      private static boolean representsInteger(String value) {
53          try {
54              Integer.parseInt(value);
55              return true;
56          } catch(NumberFormatException ne) {
57              return false;
58          }
59      }
