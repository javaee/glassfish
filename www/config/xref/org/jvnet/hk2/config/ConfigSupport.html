49  /***
50    * <p>
51    * Helper class to execute some code on configuration objects while taking
52    * care of the transaction boiler plate code.
53    * </p>
54    * <p>
55    * Programmers that wish to apply some changes to configuration objects
56    * can use these convenience methods to reduce the complexity of handling
57    * transactions.
58    * </p>
59    * <p>
60    * For instance, say a programmer need to change the HttpListener port from
61    * 8080 to 8989, it just needs to do :
62    * </p>
63    * <pre>
64    *     ... in his code somewhere ...
65    *     HttpListener httpListener = domain.get...
66    *
67    *     // If the programmer tries to modify the httpListener directly
68    *     // it will get an exception
69    *     httpListener.setPort("8989"); // will generate a PropertyVetoException
70    *
71    *     // instead he needs to use a transaction and can use the helper services
72    *     ConfigSupport.apply(new SingleConfigCode<HttpListener>() {
73    *         public Object run(HttpListener okToChange) throws PropertyException {
74    *             okToChange.setPort("8989"); // good...
75    *             httpListener.setPort("7878"); // not good, exceptions still raised...
76    *             return null;
77    *         });
78    *
79    *     // Note that after this code
80    *     System.out.println("Port is " + httpListener.getPort());
81    *     // will display 8989
82    * }
83    * </pre>
84    * @author Jerome Dochez
85    */
86  @Service

90       * Executeï¿½ some logic on one config bean of type T protected by a transaction
91       *
92       * @param code code to execute
93       * @param param config object participating in the transaction
94       * @return list of events that represents the modified config elements.
95       * @throws TransactionFailure when code did not run successfully
96       */
97      public static <T extends ConfigBeanProxy> Object apply(final SingleConfigCode<T> code, T param)
98          throws TransactionFailure {
99          
100         ConfigBeanProxy[] objects = { param };
101         return apply((new ConfigCode() {
102             @SuppressWarnings("unchecked")
103             public Object run(ConfigBeanProxy... objects) throws PropertyVetoException, TransactionFailure {
104                 return code.run((T) objects[0]);
105             }
106         }), objects);
107     }

118          if (parent==null) {
119              throw new IllegalArgumentException("parent cannot be null");
120          }

125              throw new TransactionFailure("Must use a locked parent config object for instantiating new config object", e);

131     /***
132      * Executes some logic on some config beans protected by a transaction.
133      *
134      * @param code code to execute
135      * @param objects config beans participating to the transaction
136      * @return list of property change events
137      * @throws TransactionFailure when the code did run successfully due to a
138      * transaction exception
139      */
140     public static Object apply(ConfigCode code, ConfigBeanProxy... objects)
141             throws TransactionFailure {
142         
143         // the fools think they operate on the "real" object while I am
144         // feeding them with writeable view. Only if the transaction succeed
145         // will I apply the "changes" to the real ones.
146         WriteableView[] views = new WriteableView[objects.length];
147 
148         ConfigBeanProxy[] proxies = new ConfigBeanProxy[objects.length];
149 
150         // create writeable views.
151         for (int i=0;i<objects.length;i++) {
152             proxies[i] = getWriteableView(objects[i]);
153             views[i] = (WriteableView) Proxy.getInvocationHandler(proxies[i]);
154         }
155 
156         // Of course I am not locking the live objects but the writable views.
157         // if the user try to massage the real objects, he will get
158         // a well deserved nasty exception
159         Transaction t = new Transaction();
160         for (WriteableView view : views) {
161             if (!view.join(t)) {
162                 t.rollback();
163                 throw new TransactionFailure("Cannot enlist " + view.getMasterView().getProxyType()
164                     + " in transaction", null);
165             }
166         }
167         
168         try {
169             final Object toReturn = code.run(proxies);
170             try {
171                 t.commit();
172                 if (toReturn instanceof WriteableView) {
173                     return ((WriteableView) toReturn).getMasterView();
174                 } else {
175                     return toReturn;
176                 }
177             } catch (RetryableException e) {
178                 System.out.println("Retryable...");
179                 // TODO : do something meaninful here
180                 t.rollback();
181                 return null;
182             } catch (TransactionFailure e) {
183                 t.rollback();
184                 throw e;
185             }
186 
187         } catch(TransactionFailure e) {
188             t.rollback();
189             throw e;
190         } catch (Exception e) {
191             t.rollback();
192             throw new TransactionFailure(e.getMessage(), e);
193         }
194 
195     }

197     static <T extends ConfigBeanProxy> WriteableView getWriteableView(T s, ConfigBean sourceBean)
198         throws TransactionFailure {
199 
200         WriteableView f = new WriteableView(s);
201         if (sourceBean.getLock().tryLock()) {
202             return f;
203         }
204         throw new TransactionFailure("Config bean already locked " + sourceBean, null);
205     }

256     /***
257      * sort events and dispatch the changes. There will be only one notification of event
258      * per event type, per object, meaning that if an object has had 3 attributes changes, the
259      * Changed interface implementation will get notified only once.
260      *
261      * @param events of events that resulted of a successful configuration transaction
262      * @param target the intended receiver of the changes notification
263      * @param logger to log any issues.
264      */
265     public static UnprocessedChangeEvents sortAndDispatch(PropertyChangeEvent[] events, Changed target, Logger logger) {
266         List<UnprocessedChangeEvent> unprocessed = new ArrayList<UnprocessedChangeEvent>();
267         List<Dom> added = new ArrayList<Dom>();
268         List<Dom> changed = new ArrayList<Dom>();
269 
270         for (PropertyChangeEvent event : events) {
271 
272             if (event.getOldValue()==null && event.getNewValue() instanceof ConfigBeanProxy) {
273                 // something was added
274                 try {
275                     final ConfigBeanProxy proxy =  ConfigBeanProxy.class.cast(event.getNewValue());
276                     added.add(Dom.unwrap(proxy));
277                     final NotProcessed nc = target.changed(Changed.TYPE.ADD, proxyType(proxy), proxy);
278                     if ( nc != null ) {
279                         unprocessed.add( new UnprocessedChangeEvent(event, nc.getReason() ) );
280                     }
281                 } catch (Exception e) {
282                     logger.log(Level.SEVERE, "Exception while processing config bean changes : ", e);
283                 }
284             }
285         }
286 
287         for (PropertyChangeEvent event : events) {
288 
289             try {
290                 Dom eventSource = Dom.unwrap((ConfigBeanProxy) event.getSource());
291                 if (added.contains(eventSource)) {
292                     // we don't really send the changed events for new comers.
293                     continue;
294                 }
295                 ConfigBeanProxy proxy =  null;
296                 if (event.getNewValue()==null) {
297                     try {
298                         // getOldValue() can be null, we will notify a CHANGE event
299                         proxy = ConfigBeanProxy.class.cast(event.getOldValue());
300                     } catch (ClassCastException e) {
301                         // this is ok, the old value was probably a string or something like this...
302                         // we will notify the event.getSource() that it changed.
303                     }
304                     // new value is null, but not old value, we removed something
305                     if (proxy!=null) {
306                         final NotProcessed nc = target.changed(Changed.TYPE.REMOVE, proxyType(proxy), proxy );
307                         if ( nc != null ) {
308                             unprocessed.add( new UnprocessedChangeEvent(event, nc.getReason() ) );
309                         }
310                         continue;
311                     }
312                 }
313                 // we fall back in this case, the newValue was nullm the old value was also null  or was not a ConfigBean,
314                 // we revert to just notify of the change.
315                 // when the new value is not null, we also send the CHANGE  on the source all the time, unless this was
316                 // and added config bean.
317                 if (!changed.contains(eventSource)) {
318                     proxy =  ConfigBeanProxy.class.cast(event.getSource());
319                     changed.add(eventSource);
320                     final NotProcessed nc = target.changed(Changed.TYPE.CHANGE, proxyType(proxy), proxy);
321                     if ( nc != null ) {
322                         unprocessed.add( new UnprocessedChangeEvent(event, nc.getReason() ) );
323                     }
324                 }
325             } catch (Exception e) {
326                 logger.log(Level.SEVERE, "Exception while processing config bean changes : ", e);
327             }
328         }
329         
330         return new UnprocessedChangeEvents( unprocessed );
331     }

338     public static void apply(Map<ConfigBean, Map<String, String>> mapOfChanges) throws TransactionFailure {
339 
340         Transaction t = new Transaction();
341         for (Map.Entry<ConfigBean, Map<String, String>> configBeanChange : mapOfChanges.entrySet()) {
342 
343             ConfigBean source = configBeanChange.getKey();
344             ConfigBeanProxy readableView = source.getProxy(source.getProxyType());
345             WriteableView writeable = ConfigSupport.getWriteableView(readableView, source);
346             if (!writeable.join(t)) {
347                 t.rollback();
348                 throw new TransactionFailure("Cannot enlist " + source.getProxyType() + " in transaction",null);
349             }
350             for (Map.Entry<String, String> change : configBeanChange.getValue().entrySet()) {
351                 String xmlName = change.getKey();
352                 ConfigModel.Property prop = writeable.getProperty(xmlName);
353                 if (prop==null) {
354                     throw new TransactionFailure("Unknown property name " + xmlName + " on " + source.getProxyType(), null);
355                 }
356                 if (prop.isCollection()) {
357                     try {
358                         List<String> values = (List<String>) writeable.getter(prop,
359                                 ConfigSupport.class.getDeclaredMethod("defaultPropertyValue", null).getGenericReturnType());
360                         values.add(change.getValue());                        
361                     } catch (NoSuchMethodException e) {
362                         throw new TransactionFailure("Unknown property name " + xmlName + " on " + source.getProxyType(), null);                        
363                     }
364                 } else {
365                     writeable.setter(prop, change.getValue(), String.class);
366                 }
367             }
368         }
369         try {
370             t.commit();
371         } catch (RetryableException e) {
372             System.out.println("Retryable...");
373             // TODO : do something meaninful here
374             t.rollback();
375             throw new TransactionFailure(e.getMessage(), e);
376         } catch (TransactionFailure e) {
377             System.out.println("failure, not retryable...");
378             t.rollback();
379             throw e;
380         }
381     }

383     /***
384      * Returns the list of sub-elements supported by a ConfigBean
385      * @return array of classes reprensenting the sub elements of a particular
386      * @throws ClassNotFoundException for severe errors with the model associated
387      * with the passed config bean.
388      */
389     public static Class<?>[] getSubElementsTypes(ConfigBean bean)
390         throws ClassNotFoundException {
391 
392         List<Class<?>> subTypes = new ArrayList<Class<?>>();
393         for (ConfigModel.Property element : bean.model.elements.values()) {
394             if (!element.isLeaf()) {
395                 ConfigModel elementModel =  ((ConfigModel.Node) element).model;
396                 Class<?> subType = elementModel.classLoaderHolder.get().loadClass(elementModel.targetTypeName);
397                 subTypes.add(subType);
398             } else {
399                 if (element.isCollection()) {
400                     subTypes.add(List.class);
401                 }
402             }
403         }
404         return subTypes.toArray(new Class[subTypes.size()]);
405     }

467     /***
468      * Creates a new child of the passed child and add it to the parent's live
469      * list of elements. The child is also initialized with the attributes passed
470      * where each key represent the xml property name for the attribute and the value
471      * represent the attribute's value.
472      *
473      * This code will be executed within a Transaction and can therefore throw
474      * a TransactionFailure when the creation or settings of attributes failed.
475      *
476      * Example creating a new http-listener element under http-service
477      *      ConfigBean httpService = ... // got it from somwhere.
478      *      Map<String, String> attributes = new HashMap<String, String>();
479      *      attributes.put("id", "jerome-listener");
480      *      attributes.put("enabled", "true");
481      *      ConfigSupport.createAndSet(httpService, HttpListener.class, attributes);
482      *
483      * @param parent parent config bean to which the child will be added.
484      * @param childType child type
485      * @param attributes list of attribute changes to apply to the newly created child
486      * @param runnable code that will be invoked as part of the transaction to add
487      * more attributes or elements to the newly create type
488      * @throws TransactionFailure if the creation or attribute settings failed
489      */
490     public static ConfigBean createAndSet(
491                 final ConfigBean parent,
492                 final Class<? extends ConfigBeanProxy> childType,
493                 final List<AttributeChanges> attributes,
494                 final TransactionCallBack<WriteableView> runnable)
495         throws TransactionFailure {
496 
497         ConfigBeanProxy readableView = parent.getProxy(parent.getProxyType());
498         ConfigBeanProxy readableChild = (ConfigBeanProxy)
499                 ConfigSupport.apply(new SingleConfigCode<ConfigBeanProxy>() {
500             /***
501              * Runs the following command passing the configration object. The code will be run
502              * within a transaction, returning true will commit the transaction, false will abort
503              * it.
504              *
505              * @param param is the configuration object protected by the transaction
506              * @return any object that should be returned from within the transaction code
507              * @throws java.beans.PropertyVetoException
508              *          if the changes cannot be applied
509              *          to the configuration
510              */
511             public Object run(ConfigBeanProxy param) throws PropertyVetoException, TransactionFailure {
512 
513                 // create the child
514                 ConfigBeanProxy child = ConfigSupport.createChildOf(param, childType);
515 
516                 // add the child to the parent.
517                 WriteableView writeableParent = (WriteableView) Proxy.getInvocationHandler(param);
518                 Class parentProxyType = parent.getProxyType();
519 
520                 Class<?> targetClass = null;
521                 // first we need to find the element associated with this type
522                 ConfigModel.Property element = null;
523                 for (ConfigModel.Property e : parent.model.elements.values()) {
524                     if (e.isLeaf()) {
525                         continue;
526                     }
527                     ConfigModel elementModel =  ((ConfigModel.Node) e).model;
528 
529                     if (Logger.getAnonymousLogger().isLoggable(Level.FINE)) {
530                         Logger.getAnonymousLogger().fine( "elementModel.targetTypeName = " + elementModel.targetTypeName +
531                             ", collection: " + e.isCollection() + ", childType.getName() = " + childType.getName() );
532                     }
533                             
534                     if (elementModel.targetTypeName.equals(childType.getName())) {
535                         element = e;
536                         break;
537                     }
538                     else if ( e.isCollection() ) {
539                         try {
540                             final Class<?> tempClass = childType.getClassLoader().loadClass( elementModel.targetTypeName);
541                             if ( tempClass.isAssignableFrom( childType ) ) {
542                                 element = e;
543                                 targetClass = tempClass;
544                                 break;
545                             }
546                         } catch (Exception ex ) { 
547                             throw new TransactionFailure("EXCEPTION getting class for " + elementModel.targetTypeName, ex);
548                         }
549                     }
550                 }
551                 
552                 // now depending whether this is a collection or a single leaf,
553                 // we need to process this setting differently
554                 if (element != null) {
555                     if (element.isCollection()) {
556                         // this is kind of nasty, I have to find the method that returns the collection
557                         // object because trying to do a element.get without having the parametized List
558                         // type will not work.
559                         for (Method m : parentProxyType.getMethods()) {
560                             final Class returnType = m.getReturnType();
561                             if (Collection.class.isAssignableFrom(returnType)) {
562                                 // this could be it...
563                                 if (!(m.getGenericReturnType() instanceof ParameterizedType))
564                                     throw new IllegalArgumentException("List needs to be parameterized");
565                                 final Class itemType = Types.erasure(Types.getTypeArgument(m.getGenericReturnType(), 0));
566                                 if (itemType.isAssignableFrom(childType)) {
567                                     List list = null;
568                                     try {
569                                         list = (List) m.invoke(param, null);
570                                     } catch (IllegalAccessException e) {
571                                         throw new TransactionFailure("Exception while adding to the parent", e);
572                                     } catch (InvocationTargetException e) {
573                                         throw new TransactionFailure("Exception while adding to the parent", e);
574                                     }
575                                     if (list != null) {
576                                         list.add(child);
577                                         break;
578                                     }
579                                 }
580                             }
581                         }
582                     } else {
583                         // much simpler, I can use the setter directly.
584                         writeableParent.setter(element, child, childType);
585                     }
586                 } else {
587                     throw new TransactionFailure("Parent " + parent.getProxyType() + " does not have a child of type " + childType);
588                 }
589 
590                 WriteableView writeableChild = (WriteableView) Proxy.getInvocationHandler(child);
591                 applyProperties(writeableChild, attributes);
592                 
593                 if (runnable!=null) {
594                     runnable.performOn(writeableChild);
595                 }
596                 return child;
597             }
598         }, readableView);
599         return (ConfigBean) Dom.unwrap(readableChild);
600     }

602     private static void applyProperties(WriteableView target, List<? extends AttributeChanges> changes)
603             throws TransactionFailure {
604 
605         if (changes != null) {
606             for (AttributeChanges change : changes) {
607 
608                 ConfigModel.Property prop = target.getProperty(change.name);
609                 if (prop == null) {
610                     throw new TransactionFailure("Unknown property name " + change.name + " on " + target.getProxyType());
611                 }
612                 if (prop.isCollection()) {
613                     // we need access to the List
614                     try {
615                         List list = (List) target.getter(prop, ConfigSupport.class.getDeclaredMethod("defaultPropertyValue", null).getGenericReturnType());
616                         for (String value : change.values()) {
617                             list.add(value);
618                         }
619                     } catch (NoSuchMethodException e) {
620                         throw new TransactionFailure(e.getMessage(), e);
621                     }
622                 } else {
623                     target.setter(prop, change.values()[0], String.class);
624                 }
625             }
626         }
627 
628     }

670     public static void deleteChild(
671                 final ConfigBean parent,
672                 final ConfigBean child)
673         throws TransactionFailure {
674 
675 
676         ConfigBeanProxy readableView = parent.getProxy(parent.getProxyType());
677         final Class<? extends ConfigBeanProxy> childType = child.getProxyType();
678 
679         ConfigSupport.apply(new SingleConfigCode<ConfigBeanProxy>() {
680 
681             /***
682              * Runs the following command passing the configration object. The code will be run
683              * within a transaction, returning true will commit the transaction, false will abort
684              * it.
685              *
686              * @param param is the configuration object protected by the transaction
687              * @return any object that should be returned from within the transaction code
688              * @throws java.beans.PropertyVetoException
689              *          if the changes cannot be applied
690              *          to the configuration
691              */
692             public Object run(ConfigBeanProxy param) throws PropertyVetoException, TransactionFailure {
693 
694                 // get the child
695                 ConfigBeanProxy childProxy = child.getProxy(childType);
696 
697                 // remove the child from the parent.
698                 WriteableView writeableParent = (WriteableView) Proxy.getInvocationHandler(param);
699                 Class parentProxyType = parent.getProxyType();
700 
701                 // first we need to find the element associated with this type
702                 ConfigModel.Property element = null;
703                 for (ConfigModel.Property e : parent.model.elements.values()) {
704                     ConfigModel elementModel = ((ConfigModel.Node) e).model;
705                     try {
706                         final Class<?> targetClass = parent.model.classLoaderHolder.get().loadClass(elementModel.targetTypeName);
707                         if (targetClass.isAssignableFrom(childType)) {
708                             element = e;
709                             break;
710                         }
711                     } catch(Exception ex) {
712                         // ok.
713                     }
714                 }
715                 // now depending whether this is a collection or a single leaf,
716                 // we need to process this setting differently
717                 if (element != null) {
718                     if (element.isCollection()) {
719                         // this is kind of nasty, I have to find the method that returns the collection
720                         // object because trying to do a element.get without having the parametized List
721                         // type will not work.
722                         for (Method m : parentProxyType.getMethods()) {
723                             final Class returnType = m.getReturnType();
724                             if (Collection.class.isAssignableFrom(returnType)) {
725                                 // this could be it...
726                                 if (!(m.getGenericReturnType() instanceof ParameterizedType))
727                                     throw new IllegalArgumentException("List needs to be parameterized");
728                                 final Class itemType = Types.erasure(Types.getTypeArgument(m.getGenericReturnType(), 0));
729                                 if (itemType.isAssignableFrom(childType)) {
730                                     List list = null;
731                                     try {
732                                         list = (List) m.invoke(param, null);
733                                     } catch (IllegalAccessException e) {
734                                         throw new TransactionFailure("Exception while adding to the parent", e);
735                                     } catch (InvocationTargetException e) {
736                                         throw new TransactionFailure("Exception while adding to the parent", e);
737                                     }
738                                     if (list != null) {
739                                         list.remove(childProxy);
740                                         break;
741                                     }
742                                 }
743                             }
744                         }
745                     } else {
746                         // much simpler, I can use the setter directly.
747                         writeableParent.setter(element, child, childType);
748                     }
749                 } else {
750                     throw new TransactionFailure("Parent " + parent.getProxyType() + " does not have a child of type " + childType);
751                 }
752 
753                 return child;
754             }
755         }, readableView);
756     }
