57  /**
58    * <p>
59    * Helper class to execute some code on configuration objects while taking
60    * care of the transaction boiler plate code.
61    * </p>
62    * <p>
63    * Programmers that wish to apply some changes to configuration objects
64    * can use these convenience methods to reduce the complexity of handling
65    * transactions.
66    * </p>
67    * <p>
68    * For instance, say a programmer need to change the HttpListener port from
69    * 8080 to 8989, it just needs to do :
70    * </p>
71    * <pre>
72    *     ... in his code somewhere ...
73    *     HttpListener httpListener = domain.get...
74    *
75    *     // If the programmer tries to modify the httpListener directly
76    *     // it will get an exception
77    *     httpListener.setPort("8989"); // will generate a PropertyVetoException
78    *
79    *     // instead he needs to use a transaction and can use the helper services
80    *     ConfigSupport.apply(new SingleConfigCode<HttpListener>() {
81    *         public Object run(HttpListener okToChange) throws PropertyException {
82    *             okToChange.setPort("8989"); // good...
83    *             httpListener.setPort("7878"); // not good, exceptions still raised...
84    *             return null;
85    *         });
86    *
87    *     // Note that after this code
88    *     System.out.println("Port is " + httpListener.getPort());
89    *     // will display 8989
90    * }
91    * </pre>
92    * @author Jerome Dochez
93    */
94  @Service
95  @Scoped(Singleton.class)

104      * Execute some logic on one config bean of type T protected by a transaction
105      *
106      * @param code code to execute
107      * @param param config object participating in the transaction
108      * @return list of events that represents the modified config elements.
109      * @throws TransactionFailure when code did not run successfully
110      */
111     public static <T extends ConfigBeanProxy> Object apply(final SingleConfigCode<T> code, T param)
112         throws TransactionFailure {
113         
114         ConfigBeanProxy[] objects = { param };
115         return apply((new ConfigCode() {
116             @SuppressWarnings("unchecked")
117             public Object run(ConfigBeanProxy... objects) throws PropertyVetoException, TransactionFailure {
118                 return code.run((T) objects[0]);
119             }
120         }), objects);
121     }

139     /**
140      * Executes some logic on some config beans protected by a transaction.
141      *
142      * @param code code to execute
143      * @param objects config beans participating to the transaction
144      * @return list of property change events
145      * @throws TransactionFailure when the code did run successfully due to a
146      * transaction exception
147      */    
148     public Object _apply(ConfigCode code, ConfigBeanProxy... objects)
149                 throws TransactionFailure {
150 
151         // the fools think they operate on the "real" object while I am
152         // feeding them with writeable view. Only if the transaction succeed
153         // will I apply the "changes" to the real ones.
154         WriteableView[] views = new WriteableView[objects.length];
155 
156         ConfigBeanProxy[] proxies = new ConfigBeanProxy[objects.length];
157 
158         // create writeable views.
159         for (int i=0;i<objects.length;i++) {
160             proxies[i] = getWriteableView(objects[i]);
161             views[i] = (WriteableView) Proxy.getInvocationHandler(proxies[i]);
162         }
163 
164         // Of course I am not locking the live objects but the writable views.
165         // if the user try to massage the real objects, he will get
166         // a well deserved nasty exception
167         Transaction t = new Transaction();
168         for (WriteableView view : views) {
169             if (!view.join(t)) {
170                 t.rollback();
171                 throw new TransactionFailure("Cannot enlist " + view.getMasterView().getProxyType()
172                     + " in transaction", null);
173             }
174         }
175         
176         try {
177             final Object toReturn = code.run(proxies);
178             try {
179                 t.commit();
180                 if (toReturn instanceof WriteableView) {
181                     return ((WriteableView) toReturn).getMasterView();
182                 } else {
183                     return toReturn;
184                 }
185             } catch (RetryableException e) {
186                 System.out.println("Retryable...");
187                 // TODO : do something meaninful here
188                 t.rollback();
189                 return null;
190             } catch (TransactionFailure e) {
191                 t.rollback();
192                 throw e;
193             }
194 
195         } catch (java.lang.reflect.UndeclaredThrowableException e) {
196             t.rollback();
197             Throwable throwable = e.getCause();
198             if (throwable instanceof PropertyVetoException) throw new TransactionFailure(throwable.toString(), throwable);
199             throw new TransactionFailure(e.toString(), e);
200 
201         } catch(TransactionFailure e) {
202             t.rollback();
203             throw e;
204         } catch (Exception e) {
205             t.rollback();
206             throw new TransactionFailure(e.getMessage(), e);
207         }
208 
209     }

211     static <T extends ConfigBeanProxy> WriteableView getWriteableView(T s, ConfigBean sourceBean)
212         throws TransactionFailure {
213 
214         WriteableView f = new WriteableView(s);
215         try {
216             if (sourceBean.getLock().tryLock(lockTimeOutInSeconds, TimeUnit.SECONDS)) {
217                 return f;
218             }
219         } catch(InterruptedException e) {
220             // ignore, will throw a TransactionFailure exception
221         }
222         throw new TransactionFailure("Config bean already locked " + sourceBean, null);
223     }

229      * @throws TransactionFailure if the object cannot be enrolled (probably already enrolled in

234

266     /**
267      * sort events and dispatch the changes. There will be only one notification of event
268      * per event type, per object, meaning that if an object has had 3 attributes changes, the
269      * Changed interface implementation will get notified only once.
270      *
271      * @param events of events that resulted of a successful configuration transaction
272      * @param target the intended receiver of the changes notification
273      * @param logger to log any issues.
274      */
275     public static UnprocessedChangeEvents sortAndDispatch(PropertyChangeEvent[] events, Changed target, Logger logger) {
276         if ( logger == null ) throw new IllegalArgumentException();
277         
278         List<UnprocessedChangeEvent> unprocessed = new ArrayList<UnprocessedChangeEvent>();
279         List<Dom> added = new ArrayList<Dom>();
280         List<Dom> changed = new ArrayList<Dom>();
281 
282         for (PropertyChangeEvent event : events) {
283 
284             if (event.getOldValue()==null && event.getNewValue() instanceof ConfigBeanProxy) {
285                 // something was added
286                 try {
287                     final ConfigBeanProxy proxy =  ConfigBeanProxy.class.cast(event.getNewValue());
288                     added.add(Dom.unwrap(proxy));
289                     final NotProcessed nc = target.changed(Changed.TYPE.ADD, proxyType(proxy), proxy);
290                     if ( nc != null ) {
291                         unprocessed.add( new UnprocessedChangeEvent(event, nc.getReason() ) );
292                     }
293                 } catch (Exception e) {
294                     logger.log(Level.SEVERE, "Exception while processing config bean changes : ", e);
295                 }
296             }
297         }
298 
299         for (PropertyChangeEvent event : events) {
300 
301             try {
302                 Dom eventSource = Dom.unwrap((ConfigBeanProxy) event.getSource());
303                 if (added.contains(eventSource)) {
304                     // we don't really send the changed events for new comers.
305                     continue;
306                 }
307                 ConfigBeanProxy proxy =  null;
308                 if (event.getNewValue()==null) {
309                     try {
310                         // getOldValue() can be null, we will notify a CHANGE event
311                         proxy = ConfigBeanProxy.class.cast(event.getOldValue());
312                     } catch (ClassCastException e) {
313                         // this is ok, the old value was probably a string or something like this...
314                         // we will notify the event.getSource() that it changed.
315                     }
316                     // new value is null, but not old value, we removed something
317                     if (proxy!=null) {
318                         final NotProcessed nc = target.changed(Changed.TYPE.REMOVE, proxyType(proxy), proxy );
319                         if ( nc != null ) {
320                             unprocessed.add( new UnprocessedChangeEvent(event, nc.getReason() ) );
321                         }
322                         continue;
323                     }
324                 }
325                 // we fall back in this case, the newValue was nullm the old value was also null  or was not a ConfigBean,
326                 // we revert to just notify of the change.
327                 // when the new value is not null, we also send the CHANGE  on the source all the time, unless this was
328                 // and added config bean.
329                 if (!changed.contains(eventSource)) {
330                     proxy =  ConfigBeanProxy.class.cast(event.getSource());
331                     changed.add(eventSource);
332                     final NotProcessed nc = target.changed(Changed.TYPE.CHANGE, proxyType(proxy), proxy);
333                     if ( nc != null ) {
334                         unprocessed.add( new UnprocessedChangeEvent(event, nc.getReason() ) );
335                     }
336                 }
337             } catch (Exception e) {
338                 logger.log(Level.SEVERE, "Exception while processing config bean changes : ", e);
339             }
340         }
341         
342         return new UnprocessedChangeEvents( unprocessed );
343     }

350     public void apply(Map<ConfigBean, Map<String, String>> mapOfChanges) throws TransactionFailure {
351 
352 
353         Transaction t = new Transaction();
354         try {
355             for (Map.Entry<ConfigBean, Map<String, String>> configBeanChange : mapOfChanges.entrySet()) {
356 
357                 ConfigBean source = configBeanChange.getKey();
358                 ConfigBeanProxy readableView = source.getProxy(source.getProxyType());
359                 WriteableView writeable = getWriteableView(readableView, source);
360                 if (!writeable.join(t)) {
361                     t.rollback();
362                     throw new TransactionFailure("Cannot enlist " + source.getProxyType() + " in transaction", null);
363                 }
364                 for (Map.Entry<String, String> change : configBeanChange.getValue().entrySet()) {
365                     String xmlName = change.getKey();
366                     ConfigModel.Property prop = writeable.getProperty(xmlName);
367                     if (prop == null) {
368                         throw new TransactionFailure("Unknown property name " + xmlName + " on " + source.getProxyType(), null);
369                     }
370                     if (prop.isCollection()) {
371                         try {
372                             List<String> values = (List<String>) writeable.getter(prop,
373                                     ConfigSupport.class.getDeclaredMethod("defaultPropertyValue", null).getGenericReturnType());
374                             values.add(change.getValue());
375                         } catch (NoSuchMethodException e) {
376                             throw new TransactionFailure("Unknown property name " + xmlName + " on " + source.getProxyType(), null);
377                         }
378                     } else {
379                         writeable.setter(prop, change.getValue(), String.class);
380                     }
381                 }
382             }
383         } catch(TransactionFailure e) {
384             t.rollback();
385             throw e;
386         } catch (Exception e) {
387             t.rollback();
388             throw new TransactionFailure(e.getMessage(), e);
389         }
390         try {
391             t.commit();
392         } catch (RetryableException e) {
393             System.out.println("Retryable...");
394             // TODO : do something meaninful here
395             t.rollback();
396             throw new TransactionFailure(e.getMessage(), e);
397         } catch (TransactionFailure e) {
398             System.out.println("failure, not retryable...");
399             t.rollback();
400             throw e;
401         }
402     }

404     /**
405      * Returns the list of sub-elements supported by a ConfigBean
406      * @return array of classes reprensenting the sub elements of a particular
407      * @throws ClassNotFoundException for severe errors with the model associated
408      * with the passed config bean.
409      */
410     public static Class<?>[] getSubElementsTypes(ConfigBean bean)
411         throws ClassNotFoundException {
412 
413         List<Class<?>> subTypes = new ArrayList<Class<?>>();
414         for (ConfigModel.Property element : bean.model.elements.values()) {
415             if (!element.isLeaf()) {
416                 ConfigModel elementModel =  ((ConfigModel.Node) element).model;
417                 Class<?> subType = elementModel.classLoaderHolder.get().loadClass(elementModel.targetTypeName);
418                 subTypes.add(subType);
419             } else {
420                 if (element.isCollection()) {
421                     subTypes.add(List.class);
422                 }
423             }
424         }
425         return subTypes.toArray(new Class[subTypes.size()]);
426     }

522     ConfigBean _createAndSet(
523                     final ConfigBean parent,
524                     final Class<? extends ConfigBeanProxy> childType,
525                     final List<AttributeChanges> attributes,
526                     final TransactionCallBack<WriteableView> runnable)
527             throws TransactionFailure {
528 
529         ConfigBeanProxy readableView = parent.getProxy(parent.getProxyType());
530         ConfigBeanProxy readableChild = (ConfigBeanProxy)
531                 apply(new SingleConfigCode<ConfigBeanProxy>() {
532             /**
533              * Runs the following command passing the configration object. The code will be run
534              * within a transaction, returning true will commit the transaction, false will abort
535              * it.
536              *
537              * @param param is the configuration object protected by the transaction
538              * @return any object that should be returned from within the transaction code
539              * @throws java.beans.PropertyVetoException
540              *          if the changes cannot be applied
541              *          to the configuration
542              */
543             public Object run(ConfigBeanProxy param) throws PropertyVetoException, TransactionFailure {
544 
545                 // create the child
546                 ConfigBeanProxy child = param.createChild(childType);
547                 Dom dom = Dom.unwrap(child);
548 
549                 // add the child to the parent.
550                 WriteableView writeableParent = (WriteableView) Proxy.getInvocationHandler(param);
551                 Class parentProxyType = parent.getProxyType();
552 
553                 Class<?> targetClass = null;
554                 // first we need to find the element associated with this type
555                 ConfigModel.Property element = null;
556                 for (ConfigModel.Property e : parent.model.elements.values()) {
557                     if (e.isLeaf()) {
558                         continue;
559                     }
560                     ConfigModel elementModel =  ((ConfigModel.Node) e).model;
561 
562                     if (Logger.getAnonymousLogger().isLoggable(Level.FINE)) {
563                         Logger.getAnonymousLogger().fine( "elementModel.targetTypeName = " + elementModel.targetTypeName +
564                             ", collection: " + e.isCollection() + ", childType.getName() = " + childType.getName() );
565                     }
566                             
567                     if (elementModel.targetTypeName.equals(childType.getName())) {
568                         element = e;
569                         break;
570                     }
571                     else if ( e.isCollection() ) {
572                         try {
573                             final Class<?> tempClass = elementModel.classLoaderHolder.get().loadClass(elementModel.targetTypeName);
574                             if ( tempClass.isAssignableFrom( childType ) ) {
575                                 element = e;
576                                 targetClass = tempClass;
577                                 break;
578                             }
579                         } catch (Exception ex ) { 
580                             throw new TransactionFailure("EXCEPTION getting class for " + elementModel.targetTypeName, ex);
581                         }
582                     }
583                 }
584                 
585                 // now depending whether this is a collection or a single leaf,
586                 // we need to process this setting differently
587                 if (element != null) {
588                     if (element.isCollection()) {
589                         // this is kind of nasty, I have to find the method that returns the collection
590                         // object because trying to do a element.get without having the parametized List
591                         // type will not work.
592                         for (Method m : parentProxyType.getMethods()) {
593                             final Class returnType = m.getReturnType();
594                             if (Collection.class.isAssignableFrom(returnType)) {
595                                 // this could be it...
596                                 if (!(m.getGenericReturnType() instanceof ParameterizedType))
597                                     throw new IllegalArgumentException("List needs to be parameterized");
598                                 final Class itemType = Types.erasure(Types.getTypeArgument(m.getGenericReturnType(), 0));
599                                 if (itemType.isAssignableFrom(childType)) {
600                                     List list = null;
601                                     try {
602                                         list = (List) m.invoke(param, null);
603                                     } catch (IllegalAccessException e) {
604                                         throw new TransactionFailure("Exception while adding to the parent", e);
605                                     } catch (InvocationTargetException e) {
606                                         throw new TransactionFailure("Exception while adding to the parent", e);
607                                     }
608                                     if (list != null) {
609                                         list.add(child);
610                                         break;
611                                     }
612                                 }
613                             }
614                         }
615                     } else {
616                         // much simpler, I can use the setter directly.
617                         writeableParent.setter(element, dom.<ConfigBeanProxy>createProxy(), childType);
618                     }
619                 } else {
620                     throw new TransactionFailure("Parent " + parent.getProxyType() + " does not have a child of type " + childType);
621                 }
622 
623                 WriteableView writeableChild = (WriteableView) Proxy.getInvocationHandler(child);
624                 applyProperties(writeableChild, attributes);
625 
626                 dom.addDefaultChildren();
627                 dom.register();
628 
629                 if (runnable!=null) {
630                     runnable.performOn(writeableChild);
631                 }
632                 return child;
633             }
634         }, readableView);
635         return (ConfigBean) Dom.unwrap(readableChild);
636     }

638     private void applyProperties(WriteableView target, List<? extends AttributeChanges> changes)
639             throws TransactionFailure {
640 
641         if (changes != null) {
642             for (AttributeChanges change : changes) {
643 
644                 ConfigModel.Property prop = target.getProperty(change.name);
645                 if (prop == null) {
646                     throw new TransactionFailure("Unknown property name " + change.name + " on " + target.getProxyType());
647                 }
648                 if (prop.isCollection()) {
649                     // we need access to the List
650                     try {
651                         List list = (List) target.getter(prop, ConfigSupport.class.getDeclaredMethod("defaultPropertyValue", null).getGenericReturnType());
652                         for (String value : change.values()) {
653                             list.add(value);
654                         }
655                     } catch (NoSuchMethodException e) {
656                         throw new TransactionFailure(e.getMessage(), e);
657                     }
658                 } else {
659                     target.setter(prop, change.values()[0], String.class);
660                 }
661             }
662         }
663 
664     }

739      * Unprotected child deletion, caller must start a transaction before calling this
740      * method.
741      *
742      * @param parent the parent element
743      * @param writeableParent the writeable view of the parent element
744      * @param child the child to delete
745      * @throws TransactionFailure if something goes wrong.
746      */
747     public static void _deleteChild(
748                 final ConfigBean parent,
749                 final WriteableView writeableParent,
750                 final ConfigBean child)
751         throws TransactionFailure {
752 
753 
754         final Class<? extends ConfigBeanProxy> childType = child.getProxyType();
755 
756 
757                 // get the child
758                 ConfigBeanProxy childProxy = child.getProxy(childType);
759 
760                 // get the parent type
761                 Class parentProxyType = parent.getProxyType();
762 
763                 // first we need to find the element associated with this type
764                 ConfigModel.Property element = null;
765                 for (ConfigModel.Property e : parent.model.elements.values()) {
766                     if (!(e instanceof ConfigModel.Node)) {
767                         continue;
768                     }
769 
770                     ConfigModel elementModel = ((ConfigModel.Node) e).model;
771                     try {
772                         final Class<?> targetClass = parent.model.classLoaderHolder.get().loadClass(elementModel.targetTypeName);
773                         if (targetClass.isAssignableFrom(childType)) {
774                             element = e;
775                             break;
776                         }
777                     } catch(Exception ex) {
778                         // ok.
779                     }
780                 }
781                 // now depending whether this is a collection or a single leaf,
782                 // we need to process this setting differently
783                 if (element != null) {
784                     if (element.isCollection()) {
785                         // this is kind of nasty, I have to find the method that returns the collection
786                         // object because trying to do a element.get without having the parametized List
787                         // type will not work.
788                         for (Method m : parentProxyType.getMethods()) {
789                             final Class returnType = m.getReturnType();
790                             if (Collection.class.isAssignableFrom(returnType)) {
791                                 // this could be it...
792                                 if (!(m.getGenericReturnType() instanceof ParameterizedType))
793                                     throw new IllegalArgumentException("List needs to be parameterized");
794                                 final Class itemType = Types.erasure(Types.getTypeArgument(m.getGenericReturnType(), 0));
795                                 if (itemType.isAssignableFrom(childType)) {
796                                     List list = null;
797                                     try {
798                                         list = (List) m.invoke(writeableParent.getProxy(writeableParent.<ConfigBeanProxy>getProxyType()), null);
799                                     } catch (IllegalAccessException e) {
800                                         throw new TransactionFailure("Exception while adding to the parent", e);
801                                     } catch (InvocationTargetException e) {
802                                         throw new TransactionFailure("Exception while adding to the parent", e);
803                                     }
804                                     if (list != null) {
805                                         list.remove(childProxy);
806                                         break;
807                                     }
808                                 }
809                             }
810                         }
811                     } else {
812                         // much simpler, I can use the setter directly.
813                         writeableParent.setter(element, null, childType);
814                     }
815                 } else {
816                     throw new TransactionFailure("Parent " + parent.getProxyType() + " does not have a child of type " + childType);
817                 }
818     }

875         DomDocument document = parentDom.document;

885                 return (Class<? extends ConfigBeanProxy>) childModel.classLoaderHolder.get().loadClass(childModel.targetTypeName);
886             }
887         }
