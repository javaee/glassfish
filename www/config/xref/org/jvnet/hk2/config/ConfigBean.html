80      public ConfigBean(Habitat habitat, DomDocument document, ConfigBean parent, ConfigModel model, XMLStreamReader in) {
81  
82          super(habitat, document, parent, model, in);
83          // by default all ConfigBean support the ConstrainedBeanListener interface
84          // allowing clients to register interest in attributes changing.
85          addInterceptor(ConstrainedBeanListener.class ,new ConfigBeanInterceptor<ConstrainedBeanListener>() {
86  
87              List<VetoableChangeListener> listeners = new ArrayList<VetoableChangeListener>();
88  
89              public ConstrainedBeanListener getConfiguration() {
90                  return new ConstrainedBeanListener() {

92                      public void removeVetoableChangeListener(VetoableChangeListener listener) {
93                          listeners.remove(listener);

96                      public void addVetoableChangeListener(VetoableChangeListener listener) {
97                          listeners.add(listener);
98                      }
99                  };
100             }

103                 for (VetoableChangeListener listener : listeners) {

157     protected void setter(ConfigModel.Property target, Object value) throws Exception  {
158         if (!writeLock) {
159             throw new PropertyVetoException("Instance of " + typeName() + " named '" + getKey() +
160                     "' is not locked for writing when changing attribute " + target.xmlName()
161                     + ", you must use transaction semantics to access it.", null);
162         }
163         _setter(target, value);
164     }
165 
166     void _setter(ConfigModel.Property target, Object value) throws Exception  {    
167         Object oldValue = super.getter(target, value.getClass());
168         PropertyChangeEvent evt = new PropertyChangeEvent(this, target.xmlName(), oldValue, value);
169         for (ConfigBeanInterceptor interceptor : optionalFeatures.values()) {
170             interceptor.beforeChange(evt);
171         }
172         super.setter(target, value);
173         for (ConfigBeanInterceptor interceptor : optionalFeatures.values()) {
174             interceptor.afterChange(evt, System.currentTimeMillis());
175         }
176     }
177 
178     Object _getter(ConfigModel.Property target, Type t) {
179         final Object value = super.getter(target,t);
180         for (ConfigBeanInterceptor interceptor : optionalFeatures.values()) {
181             interceptor.readValue(this, target.xmlName(), value);
182         }
183         return value;        
184     }
185 
186     protected Object getter(ConfigModel.Property target, Type t) {
187         final Object value = _getter(target, t);
188         if (value instanceof List) {
189             final ConfigBean myself = this;
190             final List valueList = (List) value;
191 
192             // we need to protect this list as it was obtained from a readable view...
193             return new AbstractList() {
194 
195                 public int size() {
196                     return valueList.size();
197                 }
198 
199                 public Object get(int index) {
200                     return valueList.get(index);
201                 }
202 
203                 public boolean add(Object o) {
204                     throw new IllegalStateException("Not part of a transaction !", null);
205                 }
206                 public Object set(int index, Object element) {
207                     throw new IllegalStateException("Not part of a transaction !", null);
208                 }
209 
210                 public Object remove(int index) {
211                     throw new IllegalStateException("Not part of a transaction !", null);
212                 }
213             };
214 
215         }
216         return value;
217     }

295      * simplistic non reentrant lock implementation, needs rework

297     final private Lock lock = new Lock() {
298         
299         public void lock() {
300             throw new UnsupportedOperationException();
301         }
302 
303         public void lockInterruptibly() throws InterruptedException {
304             throw new UnsupportedOperationException();
305         }
306 
307         public synchronized boolean tryLock() {
308             if (!writeLock) {
309                 writeLock=true;
310                 return true;
311             }
312             return false;
313         }
314 
315         public boolean tryLock(long time, TimeUnit unit) throws InterruptedException {
316             long nanosTimeout = TimeUnit.NANOSECONDS.convert(time, unit);
317             long increment = nanosTimeout/20;
318             long lastTime = System.nanoTime();
319             for (; ;) {
320                 if (tryLock()) {
321                     return true;
322                 }
323                 if (nanosTimeout < 0) {
324                     return false;
325                 }
326                 LockSupport.parkNanos(increment);
327                 long now = System.nanoTime();
328                 nanosTimeout -= now - lastTime;
329                 lastTime = now;
330                 if (Thread.interrupted())
331                     break;
332             }
333             throw new InterruptedException();
334         }
335 
336         public synchronized void unlock() {
337             writeLock = false;
338         }

342         }
