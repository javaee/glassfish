53  /***
54   * ConfigBean is the core implementation of the config beans. It has features like locking

79      public ConfigBean(Habitat habitat, DomDocument document, Dom parent, ConfigModel model, XMLStreamReader in) {
80  
81          super(habitat, document, parent, model, in);
82          // by default all ConfigBean support the ConstrainedBeanListener interface
83          // allowing clients to register interest in attributes changing.
84          addInterceptor(new ConfigBeanInterceptor<ConstrainedBeanListener>() {
85  
86              List<VetoableChangeListener> listeners = new ArrayList<VetoableChangeListener>();
87  
88              public ConstrainedBeanListener getConfiguration() {
89                  return new ConstrainedBeanListener() {
90  
91                      public void removeVetoableChangeListener(VetoableChangeListener listener) {
92                          listeners.remove(listener);
93                      }
94  
95                      public void addVetoableChangeListener(VetoableChangeListener listener) {
96                          listeners.add(listener);
97                      }
98                  };
99              }
100 
101             public void beforeChange(PropertyChangeEvent evt) throws PropertyVetoException {
102                 for (VetoableChangeListener listener : listeners) {
103                     listener.vetoableChange(evt);
104                 }
105             }
106             public void afterChange(PropertyChangeEvent evt, long timestamp) {
107             }
108 
109             public void readValue(ConfigBean source, String xmlName, Object Value) {
110             }
111         });
112     }

114     /***
115      * Returns an optional feature of the ConfigBean. Optional features are implemented
116      * by other objects and attached to this instance. Attached features can be queried
117      * using the getOptionalFeature method giving the type of the requestion optional
118      * feature.
119      * 
120      * @param featureType type of the optional feature requested.
121      * @return optional feature implementation is one is attached to this instance
122      */
123     @SuppressWarnings("unchecked")    
124     public <T> T getOptionalFeature(Class<T> featureType) {
125         if (optionalFeatures.containsKey(featureType)) {
126             return (T) optionalFeatures.get(featureType);
127         }
128         return null;
129     }

131     protected void setter(ConfigModel.Property target, Object value) throws Exception  {
132         if (!writeLock) {
133             throw new PropertyVetoException("Not part of a transaction !", null);
134         }
135         _setter(target, value);
136     }
137 
138     void _setter(ConfigModel.Property target, Object value) throws Exception  {    
139         Object oldValue = super.getter(target, value.getClass());
140         PropertyChangeEvent evt = new PropertyChangeEvent(this, target.xmlName(), oldValue, value);
141         for (ConfigBeanInterceptor interceptor : optionalFeatures.values()) {
142             interceptor.beforeChange(evt);
143         }
144         super.setter(target, value);
145         for (ConfigBeanInterceptor interceptor : optionalFeatures.values()) {
146             interceptor.afterChange(evt, System.currentTimeMillis());
147         }
148     }
149 
150     Object _getter(ConfigModel.Property target, Type t) {
151         final Object value = super.getter(target,t);
152         for (ConfigBeanInterceptor interceptor : optionalFeatures.values()) {
153             interceptor.readValue(this, target.xmlName(), value);
154         }
155         return value;        
156     }
157 
158     protected Object getter(ConfigModel.Property target, Type t) {
159         final Object value = _getter(target, t);
160         if (value instanceof List) {
161             final ConfigBean myself = this;
162             final List valueList = (List) value;
163 
164             // we need to protect this list as it was obtained from a readable view...
165             return new AbstractList() {
166 
167                 public int size() {
168                     return valueList.size();
169                 }
170 
171                 public Object get(int index) {
172                     return valueList.get(index);
173                 }
174 
175                 public boolean add(Object o) {
176                     throw new IllegalStateException("Not part of a transaction !", null);
177                 }
178                 public Object set(int index, Object element) {
179                     throw new IllegalStateException("Not part of a transaction !", null);
180                 }
181 
182                 public Object remove(int index) {
183                     throw new IllegalStateException("Not part of a transaction !", null);
184                 }
185             };
186 
187         }
188         return value;
189     }
