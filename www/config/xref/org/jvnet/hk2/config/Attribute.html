37  package org.jvnet.hk2.config;

39  import static java.lang.annotation.ElementType.FIELD;
40  import static java.lang.annotation.ElementType.METHOD;
41  import java.lang.annotation.Retention;
42  import static java.lang.annotation.RetentionPolicy.RUNTIME;
43  import java.lang.annotation.Target;
44  
45  /***
46   * Indicates that this property or the field value must be injected from
47   * an XML attribute in a configuration file.
48   *
49   * @author Kohsuke Kawaguchi
50   * @see Element
51   */
52  @Retention(RUNTIME)
53  @Target({FIELD,METHOD})
54  public @interface Attribute {
55      /***
56       * Attribute name.
57       *
58       * <h2>Default Name</h2>
59       * <p>
60       * If this value is omitted, the default name is inferred from the field/method name.
61       * First, if this is a method and the name starts with "set", then "set" will be trimmed
62       * off.
63       *
64       * <p>
65       * Then names are tokenized according to the camel case word separator,
66       * then tokens are combined with '-', then finally the whole thing is converted
67       * to the lower case.
68       *
69       * <p>
70       * Therefore, for example, a field name "httpBufferSize" would yield
71       * "http-buffer-size", and a method name "setThreadCount" would yield
72       * "thread-count" 
73       */
74      String value() default "";
75  
76      /***
77       * Indicates that this property becomes the name of the component.
78       * There can be only one key on a class.
79       */
80      boolean key() default false;
81  
82      /***
83       * Indicates that this attribute is required.
84       *
85       * <p>
86       * To specify the default value, simply use the field initializer
87       * to set it to a certain value. The field/method values are only
88       * set when the value is present.
89       */
90      boolean required() default false;
91  
92      /***
93       * Indicates that this property is a reference to another
94       * configured inhabitant.
95       *
96       * See {@link Element#reference()} for more details of the semantics.
97       *
98       * <p>
99       * When a reference property is a collection/array, then the key values
100      * are separated by ',' with surrounding whitespaces ignored. That is,
101      * it can be things like " foo , bar " (which would mean the same thing as
102      * "foo,bar".)
103      */
104     boolean reference() default false;
105 
106     /***
107      * Indicates that the variable expansion should be performed on this proeprty.
108      *
109      * See {@link Element#variableExpansion()} for more details.
110      */
111     boolean variableExpansion() default true;
112 
113     /*** Specifies the default value of the attribute.
114      * @return default value as String
115      */
116     String defaultValue() default "\u0000";
117         
118     /*** Specifies the data type. It should be the fully qualified name of
119      *  the class that identifies the real data type. For attributes that
120      *  are of type defined by basic Java primitives (or wrappers), there is
121      *  no need to specify this field. The default value is derived from
122      *  method/field declaration.
123      * @see {@link DataType}
124      * @return String specifying the name of the data type for the values of this
125      * attribute
126      */
127     Class dataType() default String.class;
128 }
