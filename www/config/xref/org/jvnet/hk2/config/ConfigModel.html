232     /**
233      * Obtains the duck method implementation from a method on the {@link ConfigBeanProxy}-derived interface.
234      */
235     public Method getDuckMethod(Method method) throws ClassNotFoundException, NoSuchMethodException {
236         synchronized (duckMethods) {
237             Method duckMethod = duckMethods.get(method);
238             if(duckMethod!=null)    return duckMethod;
239 
240             Class<?> clz = method.getDeclaringClass();
241             Class<?> duck = clz.getClassLoader().loadClass(clz.getName() + "$Duck");
242 
243             Class<?>[] types = method.getParameterTypes();
244             Class[] paramTypes = new Class[types.length+1];
245             System.arraycopy(types,0,paramTypes,1,types.length);
246             paramTypes[0] = clz;
247             duckMethod = duck.getMethod(method.getName(), paramTypes);
248             duckMethods.put(method,duckMethod);
249 
250             return duckMethod;
251         }
252     }

254     /**
255      * Obtain XML names (like "abc-def") from strings like "getAbcDef" and "hasAbcDef".
256      * <p>
257      * The conversion rule uses the model to find a good match.
258      */
259     public ConfigModel.Property toProperty(Method method) {
260         String name = method.getName();
261 
262         // check annotations first
263         Element e = method.getAnnotation(Element.class);
264         if(e!=null) {
265             String en = e.value();
266             if(en.length()>0)
267                 return elements.get(en);
268         }
269         Attribute a = method.getAnnotation(Attribute.class);
270         if(a!=null) {
271             String an = a.value();
272             if(an.length()>0)
273                 return attributes.get(an);
274         }
275         // TODO: check annotations on the getter/setter
276 
277         // first, trim off the prefix
278         for (String p : Dom.PROPERTY_PREFIX) {
279             if(name.startsWith(p)) {
280                 name = name.substring(p.length());
281                 break;
282             }
283         }
284 
285         name = camelCaseToXML(name);
286 
287         // at this point name should match XML names in the model, modulo case.
288         return findIgnoreCase(name);
289     }

291     public String trimPrefix(String name) {
292 
293         // first, trim off the prefix
294         for (String p : Dom.PROPERTY_PREFIX) {
295             if(name.startsWith(p)) {
296                 name = name.substring(p.length());
297                 break;
298             }
299         }
300         // lowecase first letter.
301         return name.toLowerCase().charAt(0) + name.substring(1);
302 
303     }

305     public String camelCaseToXML(String camelCase) {
306         
307         // tokenize by finding 'x|X' and 'X|Xx' then insert '-'.
308         StringBuilder buf = new StringBuilder(camelCase.length()+5);
309         for(String t : Dom.TOKENIZER.split(camelCase)) {
310             if(buf.length()>0)  buf.append('-');
311             buf.append(t.toLowerCase());
312         }
313         return buf.toString();
314     }

370         final ConfigModel model;
371         public Node(ConfigModel model, String xmlName) {
372             super(xmlName);

377             return false;

389         /**
390          * Coerce the given type to {@link Dom}.
391          * Only handles those types that are valid to the {@link #set(Dom, Object)} method.
392          */
393         protected final Dom toDom(Object arg) {
394             if(arg==null)
395                 return null;
396             if(arg instanceof Dom)
397                 return (Dom)arg;
398             if(arg instanceof ConfigBeanProxy)
399                 return Dom.unwrap((ConfigBeanProxy)arg);
400             throw new IllegalArgumentException("Unexpected type "+arg.getClass()+" for "+xmlName);
401         }

404     static final class CollectionNode extends Node {
405         CollectionNode(ConfigModel model, String xmlName) {
406             super(model, xmlName);
407         }
408 
409         public boolean isCollection() {
410             return true;
411         }
412 
413         public Object get(final Dom dom, Type returnType) {
414             // TODO: perhaps support more collection types?
415 
416 
417             if(!(returnType instanceof ParameterizedType))
418                 throw new IllegalArgumentException("List needs to be parameterized");
419             final Class itemType = Types.erasure(Types.getTypeArgument(returnType,0));
420 
421             final List<Dom> v = ("*".equals(xmlName)?dom.domNodeByTypeElements(itemType):dom.nodeElements(xmlName));
422 
423             if(itemType==Dom.class)
424                 // TODO: this returns a view, not a live list
425                 return v;
426             if(ConfigBeanProxy.class.isAssignableFrom(itemType)) {
427                 // return a live list
428                 return new AbstractList<Object>() {
429                     public Object get(int index) {
430                         return v.get(index).createProxy();
431                     }
432 
433                     public void add(int index, Object element) {
434                         // update the master children list, as well as this view 'v'
435                         Dom child = Dom.unwrap((ConfigBeanProxy) element);
436                         dom.insertAfter( index==0 ? null : v.get(index-1), xmlName, child);
437                         v.add(index,child);
438                     }
439 
440                     public Object remove(int index) {
441                         Dom child = v.get(index);
442                         dom.removeChild(child);
443                         v.remove(index);
444                         return child.createProxy();
445                     }
446 
447                     public Object set(int index, Object element) {
448                         Dom child = Dom.unwrap((ConfigBeanProxy) element);
449                         dom.replaceChild( v.get(index), xmlName, child );
450                         return v.set(index,child).createProxy();
451                     }
452 
453                     public int size() {
454                         return v.size();
455                     }
456                 };
457             }
458 
459             // TODO: error check needs to be improved,
460             // as itemType might be inconsistent with the actual type
461             return new AbstractList() {
462                 public Object get(int index) {
463                     return v.get(index).get();
464                 }
465 
466                 public int size() {
467                     return v.size();
468                 }
469             };
470         }
471 
472         public void set(Dom dom, Object _arg) {
473             if(!(_arg instanceof List))
474                 throw new IllegalArgumentException("Expecting a list but found "+_arg);
475             List arg = (List)_arg;
476 
477             Dom[] values = new Dom[arg.size()];
478             int i=0;
479             for (Object o : arg)
480                 values[i++] = toDom(o);
481 
482             dom.setNodeElements(xmlName,values);
483         }
484     }

528         /**
529          * Converts a single value from string to the specified target type.
530          *
531          * @return
532          *      Instance of the given 'returnType'
533          */
534         protected Object convertLeafValue(Dom parent, Class<?> returnType, String v) {
535             if(v==null)
536                 // TODO: default value handling
537                 // TODO: if primitive types, report an error
538                 return null;
539 
540             if(returnType==String.class) {
541                 return v;
542             }
543             if(returnType==Integer.class || returnType==int.class) {
544                 return Integer.valueOf(v);
545             }
546             if(returnType==Boolean.class || returnType==boolean.class) {
547                 return BOOLEAN_TRUE.contains(v);
548             }
549             throw new IllegalArgumentException("Don't know how to handle "+returnType);
550         }
551 
552         private static final Set<String> BOOLEAN_TRUE = new HashSet<String>(Arrays.asList("true","yes","on","1"));

556         CollectionLeaf(String xmlName) {
557             super(xmlName);
558         }
559 
560         public boolean isCollection() {
561             return true;
562         }
563 
564         String elementValue(Object element) {
565             return element.toString();
566         }

568         public Object get(final Dom dom, Type returnType) {
569             // TODO: perhaps support more collection types?
570             final List<String> v = dom.leafElements(xmlName);
571             if(!(returnType instanceof ParameterizedType))
572                 throw new IllegalArgumentException("List needs to be parameterized");
573             final Class itemType = Types.erasure(Types.getTypeArgument(returnType,0));
574 
575             // return a live list
576             return new AbstractList<Object>() {
577                 public Object get(int index) {
578                     return convertLeafValue(dom, itemType, v.get(index));
579                 }
580 
581                 public void add(int index, Object element) {
582                     // update the master children list, as well as this view 'v'
583                     dom.addLeafElement(xmlName, elementValue(element));
584                     v.add(index, element.toString());
585                 }
586 
587                 public Object remove(int index) {
588                     dom.removeLeafElement(xmlName, v.get(index));
589                     return v.remove(index);
590                 }
591 
592                 public Object set(int index, Object element) {
593                     dom.changeLeafElement(xmlName, v.get(index), elementValue(element));
594                     return v.set(index, element.toString());
595                 }
596 
597                 public int size() {
598                     return v.size();
599                 }
600             };
601         }
602 
603         public void set(Dom dom, Object arg) {
604             if (arg instanceof List) {
605                 String[] strings = new String[((List) arg).size()];
606                 dom.setLeafElements(xmlName, (String[]) ((List)arg).toArray(strings));
607             } else {//TODO -- I hope this is OK for now (km@dev.java.net 25 Mar 2008)
608                 throw new UnsupportedOperationException();
609             }
610         }

613     final class ReferenceCollectionLeaf extends CollectionLeaf {
614 
615         ReferenceCollectionLeaf(String xmlName) {
616             super(xmlName);
617         }
618         protected Object convertLeafValue(Dom parent, Class<?> returnType, String v) {
619             // let's look first the fast way.
620             Object candidate = parent.getHabitat().getComponent(returnType, v);
621             if (candidate!=null) {
622                 return returnType.cast(candidate);
623             }
624 
625             parent = parent.getSymbolSpaceRoot(v);
626             Dom dom = parent==null?null:parent.resolveReference(v,returnType.getName());
627             if (dom!=null) {
628                 return returnType.cast(dom.get());
629             } else {
630                 throw new IllegalArgumentException("Cannot find an instance of " + returnType + " named " + v);
631             }
632         }
633 
634         @Override
635         String elementValue(Object element) {
636             return Dom.unwrap((ConfigBeanProxy) element).getKey();
637         }

734         }

899     private Property parseValue(String elementName, DomDocument document, List<String> values) {
900         String value = values.size()>0 ? values.get(0) : null;
901         if (value==null)
902             return null;
903 
904 
905 
906         boolean collection = false;
907         if(value.startsWith("collection:")) {
908             collection = true;
909             value = value.substring(11);
910         }
911 
912         boolean reference = values.contains("reference");
913         Property prop;
914         if(value.equals("leaf")) {
915             prop = collection?reference?new ReferenceCollectionLeaf(elementName):new CollectionLeaf(elementName):
916                     reference?new ReferenceElementLeaf(elementName):new SingleLeaf(elementName);
917         } else {
918             // this element is a reference to another configured inhabitant.
919             // figure that out.
920             ConfigModel model = document.buildModel(value);
921             prop = collection?new CollectionNode(model,elementName):new SingleNode(model,elementName);
922         }
923 
924         for (String s  : values) {
925             if (s.startsWith("@")) {
926                  String annotationType = s.substring(1);
927                  prop.annotations.add(annotationType);               
928             }
929         }
930         return prop;
