176     /***
177      * Obtains the actual key value from this {@link Dom}.
178      */
179     public String getKey() {
180         String k = model.key;
181         if(k==null) return null;
182 
183         switch(k.charAt(0)) {
184         case '@':
185             return attribute(k.substring(1));
186         case '<':
187             return leafElement(k.substring(1,k.length()-1));
188         default:
189             throw new IllegalStateException("Invalid key value:"+k);
190         }
191     }

201     /*package*/ void fillAttributes(XMLStreamReader in) {
202         for( int i=in.getAttributeCount()-1; i>=0; i-- ) {
203             String n = in.getAttributeLocalName(i);
204             if(model.attributes.containsKey(n)) {
205                 if(attributes==null)
206                     attributes = new HashMap<String, String>();
207                 attributes.put(n,in.getAttributeValue(i));
208             }
209         }
210         if(attributes==null)
211             attributes = Collections.emptyMap();
212     }

267     /***
268      * Obtains the plural attribute value. Values are separate by ',' and surrounding whitespaces are ignored.
269      *
270      * @return
271      *      null if the attribute doesn't exist. This is a distinct state from the empty list,
272      *      which indicates that the attribute was there but no values were found.
273      */
274     public List<String> attributes(String name) {
275         String v = attribute(name);
276         if(v==null)     return null;
277         List<String> r = new ArrayList<String>();
278         StringTokenizer tokens = new StringTokenizer(v,",");
279         while(tokens.hasMoreTokens())
280             r.add(tokens.nextToken().trim());
281         return r;
282     }

310     /***
311      * Inserts a new {@link Dom} node right after the given DOM element.
312      *
313      * @param reference
314      *      If null, the new element will be inserted at the very beginning.
315      * @param name
316      *      The element name of the newly inserted item. "*" to indicate that the element
317      *      name be determined by the model of the new node.
318      */
319     public synchronized void insertAfter(Dom reference, String name, Dom newNode) {
320         // TODO: reparent newNode
321         if(name.equals("*"))    name=newNode.model.tagName;
322         NodeChild newChild = new NodeChild(name, newNode);
323 
324         if (children.size()==0) {
325             children = new ArrayList<Child>();
326         }
327         if(reference==null) {
328             children.add(0, newChild);
329             return;
330         }
331 
332         ListIterator<Child> itr = children.listIterator();
333         while(itr.hasNext()) {
334             Child child = itr.next();
335             if (child instanceof NodeChild) {
336                 NodeChild nc = (NodeChild) child;
337                 if(nc.dom==reference) {
338                     itr.add(newChild);
339                     habitat.addIndex(newNode, newNode.getProxyType().getName(), newNode.getKey());
340                     //habitat.add(newNode);
341                     return;
342                 }
343             }
344         }
345         throw new IllegalArgumentException(reference+" is not a valid child of "+this+". Children="+children);
346     }

348     /***
349      * Replaces an existing {@link NodeChild} with another one.
350      *
351      * @see #insertAfter(Dom, String, Dom)
352      */
353     public synchronized void replaceChild(Dom reference, String name, Dom newNode) {
354         ListIterator<Child> itr = children.listIterator();
355         while(itr.hasNext()) {
356             Child child = itr.next();   
357             if (child instanceof NodeChild) {
358                 NodeChild nc = (NodeChild) child;
359                 if(nc.dom==reference) {
360                     itr.set(new NodeChild(name,newNode));
361                     return;
362                 }
363             }
364         }
365         throw new IllegalArgumentException(reference+" is not a valid child of "+this+". Children="+children);
366     }

442     /***
443      * Given a master list and the new sub list, replace the items in the master list with the matching items
444      * from the new sub list. This process works even if the length of the new sublist is different.
445      *
446      * <p>
447      * For example, givn:
448      *
449      * <pre>
450      * replace A by A':
451      *   M=[A,B,C], S=[A'] => [A',B,C]
452      *   M=[A,B,A,B,C], S=[A',A'] => [A',B,A',B,C]
453      *
454      * when list length is different:
455      *   M=[A,A,B,C], S=[] => [B,C]
456      *   M=[A,B,C], S=[A',A'] => [A',A',B,C]
457      *   M=[B,C], S=[A',A'] => [B,C,A',A']
458      * </pre>
459      */
460     private static void stitchList(List<Child> list, String name, List<? extends Child> newSubList) {
461         // to preserve order, try to put new itesm where old items are found.
462         // if the new list is longer than the current list, we put all the extra
463         // after the last item in the sequence. That is,
464         // given [A,A,B,C] and [A',A',A'], we'll update the list to [A',A',A',B,C]
465         // The 'last' variable remembers the insertion position.
466         int last = list.size();
467 
468         ListIterator<Child> itr = list.listIterator();
469         ListIterator<? extends Child> jtr = newSubList.listIterator();
470         while(itr.hasNext()) {
471             Child child = itr.next();
472             if(child.name.equals(name)) {
473                 if(jtr.hasNext()) {
474                     itr.set(jtr.next());    // replace
475                     last = itr.nextIndex();
476                 } else
477                     itr.remove();   // remove
478             }
479         }
480 
481         // new list is longer than the current one
482         if(jtr.hasNext())
483             list.addAll(last,newSubList.subList(jtr.nextIndex(),newSubList.size()));
484     }

491     public synchronized void setLeafElements(final String name, String... values) {
492         List<Child> newChildren = new ArrayList<Child>(children);
493 
494         LeafChild[] leaves = new LeafChild[values.length];
495         for (int i = 0; i < values.length; i++)

499         children = newChildren;
500 
501         // see attribute(String,String) for the issue with this
502         getInjector().injectElement(this,name,get());
503     }

597     /***
598      * Picks up all node elements that are assignable to the given type,
599      * except those who are matched by other named elements in the model.
600      *
601      * Used to implement {@code FromElement("*")}.
602      */
603     public  List<Dom> domNodeByTypeElements(Class baseType) {
604         List<Dom> r = new ArrayList<Dom>();
605 
606         int len = children.size();
607         for( int i=0; i<len; i++ ) {
608             Child child = children.get(i);
609             if (child instanceof NodeChild) {
610                 NodeChild nc = (NodeChild) child;
611                 if(model.elements.containsKey(nc.name))
612                     continue;   // match with named
613                 if(baseType.isAssignableFrom(nc.dom.type()))
614                     r.add(nc.dom);
615             }
616         }
617         return r;
618     }

624                 return baseType.cast(elements.get(index).get());

677     /***
678      * Recursively decends the DOM tree and finds a DOM that has the given key
679      * and the type name.
680      *
681      * <p>
682      * TODO: the current algorithm does a full tree scan. Expand the model
683      * so that we can detect deadends that are statically known not to contain
684      * the kind we are looking for, and use that to cut the search space.
685      */
686     public Dom resolveReference(String key, String typeName) {
687         String keyedAs = model.keyedAs;
688         if(keyedAs!=null && keyedAs.equals(typeName) && getKey().equals(key))
689             return this; // found it
690 
691         for (Child child : children) {
692             if (child instanceof NodeChild) {
693                 NodeChild n = (NodeChild) child;
694                 Dom found = n.dom.resolveReference(key,typeName);
695                 if(found!=null) return found;
696             }
697         }
698 
699         return null;
700     }

733     public Object invoke(Object proxy, Method method, Object[] args) throws Throwable {
734         // serve java.lang.Object methods by ourselves
735         Class<?> clazz = method.getDeclaringClass();
736         if(clazz ==Object.class) {
737             try {
738                 return method.invoke(this,args);
739             } catch (InvocationTargetException e) {
740                 throw e.getTargetException();
741             }
742         }
743         if (clazz==Injectable.class) {
744             injectInto(this, args[0]);
745             return null;
746         }
747         if(method.getAnnotation(DuckTyped.class)!=null) {
748             return invokeDuckMethod(method,proxy,args);
749         }
750 
751         ConfigModel.Property p = toProperty(method);
752         if(p==null)
753             throw new IllegalArgumentException("No corresponding property found for method: "+method);
754 
755         if(args==null || args.length==0) {
756             // getter
757             return getter(p, method.getGenericReturnType());
758         } else {
759             // setter
760             setter(p, args[0]);
761             return null;
762         }
763     }

777     private Object invokeDuckMethod(Method method, Object proxy, Object[] args) throws Exception {
778         Method duckMethod = model.getDuckMethod(method);
779 
780         Object[] duckArgs;
781         if(args==null) {
782             duckArgs = new Object[]{proxy};
783         } else {
784             duckArgs = new Object[args.length+1];
785             duckArgs[0] = proxy;
786             System.arraycopy(args,0,duckArgs,1,args.length);
787         }
788 
789         try {
790             return duckMethod.invoke(null,duckArgs);
791         } catch (InvocationTargetException e) {
792             Throwable t = e.getTargetException();
793             if (t instanceof Exception)
794                 throw (Exception) t;
795             if (t instanceof Error)

798         }

809     /***
810      * Obtain XML names (like "abc-def") from strings like "getAbcDef" and "hasAbcDef".
811      * <p>
812      * The conversion rule uses the {@link #model} to find a good match.
813      */
814     protected ConfigModel.Property toProperty(Method method) {
815         String name = method.getName();
816 
817         // check annotations first
818         Element e = method.getAnnotation(Element.class);
819         if(e!=null) {
820             String en = e.value();
821             if(en.length()>0)
822                 return model.elements.get(en);
823         }
824         Attribute a = method.getAnnotation(Attribute.class);
825         if(a!=null) {
826             String an = a.value();
827             if(an.length()>0)
828                 return model.attributes.get(an);
829         }
830         // TODO: check annotations on the getter/setter
831 
832         // first, trim off the prefix
833         for (String p : PROPERTY_PREFIX) {
834             if(name.startsWith(p)) {
835                 name = name.substring(p.length());
836                 break;
837             }
838         }
839 
840         // tokenize by finding 'x|X' and 'X|Xx' then insert '-'.
841         StringBuilder buf = new StringBuilder(name.length()+5);
842         for(String t : TOKENIZER.split(name)) {
843             if(buf.length()>0)  buf.append('-');
844             buf.append(t.toLowerCase());
845         }
846         name = buf.toString();
847 
848         // at this point name should match XML names in the model, modulo case.
849         return model.findIgnoreCase(name);
850     }

895         StringBuilder buf = new StringBuilder();
