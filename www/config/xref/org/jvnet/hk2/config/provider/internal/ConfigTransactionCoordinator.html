67  /**
68   * Default implementation for {@link ConfigTransactionFactory}.

70   * @author Jeff Trent

72  @Service
73  public class ConfigTransactionCoordinator 
74      implements ConfigTransactionFactory, PreDestroy {

76    private Logger logger = Logger.getLogger(ConfigTransactionCoordinator.class.getName());
77    
78    @Inject
79    private Habitat habitat;
80  
81    @Inject(optional=true)
82    private InhabitantActivator activator;
83    
84    private ConfigTransaction theActiveOne;
85  
86    private Map<Object, MultiMap<String, ConfigByInhabitant>> beanToManagedConfigBy = new HashMap<Object, MultiMap<String, ConfigByInhabitant>>();

88    @Override
89    public synchronized ConfigTransaction getActiveTransaction(boolean create) {
90      if (create && null == theActiveOne) {
91        theActiveOne = new ConfigTransactionImpl(this);
92      }
93      
94      return theActiveOne;
95    }
96    
97    synchronized void finishedTransaction(ConfigTransaction txn) {
98      assert(theActiveOne == txn);
99      theActiveOne = null;
100   }
101 
102   @Override
103   public void preDestroy() {
104     if (null != theActiveOne) {
105       theActiveOne.rollback();
106       assert(null == theActiveOne);
107     }
108   }

110   boolean has(Object bean) {
111     return beanToManagedConfigBy.containsKey(bean);
112   }

114   MultiMap<String, ConfigByInhabitant> get(Object bean) {
115     // return a copy so that Transaction has there own working area
116     MultiMap<String, ConfigByInhabitant> orig = beanToManagedConfigBy.get(bean);
117     return (null == orig) ? null : new MultiMap<String, ConfigByInhabitant>();
118   }

120   // TODO: look for a faster implementation
121   @SuppressWarnings("unchecked")
122   Collection<ConfigByMetaInhabitant> getConfiguredByInhabitants(Class<?> clazz) {
123     HashSet<Class<?>> configured = new HashSet<Class<?>>();
124 
125     if (Proxy.isProxyClass(clazz)) {
126       for (Class<?> iface : clazz.getInterfaces()) {
127         if (null != iface.getAnnotation(Configured.class)) {
128           configured.add(iface);
129         }
130       }
131     }
132     
133     while (Object.class != clazz) {
134       String name = clazz.getName();
135       if (null != name && null != clazz.getAnnotation(Configured.class)) {
136         configured.add(clazz);
137       }
138       clazz = clazz.getSuperclass();
139     }
140     
141     ArrayList<ConfigByMetaInhabitant> configuredByList = null;
142     for (Inhabitant<?> i : habitat.getInhabitantsByContract(ConfiguredBy.class.getName())) {
143       ConfigByMetaInhabitant configuredBy = (ConfigByMetaInhabitant)i;
144       if (configured.contains(configuredBy.getConfiguredBy())) {
145         if (null == configuredByList) {
146           configuredByList = new ArrayList<ConfigByMetaInhabitant>();
147         }
148         configuredByList.add(configuredBy);
149       }
150     }
151     
152     return (null == configuredByList) ? Collections.EMPTY_LIST : Collections.unmodifiableList(configuredByList);
153   }

155   /**
156    * Called when we are committing the txn to manage new beans and the services inhabitants for those beans
157    */
158   void manage(Object bean, Set<Class<?>> beanContracts, String name, MultiMap<String, ConfigByInhabitant> managed) {
159     if (logger.isLoggable(Level.FINE)) {
160       logger.log(Level.FINE, "managing: {0}, {1}, {2}, {3}", new Object[] {bean, name, beanContracts, managed});
161     }
162     
163     // put the bean itself into the habitat
164     if (null != beanContracts && !beanContracts.isEmpty()) {
165       Inhabitant<?> configBeanInhabitant = new ExistingSingletonInhabitant<Object>(bean);
166       for (Class<?> beanContract : beanContracts) {
167         habitat.addIndex(configBeanInhabitant, beanContract.getName(), name);
168       }
169     }
170 
171     // put the configured-by instances into the habitat
172     for (Entry<String, List<ConfigByInhabitant>> cbi : managed.entrySet()) {
173       String index = cbi.getKey();
174       if (null != index) {
175         StringBuilder sb = new StringBuilder();
176         index = InhabitantsParser.parseIndex(index, sb);
177         if (null == name) {
178           name = (0 == sb.length()) ? null : sb.toString();
179         }
180       }
181       
182       for (ConfigByInhabitant i : cbi.getValue()) {
183         if (null == index) {
184           // intentionally do not register byType since that is reserved for the meta inhabitant entry
185 //          habitat.add(i);
186           habitat.addIndex(i, i.typeName(), name);
187         } else {
188           habitat.addIndex(i, index, name);
189         }
190 
191         // force activation
192         if (null == activator) {
193           Object service = i.get();
194           assert(null != service);
195         } else {
196           activator.activate(i);
197         }
198       }
199     }
200 
201     beanToManagedConfigBy.put(bean, managed);
202 
203     if (null != activator) {
204       try {
205         activator.awaitCompletion();
206       } catch (Exception e) {
207         throw new ComponentException(e);
208       }
209     }
210   }

212   /**
213    * Called when we are committing a txn to remove the beans and the associated services for those beans from the habitat
214    */
215   void unmanage(Object bean, Set<Class<?>> beanContracts, String name) {
216     if (logger.isLoggable(Level.FINE)) {
217       logger.log(Level.FINE, "unmanaging: {0}, {1}, {2}", new Object[] {bean, name, beanContracts});
218     }
219     
220     // remove the configured-by services related to this bean
221     MultiMap<String, ConfigByInhabitant> managed = beanToManagedConfigBy.remove(bean);
222     for (Entry<String, List<ConfigByInhabitant>> cbi : managed.entrySet()) {
223       String index = cbi.getKey();
224       if (null != index) {
225         index = InhabitantsParser.parseIndex(index, null);
226       }
227       
228       for (ConfigByInhabitant i : cbi.getValue()) {
229         boolean removed;
230         if (null == index) {
231           // intentionally do not register byType since that is reserved for the meta inhabitant entry
232 //          habitat.add(i);
233           removed = habitat.removeIndex(i.typeName(), i);
234         } else {
235           removed = habitat.removeIndex(index, i);
236         }
237         assert(removed);
238         
239         i.release();
240       }
241     }
242     
243     // remove the bean itself from the habitat
244     if (null != beanContracts) {
245       for (Class<?> beanContract : beanContracts) {
246         boolean removed = habitat.removeIndex(beanContract.getName(), bean);
247         assert(removed);
248         
249         // TODO: should we release it too?
250       }
251     }
252   }
