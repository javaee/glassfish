60       * A hook to perform variable replacements on the attribute/element values
61       * that are found in the configuration file.
62       * The translation happens lazily when objects are actually created, not when
63       * configuration is parsed, so this allows circular references &mdash;
64       * {@link Translator} may refer to objects in the configuration file being read.
65       */

99      /**
100      * Creates {@link ConfigModel} for the given {@link ConfigInjector} if we haven't done so.
101      */
102     /*package*/ ConfigModel buildModel(Inhabitant<? extends ConfigInjector> i) {
103         ConfigModel m = models.get(i);
104         if(m==null)
105             m = new ConfigModel(this,i,i.metadata());
106         return m;

116     /**
117      * Obtains a {@link ConfigModel} for the given class (Which should have {@link Configured} annotation on it.)
118      */
119     public ConfigModel buildModel(String fullyQualifiedClassName) {
120         Inhabitant i = habitat.getInhabitantByAnnotation(InjectionTarget.class, fullyQualifiedClassName);
121         if(i==null)
122             throw new ComponentException("ConfigInjector for %s is not found, is it annotated with @Configured",fullyQualifiedClassName);
123         return buildModel(i);
124     }

127      * Obtains the {@link ConfigModel} from the "global" element name.
128      *

131      *

135     public ConfigModel getModelByElementName(String elementName) {
136         Inhabitant<? extends ConfigInjector> i = habitat.getInhabitant(ConfigInjector.class, elementName);
137         if(i==null) return null;
138         return buildModel(i);
139     }

141     /**
142      * Calculates all @Configured interfaces subclassing the passed interface type.
143      *
144      * @param intf a @Configured interface
145      * @return List of all @Configured subclasses
146      * @throws ClassNotFoundException
147      */
148     public synchronized List<ConfigModel> getAllModelsImplementing(Class intf) throws ClassNotFoundException {
149         if (implementorsOf.size()==0) {
150             initXRef();
151         }
152         return implementorsOf.getOne(intf);   
153     }
154 
155     /**
156      * probably a bit slow, calculates all the @Configured interfaces subclassing, useful
157      * to find all possible subclasses of a type.
158      * 
159      * @throws ClassNotFoundException
160      */
161     private void initXRef() throws ClassNotFoundException {
162 
163         // force initialization of all the config models.
164         for (Inhabitant<? extends ConfigInjector> i : habitat.getInhabitants(ConfigInjector.class)) {
165             buildModel(i);
166         }
167         for (ConfigModel cm : models.values()) {
168             Class targetType = cm.classLoaderHolder.get().loadClass(cm.targetTypeName);
169             do {
170                 Class[] intfs = targetType.getInterfaces();
171                 for (Class intf : intfs) {
172                     if (intf.isAnnotationPresent(Configured.class)) {
173                         addXRef(intf, cm);
174                     }
175                 }
176                 targetType = targetType.getSuperclass();
177             } while (targetType!=null);
178         }
179     }
180 
181     private void addXRef(Class type, ConfigModel cm) {
182         List<ConfigModel> models = implementorsOf.getOne(type);
183         if (models==null) {
184             models= new ArrayList<ConfigModel>();
185             implementorsOf.add(type, models);
186         }
187         models.add(cm);
188     }

215     /*package*/
216     DataType getValidator(String dataType) {
217         synchronized(validators) {
218             DataType validator = validators.get(dataType);
219             if (validator != null)
220                 return (validator);
221         }
222         Collection<DataType> dtfh = habitat.getAllByContract(DataType.class);
223         synchronized(validators) {
224             for (DataType dt : dtfh) {
225                 validators.put(dt.getClass().getName(), dt);
226             }
227             return (validators.get(dataType));
228         }
229     }
