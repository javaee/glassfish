57  /*
58   * Custom MessageInterpolatorImpl for HK2
59   *
60   * This message interpolator is different from the default one in the following
61   * ways:
62   *
63   * 1. It uses a class specified in the "payload" argument to the annotation to
64   * find the class loader to find the resource bundle for messages. This allows
65   * classes that are in OSGi modules to specify a resource bundle for bean
66   * validation messages.
67   *
68   * 2. The "LocalStrings" resource bundle within the same package as the payload
69   * class is used to search for messages.  If a message is not found, the message
70   * is obtained from the default bean validation resource bundle.
71   *
72   * This class borrows heavily from the RI implementation of the ResourcBundleMessageInterpolator
73   * authored by Emmanuel Bernard, Hardy Ferentschik, and Gunnar Morling.
74   */
75  public class MessageInterpolatorImpl implements MessageInterpolator {
76  
77      /**
78       * The name of the default message bundle.
79       */
80      public static final String DEFAULT_VALIDATION_MESSAGES = "org.hibernate.validator.ValidationMessages";
81      /**
82       * The name of the user-provided message bundle as defined in the specification.
83       */
84      public static final String USER_VALIDATION_MESSAGES = "ValidationMessages";
85      /**
86       * Regular expression used to do message interpolation.
87       */
88      private static final Pattern MESSAGE_PARAMETER_PATTERN = Pattern.compile("(\\{[^\\}]+?\\})");
89      /**
90       * The default locale for the current user.
91       */
92      private final Locale defaultLocale = Locale.getDefault();
93      /**
94       * Step 1-3 of message interpolation can be cached. We do this in this map.
95       */
96      private final Map<LocalisedMessage, String> resolvedMessages = new WeakHashMap<LocalisedMessage, String>();
97      /**
98       * Flag indicating whether this interpolator should chance some of the interpolation steps.
99       */
100     private final boolean cacheMessages = true;
101 
102     public MessageInterpolatorImpl() { }
103 
104     @Override
105     public String interpolate(String message, Context context) {
106         // probably no need for caching, but it could be done by parameters since the map
107         // is immutable and uniquely built per Validation definition, the comparison has to be based on == and not equals though
108         return interpolate(message, context, defaultLocale);
109     }

111     /**
112      * Runs the message interpolation according to algorithm specified in JSR 303.
113      * <br/>
114      * Note:
115      * <br/>
116      * Look-ups in user bundles is recursive whereas look-ups in default bundle are not!
117      *
118      * @param message the message to interpolate
119      * @param annotationParameters the parameters of the annotation for which to interpolate this message
120      * @param locale the {@code Locale} to use for the resource bundle.
121      *
122      * @return the interpolated message.
123      */
124     @Override
125     public String interpolate(String message, Context context, Locale locale) {
126         Map<String, Object> annotationParameters = context.getConstraintDescriptor().getAttributes();
127         LocalisedMessage localisedMessage = new LocalisedMessage(message, locale);
128         String resolvedMessage = null;
129 
130         if (cacheMessages) {
131             resolvedMessage = resolvedMessages.get(localisedMessage);
132         }
133 
134         // if the message is not already in the cache we have to run step 1-3 of the message resolution
135         if (resolvedMessage == null) {
136             ResourceBundle userResourceBundle = new ContextResourceBundle(context, locale);
137             ResourceBundle defaultResourceBundle = ResourceBundle.getBundle(DEFAULT_VALIDATION_MESSAGES,
138                     locale,
139                     MessageInterpolator.class.getClassLoader());
140 
141             String userBundleResolvedMessage;
142             resolvedMessage = message;
143             boolean evaluatedDefaultBundleOnce = false;
144             do {
145                 // search the user bundle recursive (step1)
146                 userBundleResolvedMessage = replaceVariables(
147                         resolvedMessage, userResourceBundle, locale, true);
148 
149                 // exit condition - we have at least tried to validate against the default bundle and there was no
150                 // further replacements
151                 if (evaluatedDefaultBundleOnce
152                         && !hasReplacementTakenPlace(userBundleResolvedMessage, resolvedMessage)) {
153                     break;
154                 }
155 
156                 // search the default bundle non recursive (step2)
157                 resolvedMessage = replaceVariables(userBundleResolvedMessage, defaultResourceBundle, locale, false);
158                 evaluatedDefaultBundleOnce = true;
159                 if (cacheMessages) {
160                     resolvedMessages.put(localisedMessage, resolvedMessage);
161                 }
162             } while (true);
163         }
164 
165         // resolve annotation attributes (step 4)
166         resolvedMessage = replaceAnnotationAttributes(resolvedMessage, annotationParameters);
167 
168         // last but not least we have to take care of escaped literals
169         resolvedMessage = resolvedMessage.replace("\\{", "{");
170         resolvedMessage = resolvedMessage.replace("\\}", "}");
171         resolvedMessage = resolvedMessage.replace("\\\\", "\\");
172         return resolvedMessage;
173     }
174 
175     private boolean hasReplacementTakenPlace(String origMessage, String newMessage) {
176         return !origMessage.equals(newMessage);
177     }
178 
179     private String replaceVariables(String message, ResourceBundle bundle, Locale locale, boolean recurse) {
180         Matcher matcher = MESSAGE_PARAMETER_PATTERN.matcher(message);
181         StringBuffer sb = new StringBuffer();
182         String resolvedParameterValue;
183         while (matcher.find()) {
184             String parameter = matcher.group(1);
185             resolvedParameterValue = resolveParameter(
186                     parameter, bundle, locale, recurse);
187 
188             matcher.appendReplacement(sb, escapeMetaCharacters(resolvedParameterValue));
189         }
190         matcher.appendTail(sb);
191         return sb.toString();
192     }
193 
194     private String replaceAnnotationAttributes(String message, Map<String, Object> annotationParameters) {
195         Matcher matcher = MESSAGE_PARAMETER_PATTERN.matcher(message);
196         StringBuffer sb = new StringBuffer();
197         while (matcher.find()) {
198             String resolvedParameterValue;
199             String parameter = matcher.group(1);
200             Object variable = annotationParameters.get(removeCurlyBrace(parameter));
201             if (variable != null) {
202                 resolvedParameterValue = escapeMetaCharacters(variable.toString());
203             } else {
204                 resolvedParameterValue = parameter;
205             }
206             matcher.appendReplacement(sb, resolvedParameterValue);
207         }
208         matcher.appendTail(sb);
209         return sb.toString();
210     }
211 
212     private String resolveParameter(String parameterName, ResourceBundle bundle, Locale locale, boolean recurse) {
213         String parameterValue;
214         try {
215             if (bundle != null) {
216                 parameterValue = bundle.getString(removeCurlyBrace(parameterName));
217                 if (recurse) {
218                     parameterValue = replaceVariables(parameterValue, bundle, locale, recurse);
219                 }
220             } else {
221                 parameterValue = parameterName;
222             }
223         } catch (MissingResourceException e) {
224             // return parameter itself
225             parameterValue = parameterName;
226         }
227         return parameterValue;
228     }
229 
230     private String removeCurlyBrace(String parameter) {
231         return parameter.substring(1, parameter.length() - 1);
232     }
233 
234     /**
235      * @param s The string in which to replace the meta characters '$' and '\'.
236      *
237      * @return A string where meta characters relevant for {@link Matcher#appendReplacement} are escaped.
238      */
239     private String escapeMetaCharacters(String s) {
240         String escapedString = s.replace("\\", "\\\\");
241         escapedString = escapedString.replace("$", "\\$");
242         return escapedString;
243     }

245     /*
246      * A resource bundle that takes strings from the annotation context. This class
247      * looks for the "LocalStrings" resource bundle in the same package as the
248      * class that is specified in the payload of the annotation. If a String
249      * cannot be found there, then it looks in the user resource bundle as defined
250      * in JSR-303.
251      */
252 
253     private static class ContextResourceBundle extends ResourceBundle {
254         ResourceBundle contextBundle;
255         ResourceBundle userBundle;
256 
257         ContextResourceBundle(Context context, Locale locale) {
258             ConstraintDescriptor descriptor = context.getConstraintDescriptor();
259             Set<Class<? extends Payload>> payload = descriptor.getPayload();
260 
261             // Load the LogStrings.properties for the argument Locale, from the ClassLoader
262             // of the payload.
263             if (!payload.isEmpty()) {
264                 Class payloadClass = payload.iterator().next();
265                 String baseName = payloadClass.getPackage().getName() + ".LocalStrings";
266                 ClassLoader cl = payloadClass.getClassLoader();
267 
268                 try {
269                     contextBundle = ResourceBundle.getBundle(baseName, locale, cl);
270                 } catch (MissingResourceException mre) {
271                     contextBundle = null;
272                 }
273             }
274             try {
275                 userBundle = ResourceBundle.getBundle(USER_VALIDATION_MESSAGES, 
276                         locale,
277                         Thread.currentThread().getContextClassLoader());
278             } catch (MissingResourceException mre) {
279                     userBundle = null;
280             }
281 
282             if (userBundle != null) {
283                 setParent(userBundle);
284             }
285         }
286 
287         @Override
288         protected Object handleGetObject(String key) {
289             if (contextBundle != null) {
290                 return contextBundle.getObject(key);
291             }
292             return null;
293         }
294 
295         @Override
296         public Enumeration<String> getKeys() {
297             Set<String> keys = new TreeSet<String>();
298             if (contextBundle != null) {
299                 keys.addAll(Collections.list(contextBundle.getKeys()));
300             }
301             if (userBundle != null) {
302                 keys.addAll(Collections.list(userBundle.getKeys()));
303             }
304             return Collections.enumeration(keys);
305         }
306     }

308     private static class LocalisedMessage {
309 
310         private final String message;
311         private final Locale locale;
312 
313         LocalisedMessage(String message, Locale locale) {
314             this.message = message;
315             this.locale = locale;
316         }
317 
318         @Override
319         public boolean equals(Object o) {
320             if (this == o) {
321                 return true;
322             }
323             if (o == null || getClass() != o.getClass()) {
324                 return false;
325             }
326 
327             LocalisedMessage that = (LocalisedMessage) o;
328 
329             if (locale != null ? !locale.equals(that.locale) : that.locale != null) {
330                 return false;
331             }
332             if (message != null ? !message.equals(that.message) : that.message != null) {
333                 return false;
334             }
335 
336             return true;
337         }
338 
339         @Override
340         public int hashCode() {
341             int result = message != null ? message.hashCode() : 0;
342             result = 31 * result + (locale != null ? locale.hashCode() : 0);
343             return result;
344         }
345     }
