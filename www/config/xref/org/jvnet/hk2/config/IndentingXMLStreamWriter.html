37  package org.jvnet.hk2.config;
38  
39  import javax.xml.stream.XMLStreamWriter;
40  import javax.xml.stream.XMLStreamException;
41  import java.util.Stack;
42  
43  /***
44   * @author Kohsuke Kawaguchi
45   */
46  public class IndentingXMLStreamWriter extends DelegatingXMLStreamWriter {
47      private final static Object SEEN_NOTHING = new Object();
48      private final static Object SEEN_ELEMENT = new Object();
49      private final static Object SEEN_DATA = new Object();
50  
51      private Object state = SEEN_NOTHING;
52      private Stack<Object> stateStack = new Stack<Object>();
53  
54      private String indentStep = "  ";
55      private int depth = 0;
56  
57      public IndentingXMLStreamWriter(XMLStreamWriter writer) {
58          super(writer);
59      }
60  
61      /***
62       * Return the current indent step.
63       *
64       * <p>Return the current indent step: each start tag will be
65       * indented by this number of spaces times the number of
66       * ancestors that the element has.</p>
67       *
68       * @return The number of spaces in each indentation step,
69       *         or 0 or less for no indentation.
70       * @see #setIndentStep(int)
71       *
72       * @deprecated
73       *      Only return the length of the indent string.
74       */
75      public int getIndentStep() {
76          return indentStep.length();
77      }
78  
79  
80      /***
81       * Set the current indent step.
82       *
83       * @param indentStep The new indent step (0 or less for no
84       *        indentation).
85       * @see #getIndentStep()
86       *
87       * @deprecated
88       *      Should use the version that takes string.
89       */
90      public void setIndentStep(int indentStep) {
91          StringBuilder s = new StringBuilder();
92          for (; indentStep > 0; indentStep--) s.append(' ');
93          setIndentStep(s.toString());
94      }
95  
96      public void setIndentStep(String s) {
97          this.indentStep = s;
98      }
99  
100     private void onStartElement() throws XMLStreamException {
101         stateStack.push(SEEN_ELEMENT);
102         state = SEEN_NOTHING;
103         if (depth > 0) {
104             super.writeCharacters("\n");
105         }
106         doIndent();
107         depth++;
108     }
109 
110     private void onEndElement() throws XMLStreamException {
111         depth--;
112         if (state == SEEN_ELEMENT) {
113             super.writeCharacters("\n");
114             doIndent();
115         }
116         state = stateStack.pop();
117     }
118 
119     private void onEmptyElement() throws XMLStreamException {
120         state = SEEN_ELEMENT;
121         if (depth > 0) {
122             super.writeCharacters("\n");
123         }
124         doIndent();
125     }
126 
127     /***
128      * Print indentation for the current level.
129      *
130      * @exception org.xml.sax.SAXException If there is an error
131      *            writing the indentation characters, or if a filter
132      *            further down the chain raises an exception.
133      */
134     private void doIndent() throws XMLStreamException {
135         if (depth > 0) {
136             for (int i = 0; i < depth; i++)
137                 super.writeCharacters(indentStep);
138         }
139     }
140 
141 
142     public void writeStartDocument() throws XMLStreamException {
143         super.writeStartDocument();
144         super.writeCharacters("\n");
145     }
146 
147     public void writeStartDocument(String version) throws XMLStreamException {
148         super.writeStartDocument(version);
149         super.writeCharacters("\n");
150     }
151 
152     public void writeStartDocument(String encoding, String version) throws XMLStreamException {
153         super.writeStartDocument(encoding, version);
154         super.writeCharacters("\n");
155     }
156 
157     public void writeStartElement(String localName) throws XMLStreamException {
158         onStartElement();
159         super.writeStartElement(localName);
160     }
161 
162     public void writeStartElement(String namespaceURI, String localName) throws XMLStreamException {
163         onStartElement();
164         super.writeStartElement(namespaceURI, localName);
165     }
166 
167     public void writeStartElement(String prefix, String localName, String namespaceURI) throws XMLStreamException {
168         onStartElement();
169         super.writeStartElement(prefix, localName, namespaceURI);
170     }
171 
172     public void writeEmptyElement(String namespaceURI, String localName) throws XMLStreamException {
173         onEmptyElement();
174         super.writeEmptyElement(namespaceURI, localName);
175     }
176 
177     public void writeEmptyElement(String prefix, String localName, String namespaceURI) throws XMLStreamException {
178         onEmptyElement();
179         super.writeEmptyElement(prefix, localName, namespaceURI);
180     }
181 
182     public void writeEmptyElement(String localName) throws XMLStreamException {
183         onEmptyElement();
184         super.writeEmptyElement(localName);
185     }
186 
187     public void writeEndElement() throws XMLStreamException {
188         onEndElement();
189         super.writeEndElement();
190     }
191 
192     public void writeCharacters(String text) throws XMLStreamException {
193         state = SEEN_DATA;
194         super.writeCharacters(text);
195     }
196 
197     public void writeCharacters(char[] text, int start, int len) throws XMLStreamException {
198         state = SEEN_DATA;
199         super.writeCharacters(text, start, len);
200     }
201 
202     public void writeCData(String data) throws XMLStreamException {
203         state = SEEN_DATA;
204         super.writeCData(data);
205     }
