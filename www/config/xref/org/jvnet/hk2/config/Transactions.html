84                      }

87              }
88          }

97      public void preDestroy() {
98         for (Holder<ListenerNotifier<TransactionListener,  ?, Void>> listener : listeners) {
99             listener.get().stop();
100        }
101        configListenerNotifier.get().stop();
102        executor.shutdown();
103     }

105     /***
106      * Abstract notification pump, it adds jobs to the queue and process them in the order
107      * jobs were added.
108      *
109      * Jobs are just a wrapper for events of type <U> and a notification mechanism for
110      * completion notification
111      *
112      * @param <T> type of listener interface
113      * @param <U> type of events the listener methods are expecting
114      * @param <V> return type of the listener interface methods.
115      */
116     private abstract class Notifier<T, U, V> {
117         
118         private final BlockingQueue<FutureTask> pendingJobs = new ArrayBlockingQueue<FutureTask>(50);
119         private CountDownLatch latch = new CountDownLatch(1);
120 
121         /***
122          * Creates the task that will notify the listeners of a particular job.
123          * @param job contains the specifics of the notification like the events that need to be notified.
124          * @return a task that can be run and return an optional value.
125          */
126         protected abstract FutureTask<V> prepare(final Job<T, U, V> job);
127 
128         /***
129          * Adds a job to the notification pump. This job will be processed as soon as all other pending
130          * jobs have completed.
131          *
132          * @param job new notification job.
133          * @return a future on the return value.
134          */
135         public Future<V> add(final Job<T, U, V> job) {
136                 
137             // NOTE that this is put() which blocks, *not* add() which will not block and will
138             // throw an IllegalStateException if the queue is full.
139             if (latch.getCount()==0) {
140                 throw new RuntimeException("TransactionListener is inactive, yet jobs are published to it");
141             }
142             try {
143                 pendingJobs.put(prepare(job));
144             } catch (InterruptedException e ) {
145                 throw new RuntimeException(e);
146             }
147             return null;
148         }
149         
150         protected void start() {
151 
152             executor.submit(new Runnable() {
153 
154                 public void run() {
155                     while (latch.getCount()>0) {
156                         try {
157                             final FutureTask job = pendingJobs.take();
158                             // when listeners start a transaction themselves, several jobs try to get published
159                             // simultaneously so we cannot block the pump while delivering the messages. 
160                             executor.submit(new Runnable() {
161                                 public void run() {
162                                     job.run();
163                                 }
164                             });
165                         }
166 
167                         catch (InterruptedException e) {
168                             // do anything here?
169                         }
170                     }
171                 }
172                 
173             });
174         }
175 
176         void stop() {
177             latch.countDown();
178             // last event to force the close
179             pendingJobs.add(prepare(new Job<T, U, V>(null, null) {
180                 public V process(T target) {
181                     return null;
182                 }
183             }));
184         }
185     }

187     /***
188      * Default listener notification pump. One thread per listener, jobs processed in
189      * the order it was received.
190      *
191      * @param <T> type of listener interface
192      * @param <U> type of events the listener methods are expecting
193      * @param <V> return type of the listener interface methods.
194      */
195     public class ListenerNotifier<T,U,V> extends Notifier<T,U,V> {
196         
197         final T listener;
198 
199         public ListenerNotifier(T listener) {
200             this.listener = listener;
201         }
202 
203         protected FutureTask<V> prepare(final Job<T, U, V> job) {
204             return new FutureTask<V>(new Callable<V>() {
205                     public V call() throws Exception {
206                         try {
207                             if ( job.mEvents.size() != 0 ) {
208                                 return job.process(listener);
209                            }
210                         } finally {
211                             job.releaseLatch();
212                         }
213                         return null;
214                     }
215                 });
216         }   
217         
218     }

220     /***
221      * Configuration listener notification pump. All Listeners are notified within their own thread, only on thread
222      * takes care of the job pump.
223      * 
224      */
225     public class ConfigListenerNotifier extends Notifier<ConfigListener,PropertyChangeEvent,UnprocessedChangeEvents> {
226 
227         protected FutureTask<UnprocessedChangeEvents>
228             prepare(final Job<ConfigListener, PropertyChangeEvent, UnprocessedChangeEvents> job) {
229 
230         // first, calculate the recipients.
231         final Set<ConfigListener> configListeners = new HashSet<ConfigListener>();
232             if (job.mEvents != null) {
233                 for (PropertyChangeEvent event : job.mEvents) {
234                     final Dom dom = (Dom) ((ConfigView) Proxy.getInvocationHandler(event.getSource())).getMasterView();
235                     configListeners.addAll(dom.getListeners());
236 
237                     // we also notify the parent.
238                     if (dom.parent()!=null) {
239                         configListeners.addAll(dom.parent().getListeners());
240                     }
241                 }
242             }
243 
244             return new FutureTask<UnprocessedChangeEvents>(new Callable<UnprocessedChangeEvents>() {
245             public UnprocessedChangeEvents call() throws Exception {
246 
247                 try {
248                     // temporary structure to store our future notifications with pointer to the
249                     // originator config listener
250                     Map<Future<UnprocessedChangeEvents>, ConfigListener> futures = new HashMap<Future<UnprocessedChangeEvents>, ConfigListener>();
251 
252                     for (final ConfigListener listener : configListeners) {
253                         // each listener is notified in it's own thread.
254                         futures.put(executor.submit(new Callable<UnprocessedChangeEvents>() {
255                             public UnprocessedChangeEvents call() throws Exception {
256                                 UnprocessedChangeEvents e = job.process(listener);
257                                 return e;
258 
259                             }
260                         }), listener);
261                     }
262                     List<UnprocessedChangeEvents> unprocessed = new ArrayList<UnprocessedChangeEvents>(futures.size());
263                     for (Future<UnprocessedChangeEvents> future : futures.keySet()) {
264                         try {
265                             UnprocessedChangeEvents result = future.get(200, TimeUnit.SECONDS);
266                             if (result.getUnprocessed()!=null && result.getUnprocessed().size()>0) {
267                                 for (UnprocessedChangeEvent event : result.getUnprocessed()) {
268                                     Logger.getAnonymousLogger().log(Level.WARNING, "Unprocessed event : " + event);
269                                 }
270                                 unprocessed.add(result);
271                             }
272                         } catch (InterruptedException e) {
273                             Logger.getAnonymousLogger().log(Level.SEVERE, "Config Listener notification got interrupted", e);
274                         } catch (ExecutionException e) {
275                             Logger.getAnonymousLogger().log(Level.SEVERE, "Config Listener notification got interrupted", e);
276                         } catch (TimeoutException e) {
277                             ConfigListener cl = futures.get(future);
278                             Logger.getAnonymousLogger().log(Level.SEVERE, "Config Listener " + cl.getClass() + " notification took too long", e);
279                         }
280                     }
281 
282                     // all notification have been successful, I just need to notify the unprocessed events.
283                     // note these events are always synchronous so far.
284                     if (!unprocessed.isEmpty()) {
285                         Job unprocessedJob = new UnprocessedEventsJob(unprocessed, null);
286                         for (Holder<ListenerNotifier<TransactionListener, ?, Void>> listener : Transactions.this.listeners) {
287                             listener.get().add(unprocessedJob);
288                         }
289                     }
290                 } finally {
291                     job.releaseLatch();
292                 }
293 
294                 // in theory I should aggregate my unprocessed events but nobody cares.
295                 return null;
296             }
297         });
298     }

303     /***
304         A job contains an optional CountdownLatch so that a caller can learn when the
305         transaction has "cleared" by blocking until that time.
306      */
307     private abstract static class Job<T,U,V> {
308 
309         private final CountDownLatch mLatch;
310         protected final List<U> mEvents;
311         
312         public Job(List<U> events, final CountDownLatch latch ) {
313             mLatch  = latch;
314             mEvents = events;
315         }
316         
317         public void waitForLatch() throws InterruptedException {
318             if ( mLatch != null ) {
319                 mLatch.await();
320             }
321         }
322         
323         public void releaseLatch() {
324             if ( mLatch != null ) {
325                 mLatch.countDown();
326             }
327         }
328         
329         public abstract V process(T target);
330     }
331     
332     private static class TransactionListenerJob extends Job<TransactionListener, PropertyChangeEvent, Void> {
333 
334         public TransactionListenerJob(List<PropertyChangeEvent> events, CountDownLatch latch) {
335             super(events,  latch);
336         }
337         
338         @Override
339         public Void process(TransactionListener listener) {
340             try {
341                 listener.transactionCommited(mEvents);
342             } catch(Exception e) {
343                 e.printStackTrace();
344             }
345             return null;
346         }
347     }
348 
349     private static class UnprocessedEventsJob extends Job<TransactionListener, UnprocessedChangeEvents, Void> {
350 
351         public UnprocessedEventsJob(List<UnprocessedChangeEvents> events, CountDownLatch latch) {
352             super(events, latch);
353         }
354 
355         @Override
356         public Void process(TransactionListener listener) {
357             try {
358                 listener.unprocessedTransactedEvents(mEvents);
359             } catch(Exception e) {
360                 e.printStackTrace();
361             }
362             return null;
363         }
364     }

366     public class ConfigListenerJob extends Job<ConfigListener, PropertyChangeEvent, UnprocessedChangeEvents> {
367 
368         final PropertyChangeEvent[] eventsArray;
369 
370         public ConfigListenerJob(List<PropertyChangeEvent> events, CountDownLatch latch) {
371             super(events, latch);
372             eventsArray = mEvents.toArray(new PropertyChangeEvent[mEvents.size()]);            
373         }
374 
375         public UnprocessedChangeEvents process(ConfigListener target) {
376             return target.changed(eventsArray);
377         }

380     /***
381      * add a new listener to all transaction events.
382      *
383      * @param listener to be added.
384      */
385     public void addTransactionsListener(final TransactionListener listener) {
386         synchronized(listeners) {
387             listeners.add(new Holder<ListenerNotifier<TransactionListener, ?, Void>>() {
388 
389                 final ListenerNotifier<TransactionListener, PropertyChangeEvent, Void> tsListener = new ListenerNotifier<TransactionListener, PropertyChangeEvent, Void>(listener);
390                 final CountDownLatch initialized = new CountDownLatch(1);
391 
392                 public ListenerNotifier<TransactionListener, PropertyChangeEvent, Void> get() {
393                     synchronized(initialized) {
394                         if (initialized.getCount()>0) {
395                             tsListener.start();
396                             initialized.countDown();
397                         }
398                     }
399                     return tsListener;
400                 }
401             });
402         }
403     }

405     /***
406      * Removes an existing listener for transaction events
407      * @param listener the registered listener
408      * @return true if the listener unregistration was successful
409      */
410     public boolean removeTransactionsListener(TransactionListener listener) {
411         synchronized(listeners) {
412             for (Holder<ListenerNotifier<TransactionListener, ?, Void>> holder : listeners) {
413                 ListenerNotifier info = holder.get();
414                 if (info.listener==listener) {
415                     info.stop();
416                     return listeners.remove(holder);
417                 }
418             }
419         }
420         return false;
421     }

423     public List<TransactionListener> currentListeners() {
424         synchronized(listeners) {            
425             List<TransactionListener> l = new ArrayList<TransactionListener>();
426             for (Holder<ListenerNotifier<TransactionListener, ?, Void>> holder : listeners) {
427                 ListenerNotifier<TransactionListener, ?, Void> info = holder.get();
428                 l.add(info.listener);
429             }

444     /***
445      * Notification of a new transaction completion
446      *
447      * @param events accumulated list of changes
448      * @param waitTillCleared  synchronous semantics; wait until all change events are sent
449      */
450     @SuppressWarnings("cast")
451     void addTransaction(
452         final List<PropertyChangeEvent> events,
453         final boolean waitTillCleared ) {
454         
455         final List<ListenerNotifier<TransactionListener, ?, Void>> listInfos = new ArrayList<ListenerNotifier<TransactionListener, ?, Void>>();
456         for (Holder<ListenerNotifier<TransactionListener, ?, Void>> holder : listeners) {
457             ListenerNotifier<TransactionListener, ?, Void> info = holder.get();
458             listInfos.add(info);
459         }
460         
461         // create a CountDownLatch to implement waiting for events to actually be sent
462         final Job<TransactionListener, ?, Void> job = new TransactionListenerJob( events,
463                                 waitTillCleared ? new CountDownLatch(listInfos.size()) : null);
464         
465         final ConfigListenerJob configJob = new ConfigListenerJob(events,
466                 waitTillCleared? new CountDownLatch(1):null);
467         
468         // NOTE that this is put() which blocks, *not* add() which will not block and will
469         // throw an IllegalStateException if the queue is full.
470         try {
471             for (ListenerNotifier listener : listInfos) {
472                 listener.add(job);
473             }
474 
475             configListenerNotifier.get().add(configJob);
476 
477             job.waitForLatch();
478             configJob.waitForLatch();
479         } catch (InterruptedException e ) {
480             throw new RuntimeException(e);
481         }
482     }

484     public void waitForDrain() {
485         // insert a dummy Job and block until is has been processed.  This guarantees
486         // that all prior jobs have finished
487         addTransaction( new ArrayList<PropertyChangeEvent>(), true );
488         // at this point all prior transactions are guaranteed to have cleared
489     }
