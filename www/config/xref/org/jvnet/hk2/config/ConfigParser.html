59  /***

119     /***
120      * Parses a whole XML tree and builds a {@link Dom} tree.
121      *
122      * <p>
123      * This is the entry point for the root element of a configuration tree.
124      *
125      * @param in
126      *      pre-condition:  'in' is at the start element.
127      *      post-condition: 'in' is at the end element.
128      * @param document
129      *      The document that we are building right now.
130      *      Newly created {@link Dom} will belong to this document.
131      * @param parent
132      *      The parent element
133      * @return
134      *      Null if the XML element didn't yield anything (which can happen if the element is skipped.)
135      *      Otherwise fully parsed valid {@link Dom} object.
136      */
137     protected Dom handleElement(XMLStreamReader in,DomDocument document, Dom parent) throws XMLStreamException {
138         ConfigModel model = document.getModelByElementName(in.getLocalName());
139         if(model==null) {
140             String localName = in.getLocalName();
141             Logger.getAnonymousLogger().severe("Ignoring unrecognized element "+in.getLocalName() + " at " + in.getLocation());
142             // flush the sub element content from the parser
143             int depth=1;
144             while(depth>0) {
145                 final int tag = in.nextTag();
146                 if (tag==START_ELEMENT && in.getLocalName().equals(localName)) {
147                     if (Logger.getAnonymousLogger().isLoggable(Level.FINE)) {
148                         Logger.getAnonymousLogger().fine("Found child of same type "+localName+" ignoring too");
149                     }
150                     depth++;
151                 }
152                 if (tag==END_ELEMENT && in.getLocalName().equals(localName)) {
153                     if (Logger.getAnonymousLogger().isLoggable(Level.FINE)) {
154                         Logger.getAnonymousLogger().fine("closing element type " + localName);
155                     }
156                     depth--;
157                 }
158                 if (Logger.getAnonymousLogger().isLoggable(Level.FINE) && tag==START_ELEMENT) {
159                     Logger.getAnonymousLogger().fine("Jumping over " + in.getLocalName());
160                 }
161             }
162             return null;
163         }
164         return handleElement(in,document,parent,model);
165     }

167     /***
168      * Parses a whole XML tree and builds a {@link Dom} tree, by using the given model
169      * for the top-level element.
170      *
171      * <p>
172      * This is the entry point for recursively parsing inside a configuration tree.
173      * Since not every element is global, you don't always want to infer the model
174      * just from the element name (as is the case with {@link #handleElement(XMLStreamReader, DomDocument, Dom)}.
175      * 
176      * @param in
177      *      pre-condition:  'in' is at the start element.
178      *      post-condition: 'in' is at the end element.
179      * @param document
180      *      The document that we are building right now.
181      *      Newly created {@link Dom} will belong to this document.
182      * @param parent
183      *      The parent element
184      * @return
185      *      Null if the XML element didn't yield anything (which can happen if the element is skipped.)
186      *      Otherwise fully parsed valid {@link Dom} object.
187      */
188     protected Dom handleElement(XMLStreamReader in, DomDocument document, Dom parent, ConfigModel model) throws XMLStreamException {
189         final Dom dom = document.make(habitat, in, parent, model);
190 
191         // read values and fill DOM
192         dom.fillAttributes(in);
193 
194         List<Child> children=null;
195 
196         while(in.nextTag()==START_ELEMENT) {
197             String name = in.getLocalName();
198             ConfigModel.Property a = model.elements.get(name);
199 
200             if(children==null)
201                 children = new ArrayList<Child>();
202 
203             if(a==null) {
204                 // global look up
205                 Dom child = handleElement(in, document, dom);
206                 if(child!=null)
207                     children.add(new Dom.NodeChild(name, child));
208             } else
209             if(a.isLeaf()) {
210                 children.add(new Dom.LeafChild(name,in.getElementText()));
211             } else {
212                 Dom child = handleElement(in, document, dom, ((ConfigModel.Node) a).model);
213                 if(child!=null)
214                     children.add(new Dom.NodeChild(name, child));
215             }
216         }
217 
218 
219         dom.register();
220 
221         if (children==null) {
222             children = new ArrayList<Dom.Child>();
223         }
224         dom.ensureConstraints(children);
225 
226         if(!children.isEmpty())
227             dom.setChildren(children);
228 
229         dom.initializationCompleted();
230         return dom;
231     }

233     // In JDK 1.6, StAX is part of JRE, so we use no argument variant of
234     // newInstance(), where as on JDK 1.5, we use two argument version of
235     // newInstance() so that we can pass the classloader that loads
236     // XMLInputFactory to load the factory, otherwise by default StAX uses
237     // Thread's context class loader to locate the factory. See:
238     // https://glassfish.dev.java.net/issues/show_bug.cgi?id=6428
239     private static final XMLInputFactory xif =
240             XMLInputFactory.class.getClassLoader() == null ?
241                     XMLInputFactory.newInstance() :
242                     XMLInputFactory.newInstance(XMLInputFactory.class.getName(),
243                             XMLInputFactory.class.getClassLoader());
