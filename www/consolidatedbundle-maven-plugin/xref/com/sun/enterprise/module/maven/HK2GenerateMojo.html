51  	public void execute() throws MojoExecutionException {
52  		if (project.getDependencyArtifacts() != null) {
53  			List<String> consolidatedInhabitants = new ArrayList<String>();
54  
55  			// Create the consolidated inhabitant file contents by
56  			// going through all dependency artifacts
57  			for (Artifact a : (Set<Artifact>) project.getDependencyArtifacts()) {
58  				if (a.getScope() != null && a.getScope().equals("test"))
59  					continue;
60  				getLog().info("Dependency Artifact: " + a.getFile().toString());
61  
62  				try {
63  					JarFile jf = new JarFile(a.getFile());
64  					JarEntry je = jf.getJarEntry(InhabitantsFile.PATH);
65  					if (je == null)
66  						continue;
67  					getLog().debug("Dependency Artifact " + a + " has Inhabitants File: " + je);
68  
69  					Enumeration<JarEntry> entries = jf.entries();
70  					while (entries.hasMoreElements()) {
71  						JarEntry entry = entries.nextElement();
72  						if (!entry.isDirectory()
73  								&& entry.getName().startsWith(
74  										InhabitantsFile.PATH)) {
75  							getLog().info("Entry: " + entry.getName());
76  							consolidatedInhabitants.addAll(loadInhabitants(jf,
77  									entry));
78  						}
79  					}
80  				} catch (IOException iex) {
81  					iex.printStackTrace();
82  				}
83  			}
84  
85  		    writeToFile(consolidatedInhabitants);
86  		}
87  	}

89  	private List<String> loadInhabitants(JarFile jf, JarEntry e)
90  			throws IOException {
91  		List<String> l = new ArrayList<String>();
92  
93  		BufferedReader reader = null;
94  		try {
95  			DataInputStream in = new DataInputStream(jf.getInputStream(e));
96  			reader = new BufferedReader(new InputStreamReader(in));
97  			String line = null;
98  			while ((line = reader.readLine()) != null) {
99  				l.add(line);
100 			}
101 
102 		} catch (IOException iex) {
103 			iex.printStackTrace();
104 		} finally {
105 			reader.close();
106 		}
107 		return l;
108 	}

111 		// Write the consolidated inhabitant file contents
112 		// to manifestLocation
113 		
114 		//Create Manifest directory.

116 		+ "META-INF" + File.separatorChar + "inhabitants";
117 		boolean success = (new File(inhabitantsDir)).mkdirs();
118 		
119 		String fileLocation = inhabitantsDir + File.separatorChar + "default";
120 		getLog().info("Writing consolidated inhabitants to: " + fileLocation);
121 
122 		BufferedWriter out = null;
123 		try {
124 			out = new BufferedWriter(new FileWriter(fileLocation));
125 			for (String s : consolidatedInhabitants) {
126 				out.write(s);
127 				out.newLine();
128 			}
129 		} catch (IOException iex) {
130 			iex.printStackTrace();
131 		} finally {
132 			try {
133 				out.close();
134 			} catch (IOException iex) {

137 		}
