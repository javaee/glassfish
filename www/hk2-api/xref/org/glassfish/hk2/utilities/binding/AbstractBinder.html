68      /**

158     @Override
159     public void bind(DynamicConfiguration configuration) {
160         if (this.configuration != null) {
161           throw new IllegalArgumentException("Recursive configuration call detected.");
162         }
163 
164         if (configuration == null) {
165             throw new NullPointerException("configuration");
166         }
167         this.configuration = configuration;
168         try {
169             configure();
170         } finally {
171             complete();
172         }
173     }
174 
175     private <T> AbstractBindingBuilder<T> resetBuilder(AbstractBindingBuilder<T> newBuilder) {
176         if (currentBuilder != null) {
177             currentBuilder.complete(configuration(), getDefaultBinderLoader());
178         }
179 
180         currentBuilder = newBuilder;
181 
182         return newBuilder;
183     }
184 
185     private void complete() {
186         try {
187             resetBuilder(null);
188         } finally {
189             this.configuration = null;
190         }
191     }

199     /**
200      * Get the active {@link DynamicConfiguration binder factory} instance used for
201      * binding configuration. This method can only be called from within the
202      * scope of the {@link #configure()} method.
203      *
204      * @return dynamic configuration instance used for binding configuration.
205      * @throws IllegalStateException in case the method is not called from within
206      *                               an active call to {@link #configure()} method.
207      */
208     private DynamicConfiguration configuration() {
209         if (configuration == null)
210              throw new IllegalArgumentException("Dynamic configuration accessed from outside of an active binder configuration scope.");
211 
212         return configuration;
213     }

291     /**

302     private void setLoader(Descriptor descriptor) {
303         if (descriptor.getLoader() == null && descriptor instanceof DescriptorImpl) {
304             ((DescriptorImpl) descriptor).setLoader(getDefaultBinderLoader());
305         } // else who knows?
306     }
307 
308     private HK2Loader getDefaultBinderLoader() {
309         if (defaultLoader == null) {
310             ClassLoader loader = this.getClass().getClassLoader();
311             final ClassLoader binderClassLoader = loader == null ? ClassLoader.getSystemClassLoader() : loader;
312             defaultLoader = new HK2Loader() {
313                 @Override
314                 public Class<?> loadClass(String className) throws MultiException {
315                     try {
316                         return binderClassLoader.loadClass(className);
317                     } catch (ClassNotFoundException e) {
318                         throw new MultiException(e);
319                     }
320                 }
321             };
322         }
323         return defaultLoader;
324     }
