56  /**
57   * This is a writeable version of the Injectee interface.  Using this
58   * implementation may make your code more portable, as new methods
59   * added to the interface will be reflected in this class.
60   * 
61   * @author jwells
62   *
63   */
64  public class InjecteeImpl implements Injectee, Serializable {
65      /**
66       * For serialization
67       */
68      private static final long serialVersionUID = -1501119884265443727L;
69      
70      private Type requiredType;
71      private Set<Annotation> qualifiers;
72      private int position;
73      private Class<?> pClass;
74      private AnnotatedElement parent;
75      private boolean isOptional = false;
76      private boolean isSelf = false;
77      private Unqualified unqualified = null;
78      
79      /**
80       * None of the fields of the returned object will be set
81       */
82      public InjecteeImpl() {
83      }
84      
85      /**
86       * Only the requiredType field will be set
87       * 
88       * @param requiredType The possibly null required type
89       */
90      public InjecteeImpl(Type requiredType) {
91          this.requiredType = requiredType;
92      }
93      
94      /**
95       * This is the copy constructor, which will copy all the values from the incoming Injectee
96       * @param copyMe The non-null Injectee to copy the values from
97       */
98      public InjecteeImpl(Injectee copyMe) {
99          requiredType = copyMe.getRequiredType();
100         position = copyMe.getPosition();
101         parent = copyMe.getParent();
102         qualifiers = Collections.unmodifiableSet(copyMe.getRequiredQualifiers());
103         isOptional = copyMe.isOptional();
104         isSelf = copyMe.isSelf();
105         // unqualified = copyMe.getUnqualified();
106         
107         if (parent instanceof Field) {
108             pClass = ((Field) parent).getDeclaringClass();
109         }
110         else if (parent instanceof Constructor) {
111             pClass = ((Constructor<?>) parent).getDeclaringClass();
112         }
113         else {
114             pClass = ((Method) parent).getDeclaringClass();
115         }
116     }
