110 	 * Does a deep copy of the incoming descriptor
111 	 * 
112 	 * @param copyMe The descriptor to copy
113 	 */
114 	public DescriptorImpl(Descriptor copyMe) {
115 	    name = copyMe.getName();
116         scope = copyMe.getScope();
117         implementation = copyMe.getImplementation();
118         descriptorType = copyMe.getDescriptorType();
119         descriptorVisibility = copyMe.getDescriptorVisibility();
120         loader = copyMe.getLoader();
121         rank = copyMe.getRanking();
122         proxiable = copyMe.isProxiable();
123         id = copyMe.getServiceId();
124         locatorId = copyMe.getLocatorId();
125         baseDescriptor = copyMe.getBaseDescriptor();
126         
127 	    if (copyMe.getAdvertisedContracts() != null) {
128 	        contracts.addAll(copyMe.getAdvertisedContracts());
129 	    }
130 		
131 	    if (copyMe.getQualifiers() != null) {
132 		    qualifiers.addAll(copyMe.getQualifiers());
133 	    }
134 		
135 	    if (copyMe.getMetadata() != null) {
136 		    metadatas.putAll(ReflectionHelper.deepCopyMetadata(copyMe.getMetadata()));
137 	    }
138 	}

452 	public int hashCode() {
453 	    int retVal = 0;
454 	    
455 	    if (implementation != null) {
456 	        retVal ^= implementation.hashCode();
457 	    }
458 	    if (contracts != null) {
459 	        for (String contract : contracts) {
460 	            retVal ^= contract.hashCode();
461 	        }
462 	    }
463 	    if (name != null) {
464 	        retVal ^= name.hashCode();
465 	    }
466 	    if (scope != null) {
467 	        retVal ^= scope.hashCode();
468 	    }
469 	    if (qualifiers != null) {
470 	        for (String qualifier : qualifiers) {
471 	            retVal ^= qualifier.hashCode();
472 	        }
473 	    }
474 	    if (descriptorType != null) {
475 	        retVal ^= descriptorType.hashCode();
476 	    }
477 	    if (descriptorVisibility != null) {
478             retVal ^= descriptorVisibility.hashCode();
479         }
480 	    if (metadatas != null) {
481 	        for (Map.Entry<String, List<String>> entries : metadatas.entrySet()) {
482 	            retVal ^= entries.getKey().hashCode();
483 	            
484 	            for (String value : entries.getValue()) {
485 	                retVal ^= value.hashCode();
486 	            }
487 	        }
488 	    }
489 	    if (proxiable != null) {
490 	        if (proxiable.booleanValue()) {
491 	            retVal ^= 1;
492 	        }
493 	        else {
494 	            retVal ^= -1;
495 	        }
496 	    }
497 	    
498 	    return retVal;
499 	}

501 	private static boolean safeEquals(Object a, Object b) {
502 	    if (a == b) return true;
503 	    if (a == null) return false;
504 	    if (b == null) return false;
505 	    return a.equals(b);
506 	}

508 	private static <T> boolean equalOrderedCollection(Collection<T> a, Collection<T> b) {
509 	    if (a == b) return true;
510 	    if (a == null) return false;
511 	    if (b == null) return false;
512 	    
513 	    if (a.size() != b.size()) return false;
514 	    
515 	    Object aAsArray[] = a.toArray();
516 	    Object bAsArray[] = b.toArray();
517 	    
518 	    for (int lcv = 0; lcv < a.size(); lcv++) {
519 	        if (!safeEquals(aAsArray[lcv], bAsArray[lcv])) return false;
520 	    }
521 	    
522 	    return true;
523 	}

525 	private static <T> boolean equalMetadata(Map<String, List<String>> a, Map<String, List<String>> b) {
526         if (a == b) return true;
527         if (a == null) return false;
528         if (b == null) return false;
529         
530         if (a.size() != b.size()) return false;
531         
532         for (Map.Entry<String, List<String>> entry : a.entrySet()) {
533             String aKey = entry.getKey();
534             List<String> aValue = entry.getValue();
535             
536             List<String> bValue = b.get(aKey);
537             if (bValue == null) return false;
538             
539             if (!equalOrderedCollection(aValue, bValue)) return false;
540         }
541         
542         return true;
543     }

545 	public boolean equals(Object a) {
546 	    if (a == null) return false;
547 	    if (!(a instanceof Descriptor)) return false;
548 	    Descriptor d = (Descriptor) a;
549 	    
550 	    if (!safeEquals(implementation, d.getImplementation())) return false;
551 	    if (!equalOrderedCollection(contracts, d.getAdvertisedContracts())) return false;
552 	    if (!safeEquals(name, d.getName())) return false;
553 	    if (!safeEquals(scope, d.getScope())) return false;
554 	    if (!equalOrderedCollection(qualifiers, d.getQualifiers())) return false;
555 	    if (!safeEquals(descriptorType, d.getDescriptorType())) return false;
556 	    if (!safeEquals(descriptorVisibility, d.getDescriptorVisibility())) return false;
557 	    if (!equalMetadata(metadatas, d.getMetadata())) return false;
558 	    if (!safeEquals(proxiable, d.isProxiable())) return false;
559 	    
560 	    return true;
561 	}

563 	/**
564 	 * Will pretty print a descriptor
565 	 * 
566 	 * @param sb The string buffer put the pretty print into
567 	 * @param d The descriptor to write
568 	 */
569 	public static void pretty(StringBuffer sb, Descriptor d) {
570 	    if (sb == null || d == null) return;
571 	    
572         sb.append("\n\timplementation=" + d.getImplementation());
573         
574         if (d.getName() != null) {
575             sb.append("\n\tname=" + d.getName());
576         }
577         
578         sb.append("\n\tcontracts=");
579         sb.append(ReflectionHelper.writeSet(d.getAdvertisedContracts()));
580         
581         sb.append("\n\tscope=" + d.getScope());
582         
583         sb.append("\n\tqualifiers=");
584         sb.append(ReflectionHelper.writeSet(d.getQualifiers()));
585         
586         sb.append("\n\tdescriptorType=" + d.getDescriptorType());
587         
588         sb.append("\n\tdescriptorVisibility=" + d.getDescriptorVisibility());
589         
590         sb.append("\n\tmetadata=");
591         sb.append(ReflectionHelper.writeMetadata(d.getMetadata()));
592         
593         sb.append("\n\trank=" + d.getRanking());
594         
595         sb.append("\n\tloader=" + d.getLoader());
596         
597         sb.append("\n\tproxiable=" + d.isProxiable());
598         
599         sb.append("\n\tid=" + d.getServiceId());
600         
601         sb.append("\n\tlocatorId=" + d.getLocatorId());
602         
603         sb.append("\n\tidentityHashCode=" + System.identityHashCode(d));
604 	    
605 	}

617 	/**
618 	 * This writes this object to the data output stream in a human-readable
619 	 * format, excellent for writing out data files
620 	 * 
621 	 * @param out The output stream to write this object out to
622 	 * @throws IOException on failure
623 	 */
624 	public void writeObject(PrintWriter out) throws IOException {
625 	
626         out.print(START_START);
627         
628         // Implementation
629         if (implementation != null) {
630             out.print(implementation);
631         }
632         
633         out.println(END_START);
634         
635         // Contracts
636         if (contracts != null && !contracts.isEmpty()) {
637             out.println(CONTRACT_KEY + ReflectionHelper.writeSet(contracts));
638         }
639         
640         if (name != null) {
641             out.println(NAME_KEY + name);
642         }
643         
644         if (scope != null && !scope.equals(PerLookup.class.getName())) {
645             out.println(SCOPE_KEY + scope);
646         }
647         
648         if (qualifiers != null && !qualifiers.isEmpty()) {
649             out.println(QUALIFIER_KEY + ReflectionHelper.writeSet(qualifiers));
650         }
651         
652         if (descriptorType != null && descriptorType.equals(DescriptorType.PROVIDE_METHOD)) {
653             out.println(TYPE_KEY + PROVIDE_METHOD_DT);
654         }
655         
656         if (descriptorVisibility != null && descriptorVisibility.equals(DescriptorVisibility.LOCAL)) {
657             out.println(VISIBILITY_KEY + LOCAL_DT);
658         }
659         
660         if (rank != 0) {
661             out.println(RANKING_KEY + rank);
662         }
663         
664         if (proxiable != null) {
665             out.println(PROXIABLE_KEY + proxiable.booleanValue());
666         }
667         
668         if (metadatas != null && !metadatas.isEmpty()) {
669             out.println(METADATA_KEY + ReflectionHelper.writeMetadata(metadatas));
670         }
671         
672         out.println();  // This demarks the end of the section
673     }

675 	/**
676 	 * This can be used to read in instances of this object that were previously written out with
677 	 * writeObject.  Useful for reading from external data files
678 	 * 
679 	 * @param in The reader to read from
680 	 * @return true if a descriptor was read, false otherwise.  This is useful if reading a file that might have comments at the end
681 	 * @throws IOException on failure
682 	 */
683 	public boolean readObject(BufferedReader in) throws IOException {
684         String line = in.readLine();
685         
686         boolean sectionStarted = false;
687         while (line != null) {
688             String trimmed = line.trim();
689             
690             if (!sectionStarted) {
691                 if (trimmed.startsWith(START_START)) {
692                     sectionStarted = true;
693                     
694                     int endStartIndex = trimmed.indexOf(END_START_CHAR, 1);
695                     if (endStartIndex < 0) {
696                         throw new IOException("Start of implementation ends without ] character: " +
697                             trimmed);
698                     }
699                     
700                     if (endStartIndex > 1) {
701                         implementation = trimmed.substring(1, endStartIndex);
702                     }
703                 }
704             }
705             else {
706                 if (trimmed.length() <= 0) {
707                     // A blank line indicates end of object
708                     return true;
709                 }
710                 
711                 int equalsIndex = trimmed.indexOf('=');
712                 
713                 if (equalsIndex >= 1) {
714                     
715                     String leftHandSide = trimmed.substring(0, equalsIndex + 1);  // include the =
716                     String rightHandSide = trimmed.substring(equalsIndex + 1);
717                     
718                     if (leftHandSide.equalsIgnoreCase(CONTRACT_KEY)) {
719                         contracts = new LinkedHashSet<String>();
720
