44  import java.lang.annotation.Annotation;

48  /**
49   * ServiceLocator is the registry for HK2 services
50   * <p>
51   * Services can be looked up from this instance
52   */
53  @Contract
54  public interface ServiceLocator {
55      /**
56       * Gets the best service from this locator that implements
57       * this contract or has this implementation
58       * <p>
59       * Use this method only if destroying the service is not important
60       * 
61       * @param contractOrImpl May not be null, and is the contract
62       * or concrete implementation to get the best instance of
63       * @param qualifiers The set of qualifiers that must match this service
64       * definition
65       * @return An instance of the contract or impl.  May return
66       * null if there is no provider that provides the given
67       * implementation or contract
68       * @throws MultiException if there was an error during service creation
69       */
70      public <T> T getService(Type contractOrImpl, Annotation... qualifiers) throws MultiException;
71      
72      /**
73       * Gets the best service from this locator that implements
74       * this contract or has this implementation and has the given
75       * name
76       * <p>
77       * Use this method only if destroying the service is not important
78       * 
79       * @param contractOrImpl May not be null, and is the contract
80       * or concrete implementation to get the best instance of
81       * @param name May not be null, and is the name of the
82       * implementation to be returned
83       * @param qualifiers The set of qualifiers that must match this service
84       * definition
85       * @return An instance of the contract or impl.  May return
86       * null if there is no provider that provides the given
87       * implementation or contract
88       * @throws MultiException if there was an error during service creation
89       */
90      public <T> T getService(Type contractOrImpl, String name, Annotation... qualifiers) throws MultiException;
91      
92      /**
93       * Gets the all the services from this locator that implements
94       * this contract or has this implementation
95       * <p>
96       * Use this method only if destroying the service is not important
97       * 
98       * @param contractOrImpl May not be null, and is the contract
99       * or concrete implementation to get the best instance of
100      * @param qualifiers The set of qualifiers that must match this service
101      * definition
102      * @return A list of services implementing this contract
103      * or concrete implementation.  May not return null, but
104      * may return an empty list
105      * @throws MultiException if there was an error during service creation
106      */
107     public <T> List<T> getAllServices(Type contractOrImpl,
108             Annotation... qualifiers) throws MultiException;
109     
110     /**
111      * Gets the all the services from this locator that has the given
112      * qualifier or qualifiers
113      * <p>
114      * Use this method only if destroying the services is not important
115      * 
116      * @param qualifier May not be null, and is a qualifier that must
117      * match the service definition
118      * @param qualifiers The set of qualifiers that must match this service
119      * definition
120      * @return A list of services implementing this contract
121      * or concrete implementation.  May not return null, but
122      * may return an empty list
123      * @throws MultiException if there was an error during service creation
124      */
125     public <T> List<T> getAllServices(Annotation qualifier,
126             Annotation... qualifiers) throws MultiException;
127     
128     /**
129      * Gets the all the services from this locator that implements
130      * this contract or has this implementation
131      * <p>
132      * Use this method only if destroying the service is not important
133      * <p>
134      * This method should also be used with care to avoid classloading
135      * a large number of services
136      * 
137      * @param searchCriteria The returned service will match the Filter
138      * (in other words, searchCriteria.matches returns true).  May not
139      * be null
140      * @return A list of services matching this filter.  May not return null,
141      * but may return an empty list
142      * @throws MultiException if there was an error during service creation
143      */
144     public List<?> getAllServices(Filter searchCriteria) throws MultiException;
145     
146     /**
147      * Gets a service handle that can be used to get and destroy the returned
148      * service.  If a service, and all per lookup services must be destroyed then
149      * this method should be used to destroy the object
150      * <p>
151      * It is assumed that this method is called by the top level code.  All injection
152      * points created because of this invocation must use the
153      * getServiceHandle(ActiveDescriptor<T>, ServiceHandle<T>)
154      * method to retrieve objects, so that they can be destroyed in the proper sequence
155      * 
156      * @param contractOrImpl May not be null, and is the contract
157      * or concrete implementation to get the best instance of
158      * @param qualifiers The set of qualifiers that must match this service
159      * definition
160      * @return Will return root as a convenience
161      * @throws MultiException if there was an error during service creation
162      */
163     public <T> ServiceHandle<T> getServiceHandle(Type contractOrImpl, Annotation... qualifiers) throws MultiException;
164     
165     /**
166      * Gets a service handle that can be used to get and destroy the returned
167      * service
168      * 
169      * @param contractOrImpl May not be null, and is the contract
170      * or concrete implementation to get the best instance of
171      * @param name The name to use to further qualify the search (may be null)
172      * @param qualifiers The set of qualifiers that must match this service
173      * definition
174      * @return Will the service handle for the best service matching the
175      * given criteria, or null if no matching service could be found
176      * @throws MultiException if there was an error during service creation
177      * @throws IllegalArgumentException if contractOrImpl is null
178      */
179     public <T> ServiceHandle<T> getServiceHandle(Type contractOrImpl, String name,
180             Annotation... qualifiers) throws MultiException;
181     
182     /**
183      * Gets service handles that can be used to get and destroy the returned
184      * services
185      * <p>
186      * 
187      * @param contractOrImpl May not be null, and is the contract
188      * or concrete implementation to get the best instance of
189      * @param qualifiers The set of qualifiers that must match this service
190      * definition
191      * @return A non-null but possibly empty list of service handles matching
192      * the given criteria
193      * @throws MultiException if there was an error during service creation
194      * @throws IllegalArgumentException if contractOrImpl is null
195      */
196     public List<ServiceHandle<?>> getAllServiceHandles(Type contractOrImpl,
197             Annotation... qualifiers) throws MultiException;
198     
199     /**
200      * Gets service handles that can be used to get and destroy the returned
201      * services
202      * <p>
203      * 
204      * @param qualifier May not be null, and is a qualifier that must
205      * match the service definition
206      * @param qualifiers The set of qualifiers that must match this service
207      * definition
208      * @return A non-null but possibly empty list of service handles matching
209      * the given criteria
210      * @throws MultiException if there was an error during service creation
211      * @throws IllegalArgumentException if contractOrImpl is null
212      */
213     public List<ServiceHandle<?>> getAllServiceHandles(Annotation qualifier,
214             Annotation... qualifiers) throws MultiException;
215     
216     /**
217      * Gets a service handle that can be used to get and destroy the returned
218      * service.  If a service, and all per lookup services must be destroyed then
219      * this method should be used to destroy the object
220      * <p>
221      * It is assumed that this method is called by the top level code.  All injection
222      * points created because of this invocation must use the
223      * getServiceHandle(ActiveDescriptor<T>, ServiceHandle<T>)
224      * method to retrieve objects, so that they can be destroyed in the proper sequence
225      * 
226      * @param searchCriteria A filter to use when determining which services should apply 
227      * @return A list of handles in ranked order that match the given filter
228      * @throws MultiException if there was an error during service creation
229      */
230     public List<ServiceHandle<?>> getAllServiceHandles(Filter searchCriteria) throws MultiException;
231     
232     /**
233      * Gets the list of descriptors that match the given filter
234      * 
235      * @param filter A filter to use when determining which services should apply
236      * @return A list of descriptors in ranked order that match the given filter
237      */
238     public List<ActiveDescriptor<?>> getDescriptors(Filter filter);
239     
240     /**
241      * Gets the descriptor that best matches this filter, taking ranking
242      * and service id into account
243      * 
244      * @param filter The filter to use to retrieve the set of descriptors
245      * @return The best descriptor matching the filter, or null if there
246      * is no descriptor that matches the filter
247      */
248     public ActiveDescriptor<?> getBestDescriptor(Filter filter);
249     
250     /**
251      * Converts a descriptor to an ActiveDescriptor.  Will use the registered
252      * HK2Loaders to perform this action
253      * 
254      * @param descriptor The descriptor to convert, may not be null
255      * @param injectee The injectee on behalf of whom this descriptor is being injected.  May
256      * be null if the injectee is unknown
257      * @return The active descriptor as loaded with the first valid {@link HK2Loader}
258      * @throws MultiException if there were errors when loading or analyzing the class
259      */
260     public ActiveDescriptor<?> reifyDescriptor(Descriptor descriptor, Injectee injectee) throws MultiException;
261     
262     /**
263      * Converts a descriptor to an ActiveDescriptor.  Will use the registered
264      * HK2Loaders to perform this action
265      * 
266      * @param descriptor The descriptor to convert, may not be null
267      * @return The active descriptor as loaded with the first valid {@link HK2Loader}
268      * @throws MultiException if there were errors when loading or analyzing the class
269      */
270     public ActiveDescriptor<?> reifyDescriptor(Descriptor descriptor) throws MultiException;
271     
272     /**
273      * This method will first find a descriptor for this injectee, and then
274      * reify that descriptor.  If multiple descriptors are found, they will
275      * be reified in ranking order until an ActiveDescriptor matching the Injectee is
276      * found.  
277      * 
278      * @param injectee the injection point for whom to find the ActiveDescriptor
279      * @return The active descriptor for this injection point
280      * @throws MultiException if there were errors when loading or analyzing the class
281      */
282     public ActiveDescriptor<?> getInjecteeDescriptor(Injectee injectee) throws MultiException;
283     
284     /**
285      * Gets a service handle that can be used to get and destroy the returned
286      * service.  If a service, and all per lookup services must be destroyed then
287      * this method should be used to destroy the object
288      * <p>
289      * It is assumed that this method is called by the top level code.  All injection
290      * points created because of this invocation must use the
291      * getServiceHandle(ActiveDescriptor<T>, ServiceHandle<T>)
292      * method to retrieve objects, so that they can be destroyed in the proper sequence
293      * 
294      * @param activeDescriptor The service handle that can be used to get and destroy
295      * this service
296      * @param injectee The injectee on behalf of whom this descriptor is being injected.  May
297      * be null if the injectee is unknown
298      * @return Will return root as a convenience
299      * @throws MultiException if there was an error during service creation
300      */
301     public <T> ServiceHandle<T> getServiceHandle(ActiveDescriptor<T> activeDescriptor, Injectee injectee) throws MultiException;
302     
303     /**
304      * Gets a service handle that can be used to get and destroy the returned
305      * service.  If a service, and all per lookup services must be destroyed then
306      * this method should be used to destroy the object
307      * <p>
308      * It is assumed that this method is called by the top level code.  All injection
309      * points created because of this invocation must use the
310      * getServiceHandle(ActiveDescriptor<T>, ServiceHandle<T>)
311      * method to retrieve objects, so that they can be destroyed in the proper sequence
312      * 
313      * @param activeDescriptor The service handle that can be used to get and destroy
314      * this service
315      * @return Will return root as a convenience
316      * @throws MultiException if there was an error during service creation
317      */
318     public <T> ServiceHandle<T> getServiceHandle(ActiveDescriptor<T> activeDescriptor) throws MultiException;
319     
320     /**
321      * This method should be called by code getting injectee's on behalf of some
322      * root object.  In this way the objects associated with the root object can
323      * be destroyed in the proper sequence
324      * 
325      * @param activeDescriptor The descriptor whose service to create
326      * @param root The ultimate parent of this service creation.  May be null
327      * @return The service matching this descriptor
328      * @throws MultiException if there was an error during service creation
329      */
330     public <T> T getService(ActiveDescriptor<T> activeDescriptor, ServiceHandle<?> root) throws MultiException;
331   
332     /**
333      * Returns the name of this ServiceLocator
334      * @return The name of this ServiceLocator, will not return null
335      */
336     public String getName();
337     
338     /**
339      * This returns the unique locator ID for this locator.  The locator ID will
340      * be assigned at the time of creation and must be a monotonacally increasing
341      * number (starting at zero)
342      * @return The identifier for this service locator
343      */
344     public long getLocatorId();
345   
346     /**
347      * This method will shutdown every service associated with this ServiceLocator.
348      * Those services that have a preDestroy shall have their preDestroy called
349      */
350     public void shutdown();
351     
352     /**
353      * This method will analyze the given class, and create it if can.  The object
354      * created in this way will not be managed by HK2.  It is the responsibility of
355      * the caller to ensure that any lifecycle this object has is honored
356      * 
357      * @param createMe The class to create, may not be null
358      * @return An instance of the object
359      */
360     public <T> T create(Class<T> createMe);
361     
362     /**
363      * This will analyze the given object and inject into its fields and methods.
364      * The object injected in this way will not be managed by HK2
365      * 
366      * @param injectMe The object to be analyzed and injected into
367      */
368     public void inject(Object injectMe);
369     
370     /**
371      * This will analyze the given object and call the postConstruct method.
372      * The object given will not be managed by HK2
373      * 
374      * @param postConstructMe The object to postConstruct
375      */
376     public void postConstruct(Object postConstructMe);
377     
378     /**
379      * This will analyze the given object and call the preDestroy method.
380      * The object given will not be managed by HK2
381      * 
382      * @param preDestroyMe The object to preDestroy
383      */
384     public void preDestroy(Object preDestroyMe);
385     
386     /**
387      * This method creates, injects and post-constructs an object with the given
388      * class. This is equivalent to calling the {@link ServiceLocator#create(Class)}
389      * method followed by the {@link ServiceLocator#inject(Object)} method followed
390      * by the {@link ServiceLocator#postConstruct(Object)} method.
391      * <p>
392      * The object created is not managed by the locator.
393      *
394      * @param createMe The non-null class to create this object from
395      * @return An instance of the object that has been created, injected and post constructed
396      * @throws MultiException if there was an error when creating or initializing the object
397      */
398     public <U> U createAndInitialize(Class<U> createMe);
