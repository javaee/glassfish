40  package org.glassfish.hk2.api;
41  
42  /**
43   * This class is used to add {@link Descriptor}s to a {@link ServiceLocator}
44   * instance.  No operation performed on this object will be reflected
45   * in the {@link ServiceLocator} until the commit method is called.
46   * 
47   * @author jwells
48   *
49   */
50  public interface DynamicConfiguration {
51      /**
52       * This method will bind the given descriptor to this Module.
53       * If the descriptor given is not an ActiveDescriptor then a
54       * non-reified ActiveDescriptor will be returned with the system
55       * provided fields set.  If the descriptor given is a reified
56       * ActiveDescriptor then the descriptor returned will be a
57       * reified ActiveDescriptor that takes all values except for the
58       * id from the given descriptor
59       * 
60       * @param key May not be null.  Will be used to derive the various
61       * key fields associated with the given provider
62       * @return The entry as added to the service registry, with fields
63       * of the Descriptor filled in by the system as appropriate
64       * @throws IllegalArgumentException if there is an error in the key
65       */
66      public <T> ActiveDescriptor<T> bind(Descriptor key);
67      
68      /**
69       * This method will bind the descriptors found in the
70       * {@link FactoryDescriptors}.  This method will first
71       * validate the descriptors from the {@link FactoryDescriptors}
72       * and then simply bind them into this configuration as
73       * two independent descriptors. 
74       * 
75       * @param factoryDescriptors A description of a factory service
76       * and the type the factory service provides.  May not be null
77       * @return The descriptors returned from this object may be cast
78       * to ActiveDescriptor and will contain all the fields of the descriptors
79       * filled in by the system 
80       * @throws IllegalArgumentException if there is an error in the input parameter
81       */
82      public FactoryDescriptors bind(FactoryDescriptors factoryDescriptors);
83      
84      /**
85       * This allows third party systems to add reified active descriptors to the system.
86       * The active descriptor given must be fully reified (isReified must return true) and
87       * the create and destroy methods must be implemented.
88       * 
89       * @param activeDescriptor The reified active descriptor to be added to the system.  The
90       * system will not attempt to reify this descriptor itself
91       * @return The entry as added to the service registry, with fields
92       * of the Descriptor filled in by the system as appropriate
93       * @throws IllegalArgumentException if the descriptor is not reified
94       */
95      public <T> ActiveDescriptor<T> addActiveDescriptor(ActiveDescriptor<T> activeDescriptor)
96              throws IllegalArgumentException;
97      
98      /**
99       * This adds an active descriptor to the system based completely on the analysis
100      * of the given class.  The class itself and all interfaces marked contract will
101      * be in the list of advertised services.  The scope and qualifiers will be taken
102      * from the annotations on the class.
103      * 
104      * @param rawClass The class to analyze, must not be null 
105      * @return The active (reified) descriptor that has been added to the system, with
106      * all fields filled in based on the rawClass
107      * @throws MultiException If this class cannot be a service
108      * @throws IllegalArgumentException if rawClass is null
109      */
110     public <T> ActiveDescriptor<T> addActiveDescriptor(Class<T> rawClass)
111             throws MultiException, IllegalArgumentException;
112     
113     /**
114      * This filter will added to the list of filters in this Configuration that will
115      * determine which Descriptors will be removed from the system.  Only services directly
116      * from this Configuration objects' associated ServiceLocator will be given to this Filter
117      * (it will not be given descriptors from the ServiceLocators parent).  The descriptors passed
118      * into this filter may be cast to {@link ActiveDescriptor}.  The descriptors passed into this
119      * filter may or may not have been reified.  This filter should not reify passed in descriptors.
120      * <p>
121      * And descriptor for which this filter returns true will be removed from the
122      * {@link ServiceLocator} prior to any additions that are performed with this
123      * Configuration object.  Hence a Configuration can remove and add a descriptor of the
124      * same type in one commit.
125      * <p>
126      * In order to unbind a filter the caller of commit must pass the LOOKUP validators and the
127      * UNBIND validators.
128      * 
129      * @param unbindFilter This filter will be added to the list of filters that this
130      * configuration object will use to determine which descriptors to unbind from the system.
131      * May not be null
132      * @throws IllegalArgumentException if unbindFilter is null
133      */
134     public void addUnbindFilter(Filter unbindFilter) throws IllegalArgumentException;
135     
136     /**
137      * This causes the configuration to get committed.  This
138      * method may only be called once
139      * 
140      * @throws MultiException If errors were found in the commit process
141      * @throws IllegalStateException if this method was called more than once
142      */
143     public void commit() throws MultiException;
144 
145 }
