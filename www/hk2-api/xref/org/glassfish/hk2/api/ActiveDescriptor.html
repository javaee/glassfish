40  package org.glassfish.hk2.api;
41  
42  import java.lang.annotation.Annotation;
43  import java.lang.reflect.Type;
44  import java.util.List;
45  import java.util.Set;
46  
47  /**
48   * An ActiveDescriptor contains information about a Descriptor after it has been reified.
49   * Most of the methods in an ActiveDescriptor cannot be called until the isReified method
50   * return true.  Prior to that the information is not yet known.  ActiveDescriptors
51   * can be reified with the {@link ServiceLocator}.reifyDescriptor method.
52   * <p>
53   * ActiveDescriptors may also be provided to the system pre-reified, which is useful
54   * when the Service described may be produced by a third-party system.  In this case
55   * the ActiveDescriptors create method must use the third-party system in order to
56   * create instances of the described service
57   * 
58   * @author jwells
59   * 
60   * @param <T> This is the type produced by the cache and consumed by the cache
61   */
62  public interface ActiveDescriptor<T> extends Descriptor, SingleCache<T> {
63      /**
64       * This method returns true if this descriptor has been reified
65       * (class loaded).  If this method returns false then the other methods
66       * in this interface will throw an IllegalStateException.  Once this
67       * method returns true it may be
68       * 
69       * @return true if this descriptor has been reified, false otherwise
70       */
71      public boolean isReified();
72      
73      /**
74       * The implementation class that should be used
75       * to generate new instances of this descriptor.
76       * <p>
77       * If the class returned is a Factory, then the
78       * factory is used to create instances.  In this case
79       * the system will get an instance of the factory and
80       * use it to create the instances
81       *  
82       * @return The class that directly implements the contract
83       * types, or the class that is the factory for an object
84       * that implements the contract types
85       */
86      public Class<?> getImplementationClass();
87      
88      /**
89       * The set of types that this ActiveDescriptor must produce.
90       * These types may be Classes or ParameterizedTypes, and
91       * may be no other subclass of Type
92       * 
93       * @return the set of types this ActiveDescriptor must implement
94       * or extend
95       */
96      public Set<Type> getContractTypes();
97      
98      /**
99       * Returns the scope that this ActiveDescriptor belongs to
100      * 
101      * @return The scope of this ActiveDescriptor
102      */
103     public Class<? extends Annotation> getScopeAnnotation();
104     
105     /**
106      * The full set of qualifiers that this ActiveDescriptor
107      * provides
108      * 
109      * @return The set of annotations that this ActiveDescriptor
110      * provides
111      */
112     public Set<Annotation> getQualifierAnnotations();
113     
114     /**
115      * Returns the full list of Injectees this class has.  These
116      * references will be resolved prior to the class being constructed,
117      * even if these injectees are field or method injectees.
118      * <p>
119      * If this descriptor is describing a factory created type then
120      * this list must have zero length
121      * 
122      * @return Will not return null, but may return an empty list.  The set
123      * of Injectees that must be resolved before this ActiveDescriptor can
124      * be constructed
125      */
126     public List<Injectee> getInjectees();
127     
128     /**
129      * If this ActiveDescriptor has DescriptorType of PROVIDE_METHOD then
130      * this field will return the ServiceId of its associated Factory
131      * service.  Otherwise this method should return null
132      * 
133      * @return The service ID of the associated factory service
134      */
135     public Long getFactoryServiceId();
136     
137     /**
138      * If this ActiveDescriptor has DescriptorType of PROVIDE_METHOD then
139      * this field will return the ServiceId of its associated Factory
140      * service.  Otherwise this method should return null
141      * 
142      * @return The locator ID of the associated factory service
143      */
144     public Long getFactoryLocatorId();
145     
146     /**
147      * Creates an instance of the ActiveDescriptor.  All of the
148      * Injectee's must be created prior to instantiation, and
149      * associated with the ExtendedProvider so that they can be
150      * destroyed properly
151      * 
152      * @param root The root service handle, which can be used
153      * to associated all the PerLookup objects with this creation
154      * 
155      * @return An instance of this ActiveDescriptor
156      */
157     public T create(ServiceHandle<?> root);
158     
159     /**
160      * Disposes this instance.  All the PerLookup objects that
161      * were created for this instance will be destroyed after this
162      * object has been destroyed
163      * 
164      * @param instance The instance to destroy
165      */
166     public void dispose(T instance);
167     
168     
169 }
