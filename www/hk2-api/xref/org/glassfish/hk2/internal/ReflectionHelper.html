84      /**
85       * Given the type parameter gets the raw type represented
86       * by the type, or null if this has no associated raw class
87       * @param type The type to find the raw class on
88       * @return The raw class associated with this type
89       */
90      public static Class<?> getRawClass(Type type) {
91          if (type == null) return null;
92          
93          if (type instanceof Class) {
94              return (Class<?>) type;
95          }
96          
97          if (type instanceof ParameterizedType) {
98              Type rawType = ((ParameterizedType) type).getRawType();
99              if (rawType instanceof Class) {
100                 return (Class<?>) rawType;
101             }
102         }
103         
104         return null;
105     }

107     private static String getNamedName(Named named, Class<?> implClass) {
108         String name = named.value();
109         if (name != null && !name.equals("")) return name;
110         
111         String cn = implClass.getName();
112             
113         int index = cn.lastIndexOf(".");
114         if (index < 0) return cn;
115         
116         return cn.substring(index + 1);
117     }

135     /**
136      * Returns the set of types this class advertises
137      * @param t the object we are analyzing
138      * @return The type itself and the contracts it implements
139      */
140     public static Set<Type> getAdvertisedTypesFromObject(Object t) {
141         Set<Type> retVal = new LinkedHashSet<Type>();
142         if (t == null) return retVal;
143         
144         Class<?> clazz = t.getClass();
145         
146         retVal.add(clazz);
147         
148         Type genericSuperclass = clazz.getGenericSuperclass();
149         while (genericSuperclass != null) {
150             Class<?> rawClass = getRawClass(genericSuperclass);
151             if (rawClass == null) break;
152             
153             if (rawClass.isAnnotationPresent(Contract.class)) {
154                 retVal.add(genericSuperclass);
155             }
156             
157             genericSuperclass = rawClass.getGenericSuperclass();
158         }
159         
160         while (clazz != null) {
161             Type genericInterfaces[] = clazz.getGenericInterfaces();
162             for (Type genericInterface : genericInterfaces) {
163                 Class<?> rawClass = getRawClass(genericInterface);
164                 if (rawClass == null) continue;
165             
166                 if (rawClass.isAnnotationPresent(Contract.class)) {
167                     retVal.add(genericInterface);
168                 }
169             }
170             
171             clazz = clazz.getSuperclass();
172         }
173         
174         return retVal;
175     }

178      * Returns the set of types this class advertises
179      * @param clazz the class we are analyzing
180      * @return The type itself and the contracts it implements
181      */
182     public static Set<String> getContractsFromClass(Class<?> clazz) {
183         Set<String> retVal = new LinkedHashSet<String>();
184         if (clazz == null) return retVal;
185         
186         retVal.add(clazz.getName());
187         
188         Class<?> extendsClasses = clazz.getSuperclass();
189         while (extendsClasses != null) {
190             if (extendsClasses.isAnnotationPresent(Contract.class)) {
191                 retVal.add(extendsClasses.getName());
192             }
193             
194             extendsClasses = extendsClasses.getSuperclass();
195         }
196         
197         while (clazz != null) {
198             Class<?> interfaces[] = clazz.getInterfaces();
199             for (Class<?> iFace : interfaces) {
200                 if (iFace.isAnnotationPresent(Contract.class)) {
201                     retVal.add(iFace.getName());
202                 }
203             }
204             
205             clazz = clazz.getSuperclass();
206         }
207         
208         return retVal;
209     }

329     /**
330      * Writes a set in a way that can be read from an input stream as well.  The values in
331      * the set may not contain the characters "{},"
332      * 
333      * @param line The line to read
334      * @param addToMe The set to add the strings to
335      * @return The number of characters read until the end of the set
336      * @throws IOException On a failure
337      */
338     private static int internalReadSet(char asChars[], int startIndex, Collection<String> addToMe) throws IOException {
339         int dot = startIndex;
340         int startOfSet = -1;
341         while (dot < asChars.length) {
342             if (asChars[dot] == '{') {
343                 startOfSet = dot;
344                 dot++;
345                 break;
346             }
347             dot++;
348         }
349         
350         if (startOfSet == -1) {
351             throw new IOException("Unknown set format, no initial { character : " + new String(asChars));
352         }
353         
354         StringBuffer elementBuffer = new StringBuffer();
355         int endOfSet = -1;
356         while (dot < asChars.length) {
357             char dotChar = asChars[dot];
358             
359             if (dotChar == '}') {
360                 addToMe.add(elementBuffer.toString());
361                 
362                 endOfSet = dot;
363                 break;  // Done!
364             }
365             
366             if (dotChar == ',') {
367                 // Terminating a single element
368                 addToMe.add(elementBuffer.toString());
369                 
370                 elementBuffer = new StringBuffer();
371             }
372             else {
373                 // This character is either an escape character or a real character
374                 if (dotChar != '\\') {
375                     elementBuffer.append(dotChar);
376                 }
377                 else {
378                     // This is an escape character
379                     if (dot + 1 >= asChars.length) {
380                         // This is an error, escape at end of buffer
381                         break;
382                     }
383                     
384                     dot++;  // Moves it forward
385                     dotChar = asChars[dot];
386                     
387                     if (dotChar == 'n') {
388                         elementBuffer.append('\n');
389                     }
390                     else if (dotChar == 'r') {
391                         elementBuffer.append('\r');
392                     }
393                     else {
394                         elementBuffer.append(dotChar);
395                     }
396                 }
397                 
398             }
399             
400             dot++;
401         }
402         
403         if (endOfSet == -1) {
404             throw new IOException("Unknown set format, no ending } character : " + new String(asChars));
405         }
406         
407         return dot - startIndex;
408     }

410     private static int readKeyStringListLine(char asChars[], int startIndex, Map<String, List<String>> addToMe) throws IOException {
411         int dot = startIndex;
412         
413         int equalsIndex = -1;
414         while (dot < asChars.length) {
415             char dotChar = asChars[dot];
416             
417             if (dotChar == '=') {
418                 equalsIndex = dot;
419                 break;
420             }
421             
422             dot++;
423         }
424         
425         if (equalsIndex < 0) {
426             throw new IOException("Unknown key-string list format, no equals: " + new String(asChars));
427         }
428         
429         String key = new String(asChars, startIndex, (equalsIndex - startIndex));  // Does not include the =
430         dot++;  // Move it past the equals
431         
432         if (dot >= asChars.length) {
433             // Key with no values, this is illegal
434             throw new IOException("Found a key with no value, " + key + " in line " + new String(asChars));
435             
436         }
437         
438         LinkedList<String> listValues = new LinkedList<String>();
439         
440         int addOn = internalReadSet(asChars, dot, listValues);
441         if (!listValues.isEmpty()) {
442             addToMe.put(key, listValues);
443         }
444         
445         dot += addOn + 1;
446         if (dot < asChars.length) {
447             char skipComma = asChars[dot];
448             if (skipComma == ',') {
449                 dot++;
450             }
451         }
452         
453         return dot - startIndex;  // The +1 gets us to the next character in the stream
454     }

459      * @param addToMe The set to add the strings to

473     private static String escapeString(String escapeMe) {
474         char asChars[] = new char[escapeMe.length()];
475         
476         escapeMe.getChars(0, escapeMe.length(), asChars, 0);
477         
478         StringBuffer sb = new StringBuffer();
479         for (int lcv = 0; lcv < asChars.length; lcv++) {
480             char candidateChar = asChars[lcv];
481             
482             if (ESCAPE_CHARACTERS.contains(candidateChar)) {
483                 sb.append('\\');
484                 sb.append(candidateChar);
485             }
486             else if (REPLACE_CHARACTERS.containsKey(candidateChar)) {
487                 char replaceWithMe = REPLACE_CHARACTERS.get(candidateChar);
488                 sb.append('\\');
489                 sb.append(replaceWithMe);
490             }
491             else {
492                 sb.append(candidateChar);
493             }
494         }
495         
496         return sb.toString();
497     }

585      * Adds a value to the list of values associated with this key
586      * 
587      * @param metadatas The base metadata object
588      * @param key The key to which to add the value.  May not be null
589      * @param value The value to add.  May not be null
590      */
591     public static void addMetadata(Map<String, List<String>> metadatas, String key, String value) {
592         if (key == null || value == null) return;
593         if (key.indexOf('=') >= 0) {
594             throw new IllegalArgumentException("The key field may not have an = in it:" + key);
595         }
596         
597         List<String> inner = metadatas.get(key);
598         if (inner == null) {
599             inner = new LinkedList<String>();
600             metadatas.put(key, inner);
601         }
602         
603         inner.add(value);
604     }

607      * Removes the given value from the given key
608      * 
609      * @param metadatas The base metadata object
610      * @param key The key of the value to remove.  May not be null
611      * @param value The value to remove.  May not be null
612      * @return true if the value was removed

614     public static boolean removeMetadata(Map<String, List<String>> metadatas, String key, String value) {
615         if (key == null || value == null) return false;
616         
617         List<String> inner = metadatas.get(key);
618         if (inner == null) return false;
619         
620         boolean retVal = inner.remove(value);

622         
623         return retVal;

638     /**
639      * This method does a deep copy of the incoming meta-data, (which basically means we will
640      * also make copies of the value list)
641      * 
642      * @param copyMe The guy to copy (if null, null will be returned)
643      * @return A deep copy of the metadata
644      */
645     public static Map<String, List<String>> deepCopyMetadata(Map<String, List<String>> copyMe) {
646         if (copyMe == null) return null;
647         
648         Map<String, List<String>> retVal = new LinkedHashMap<String, List<String>>();
649         
650         for (Map.Entry<String, List<String>> entry : copyMe.entrySet()) {
651             String key = entry.getKey();
652             if (key.indexOf('=') >= 0) {
653                 throw new IllegalArgumentException("The key field may not have an = in it:" + key);
654             }
655             
656             List<String> values = entry.getValue();
657             LinkedList<String> valuesCopy = new LinkedList<String>();
658             for (String value : values) {
659                 valuesCopy.add(value);
660             }
661             
662             retVal.put(key, valuesCopy);
663         }
664         
665         return retVal;
666     }
