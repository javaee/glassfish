55  /**
56   * The implementation of the {@link ServiceLocatorFactory} that looks
57   * in the OSGi service registry or the META-INF/services for the implementation
58   * to use.  Failing those things, it uses the standard default locator
59   * generator, which is found in auto-depends, which is the 99.9% case
60   * 
61   * @author jwells
62   */
63  public class ServiceLocatorFactoryImpl extends ServiceLocatorFactory {
64    private final ServiceLocatorGenerator defaultGenerator;
65    private final Object lock = new Object();
66    private final HashMap<String, ServiceLocator> serviceLocators = new HashMap<String, ServiceLocator>();
67    private static int name_count = 0;
68    private static final String GENERATED_NAME_PREFIX = "__HK2_Generated_";
69  
70    private static Logger logger = Logger.getLogger(ServiceLocatorFactoryImpl.class.getPackage().getName());
71  
72      /**
73     * This will create a new set of name to locator mappings
74     */
75    public ServiceLocatorFactoryImpl() {
76        defaultGenerator = AccessController.doPrivileged(new PrivilegedAction<ServiceLocatorGenerator>() {
77  
78          @Override
79          public ServiceLocatorGenerator run() {
80              try {
81                  return getGenerator();
82              }
83              catch (Throwable th) {
84                  Logger.getLogger(ServiceLocatorFactoryImpl.class.getName()).warning("Error finding implementation of hk2: " + th.getMessage());
85                  // Thread.dumpStack();
86                  return null;
87              }
88          }
89            
90        });
91    }
92    
93    private static ServiceLocatorGenerator getGenerator() {
94        Iterable<? extends ServiceLocatorGenerator> generators = ServiceLoader.lookupProviderInstances(ServiceLocatorGenerator.class);
95        if (generators != null) {
96            // non-null indicates we are in OSGi environment
97            // So, we will return whatever we find. If we don't find anything here, then we assume it is a
98            // configuration error and return null.
99            // Since org.glassfish.hk2.osgiresourcelocator.ServiceLoader never throws ServiceConfigurationError,
100           // there is no need to catch it and try next item in the iterator.
101           final Iterator<? extends ServiceLocatorGenerator> iterator = generators.iterator();
102           return iterator.hasNext() ? iterator.next() : null;
103       } else {
104           // We are in non-OSGi environment, let's use JDK ServiceLoader instead.
105           // Make sure we use our current loader to locate the service as opposed to some arbitrary TCL
106           final ClassLoader classLoader = ServiceLocatorFactoryImpl.class.getClassLoader();
107           Iterator<ServiceLocatorGenerator> providers = java.util.ServiceLoader.load(ServiceLocatorGenerator.class,
108                   classLoader).iterator();
109           while (providers.hasNext()) {
110               try {
111                   return providers.next();
112               } catch (ServiceConfigurationError sce) {
113                   // This can happen. See the exception javadoc for more details.
114                   logger.logp(Level.FINE, "ServiceLocatorFactoryImpl", "getGenerator",
115                           "Exception while looking up service locator generator", sce);
116                   // We will try the next one
117               }
118           }
119       }
120       logger.warning("Cannot find a default implementation of the HK2 ServiceLocatorGenerator");
121       return null;
122   }

136   public ServiceLocator find(String name) {
137     synchronized (lock) {
138       return serviceLocators.get(name);

142   /* (non-Javadoc)
143    * @see org.glassfish.hk2.api.ServiceLocatorFactory#destroy(java.lang.String)
144    */
145   @Override
146   public void destroy(String name) {
147       ServiceLocator killMe = null;
148     
149       synchronized (lock) {
150           killMe = serviceLocators.remove(name);
151       }
152     
153       if (killMe != null) {
154           killMe.shutdown();
155       }
156   }
157   
158   public void destroy(ServiceLocator locator) {
159       if (locator == null) return;
160       
161       locator.shutdown();
162       
163       destroy(locator.getName());  // Removes it from our DB
164   }

175     /* (non-Javadoc)
176      * @see org.glassfish.hk2.api.ServiceLocatorFactory#create(java.lang.String, org.glassfish.hk2.api.ServiceLocator, org.glassfish.hk2.extension.ServiceLocatorGenerator)
177      */
178     @Override
179     public ServiceLocator create(String name, ServiceLocator parent,
180             ServiceLocatorGenerator generator) {
181  
182         synchronized (lock) {
183             ServiceLocator retVal;
184 
185             if (name == null) {
186                 name = GENERATED_NAME_PREFIX + name_count++;
187                 return internalCreate(name, parent, generator);
188             }
189 
190             retVal = serviceLocators.get(name);
191             if (retVal != null) return retVal;
192             retVal = internalCreate(name, parent, generator);
193             serviceLocators.put(name, retVal);
194 
195             return retVal;
196         }
197     }
198 
199     private ServiceLocator internalCreate(String name, ServiceLocator parent, ServiceLocatorGenerator generator) {
200         if (generator == null) {
201             if (defaultGenerator == null) {
202                 throw new IllegalStateException("No generator was provided and there is no default generator registered");
203             }
204             generator = defaultGenerator;
205         }
206         return generator.create(name, parent);
207     }
