41  package org.jvnet.hk2.annotations;
42  
43  import java.lang.annotation.*;
44  
45  /**
46   * Decorates a {@link java.lang.reflect.Method} with all the annotation which types are
47   * specified using the {@link org.jvnet.hk2.annotations.Decorate#with()}.
48   *
49   * @author Jerome Dochez
50   */
51  @Retention(RetentionPolicy.RUNTIME)
52  @Target(ElementType.TYPE)
53  public @interface Decorate {
54  
55      /**
56       * Returns the decorated type
57       *
58       * @return the decorated type
59       */
60      Class<?> targetType();
61  
62      /**
63       * Returns the decorated method name defined in the Class returned by
64       * {@link org.jvnet.hk2.annotations.Decorate#targetType()}
65       *
66       * @return the decorated method name
67       */
68      String methodName();
69  
70      /**
71       * Returns a list of annotation types that are defined on the same
72       * annotated element and decorate the method identified by the
73       * {@link org.jvnet.hk2.annotations.Decorate#targetType()} and
74       * {@link org.jvnet.hk2.annotations.Decorate#methodName()}.
75       *
76       * @return list of annotate types that decorates the {@link java.lang.reflect.Method}
77       */
78      Class<? extends Annotation>[] with();
79  }
