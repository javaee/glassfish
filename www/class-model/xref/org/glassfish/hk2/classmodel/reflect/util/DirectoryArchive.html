53  /**
54   * Directory base archive abstraction
55   */
56  public class DirectoryArchive extends AbstractAdapter {
57      public final File directory;
58      public final Parser parser;
59  
60      public DirectoryArchive(Parser parser, File directory) {
61          this.directory = directory;
62          this.parser = parser;
63      }
64  
65      @Override
66      public String toString() {
67        return getURI().toString();
68      }
69      
70      @Override
71      public URI getURI() {
72          return directory.toURI();
73      }
74  
75  
76      @Override
77      public Manifest getManifest() throws IOException {
78          File manifest = new File(directory, JarFile.MANIFEST_NAME);
79          if (manifest.exists()) {
80              InputStream is = new BufferedInputStream(new FileInputStream(manifest));
81              try {
82                  return new Manifest(is);
83              } finally {
84                  is.close();
85              }
86          }
87          return null;
88      }
89  
90      @Override
91      public void onSelectedEntries(Selector selector, EntryTask task, Logger logger) throws IOException {
92          parse(directory, selector, task, logger);
93      }
94  
95      private void parse(File dir, Selector selector, EntryTask task, Logger logger) throws IOException {
96          File [] listFiles = dir.listFiles();
97          if (null == listFiles) {
98              System.err.println("listFiles() is null for: " + dir);
99              return;
100         }
101       
102         ByteBuffer buffer = ByteBuffer.allocate(52000);
103 
104         for (File f : listFiles) {
105             Entry ae = new Entry(mangle(f), f.length(), f.isDirectory());
106             if (!f.isDirectory()) {
107                 if (ae.name.endsWith(".jar")) {
108                     JarArchive ja = null;
109                     try {
110                         ja = new JarArchive(parser, f.toURI());
111                         ja.onSelectedEntries(selector, task, logger);
112                     } finally {
113                         ja.close();
114                     }
115                     continue;
116                 }
117                 if (!selector.isSelected(ae))
118                     continue;
119                 InputStream is = null;
120                 try {
121                     try {
122                         is = new FileInputStream(f);
123                         task.on(ae, is);
124                     } catch(Exception e) {
125                         logger.log(Level.SEVERE, "Exception while processing " + f.getName() +
126                                 " of size " + f.length(), e);
127                     }
128 
129                 } finally {
130                     if (is!=null) {
131                         is.close();
132                     }
133                 }
134             } else {
135                 parse(f, selector, task, logger);
136             }
137 
138         }
139     }
140 
141     private String mangle(File f) {
142         String relativePath = f.getAbsolutePath().substring(directory.getAbsolutePath().length()+1);
143         return relativePath.replace(File.separatorChar, '/');
144     }
145 
146     @Override
147     public void close() throws IOException {
148     }
