40  package org.glassfish.hk2.classmodel.reflect.impl;
41  
42  import org.glassfish.hk2.classmodel.reflect.*;
43  import org.glassfish.hk2.classmodel.reflect.util.ParsingConfig;
44  
45  import java.io.File;
46  import java.io.IOException;
47  import java.util.*;
48  import java.net.URI;
49  
50  /**
51   * Implementation of the Type abstraction.
52   *
53   * @author Jerome Dochez
54   */
55  public class TypeImpl extends AnnotatedElementImpl implements Type {
56  
57      final TypeProxy<Type> sink;
58      final List<MethodModel> methods = new ArrayList<MethodModel>();
59      final Set<URI> definingURIs= new HashSet<URI>();
60  
61  
62      public TypeImpl(String name, TypeProxy<Type> sink) {
63          super(name);
64          this.sink = sink;
65      }
66  
67      @Override
68      public Collection<URI> getDefiningURIs() {
69          return Collections.unmodifiableSet(definingURIs);
70      }
71  
72      synchronized void addDefiningURI(URI uri) {
73          definingURIs.add(uri);
74          try {
75              File file = new File(uri);
76  //            assert(file.exists()) : file + " does not exist";
77              definingURIs.add(file.getCanonicalFile().toURI());
78          } catch (IOException e) {
79              // ignore, this is a safeguard for confused user's code that do not
80              // deal well with file path.
81          }
82      }
83  
84      @Override
85      public boolean wasDefinedIn(Collection<URI> uris) {
86          for (URI uri : uris) {
87              if (definingURIs.contains(uri)) {
88                  return true;
89              }
90          }
91          return false;
92      }
93  
94      synchronized void addMethod(MethodModelImpl m) {
95          methods.add(m);
96      }
97  
98      @Override
99      public Collection<MethodModel> getMethods() {
100         return Collections.unmodifiableList(methods);
101     }
102 
103     TypeProxy getProxy() {
104         return sink;
105     }
106 
107     @Override
108     public Collection<Member> getReferences() {
109         return sink.getRefs();
110     }
111 
112     @Override
113     protected void print(StringBuffer sb) {
114         super.print(sb);    //To change body of overridden methods use File | Settings | File Templates.
115         sb.append(", subclasses=[");
116         for (AnnotatedElement cm : sink.getSubTypeRefs()) {
117             sb.append(" ").append(cm.getName());
118         }
119         sb.append("]");
120     }
121 }
