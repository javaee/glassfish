66      public ModelClassVisitor(ParsingContext ctx, URI definingURI, String entryName) {
67          this.ctx = ctx;
68          this.definingURI = definingURI;
69          this.entryName = entryName;
70          typeBuilder = ctx.getTypeBuilder(definingURI);
71      }

73      @Override
74      public void visit(int version, int access, String name, String signature, String superName, String[] interfaces) {        
75  
76          String parentName = (superName!=null?org.objectweb.asm.Type.getObjectType(superName).getClassName():null);
77          TypeProxy parent = (parentName!=null?typeBuilder.getHolder(parentName, typeBuilder.getType(access)):null);
78          if (parent!=null) {
79              // put a temporary parent until we eventually visit it. 
80              parent.set(typeBuilder.getType(access, parentName, null));
81          }
82          String className = org.objectweb.asm.Type.getObjectType(name).getClassName();
83          URI classDefURI=null;
84          try {
85              int index = entryName.length() - name.length() - 6;
86              String newPath=(index>0?definingURI.getPath() + entryName.substring(0, index):definingURI.getPath());
87              classDefURI = new URI(definingURI.getScheme(), newPath, definingURI.getFragment());
88  
89          } catch (URISyntaxException e) {
90              e.printStackTrace();  //To change body of catch statement use File | Settings | File Templates.
91          }
92          type = ctx.getTypeBuilder(classDefURI).getType(access, className, parent);
93          deepVisit =ctx.getConfig().getInjectionTargetAnnotations().isEmpty();
94  
95          // reverse index
96          if (parent!=null) {
97              parent.getSubTypeRefs().add(type);
98          }
99  
100 
101         try {
102             ClassModelImpl classModel = (ClassModelImpl) type;
103             for (String intf : interfaces) {
104                 String interfaceName = org.objectweb.asm.Type.getObjectType(intf).getClassName();
105                 TypeProxy<InterfaceModel> typeProxy = typeBuilder.getHolder(interfaceName, InterfaceModel.class);
106                 classModel.isImplementing(typeProxy);
107                 typeProxy.getImplementations().add(classModel);
108 
109             }
110         } catch(ClassCastException e) {
111             // ignore
112         }
113 
114     }

126     @Override
127     public AnnotationVisitor visitAnnotation(String desc, boolean visible) {
128         
129         desc = unwrap(desc);
130 
131 
132         final AnnotationTypeImpl at = (AnnotationTypeImpl) typeBuilder.getType(Opcodes.ACC_ANNOTATION, desc, null );
133         final AnnotationModelImpl am = new AnnotationModelImpl(type, at);
134 
135         // reverse index
136         at.getAnnotatedElements().add(type);
137 
138         // forward index
139         type.addAnnotation(am);
140 
141         if (ctx.getConfig().getInjectionTargetAnnotations().contains(desc)) {
142             System.out.println("Inspecting fields of " + type.getName());
143             deepVisit =true;
144         }
145         visitingContext.annotation=am;
146         return annotationVisitor;
147     }

159     @Override
160     public FieldVisitor visitField(int access, final String name, final String desc, final String signature, final Object value) {
161 
162         if (!deepVisit) {
163             return null;
164         }
165 
166         ClassModelImpl cm;
167         try {
168              cm = (ClassModelImpl) type;
169         } catch (Exception e) {
170             return null;
171         }
172 
173         org.objectweb.asm.Type asmType = org.objectweb.asm.Type.getType(desc);
174 //        if (type==org.objectweb.asm.Type.INT_TYPE || type==org.objectweb.asm.Type.BOOLEAN_TYPE) {
175 //            return null;
176 //        }
177 
178         TypeProxy<?> fieldType =  typeBuilder.getHolder(asmType.getClassName());
179         final FieldModelImpl field = typeBuilder.getFieldModel(name, fieldType, cm);
180         visitingContext.field = field;
181 
182         // reverse index.
183         fieldType.getRefs().add(field);
184 
185         // forward index
186         cm.addField(field);
187         return fieldVisitor;
188     }

190     @Override
191     public MethodVisitor visitMethod(int access, String name, String desc, String signature, String[] exceptions) {
192         if (!deepVisit) {
193             return null;
194         }
195 
196         ExtensibleType cm;
197         try {
198              cm = (ExtensibleType) type;
199         } catch (Exception e) {
200             return null;
201         }
202 
203         final MethodModelImpl method = new MethodModelImpl(name, cm);
204         type.addMethod(method);
205         visitingContext.method = method;
206         return methodVisitor;
207     }

225     private class ModelMethodVisitor extends EmptyVisitor implements FieldVisitor {
226 
227         private final VisitingContext context;
228 
229         private ModelMethodVisitor(VisitingContext context) {
230             this.context = context;
231         }
232 
233         @Override
234         public AnnotationVisitor visitAnnotation(String desc, boolean visible) {
235 
236             if (context.method==null) {
237                 // probably an annotation method, ignore
238                 return null;
239             }
240             AnnotationTypeImpl annotationType = (AnnotationTypeImpl) typeBuilder.getType(Opcodes.ACC_ANNOTATION, unwrap(desc), null);
241             AnnotationModelImpl annotationModel = new AnnotationModelImpl(context.method, annotationType);
242 
243             // reverse index.
244             annotationType.getAnnotatedElements().add(context.method);
245 
246             // forward index
247             context.method.addAnnotation(annotationModel);
248             context.annotation = annotationModel;
249             return annotationVisitor;
250         }
251 
252         @Override
253         public void visitEnd() {
254             context.method=null;
255         }
256     }
