65      /**
66       * Context builder

68      public static class Builder {
69          Logger logger=Logger.getLogger("org.glassfish.hk2.classmodel");
70          ExecutorService executorService=null;
71          ArchiveSelector archiveSelector=null;
72          ParsingConfig config=null;
73          ResourceLocator locator=null;
74  
75          /**
76           * Returns the configured or default logger for the class-model library.
77           *
78           * @return the current logger associated with this builder, either
79           * set using {@link #logger(java.util.logging.Logger)} method, either
80           * using the default logger for this library.
81           */
82          public Logger logger() {
83              return logger;
84          }

86          /**
87           * Sets the logger to be used during the parsing activity.
88           *
89           * @param logger a logger instance
90           * @return itself
91           */
92          public Builder logger(Logger logger) {
93              this.logger = logger;
94              return this;
95          }
96  
97          /**
98           * Sets the executor service to be used to spawn threads during the
99           * parsing activity. The parsing activity is an asynchronous process
100          * that can choose to spawn threads to handle sub part of the handling
101          * process.
102          *
103          * @param service the executor service to be used during the parsing
104          * activity
105          * @return itself
106          */
107         public Builder executorService(ExecutorService service) {
108             this.executorService = service;
109             return this;
110         }
111 
112         /**
113          * Sets the archive selector that can selects which jar should be parsed
114          * during the parsing activity. This is particularly useful when the
115          * parser is configured to parse an entire directory of jars but only
116          * needs to actually parse those jars depending on some environmental
117          * condition or if the jar shows a particular stigma (like a jar entry
118          * existence).
119          *
120          * @param selector the archive selector.
121          * @return itself
122          */
123         public Builder archiveSelector(ArchiveSelector selector) {
124             this.archiveSelector =  selector;
125             return this;
126         }
127 
128         /**
129          * Sets the resource locator that can be used to load and parse extra
130          * types that were referenced during the parsing but could not be parsed
131          * due to their absence from the input archive set. The parser will call
132          * the {@link ResourceLocator} to give a chance to the caller to selectively
133          * add such unvisited types to the parsing activity
134          *
135          * @param locator a resource locator instance
136          * @return itself
137          */
138         public Builder locator(ResourceLocator locator) {
139             this.locator = locator;
140             return null;
141         }
142 
143         /**
144          * Sets the parsing config that can be used to select which types should
145          * be exhaustively visited (fields + methods visits) or not.
146          *
147          * @param config the config instance
148          * @return itself
149          */
150         public Builder config(ParsingConfig config) {
151             this.config = config;
152             return this;
153         }
154 
155         /**
156          * Build the final ParsingContext with the provided configuration.
157          *

159          */
160         public ParsingContext build() {
161             return new ParsingContext(this);
162         }

167     final TypesCtr types = new TypesCtr();
168     final ExecutorService executorService;
169     final ArchiveSelector archiveSelector;
170     final Logger logger;
171     final ParsingConfig config;
172     final ResourceLocator locator;
173 
174     private ParsingContext(Builder builder) {
175 //        Runtime runtime = Runtime.getRuntime();
176         this.executorService = builder.executorService;
177         this.archiveSelector = builder.archiveSelector;
178         this.logger = builder.logger;
179         this.locator = builder.locator;
180         this.config = builder.config!=null?builder.config:new ParsingConfig() {
181             final Set<String> emptyList = Collections.emptySet();
182             @Override
183             public Set<String> getAnnotationsOfInterest() {
184                 return emptyList;
185             }
186 
187             @Override
188             public Set<String> getTypesOfInterest() {
189                 return emptyList;
190             }
191 
192             @Override
193             public boolean modelUnAnnotatedMembers() {
194                 return false;
195             }
196         };
197     }

200     Map<URI, TypeBuilder> builders = new HashMap<URI, TypeBuilder>();
201 
202     public synchronized TypeBuilder getTypeBuilder(URI definingURI) {
203         TypeBuilder builder = builders.get(definingURI);
204         if (builder==null) {
205             builder = new TypesImpl(types, definingURI);
206             builders.put(definingURI, builder);
207         }
208         return builder;
209     }

217      */
218     public Types getTypes() {
