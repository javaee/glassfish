41  package com.sun.enterprise.module.maven;
42  
43  import org.apache.maven.plugin.AbstractMojo;
44  import org.apache.maven.plugin.MojoExecutionException;
45  import org.apache.maven.plugin.MojoFailureException;
46  import org.apache.maven.project.MavenProject;
47  import org.apache.maven.shared.osgi.Maven2OsgiConverter;
48  import org.apache.maven.artifact.versioning.DefaultArtifactVersion;
49  
50  /**
51   * Converts the project version into the OSGi format and
52   * set that to "project.osgi.version" property.
53   * It can be configured to drop certain portions from the
54   * version. See {@link #dropVersionComponent}.
55   *
56   * @author Kohsuke Kawaguchi
57   * @author Sanjeeb.Sahoo@Sun.COM
58   * @goal compute-osgi-version
59   * @threadSafe
60   * @phase validate
61   * @requiresProject
62   */
63  public class OsgiVersionMojo extends AbstractMojo {
64      /**
65       * The maven project.
66       *
67       * @parameter expression="${project}"
68       * @required
69       * @readonly
70       */
71      protected MavenProject project;
72  
73      /**
74       * Flag used to determine what components of the version will be used
75       * in OSGi version.
76       * An OSGi version has four parts as shown below:
77       * major.minor.micro.qualifer.
78       * It is not always desirable to use all four parts while
79       * exporting packages. In fact, maven version and OSGi version
80       * behave just opposite during version comparison as shown below:
81       * a maven version 1.2.3-SNAPSHOT is mapped to OSGi version 1.2.3.SNAPSHOT.
82       * In maven, 1.2.3 > 1.2.3-SNAPSHOT, but in OSGi, 1.2.3 < 1.2.3.SNAPSHOT.
83       * So, it is highly desirable to drop qualifier while computing the version.
84       * Instead of hardcoding the policy, we let user tell us what portions will
85       * be used in the OSGi version. If they ask us to drop minor, then only
86       * major will be used. Similarly, if they ask us to drop qualifier, then
87       * major, minor and micro portions will be used.
88       * @parameter
89       */
90      protected String dropVersionComponent;
91  
92      /**
93       * @parameter default-value="project.osgi.version"
94       */
95      protected String versionPropertyName;
96  
97      private enum VERSION_COMPONENT {major, minor, micro, qualifier};
98  
99      /**
100      * @component
101      */
102     protected Maven2OsgiConverter converter;
103 
104     public void execute() throws MojoExecutionException, MojoFailureException {
105         DefaultArtifactVersion projectVersion =
106                 new DefaultArtifactVersion(project.getVersion());
107         VERSION_COMPONENT compToDrop = dropVersionComponent == null ?
108                 null : VERSION_COMPONENT.valueOf(dropVersionComponent);
109 
110         DefaultArtifactVersion newVersion = projectVersion;
111         if (compToDrop != null) {
112             switch (compToDrop) {
113                 case major: {
114                     newVersion = new DefaultArtifactVersion("0.0.0");
115                     break;
116                 }
117                 case minor: {
118                     final int major = projectVersion.getMajorVersion();
119                     newVersion = new DefaultArtifactVersion(major +"");
120                     break;
121                 }
122                 case micro: {
123                     final int major = projectVersion.getMajorVersion();
124                     final int minor = projectVersion.getMinorVersion();
125                     newVersion = new DefaultArtifactVersion(major + "." + minor);
126                     break;
127                 }
128                 case qualifier: {
129                     final int major = projectVersion.getMajorVersion();
130                     final int minor = projectVersion.getMinorVersion();
131                     final int micro = projectVersion.getIncrementalVersion();
132                     newVersion = new DefaultArtifactVersion(major + "." + minor + "." + micro);
133                     break;
134                 }
135             }
136         }
137         String v = converter.getVersion(newVersion.toString());
138 
139         getLog().debug("OSGi Version for "+project.getVersion()+" is "+v);
140         getLog().debug("It is set in project property called "+ versionPropertyName);
141         project.getProperties().put(versionPropertyName,v);
142     }
