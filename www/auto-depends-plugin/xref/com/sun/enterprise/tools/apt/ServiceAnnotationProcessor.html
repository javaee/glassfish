69  /**
70   * This class is proccessing the @{@link Service} annotation and generates
71   * META-INF/services style text file for each interface annotated with @Contract
72   *
73   * @author Jerome Dochez
74   */
75  public class ServiceAnnotationProcessor implements AnnotationProcessor, RoundCompleteListener {
76  
77      private final boolean debug;
78      private final AnnotationProcessorEnvironment env;
79      private Map<String, ServiceFileInfo> serviceFiles = new HashMap<String, ServiceFileInfo>();
80  
81      /**
82       * Creates a new instance of ServiceAnnotationProcessor
83       */
84      public ServiceAnnotationProcessor(AnnotationProcessorEnvironment env) {
85          this.env = env;
86          env.addListener(this);
87  
88          debug = env.getOptions().containsKey("-Adebug");
89          if (debug) {
90              env.getMessager().printNotice(env.getOptions().toString());
91          }
92          
93          // load all existing meta-inf files...
94          loadExistingMetaInfFiles();
95      }

98      /**
99       * Loads all existing META-INF/services file from our destination directory
100      * This is usuful because during incremental builds, not all source files
101      * are recompiled, henve we cannot rewrite the META-INF/services file from
102      * scratch each time, but append/remove entries from it as necessary.
103      */
104     protected void loadExistingMetaInfFiles() {
105         String outDirectory = env.getOptions().get("-s");
106         if (outDirectory==null) {
107             outDirectory = System.getProperty("user.dir");
108         }
109         File outDir = new File(new File(outDirectory),"META-INF/services").getAbsoluteFile();
110         if (debug) {
111             env.getMessager().printNotice("Output dir is " + outDir.getAbsolutePath());
112         }
113 
114         if (!outDir.exists()) {
115             return;
116         }
117         for (File file : outDir.listFiles()) {
118             if(file.isDirectory())  continue;
119             Set<String> entries = new HashSet<String>();
120             try {
121                 FileReader reader = new FileReader(file);
122                 LineNumberReader lineReader = new LineNumberReader(reader);
123                 String line = lineReader.readLine();
124                 while (line != null) {
125                     entries.add(line);
126                     line = lineReader.readLine();
127                 }
128             } catch (IOException e) {
129                 env.getMessager().printError(e.getMessage());
130             }
131             ServiceFileInfo info = new ServiceFileInfo(file.getName(), entries);
132             serviceFiles.put(file.getName(), info);
133         }
134     }

136     /**
137      * Annotation processor entry point, we are using a visitor pattern the visit
138      * only the class declaration.
139      */
140     public void process() {
141         DeclarationVisitor scanner = DeclarationVisitors.getDeclarationScanner(
142             new ListClassVisitor(),
143             DeclarationVisitors.NO_OP);
144 
145         for (TypeDeclaration typeDecl : env.getSpecifiedTypeDeclarations())
146             typeDecl.accept(scanner);
147     }
148 
149     /**
150      * Invoked on all class declaration, use the cached META-INF/servies
151      * information and the mirror APIs to find out if classes need to be
152      * added or removed from the generated service file.
153      */
154     private class ListClassVisitor extends SimpleDeclarationVisitor {
155         public void visitClassDeclaration(ClassDeclaration d) {
156             if (debug) {
157                 env.getMessager().printNotice("Visiting " + d.getQualifiedName());
158             }
159             Service service = d.getAnnotation(Service.class);
160             if (debug) {
161                 env.getMessager().printNotice("Service annotation = " + service);
162             }
163             if (service != null) {
164 
165                 // look for @ContractProvided interface
166                 checkContractProvided(d);
167                 
168                 // look for contract in interfaces
169                 for (InterfaceType intf : d.getSuperinterfaces()) {
170                     checkContract(intf.getDeclaration(), d);
171                 }
172 
173                 // look for contract in super classes
174                 ClassDeclaration sd = d;
175                 while(sd.getSuperclass()!=null) {
176                     sd = sd.getSuperclass().getDeclaration();
177                     checkContract(sd,d);
178                 }
179             } else {
180                 // we need to check if that class previously add an @Service annotation so
181                 // we remove the entry from the META-INF file
182                 for (ServiceFileInfo info : serviceFiles.values()) {
183                     if (debug) {
184                         env.getMessager().printNotice("Checking against " + info.getServiceName());
185                     }
186                     for (String implementor : info.getImplementors()) {
187                         if (implementor.equals(d.getQualifiedName())) {
188                             if (debug) {
189                                 env.getMessager().printNotice("Need to remove " + implementor);
190                             }
191                             info.getImplementors().remove(implementor);
192                             try {
193                                 info.createFile(env);
194                             } catch(IOException ioe) {
195                                 env.getMessager().printError(ioe.getMessage());
196                             }
197                             return;    
198                         }
199                     }
200                 }
201             }
202 
203             // check for meta annotations
204             for(AnnotationMirror a : d.getAnnotationMirrors()) {
205                 AnnotationTypeDeclaration atd = a.getAnnotationType().getDeclaration();
206                 Contract c = atd.getAnnotation(Contract.class);
207                 if(c!=null) {
208                     // this is a contract annotation
209                     createContractImplementation(atd.getQualifiedName(),d);
210                 }
211             }
212         }
213 
214 
215         private void checkContractProvided(ClassDeclaration impl) {
216             ContractProvided provided = impl.getAnnotation(ContractProvided.class);
217 
218             String intfName = null;
219             if (provided != null) {
220 
221                 try {
222                     provided.value();
223                 } catch (MirroredTypeException e) {
224                     intfName = e.getQualifiedName();
225                 }
226                 if (debug) {
227                     System.out.println("Provided is " + intfName);
228                 }
229                 createContractImplementation(intfName, impl);
230             }
231 
232         }
233 
234         private void checkContract(TypeDeclaration type, ClassDeclaration impl) {
235             Contract contract = type.getAnnotation(Contract.class);
236 
237             if (contract != null)
238                 createContractImplementation(type.getQualifiedName(), impl);
239         }
240     }
241 
242     /**
243      * Records the contract&lt;->service relationship.
244      *
245      * @param contractName
246      *      FQCN of the contract type/annotation.
247      * @param impl
248      *      Implementation class.
249      */
250     private void createContractImplementation(String contractName, ClassDeclaration impl) {
251         ServiceFileInfo info;
252         if (!serviceFiles.containsKey(contractName)) {
253             info = new ServiceFileInfo(contractName, new HashSet<String>());
254             serviceFiles.put(contractName, info);
255         } else {
256             info = serviceFiles.get(contractName);
257         }
258 
259         if (info.getImplementors().add(impl.getQualifiedName())) {
260             try {
261                 info.createFile(env);
262             } catch (IOException ioe) {
263                 env.getMessager().printError(ioe.getMessage());
264             }
265         }
266     }

268     /**
269      * Notification of round complete, I use this to actually write the
270      * service file content and close the file.
271      * 
272      * @param e
273      */
274     public void roundComplete(RoundCompleteEvent e) {
275         for (ServiceFileInfo info : serviceFiles.values()) {
276             if (info.isDirty()) {
277                 if (debug) {
278                     env.getMessager().printNotice("Creating META-INF/services "+ info.getServiceName() + " file");
279                 }
280                 PrintWriter writer = info.getWriter();
281                 for (String implementor : info.getImplementors()) {
282                     if (debug) {
283                         env.getMessager().printNotice(" Implementor " + implementor);
284                     }
285                     writer.println(implementor);
286                 }
287                 writer.close();
288             }
289         }
290     }
