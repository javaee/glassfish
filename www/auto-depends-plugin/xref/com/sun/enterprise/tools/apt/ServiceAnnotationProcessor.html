52  /***
53   * This class is proccessing the @Service annotation and generates
54   * META-INF/services style text file for each interface annotated with @Contract
55   *
56   * @author Jerome Dochez
57   */
58  public class ServiceAnnotationProcessor implements AnnotationProcessor, RoundCompleteListener {
59  
60      private final boolean debug;
61      private final AnnotationProcessorEnvironment env;
62      private Map<String, ServiceFileInfo> serviceFiles = new HashMap<String, ServiceFileInfo>();
63  
64      /***
65       * Creates a new instance of ServiceAnnotationProcessor
66       */
67      public ServiceAnnotationProcessor(AnnotationProcessorEnvironment env) {
68          this.env = env;
69          env.addListener(this);
70  
71          debug = env.getOptions().containsKey("-Adebug");
72          if (debug) {
73              env.getMessager().printNotice(env.getOptions().toString());
74          }
75          
76          // load all existing meta-inf files...
77          loadExistingMetaInfFiles();
78      }
79  
80  
81      /***
82       * Loads all existing META-INF/services file from our destination directory
83       * This is usuful because during incremental builds, not all source files
84       * are recompiled, henve we cannot rewrite the META-INF/services file from
85       * scratch each time, but append/remove entries from it as necessary.
86       */
87      protected void loadExistingMetaInfFiles() {
88          String outDirectory = env.getOptions().get("-s");
89          if (outDirectory==null) {
90              outDirectory = System.getProperty("user.dir");
91          }
92          File outDir = new File(new File(outDirectory),"META-INF/services").getAbsoluteFile();
93          if (debug) {
94              env.getMessager().printNotice("Output dir is " + outDir.getAbsolutePath());
95          }
96  
97          if (!outDir.exists()) {
98              return;
99          }
100         for (File file : outDir.listFiles()) {
101             if(file.isDirectory())  continue;
102             Set<String> entries = new HashSet<String>();
103             try {
104                 FileReader reader = new FileReader(file);
105                 LineNumberReader lineReader = new LineNumberReader(reader);
106                 String line = lineReader.readLine();
107                 while (line != null) {
108                     entries.add(line);
109                     line = lineReader.readLine();
110                 }
111             } catch (IOException e) {
112                 env.getMessager().printError(e.getMessage());
113             }
114             ServiceFileInfo info = new ServiceFileInfo(file.getName(), entries);
115             serviceFiles.put(file.getName(), info);
116         }
117     }

119     /***
120      * Annotation processor entry point, we are using a visitor pattern the visit
121      * only the class declaration.
122      */
123     public void process() {
124 
125         for (TypeDeclaration typeDecl : env.getSpecifiedTypeDeclarations())
126             typeDecl.accept(DeclarationVisitors.getDeclarationScanner(new ListClassVisitor(),
127                     DeclarationVisitors.NO_OP));
128     }
129 
130     /***
131      * Invoked on all class declaration, use the cached META-INF/servies
132      * information and the mirror APIs to find out if classes need to be
133      * added or removed from the generated service file.
134      */
135     private class ListClassVisitor extends SimpleDeclarationVisitor {
136         
137         public void visitClassDeclaration(ClassDeclaration d) {
138             if (debug) {
139                 env.getMessager().printNotice("Visiting " + d.getQualifiedName());
140             }
141             Service service = d.getAnnotation(Service.class);
142             if (debug) {
143                 env.getMessager().printNotice("Service annotation = " + service);
144             }
145             if (service != null) {
146 
147                 // look for @ContractProvided interface
148                 checkContractProvided(d);
149                 
150                 // look for contract in interfaces
151                 for (InterfaceType intf : d.getSuperinterfaces()) {
152                     checkContract(intf.getDeclaration(), d);
153                 }
154 
155                 // look for contract in super classes
156                 ClassDeclaration sd = d;
157                 while(sd.getSuperclass()!=null) {
158                     sd = sd.getSuperclass().getDeclaration();
159                     checkContract(sd,d);
160                 }
161             } else {
162                 // we need to check if that class previously add an @Service annotation so
163                 // we remove the entry from the META-INF file
164                 for (ServiceFileInfo info : serviceFiles.values()) {
165                     if (debug) {
166                         env.getMessager().printNotice("Checking against " + info.getServiceName());
167                     }
168                     for (String implementor : info.getImplementors()) {
169                         if (implementor.equals(d.getQualifiedName())) {
170                             if (debug) {
171                                 env.getMessager().printNotice("Need to remove " + implementor);
172                             }
173                             info.getImplementors().remove(implementor);
174                             try {
175                                 info.createFile(env);
176                             } catch(IOException ioe) {
177                                 env.getMessager().printError(ioe.getMessage());
178                             }
179                             return;    
180                         }
181                     }
182                 }
183             }
184 
185             // check for meta annotations
186             for(AnnotationMirror a : d.getAnnotationMirrors()) {
187                 AnnotationTypeDeclaration atd = a.getAnnotationType().getDeclaration();
188                 Contract c = atd.getAnnotation(Contract.class);
189                 if(c!=null) {
190                     // this is a contract annotation
191                     createContractImplementation(atd.getQualifiedName(),d.getQualifiedName());
192                 }
193             }
194         }
195 
196 
197         private void checkContractProvided(ClassDeclaration impl) {
198             ContractProvided provided = impl.getAnnotation(ContractProvided.class);
199 
200             String intfName = null;
201             if (provided != null) {
202 
203                 try {
204                     provided.value();
205                 } catch (MirroredTypeException e) {
206                     intfName = e.getQualifiedName();
207                 }
208                 if (debug) {
209                     System.out.println("Provided is " + intfName);
210                 }
211                 createContractImplementation(intfName, impl.getQualifiedName());
212             }
213 
214         }
215 
216         private void checkContract(TypeDeclaration type, ClassDeclaration impl) {
217             Contract contract = type.getAnnotation(Contract.class);
218 
219             String intfName = null;
220 
221             if (contract != null || (intfName != null && intfName.equals(type.getQualifiedName()))) {
222                 createContractImplementation(type.getQualifiedName(), impl.getQualifiedName());
223             }
224         }
225     }
226 
227     /***
228      * Records the contract&lt;->service relationship.
229      */
230     private void createContractImplementation(String contractName, String implementationName) {
231         ServiceFileInfo info;
232         if (!serviceFiles.containsKey(contractName)) {
233             info = new ServiceFileInfo(contractName, new HashSet<String>());
234             serviceFiles.put(contractName, info);
235         } else {
236             info = serviceFiles.get(contractName);
237         }
238 
239         if (!info.getImplementors().contains(implementationName)) {
240             info.getImplementors().add(implementationName);
241             try {
242                 info.createFile(env);
243             } catch (IOException ioe) {
244                 env.getMessager().printError(ioe.getMessage());
245             }
246 
247         }
248     }

250     /***
251      * Notification of round complete, I use this to actually write the
252      * service file content and close the file.
253      * 
254      * @param e
255      */
256     public void roundComplete(RoundCompleteEvent e) {
257         final boolean debug = env.getOptions().containsKey("-Adebug");
258         for (ServiceFileInfo info : serviceFiles.values()) {
259             if (info.isDirty()) {
260                 if (debug) {
261                     env.getMessager().printNotice("Creating META-INF/services "+ info.getServiceName() + " file");
262                 }
263                 PrintWriter writer = info.getWriter();
264                 for (String implementor : info.getImplementors()) {
265                     if (debug) {
266                         env.getMessager().printNotice(" Implementor " + implementor);
267                     }
268                     writer.println(implementor);
269                 }
270                 writer.close();
271             }
272         }
273     }
