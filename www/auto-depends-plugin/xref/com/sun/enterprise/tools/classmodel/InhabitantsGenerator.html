104   /**
105    * Construction with all of the parameters needed to generate.
106    * 
107    * @param descriptor optionally a preconfigured inhabitants descriptor
108    * @param inhabitantsSourceFiles required set of inhabitants source files (directories | jars)
109    * @param inhabitantsClassPath the fully qualified classpath in order to resolve class-model
110    */
111   public InhabitantsGenerator(InhabitantsDescriptor descriptor,
112       ClassPath inhabitantsSourceFiles,
113       ClassPath inhabitantsClassPath) {
114     
115     inhabitantsClassPath = filterIgnores(inhabitantsClassPath);
116     
117     // TODO:
118 //    // have caching on all of the time
119 //    ClassPathAdvisor advisor = new FileCachingClassPathAdvisor();
120     ClassPathAdvisor advisor = null;
121     
122     logger.log(Level.FINE, "working classpath: {0}", inhabitantsClassPath);
123     this.ipcGen = InhabitantsParsingContextGenerator.
124           create(null, createExecutorService(), inhabitantsClassPath, advisor);
125     
126     if (null != descriptor) {
127       this.descriptor = descriptor;
128     } else {
129       this.descriptor = new InhabitantsDescriptor();
130       this.descriptor.setComment("by " + getClass().getName());
131     }
132   
133     try {
134       logger.log(Level.FINE, "Parsing: {0}", inhabitantsSourceFiles);
135       ipcGen.parse(inhabitantsSourceFiles.getFileEntries());
136     } catch (IOException e) {
137       throw new RuntimeException(e);
138     }
139     
140     codeSourceFilter = new CodeSourceFilter(inhabitantsSourceFiles);
141   }

148   private ClassPath filterIgnores(ClassPath inhabitantsClassPath) {
149     LinkedHashSet<File> newFiles = new LinkedHashSet<File>();
150 
151     Set<File> entries = new LinkedHashSet<File>(inhabitantsClassPath.getFileEntries());
152     for (File file : entries) {
153       if (!IGNORE.contains(file.getName())) {
154         logger.log(Level.FINE, "accepting {0}", file);
155         newFiles.add(file);
156       } else {
157         logger.log(Level.FINE, "filtering {0}", file);
158       }
159     }
160     
161     ClassPath newClassPath = ClassPath.create(null, newFiles);
162     return newClassPath;
163   }

165   public void generate(File targetInhabitantFile, boolean sort) throws IOException {
166     File parent = targetInhabitantFile.getParentFile();
167     if (null != parent) {
168       parent.mkdirs();
169     }
170     
171     ByteArrayOutputStream out = null;
172     PrintWriter w;
173     if (sort) {
174       out = new ByteArrayOutputStream();
175       w = new PrintWriter(out);
176     } else {
177       w = new PrintWriter(targetInhabitantFile, "UTF-8");
178     }
179     
180     try {
181       generate(w);
182     } finally {
183       w.close();
184     }
185     
186     if (descriptor.isEmpty()) {
187       targetInhabitantFile.delete();
188       return;
189     }
190 
191     if (sort) {
192       String sorterdInhabitants = Utilities.sortInhabitantsDescriptor(out.toString(), sort);
193       FileOutputStream fos = new FileOutputStream(targetInhabitantFile);
194       fos.write(sorterdInhabitants.getBytes());
195       fos.close();
196     }
197   }

199   public void generate(PrintWriter writer) throws IOException {
200     descriptor.clear();
201 
202     InhabitantsParserDescriptorWriter ip = new InhabitantsParserDescriptorWriter();
203     InhabitantsFeed feed = InhabitantsFeed.create(new Habitat(), ip);
204     feed.populate(ipcGen);
205 
206     // flush the last inhabitant to the descriptor
207     ip.flush();
208     
209     // i/o the descriptor(s) out
210     descriptor.write(writer);
211   }

252   static ClassPath getFullInhabitantsClassPath() {
253     String arg;
254     ClassPath inhabitantsClassPath = null;
255     arg = System.getProperty(PARAM_INHABITANTS_CLASSPATH);
256     if (null == arg || arg.isEmpty()) {
257       inhabitantsClassPath = ClassPath.create(null, false);
258       System.err.println("WARNING: sysprop " + PARAM_INHABITANTS_CLASSPATH + 
259           " is missing; defaulting to system classpath; this may result in an invalid inhabitants file being created.");
260       if (logger.isLoggable(Level.FINER)) {
261         logger.log(Level.FINER, "classpath={0}", inhabitantsClassPath.getFileEntries());
262       }
263     } else {
264       inhabitantsClassPath = ClassPath.create(null, arg);
265     }
266     return inhabitantsClassPath;
267   }

269   static ClassPath getScopedInhabitantCodeSources() {
270     String arg;
271     arg = System.getProperty(PARAM_INHABITANTS_SOURCE_FILES);
272     if (null == arg || arg.isEmpty()) {
273       System.err.println("ERROR: sysprop " + PARAM_INHABITANTS_SOURCE_FILES + " is expected");
274       System.exit(-1);
275     }
276     List<File> sourceFiles = new ArrayList<File>();
277     String [] sourceFileNames = arg.split(File.pathSeparator);
278     for (String sourceFile : sourceFileNames) {
279       File source = new File(sourceFile);
280       if (source.exists()) {
281         sourceFiles.add(source);
282       } else {
283         System.err.println("WARNING: can't find " + sourceFile);
284       }
285     }
286     ClassPath inhabitantsSourceFiles = ClassPath.create(null, sourceFiles);
287     return inhabitantsSourceFiles;
288   }

291   static File getInhabitantFile(String key, boolean mustExist) {
292     String arg = System.getProperty(key);
293     if (null == arg || arg.isEmpty()) {
294       System.err.println("ERROR: sysprop " + key + " is expected");
295       System.exit(-1);
296     }
297     File targetInhabitantFile = new File(arg);
298     if (mustExist && !targetInhabitantFile.exists()) {
299       System.err.println("ERROR: " + targetInhabitantFile + " does not exist");
300       System.exit(-1);
301     }
302     return targetInhabitantFile;
303   }

306   /**
307    * Marshals descriptor lines from parsed inhabitants.
308    */
309   private class InhabitantsParserDescriptorWriter extends InhabitantsParser {
310 
311     private Inhabitant<?> pendingInhabitant;
312     private List<String> pendingUnamedContracts;
313     
314     private InhabitantsParserDescriptorWriter() {
315       super(null);
316     }
317     
318     /**
319      * Writes out any last pending inhabitant
320      */
321     public void flush() {
322       if (null != pendingInhabitant) {
323         logger.log(Level.INFO, "adding inhabitant {0} with contracts {1}", new Object[] {pendingInhabitant, pendingUnamedContracts});
324         descriptor.putAll(pendingInhabitant.typeName(), null, pendingUnamedContracts, null, pendingInhabitant.metadata());
325         pendingUnamedContracts = null;
326         pendingInhabitant = null;
327       }
328     }
329 
330     /**
331      * Controls the filtering.  This decides whether add(i) or addIndex(...) is ultimately called.
332      */
333     @Override
334     protected boolean isFilteredInhabitant(String typeName) {
335       if (null == codeSourceFilter || codeSourceFilter.matches(typeName)) {
336         logger.log(Level.FINE, "accepting {0}", typeName);
337         return false; // true==(ignore it); false==(include it)
338       } else {
339         logger.log(Level.FINE, "filtering {0}", typeName);
340         return true;
341       }
342     }
343 
344     /**
345      * The idea is to put the inhabitant into the descriptors instead of the habitat here
346      */
347     @Override
348     public void add(Inhabitant<?> i) {
349 //      System.out.println("add\t" + i + " " + i.metadata());
350 
351       // flush any previous inhabitant definition
352       flush();
353       
354       pendingInhabitant = i;
355     }
356     
357     /**
358      * The idea is to merge the inhabitant index into the descriptors instead of the habitat here
359      */
360     @Override
361     public void addIndex(Inhabitant<?> i, String typeName, String name) {
362       // don't flush since we are building up the inhabitant descriptor definition
363       
364 //      System.out.println("addIndex\t" + i + "; " + typeName + "; " + name);
365 
366       if (null == pendingUnamedContracts) {
367         pendingUnamedContracts = new ArrayList<String>();
368       }
369 
370       if (null == name) {
371         pendingUnamedContracts.add(typeName);
372       } else {
373         pendingUnamedContracts.add(typeName + ":" + name);
374       }
375     }
376   }
