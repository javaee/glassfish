56  /**
57   * Provides ability to filter an inhabitants file to only include what
58   * is designated by a specified classpath.
59   * 
60   * @author Jeff Trent
61   */
62  public class InhabitantsFilter extends Constants {
63  
64    private static Logger logger = Logger.getLogger(InhabitantsFilter.class.getName());
65    
66    protected InhabitantsFilter() {}
67  
68    /**
69     * Requires {@link Constants#PARAM_INHABITANT_SOURCE_FILE}, {@link Constants#PARAM_INHABITANT_TARGET_FILE},
70     * {@link Constants#PARAM_INHABITANTS_SOURCE_FILES}, and {@link Constants#PARAM_INHABITANTS_SORTED}
71     * to be passed.
72     * 
73     * @param args not used
74     * @throws Exception
75     */
76    public static void main(String[] args) throws Exception {
77      File sourceInhabitantFile = InhabitantsGenerator.getInhabitantFile(PARAM_INHABITANT_SOURCE_FILE, false);
78      if (!sourceInhabitantFile.exists()) {
79        logger.log(Level.INFO, "Nothing to do.");
80        return;
81      }
82      File targetInhabitantFile = InhabitantsGenerator.getInhabitantFile(PARAM_INHABITANT_TARGET_FILE, false);
83      ClassPath targetClassPath = InhabitantsGenerator.getScopedInhabitantCodeSources();
84      CodeSourceFilter filter = new CodeSourceFilter(targetClassPath);
85      logger.log(Level.FINE, "filter is {0}", filter);
86      
87      InhabitantsDescriptor inDescriptor = new InhabitantsDescriptor();
88      logger.log(Level.FINE, "source file is {0}", sourceInhabitantFile);
89      inDescriptor.load(sourceInhabitantFile);
90  
91      InhabitantsDescriptor outDescriptor = new InhabitantsDescriptor();
92      outDescriptor.enableDateOutput(false);
93      
94      boolean sorted = Boolean.getBoolean(PARAM_INHABITANTS_SORTED);
95      
96      process(inDescriptor, outDescriptor, filter);
97      writeInhabitantsFile(targetInhabitantFile, outDescriptor, sorted);
98    }
99  
100   /**
101    * Builds the outDescriptor based on filtered inDescriptor contents.
102    * 
103    * @param inDescriptor the source input descriptor
104    * @param outDescriptor the destination descriptor
105    * @param filter the filter
106    */
107   public static void process(InhabitantsDescriptor inDescriptor,
108       InhabitantsDescriptor outDescriptor,
109       CodeSourceFilter filter) {
110     
111     for (Entry<String, String> entry : inDescriptor.entrySet()) {
112       String key = entry.getKey();
113       String value = entry.getValue();
114       String clazz = classOf(value);
115       if (null == filter || filter.matches(clazz)) {
116         outDescriptor.put(key, value);
117       }
118     }
119   }
120 
121   /**
122    * Writes the inhabitants file to disk.
123    * 
124    * @param targetInhabitantFile
125    * @param outDescriptor
126    * @param sorted
127    */
128   public static void writeInhabitantsFile(
129       File targetInhabitantFile,
130       InhabitantsDescriptor outDescriptor,
131       boolean sorted) throws FileNotFoundException, IOException {
132     if (outDescriptor.isEmpty()) {
133       targetInhabitantFile.delete();
134     } else {
135       logger.log(Level.INFO, "writing file {0}", targetInhabitantFile);
136       
137       if (!sorted) {
138         outDescriptor.write(targetInhabitantFile);
139       } else {
140         File parent = targetInhabitantFile.getParentFile();
141         if (null != parent) {
142           parent.mkdirs();
143         }
144         
145         ByteArrayOutputStream out = new ByteArrayOutputStream();
146         PrintWriter w = new PrintWriter(out);
147         outDescriptor.write(w);
148         w.close();
149         
150         String sorterdInhabitants = Utilities.sortInhabitantsDescriptor(out.toString(), sorted);
151         FileOutputStream fos = new FileOutputStream(targetInhabitantFile);
152         fos.write(sorterdInhabitants.getBytes());
153         fos.close();
154       }
155     }
156   }
157 
158   static String classOf(String value) {
159     return Utilities.split(value).getOne("class");
160   }
