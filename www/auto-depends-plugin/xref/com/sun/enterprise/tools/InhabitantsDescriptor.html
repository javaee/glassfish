66  /**

113   /**
114    * Loads an existing file.
115    */
116   public void load(File f) throws IOException {
117     FileInputStream in = new FileInputStream(f);
118     try {
119       InhabitantsScanner scanner = new InhabitantsScanner(in, f.getPath());
120       for (InhabitantParser kvpp : scanner) {
121         put(kvpp.getImplName(), kvpp.getLine());
122       }
123     } finally {
124       if (null != in) {
125         in.close();
126       }
127     }
128   }

137   @SuppressWarnings("unchecked")
138   public String putAll(String service,
139       Collection<String> contracts,
140       Collection<String> annotations,
141       String name,
142       Object metaObj) {
143     StringBuilder buf = new StringBuilder();
144     buf.append(InhabitantsFile.CLASS_KEY).append('=').append(service);
145     
146     if (null != contracts) {
147       for (String contract : contracts) {
148         buf.append(",").append(INDEX_KEY).append("=").append(contract);
149         if (null != name) {
150           buf.append(":").append(name);
151         }
152       }
153     }
154     
155     if (null != annotations) {
156       for (String contract : annotations) {
157         buf.append(",").append(INDEX_KEY).append("=").append(contract);
158       }
159     }
160     
161     if (metaObj instanceof Map) {
162       Map<String, String> meta = (Map<String, String>)metaObj;
163       for (Map.Entry<String, String> entry : meta.entrySet()) {
164         buf.append(",").append(entry.getKey()).append("=").append(entry.getValue());
165       }
166     } else if (metaObj instanceof MultiMap) {
167       MultiMap<String, String> meta = (MultiMap<String, String>)metaObj;
168       for (Map.Entry<String, List<String>> entry : meta.entrySet()) {
169         String key = entry.getKey();
170         List<String> vals = new ArrayList<String>(entry.getValue());
171         Collections.sort(vals);
172         for (String val : vals) {
173           buf.append(",").append(key).append("=").append(val);
174         }
175       }
176     }
177     
178     return put(service, buf.toString());
179   }

188   /**
189    * Writes the descriptor to a file.
190    */
191   public void write(File outputDir, Messager messager, String habitatName) {
192     File out = new File(new File(outputDir, InhabitantsFile.PATH), habitatName);
193     
194     if (!dirty) {
195       if (out.exists()) {
196         out.delete();
197       }
198       
199       return; // no need to write.
200     }
201 
202     try {
203       write(out);
204     } catch (IOException e) {
205       if (null == messager) {
206         logger.warning("Failed to write inhabitants file " + habitatName);
207       } else {
208         messager.printError("Failed to write inhabitants file " + habitatName, e);
209       }
210     }
211   }

213   public void write(File out) throws IOException {
214     File parent = out.getParentFile();
215     if (null != parent) {
216       parent.mkdirs();
217     }
218     
219     PrintWriter w = new PrintWriter(out, "UTF-8");
220     try {
221       write(w);
222     } finally {
223       w.close();
224     }
225   }

227   @SuppressWarnings("deprecation")
228   public void write(PrintWriter w) {
229     if (dateEnabled) {
230       w.println("# generated on " + new Date().toGMTString());
231     }
232     
233     if (commentEnabled && null != comment) {
234       w.println(comment);
235     }
236     
237     for (String line : values()) {
238       w.println(line);
239     }
240   }

242   @Override
243   public String toString() {
244     ByteArrayOutputStream os = new ByteArrayOutputStream();
245     PrintWriter w = new PrintWriter(os);
246     write(w);
247     w.close();
