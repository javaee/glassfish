40  package org.jvnet.hk2.config;
41  
42  import javax.xml.stream.XMLStreamWriter;
43  import javax.xml.stream.XMLStreamException;
44  import java.util.Stack;
45  
46  /**
47   * @author Kohsuke Kawaguchi
48   */
49  public class IndentingXMLStreamWriter extends DelegatingXMLStreamWriter {
50      private final static Object SEEN_NOTHING = new Object();
51      private final static Object SEEN_ELEMENT = new Object();
52      private final static Object SEEN_DATA = new Object();
53  
54      private Object state = SEEN_NOTHING;
55      private Stack<Object> stateStack = new Stack<Object>();
56  
57      private String indentStep = "  ";
58      private int depth = 0;
59  
60      public IndentingXMLStreamWriter(XMLStreamWriter writer) {
61          super(writer);
62      }
63  
64      /**
65       * Return the current indent step.
66       *
67       * <p>Return the current indent step: each start tag will be
68       * indented by this number of spaces times the number of
69       * ancestors that the element has.</p>
70       *
71       * @return The number of spaces in each indentation step,
72       *         or 0 or less for no indentation.
73       * @see #setIndentStep(int)
74       *
75       * @deprecated
76       *      Only return the length of the indent string.
77       */
78      public int getIndentStep() {
79          return indentStep.length();
80      }
81  
82  
83      /**
84       * Set the current indent step.
85       *
86       * @param indentStep The new indent step (0 or less for no
87       *        indentation).
88       * @see #getIndentStep()
89       *
90       * @deprecated
91       *      Should use the version that takes string.
92       */
93      public void setIndentStep(int indentStep) {
94          StringBuilder s = new StringBuilder();
95          for (; indentStep > 0; indentStep--) s.append(' ');
96          setIndentStep(s.toString());
97      }
98  
99      public void setIndentStep(String s) {
100         this.indentStep = s;
101     }
102 
103     private void onStartElement() throws XMLStreamException {
104         stateStack.push(SEEN_ELEMENT);
105         state = SEEN_NOTHING;
106         if (depth > 0) {
107             super.writeCharacters("\n");
108         }
109         doIndent();
110         depth++;
111     }
112 
113     private void onEndElement() throws XMLStreamException {
114         depth--;
115         if (state == SEEN_ELEMENT) {
116             super.writeCharacters("\n");
117             doIndent();
118         }
119         state = stateStack.pop();
120     }
121 
122     private void onEmptyElement() throws XMLStreamException {
123         state = SEEN_ELEMENT;
124         if (depth > 0) {
125             super.writeCharacters("\n");
126         }
127         doIndent();
128     }
129 
130     /**
131      * Print indentation for the current level.
132      *
133      * @exception org.xml.sax.SAXException If there is an error
134      *            writing the indentation characters, or if a filter
135      *            further down the chain raises an exception.
136      */
137     private void doIndent() throws XMLStreamException {
138         if (depth > 0) {
139             for (int i = 0; i < depth; i++)
140                 super.writeCharacters(indentStep);
141         }
142     }
143 
144 
145     public void writeStartDocument() throws XMLStreamException {
146         super.writeStartDocument();
147         super.writeCharacters("\n");
148     }
149 
150     public void writeStartDocument(String version) throws XMLStreamException {
151         super.writeStartDocument(version);
152         super.writeCharacters("\n");
153     }
154 
155     public void writeStartDocument(String encoding, String version) throws XMLStreamException {
156         super.writeStartDocument(encoding, version);
157         super.writeCharacters("\n");
158     }
159 
160     public void writeStartElement(String localName) throws XMLStreamException {
161         onStartElement();
162         super.writeStartElement(localName);
163     }
164 
165     public void writeStartElement(String namespaceURI, String localName) throws XMLStreamException {
166         onStartElement();
167         super.writeStartElement(namespaceURI, localName);
168     }
169 
170     public void writeStartElement(String prefix, String localName, String namespaceURI) throws XMLStreamException {
171         onStartElement();
172         super.writeStartElement(prefix, localName, namespaceURI);
173     }
174 
175     public void writeEmptyElement(String namespaceURI, String localName) throws XMLStreamException {
176         onEmptyElement();
177         super.writeEmptyElement(namespaceURI, localName);
178     }
179 
180     public void writeEmptyElement(String prefix, String localName, String namespaceURI) throws XMLStreamException {
181         onEmptyElement();
182         super.writeEmptyElement(prefix, localName, namespaceURI);
183     }
184 
185     public void writeEmptyElement(String localName) throws XMLStreamException {
186         onEmptyElement();
187         super.writeEmptyElement(localName);
188     }
189 
190     public void writeEndElement() throws XMLStreamException {
191         onEndElement();
192         super.writeEndElement();
193     }
194 
195     public void writeCharacters(String text) throws XMLStreamException {
196         state = SEEN_DATA;
197         super.writeCharacters(text);
198     }
199 
200     public void writeCharacters(char[] text, int start, int len) throws XMLStreamException {
201         state = SEEN_DATA;
202         super.writeCharacters(text, start, len);
203     }
204 
205     public void writeCData(String data) throws XMLStreamException {
206         state = SEEN_DATA;
207         super.writeCData(data);
208     }
