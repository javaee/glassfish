61       * A hook to perform variable replacements on the attribute/element values
62       * that are found in the configuration file.
63       * The translation happens lazily when objects are actually created, not when
64       * configuration is parsed, so this allows circular references &mdash;
65       * {@link Translator} may refer to objects in the configuration file being read.

123     /**
124      * Obtains a {@link ConfigModel} for the given class (Which should have {@link Configured} annotation on it.)
125      */
126     public ConfigModel buildModel(String fullyQualifiedClassName) {
127         
128         ActiveDescriptor<? extends ConfigInjector<?>> desc;
129         synchronized (cache) {
130             desc = cache.get(fullyQualifiedClassName);
131             if (desc == null) {
132                 desc = (ActiveDescriptor<? extends ConfigInjector<?>>)
133                     habitat.getBestDescriptor(new InjectionTargetFilter(fullyQualifiedClassName));
134                 
135                 if (desc == null) {
136                     throw new ConfigurationException("ConfigInjector for %s is not found, is it annotated with @Configured",fullyQualifiedClassName);
137                 }
138             
139                 cache.put(fullyQualifiedClassName, desc);
140             }
141         }
142         
143         return buildModel(desc);
144     }

155     public ConfigModel getModelByElementName(String elementName) {
156         ServiceHandle<? extends ConfigInjector> i = habitat.getServiceHandle(ConfigInjector.class, elementName);
157         if(i==null) return null;
158         return buildModel(i.getActiveDescriptor());
159     }

161     private class InjectionTargetFilter
162         implements IndexedFilter {
163 
164         String targetName;
165 
166         private InjectionTargetFilter(String targetName) {
167             this.targetName = targetName;
168         }
169 
170         @SuppressWarnings("unchecked")
171         @Override
172         public boolean matches(Descriptor d) {
173             if (d.getQualifiers().contains(InjectionTarget.class.getName())) {
174                 List<String> list = d.getMetadata().get("target");
175                 if (list == null) return false;
176                 
177                 String value = list.get(0) ;
178                 
179                 // No need to synchronize on cache, it is already synchronized
180                 cache.put(value, (ActiveDescriptor<? extends ConfigInjector<?>>) d);
181                 
182                 if (value.equals(targetName)) {
183                     return true;
184                 }
185             }
186 
187             return false;
188         }
189 
190         @Override
191         public String getAdvertisedContract() {
192             return ConfigInjector.class.getName();
193         }
194 
195         @Override
196         public String getName() {
197             return null;
198         }
199     }

201     /**
202      * Calculates all @Configured interfaces subclassing the passed interface type.
203      *
204      * @param intf a @Configured interface
205      * @return List of all @Configured subclasses
206      * @throws ClassNotFoundException
207      */
208     public synchronized List<ConfigModel> getAllModelsImplementing(Class intf) throws ClassNotFoundException {
209         if (implementorsOf.size()==0) {
210             initXRef();
211         }
212         return implementorsOf.getOne(intf);   
213     }
214 
215     /**
216      * probably a bit slow, calculates all the @Configured interfaces subclassing, useful
217      * to find all possible subclasses of a type.
218      * 
219      * @throws ClassNotFoundException
220      */
221     private void initXRef() throws ClassNotFoundException {
222 
223         // force initialization of all the config models.
224         for (ServiceHandle<?> i : habitat.getAllServiceHandles(ConfigInjector.class)) {
225             buildModel((ActiveDescriptor<? extends ConfigInjector>) i.getActiveDescriptor());
226         }
227         for (ConfigModel cm : models.values()) {
228             Class targetType = cm.classLoaderHolder.loadClass(cm.targetTypeName);
229             do {
230                 Class[] intfs = targetType.getInterfaces();
231                 for (Class intf : intfs) {
232                     if (intf.isAnnotationPresent(Configured.class)) {
233                         addXRef(intf, cm);
234                     }
235                 }
236                 targetType = targetType.getSuperclass();
237             } while (targetType!=null);
238         }
239     }
240 
241     private void addXRef(Class type, ConfigModel cm) {
242         List<ConfigModel> models = implementorsOf.getOne(type);
243         if (models==null) {
244             models= new ArrayList<ConfigModel>();
245             implementorsOf.add(type, models);
246         }
247         models.add(cm);
248     }

277     /*package*/
278     DataType getValidator(String dataType) {
279         synchronized(validators) {
280             DataType validator = validators.get(dataType);
281             if (validator != null)
282                 return (validator);
283         }
284         Collection<DataType> dtfh = habitat.getAllServices(DataType.class);
285         synchronized(validators) {
286             for (DataType dt : dtfh) {
287                 validators.put(dt.getClass().getName(), dt);
288             }
289             return (validators.get(dataType));
290         }
291     }
