66  /**
67   * Jsr-330 Injection Resolver
68   * 
69   * @author Jeff Trent
70   * 
71   * @since 3.1
72   */
73  @SuppressWarnings("unchecked")
74  public class Jsr330InjectionResolver extends
75      InjectionResolver<javax.inject.Inject> {
76  
77    @Inject
78    private Habitat habitat;
79  
80    public Jsr330InjectionResolver() {
81      super(javax.inject.Inject.class);
82    }
83  
84    public Jsr330InjectionResolver(Habitat h) {
85      super(javax.inject.Inject.class);
86      this.habitat = h;
87    }
88  
89    @Override
90    public <V> V getValue(Object component,
91          Inhabitant<?> onBehalfOf,
92          AnnotatedElement target,
93          Type gtype,
94          Class<V> type) throws ComponentException {
95      Object result;
96  
97      if (Types.isSubClassOf(type, Provider.class)) {
98        gtype = getGenericType(gtype, target);
99        result = getHolderInjectValue(habitat, onBehalfOf, gtype);
100     } else {
101       result = get(habitat, onBehalfOf, target, type);
102     }
103 
104     return (V)result;
105   }
106 
107   protected Type getGenericType(Type gtype, AnnotatedElement target) {
108     if (null != gtype) {
109       return gtype;
110     }
111     
112     if (target instanceof Field) {
113       return ((Field)target).getGenericType();
114     } else if (target instanceof Method) {
115       Type gpTypes[] = ((Method)target).getGenericParameterTypes();
116       if (1 == gpTypes.length) {
117         return gpTypes[0];
118       }
119     }
120     
121     throw new ComponentException("unknown type: " + target);
122   }

124   protected static <V> Provider<V> getHolderInjectValue(
125       Habitat habitat,
126       Object onBehalfOf,
127       Type paramType) throws ComponentException {
128     if (!ParameterizedType.class.isInstance(paramType)) {
129       throw new ComponentException("unknown type: " + paramType.toString());
130     }
131 
132     Type [] types = ((ParameterizedType)paramType).getActualTypeArguments();
133     if (null == types || 1 != types.length) {
134       throw new ComponentException(paramType.toString() + " on " + onBehalfOf);
135     }
136     
137     Class<V> paramClass = Types.erasure(types[0]);
138     final Inhabitant<V> inhabitant = getInhabitant(habitat, onBehalfOf, paramClass);
139     
140     return new BasicProviderImpl(inhabitant);
141   }
142   
143   protected static <V> Inhabitant<V> getInhabitant(
144       Habitat habitat,
145       Object onBehalfOf,
146       Class<V> type) {
147     Inhabitant<V> result = habitat.getInhabitant(type, null);
148     if (null == result) {
149       result = habitat.getInhabitantByType(type);
150     }
151 
152     if (null == result) {
153       throw new ComponentException("unable to resolve %s", type.toString(), onBehalfOf);
154     }
155     
156     return result;
157   }
158 
159   protected static <V> V get(
160       Habitat habitat,
161       Object onBehalfOf,
162       AnnotatedElement target,
163       Class<V> type) {
164     Annotation annotations[] = qualifiers(target.getAnnotations());
165     
166     Named named = target.getAnnotation(Named.class);
167     String name = (null == named) ? null : named.value();
168     
169     V result = null;
170     
171     if (null != name) {
172       result = getServiceInjectValue(habitat, type, name);
173     }
174     
175     if (null == result) {
176       if (null != annotations) {
177         result = getServiceInjectValue(habitat, type, annotations);
178       }
179     }
180     
181     if (null == result) {
182       result = getComponentInjectValue(habitat, type);
183     }
184 
185     if (null == result) {
186       if (null == name && null == annotations) {
187         result = getServiceInjectValue(habitat, type, (String)null);
188       }
189     }
190     
191     if (null == result) {
192       throw new ComponentException("unable to resolve %s for %s", type.toString(), onBehalfOf);
193     }
194     
195     return result;
196   }
197 
198   private static Annotation[] qualifiers(Annotation[] annotations) {
199     if (null == annotations || 0 == annotations.length) {
200       return null;
201     }
202     
203     ArrayList<Annotation> result = null;
204     for (Annotation ann : annotations) {
205       if (null != ann.annotationType().getAnnotation(Qualifier.class)) {
206         if (null == result) {
207           result = new ArrayList<Annotation>();
208         }
209         result.add(ann);
210       }
211     }
212     
213     return (null == result) ? null : result.toArray(annotations);
214   }

216   protected static <V> V getServiceInjectValue(Habitat habitat,
217         Class<V> type,
218         String name) throws ComponentException {
219     V result = habitat.getComponent(type, name);
220     return result;
221   }
222 
223   protected static <V> V getServiceInjectValue(Habitat habitat,
224       Class<V> type,
225       Annotation[] annotations) throws ComponentException {
226     Collection<Inhabitant<?>> candidates = habitat.getInhabitantsByContract(type.getName());
227     for (Inhabitant<?> candidate : candidates) {
228       if (matched(annotations, candidate.metadata())) {
229         return (V) candidate.get();
230       }
231     }
232     return null;
233   }
234   
235   protected static boolean matched(Annotation[] annotations,
236       MultiMap<String, String> candidateMd) {
237     for (Annotation ann : annotations) {
238       if (null != ann) {
239         String name = ann.annotationType().getName();
240         if (!candidateMd.contains(Constants.QUALIFIER, name)) {
241           return false;
242         }
243       }
244     }
245     return true;
246   }
247 
248   protected static <V> V getComponentInjectValue(
249       Habitat habitat,
250       Class<V> type) throws ComponentException {
251     V result = habitat.getByType(type);
252     return result;
253   }
