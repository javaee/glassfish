70  /**
71   * Default Jsr330Processor for locating and installing all
72   * Jsr330Bindings into the currently running habitat.
73   * 
74   * This implementation leverages the {@link RunLevelServicee}
75   * to be immediately run as part of startup if this module/class
76   * is in the classpath.
77   * 
78   * @author Jeff Trent
79   * 
80   * @since 3.1
81   */
82  //TODO: need to implement service ranking in Hk2
83  @Service(metadata=org.jvnet.hk2.component.Constants.SERVICE_RANKING + "=" + Integer.MIN_VALUE)
84  @RunLevel(RunLevel.KERNEL_RUNLEVEL)
85  public class Jsr330ProcessorImpl implements Jsr330Processor, PostConstruct {

88    private Habitat habitat;

90    @Inject
91    private Jsr330BindingLocator locator;

93    private boolean installed;

95    public Jsr330ProcessorImpl() {
96    }

98    public Jsr330ProcessorImpl(Habitat h) {
99      this.habitat = h;

102   @SuppressWarnings("unchecked")
103   @Override
104   public void postConstruct() {
105     // TODO: this should not be necessary
106     if (!installed) {
107       habitat.addIndex(new ExistingSingletonInhabitant<InjectionResolver>(InjectionResolver.class,
108                 new Jsr330InjectionResolver(habitat)),
109                 InjectionResolver.class.getName(), null);
110       installed = true;
111     }
112   
113     installAllBindings();
114   }

116   @Override
117   public void installAllBindings() {
118     Collection<? extends Jsr330Binding> bindings = locator.getAllBindings();
119     assert(null != bindings);
120     for (Jsr330Binding binding : bindings) {
121       install(binding);
122     }
123   }

125   public void install(Jsr330Binding binding) {
126     String typeName = binding.getServiceClassName();
127     
128     Set<String> indicies = new HashSet<String>();
129     Collection<String> ccns = binding.getContractClassNames();
130     if (null != ccns) {
131       indicies.addAll(ccns);
132     }
133     
134     Collection<Class<? extends Annotation>> qualifiers = binding.getServiceQualifiers();
135     if (null != qualifiers) {
136       for (Class<? extends Annotation> ann : qualifiers) {
137         if (null != ann) {
138           indicies.add(ann.getName());
139         }
140       }
141     }
142     
143     Collection<Named> named = binding.getServiceNames();
144     Set<String> names = null;
145     if (null != named) {
146       names = new HashSet<String>();
147       for (Named name : named) {
148         if (null != name) {
149           names.add(name.value());
150         }
151       }
152     }
153     
154     Scope scope = getScope(binding);
155     
156     MultiMap<String, String> metadata = getMetadata(binding);
157 
158     Holder<ClassLoader> clh = new Holder.Impl<ClassLoader>(binding.getClass().getClassLoader());
159     
160     Inhabitant<?> i =  createInhabitant(clh, typeName, indicies, qualifiers, scope, metadata);
161     InhabitantData data = createInhabitantData(i, binding, indicies, names, scope, metadata);
162     add(data);
163   }
164 
165   @SuppressWarnings("unchecked")
166   protected Inhabitant<?> createInhabitant(Holder<ClassLoader> clh,
167       String typeName,
168       Set<String> indicies,
169       Collection<Class<? extends Annotation>> annotations,
170       Scope scope,
171       MultiMap<String, String> metadata) {
172     merge(metadata, annotations);
173     Inhabitant<?> i = new Jsr330LazyInhabitant(habitat,
174           clh,
175           typeName,
176           scope,
177           metadata);
178     return i;
179   }
180 
181   protected void merge(MultiMap<String, String> metadata,
182       Collection<Class<? extends Annotation>> annotations) {
183     if (null != annotations) {
184       for (Class<? extends Annotation> ann : annotations) {
185         if (null != ann) {
186           metadata.add(Constants.QUALIFIER, ann.getName());
187         }
188       }
189     }
190   }
191 
192   protected InhabitantData createInhabitantData(Inhabitant<?> i,
193       Jsr330Binding binding, 
194       Set<String> indicies,
195       Set<String> names,
196       Scope scope,
197       MultiMap<String, String> metadata) {
198     return new InhabitantData(i, indicies, names, scope, metadata);
199   }

201   /**
202    * Returns the Hk2 scope giving the binding.
203    *  
204    * @param binding the binding
205    * 
206    * @return the Hk2 Scope
207    */
208   protected Scope getScope(Jsr330Binding binding) {
209     javax.inject.Scope scope = binding.getServiceScope();
210     if (null == scope) {
211       return new PerLookup();
212     }
213     
214     if (javax.inject.Singleton.class.isInstance(scope)) {
215       throw new IllegalArgumentException(scope.toString());
216     }
217     
218     return new Singleton();
219   }
220 
221   /**
222    * Returns the Hk2 metadata given the binding.
223    * 
224    * @param binding the binding
225    * 
226    * @return the Hk2 metadata
227    */
228   protected MultiMap<String, String> getMetadata(Jsr330Binding binding) {
229     MultiMap<String, String> metadata = null;
230     
231     if (binding instanceof Jsr330BindingExt) {
232       metadata = ((Jsr330BindingExt)binding).getMetadata();
233     }
234     
235     return (null == metadata) ? new MultiMap<String, String>() : metadata;
236   }
237 
238   protected void add(InhabitantData data) {
239     habitat.add(data.i);
240 
241     Set<String> names = data.names;
242     if (null == names) {
243       names = Collections.singleton(null);
244     }
245     
246     for (String v : data.indicies) {
247       // register inhabitant to the index
248       for (String name : names) {
249         habitat.addIndex(data.i, v, name);
250       }
251     }
252   }
