70  /**
71   * This class should be extended by test classes in order to get an automatically
72   * filled in ServiceLocator.  By default the testLocator will inspect the package
73   * of the test to find any classes annotated with &#64;Service.  The locator will
74   * also be able to do second-chance advertisement of services that were injected.
75   * The default ServiceLocator will also have an error handler that causes any classloading
76   * failure to get rethrown up to the lookup call, since this can sometimes cause
77   * confusion.
78   * <p>
79   * This behavior can be customized by overriding the before method and calling the super
80   * of one of the other methods available for customization
81   * 
82   * @author jwells
83   */
84  public class HK2Runner {
85      private final static String CLASS_PATH_PROP = "java.class.path";
86      private final static String DOT_CLASS = ".class";
87      
88      /**
89       * Test classes can use this service locator as their private test locator
90       */
91      protected ServiceLocator testLocator;
92      
93      /**
94       * The verbosity of this runner
95       */
96      private boolean verbose = false;
97      
98      /**
99       * This will generate the default testLocator for this test
100      * class, which will search the package of the test itself for
101      * classes annotated with &#64;Service.
102      */
103     @Before
104     public void before() {
105         initialize(this.getClass().getName(), null, null);
106     }
107 
108     /**
109      * This method initializes the service locator with services from the given list
110      * of packages (in "." format) and with the set of classes given.
111      * 
112      * @param name The name of the service locator to create.  If there is already a
113      * service locator of this name then the remaining fields will be ignored and the existing
114      * locator with this name will be returned.  May not be null
115      * @param packages The list of packages (in "." format, i.e. "com.acme.test.services") that
116      * we should hunt through the classpath for in order to find services.  If null this is considered
117      * to be the empty set
118      * @param clazzes A set of classes that should be analyzed as services, whether they declare
119      * &#64;Service or not.  If null this is considered to be the empty set
120      */
121     protected void initialize(String name, List<String> packages, List<Class<?>> clazzes) {
122         if (name == null) throw new IllegalArgumentException();
123         if (packages == null) packages = new LinkedList<String>();
124         if (clazzes == null) clazzes = new LinkedList<Class<?>>();
125         
126         ServiceLocator found = ServiceLocatorFactory.getInstance().find(name);
127         if (found != null) {
128             testLocator = found;
129             
130             testLocator.inject(this);
131             return;
132         }
133         
134         testLocator = ServiceLocatorFactory.getInstance().create(name);
135         
136         DynamicConfigurationService dcs = testLocator.getService(DynamicConfigurationService.class);
137         DynamicConfiguration config = dcs.createDynamicConfiguration();
138         
139         config.addActiveDescriptor(ErrorServiceImpl.class);
140         config.addActiveDescriptor(JustInTimeInjectionResolverImpl.class);
141         
142         addServicesFromDefault(config);
143         
144         addServicesFromPackage(config, packages);
145         
146         for (Class<?> clazz : clazzes) {
147             config.addActiveDescriptor(clazz);
148         }
149         
150         config.commit();
151         
152         testLocator.inject(this);
153     }

159     private void addServicesFromDefault(final DynamicConfiguration config) {
160         AccessController.doPrivileged(new PrivilegedAction<Object>() {
161 
162             @Override
163             public Object run() {
164                 internalAddServicesFromDefault(config);
165                 return null;
166             }
167

171     private void internalAddServicesFromDefault(DynamicConfiguration config) {
172         ClassLoader loader = this.getClass().getClassLoader();
173         
174         Enumeration<URL> resources;
175         try {
176             resources = loader.getResources("META-INF/hk2-locator/default");
177         }
178         catch (IOException ioe) {
179             ioe.printStackTrace();
180             
181             return;
182         }
183         
184         while (resources.hasMoreElements()) {
185             URL url = resources.nextElement();
186            
187             try {
188                 InputStream urlStream = url.openStream();
189                 
190                 BufferedReader reader = new BufferedReader(new InputStreamReader(urlStream));
191                 
192                 boolean goOn = true;
193                 while (goOn) {
194                     DescriptorImpl bindMe = new DescriptorImpl();
195                 
196                     goOn = bindMe.readObject(reader);
197                     if (goOn == true) {
198                         config.bind(bindMe);
199                     }
200                 }
201                 
202                 reader.close();
203             }
204             catch (IOException ioe) {
205                 continue;
206             }
207             
208             
209         }
210         
211     }

225     private void internalAddServicesFromPackage(DynamicConfiguration config, List<String> packages) {
226         if (packages.isEmpty()) return;
227         
228         String classPath = System.getProperty(CLASS_PATH_PROP);
229         StringTokenizer st = new StringTokenizer(classPath, File.pathSeparator);
230         
231         while(st.hasMoreTokens()) {
232             String pathElement = st.nextToken();
233             
234             addServicesFromPathElement(config, packages, pathElement);
235         }
236         
237     }

239     private void addServicesFromPathElement(DynamicConfiguration config, List<String> packages, String element) {
240         File fileElement = new File(element);
241         if (!fileElement.exists()) return;
242         
243         if (fileElement.isDirectory()) {
244             addServicesFromPathDirectory(config, packages, fileElement);
245         }
246         else {
247             addServicesFromPathJar(config, packages, fileElement);
248         }
249     }

251     private void addServicesFromPathDirectory(DynamicConfiguration config, List<String> packages, File directory) {
252         for (String pack : packages) {
253             File searchDir = new File(directory, convertToFileFormat(pack));
254             if (!searchDir.exists()) continue;
255             if (!searchDir.isDirectory()) continue;
256             
257             File candidates[] = searchDir.listFiles(new FilenameFilter() {
258 
259                 @Override
260                 public boolean accept(File dir, String name) {
261                     if (name == null) return false;
262                     if (name.endsWith(DOT_CLASS)) return true;
263                     return false;
264                 }
265                 
266             });
267             
268             if (candidates == null) continue;
269             
270             for (File candidate : candidates) {
271                 try {
272                     FileInputStream fis = new FileInputStream(candidate);
273                     
274                     addClassIfService(config, fis);
275                 }
276                 catch (IOException ioe) {
277                     // Just don't add it
278                 }
279             }
280         }
281         
282     }

284     private void addServicesFromPathJar(DynamicConfiguration config, List<String> packages, File jar) {
285         JarFile jarFile;
286         try {
287             jarFile = new JarFile(jar);
288         }
289         catch (IOException ioe) {
290             // Not a jar file, forget it
291             return;
292         }
293         
294         for (String pack : packages) {
295             String packAsFile = convertToFileFormat(pack);
296             int packAsFileLen = packAsFile.length() + 1;
297             
298             Enumeration<JarEntry> entries = jarFile.entries();
299             while (entries.hasMoreElements()) {
300                 JarEntry entry = entries.nextElement();
301                 
302                 String entryName = entry.getName();
303                 if (!entryName.startsWith(packAsFile)) {
304                     // Not in the correct directory
305                     continue;
306                 }
307                 if (entryName.substring(packAsFileLen).contains("/")) {
308                     // Next directory down
309                     continue;
310                 }
311                 if (!entryName.endsWith(DOT_CLASS)) {
312                     // Not a class
313                     continue;
314                 }
315                 
316                 try {
317                     addClassIfService(config, jarFile.getInputStream(entry));
318                 }
319                 catch (IOException ioe) {
320                     // Simply don't add it if we can't read it
321                 }
322             }
323         }
324     }

326     private void addClassIfService(DynamicConfiguration config, InputStream is) throws IOException {
327         ClassReader reader = new ClassReader(is);
328         
329         ClassVisitorImpl cvi = new ClassVisitorImpl(config, verbose);
330         
331         reader.accept(cvi, ClassReader.SKIP_CODE | ClassReader.SKIP_DEBUG | ClassReader.SKIP_FRAMES);
332         
333     }
334     
335     private static String convertToFileFormat(String clazzFormat) {
336         return clazzFormat.replaceAll("\\.", "/");
337     }
