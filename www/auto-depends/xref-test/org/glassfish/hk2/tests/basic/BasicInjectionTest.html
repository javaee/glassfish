254         assertInjectedInstance(FactoryProvidedContractBImpl.class, b);
255 
256         final Provider<FactoryProvidedContractB> pb = services.forContract(FactoryProvidedContractB.class).getProvider();
257         assertInjectedProvider(FactoryProvidedContractBImpl.class, pb);
258 
259         // binding defined using factory class and qualifier annotation
260         final FactoryProvidedContractC c_a = services.forContract(FactoryProvidedContractC.class).annotatedWith(MarkerA.class).get();
261         assertInjectedInstance(FactoryProvidedContractCAImpl.class, c_a);
262 
263         final Provider<FactoryProvidedContractC> pc_a = services.forContract(FactoryProvidedContractC.class).annotatedWith(MarkerA.class).getProvider();
264         assertInjectedProvider(FactoryProvidedContractCAImpl.class, pc_a);
265 
266         // binding defined using factory instance and qualifier annotation
267         final FactoryProvidedContractC c_b = services.forContract(FactoryProvidedContractC.class).annotatedWith(MarkerB.class).get();
268         assertInjectedInstance(FactoryProvidedContractCBImpl.class, c_b);
269 
270         final Provider<FactoryProvidedContractC> pc_b = services.forContract(FactoryProvidedContractC.class).annotatedWith(MarkerB.class).getProvider();
271         assertInjectedProvider(FactoryProvidedContractCBImpl.class, pc_b);
272 
273         // verifying null is returned for non-annotated binding that was not defined
274         final FactoryProvidedContractC c_default = services.forContract(FactoryProvidedContractC.class).get();
275         assertNull("No binding defined for the non-annotated contract. Provisioned instance should be null.", c_default);
276 
277         final Provider<FactoryProvidedContractC> pc_default = services.forContract(FactoryProvidedContractC.class).getProvider();
278         assertTrue("No binding defined for the non-annotated contract. Provider or returned instance should be null.", pc_default == null || pc_default.get() == null);
