53  /**
54   * Inhabitant scanner based on introspection information rather than statically
55   * generated inhabitant file
56   */
57  public class InhabitantIntrospectionScanner implements Iterable<InhabitantParser> {
58  
59      final Types types;
60    
61      final Iterator<AnnotatedElement> inhabitantAnnotations;
62      Iterator<AnnotatedElement> current;
63  
64      @SuppressWarnings("unchecked")
65      public InhabitantIntrospectionScanner(ParsingContext context) {
66          types = context.getTypes();
67          AnnotationType am = types.getBy(AnnotationType.class, InhabitantAnnotation.class.getName());
68          if (am==null) {                                  
69              inhabitantAnnotations = Collections.EMPTY_LIST.iterator();
70          } else {
71              Collection<AnnotatedElement> ccc = am.allAnnotatedTypes();
72              inhabitantAnnotations = ccc.iterator();
73          }
74          fetch();
75      }
76  
77      /*
78       * puts current in the first non empty iterator.
79       */
80      @SuppressWarnings("unchecked")
81      private void fetch() {
82          if (!inhabitantAnnotations.hasNext()) {
83              current = Collections.EMPTY_LIST.iterator();
84              return;
85          }
86          do {
87              AnnotationType am = AnnotationType.class.cast(inhabitantAnnotations.next());
88              current = am.allAnnotatedTypes().iterator();
89          } while (!current.hasNext() && inhabitantAnnotations.hasNext());
90      }

92      public boolean isContract(AnnotatedElement type) {
93          // must be annotated with @Contract
94          if (null == type) return false;
95          return type.getAnnotation(Contract.class.getName())!=null;
96      }

98      public void findClassContracts(ClassModel cm, Set<String> interfaces, Set<String> annotationTypeInterfaces) {
99          for (InterfaceModel im : cm.getInterfaces()) {
100             if (isContract(im)) {
101                 interfaces.add(im.getName());
102             }
103         }
104         findContractsFromAnnotations(cm, interfaces, annotationTypeInterfaces);
105     }
106 
107     public void findInterfaceContracts(InterfaceModel im, Set<String> interfaces, Set<String> annInterfaces) {
108         if (im.getParent()!=null) {
109             findInterfaceContracts(im.getParent(), interfaces, annInterfaces);
110         }
111         if (isContract(im)) {
112             if (im instanceof AnnotationType) {
113               String name = im.getName();
114               if (!name.equals(FactoryFor.class.getName())) {
115                 annInterfaces.add(name);
116               }
117             } else {
118               interfaces.add(im.getName());
119             }
120         }
121         findContractsFromAnnotations(im, interfaces, annInterfaces);
122     }
123 
124     public void findContractsFromAnnotations(AnnotatedElement ae, Set<String> interfaces, Set<String> annInterfaces) {
125         for (AnnotationModel am : ae.getAnnotations()) {
126             AnnotationType at = am.getType();
127             findInterfaceContracts(at, interfaces, annInterfaces);
128             
129             String name = at.getName();
130             if (name.equals(ContractProvided.class.getName())) {
131               String val = scrub(am.getValues().get("value"));
132               if (null != val) {
133                 interfaces.add(val);
134               }
135             } else if (name.equals(FactoryFor.class.getName())) {
136               String val = scrub(am.getValues().get("value"));
137               if (null != val) {
138                 annInterfaces.add(name + ":" + val);
139               }
140             }
141         }
142     }

144     public static String scrub(Object obj) {
145         if (null == obj) {
146             return null;
147         }
148         if (obj instanceof String) {
149           return (String)obj;
150         }
151         String mangled = obj.toString();
152         if (mangled.startsWith("L") && mangled.endsWith(";")) {
153             mangled = mangled.substring(1, mangled.length()-1);
154         }
155         return mangled.replace("/", ".");
156     }

158     public void findContracts(ClassModel cm, Set<String> interfaces, Set<String> annotationTypeInterfaces) {
159         for (InterfaceModel im : cm.getInterfaces()) {
160             getAllContractInterfaces(im, interfaces);
161         }
162 
163         findContractsFromAnnotations(cm, interfaces, annotationTypeInterfaces);
164 
165         // walk parent chain too
166         ClassModel parent = cm.getParent();
167         if (null == parent) {
168           // at this point, we check all interfaces to see if they have super interfaces
169           Set<String> newIfaces = new LinkedHashSet<String>();
170           for (String ifaceName : interfaces) {
171             InterfaceModel iface = types.getBy(InterfaceModel.class, ifaceName);
172             getAllContractInterfaces(iface, newIfaces);
173           }
174           interfaces.addAll(newIfaces);
175         } else if (!parent.getName().equals(Object.class.getName())) {
176           findContracts(parent, interfaces, annotationTypeInterfaces);
177         }
178     }

180     /**
181      * add all @Contract annotated interfaces extended by the passed interface model
182      * to the provided list.
183      *
184      * @param im interface model to get all @Contract annotated extended interfaces from
185      * @param interfaces list of interfaces im is extending
186      */
187     private void getAllContractInterfaces(InterfaceModel im, Collection<String> interfaces) {
188         
189         if (im==null) return;
190         
191         if (isContract(im)) {
192             interfaces.add(im.getName());
193         }
194         for (InterfaceModel implementedIntf : im.getInterfaces()) {
195             getAllContractInterfaces(implementedIntf, interfaces);
196         }
197     }

199     public Iterator<InhabitantParser> iterator() {
200         return new Iterator<InhabitantParser>() {
201             public boolean hasNext() {
202                 return current.hasNext();
203             }
204 
205             public InhabitantParser next() {
206                 final AnnotatedElement ae = current.next();
207                 InhabitantParser ip = new InhabitantParser() {
208                     @Override
209                     public Iterable<String> getIndexes() {
210                         if (ae instanceof ClassModel) {
211                             final ClassModel cm = (ClassModel) ae;
212 
213                             final LinkedHashSet<String> implInterfaces = new LinkedHashSet<String>();
214                             final LinkedHashSet<String> implAnnotationInterfaces = new LinkedHashSet<String>();
215                             findContracts(cm, implInterfaces, implAnnotationInterfaces);
216                             
217                             final Iterator<String> interfaces = implInterfaces.iterator();
218                             final Iterator<String> annInterfaces = implAnnotationInterfaces.iterator();
219                             return new Iterable<String>() {
220                                 public Iterator<String> iterator() {
221                                     return new Iterator<String>() {
222                                         public boolean hasNext() {
223                                             return interfaces.hasNext() || annInterfaces.hasNext();
224                                         }
225 
226                                         public String next() {
227                                             if (interfaces.hasNext()) {
228                                               final AnnotationModel am = cm.getAnnotation(Service.class.getName());
229                                               String contract = interfaces.next();
230                                               String name = (String) am.getValues().get("name");
231                                               if (name==null || name.isEmpty()) {
232                                                   return contract;
233                                               } else {
234                                                   return contract+":"+name;
235                                               }
236                                             } else {
237                                               String contract = annInterfaces.next();
238                                               return contract;
239                                             }
240                                         }
241 
242                                         public void remove() {
243                                             throw new UnsupportedOperationException();
244                                         }
245                                     };
246                                 }
247                             };
248 
249                         }
250                         // this should be an error...
251                         return (new ArrayList<String>());
252                     }

257                     }

262                     }

266                         throw new UnsupportedOperationException();
267                     }

274                     @Override
275                     public MultiMap<String, String> getMetaData() {
276                         MultiMap<String, String> mm = new MultiMap<String, String>();
277                         final AnnotationModel am = ae.getAnnotation(Service.class.getName());
278                         if (null != am) {
279                           Object metaObj = am.getValues().get("metadata");
280                           if (null != metaObj) {
281                             String meta = metaObj.toString();
282                             String [] split = meta.split(",");
283                             for (String entry : split) {
284                               String [] split2 = entry.split("=");
285                               if (2 == split2.length) {
286                                 mm.add(split2[0], split2[1]);
287                               }
288                             }
289                           }
290                         }
291                         return mm;
292                     }
293                 };
294                 if (!current.hasNext()) {
295                     fetch();
296                 }
297                 return ip;
298             }

302                 throw new UnsupportedOperationException();
303             }
304         };
