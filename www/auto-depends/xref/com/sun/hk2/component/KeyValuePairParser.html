37  package com.sun.hk2.component;
38  
39  import java.util.Iterator;
40  import java.util.StringTokenizer;
41  
42  /***
43   * Parses a string like <tt>key=value,key=value,key=value</tt>.
44   *
45   * <p>
46   * More specifically the format of the line satisfies the following BNF constructions:
47   *
48   * <pre>
49   * LINE = TOKEN ( ',' LINE )?               // LINE is ','-separated TOKENs
50   * TOKEN = KEY ( '=' VALUE )?               // TOKEN is a key/value pair. value is optional.
51   * KEY = [^,=]+                             // KEY is a non-empty string that doesn't contain ',' nor '='
52   * VALUE = '"' ([^"]| '\' ANYCHAR )* '"'    // VALUE is any string surrounded by quotes/doublequotes (where \ is used as escape), or ...
53   * VALUE = ''' ([^']| '\' ANYCHAR )* '''
54   * VALUE = [^,]*                //       possibly empty string that doesn't contain ','
55   * </pre>
56   *
57   * <p>
58   * This class works like {@link StringTokenizer}; each time {@link #parseNext()}
59   * is invoked, the parser "moves" to the next key/value pair, which
60   * you can then obtain with {@link #getKey()} and {@link #getValue()}.
61   * The {@link #hasNext()} method shall be used to check if there's a next key/value
62   * pair after the current position.
63   *
64   * TODO: improved error check
65   * @author Kohsuke Kawaguchi
66   */
67  public final class KeyValuePairParser {
68      private String str;
69      private int idx;
70  
71      private String key;
72      private String value;
73      private final StringBuilder buf = new StringBuilder();
74  
75      public KeyValuePairParser() {
76      }
77  
78      public KeyValuePairParser(String s) {
79          set(s);
80      }
81  
82      /***
83       * Resets the parser to parse the given string that looks like "key=value,key=value,..."
84       */
85      public void set(String str) {
86          this.str = str;
87          idx = 0;
88      }
89  
90      public boolean hasNext() {
91          return idx<str.length();
92      }
93  
94      private int indexOf(char separator) {
95          int i = str.indexOf(separator, idx + 1);
96          if(i<0) return str.length();
97          return i;
98      }
99  
100     /***
101      * Returns true if the current character is ch.
102      */
103     private boolean isAt(char ch) {
104         return idx<str.length() && str.charAt(idx)==ch;
105     }
106 
107     private char current() {
108         return str.charAt(idx);
109     }
110 
111     public void parseNext() {
112         // parse key
113         int del = Math.min(indexOf('='),indexOf(','));
114         key = str.substring(idx,del);
115         value = null;
116         idx=del;
117 
118         if(idx==str.length())
119             return; // key only, and we are at the end of the line
120 
121         if(current()==',') {
122             // key only, so consume ',' and return.
123             idx++;
124             return;
125         }
126 
127         // we have a value. now consume '='
128         assert current()=='=';
129         idx++;
130 
131         if(isAt('\'') || isAt('"')) {
132             char quote = current();
133             // quoted string. now let's find the end:
134             idx++;
135             buf.setLength(0);
136             while(true) {
137                 if(isAt(quote)) {
138                     value = buf.toString();
139                     // consume endquote and ','. this may put as beyond EOL but that's fine.
140                     idx+=2;
141                     return;
142                 }
143                 if(isAt('//'))
144                     idx++;
145                 buf.append(current());
146                 idx++;
147             }
148         }
149 
150         // unquoted string
151         int end = indexOf(',');
152         value = str.substring(idx,end);
153         idx=end+1;
154     }
155 
156 
157     public String getKey() {
158         return key;
159     }
160 
161     public String getValue() {
162         return value;
163     }
164 
165     /***
166      * Moves the parser to the head of the key/value pair, to reparse the same string.
167      */
168     public void rewind() {
169         idx=0;
170     }
171 
172     /***
173      * Skips until we find the value of the given key,
174      * and leaves the parser at that point.
175      * <p>
176      * This method is not terribly efficient, but it is convenient.
177      */
178     public String find(String key) {
179         while(hasNext()) {
180             parseNext();
181             if(getKey().equals(key))
182                 return getValue();
183         }
184         return null;
185     }
186 
187     /***
188      * Finds all the values of the given key.
189      * Calling the iterator will rewind the parser.
190      */
191     public Iterable<String> findAll(final String key) {
192         return new Iterable<String>() {
193             public Iterator<String> iterator() {
194                 rewind();
195                 return new Iterator<String>() {
196                     private String next;
197                     private void fetch() {
198                         if(next==null)
199                             next = find(key);
200                     }
201                     public boolean hasNext() {
202                         fetch();
203                         return next!=null;
204                     }
205 
206                     public String next() {
207                         fetch();
208                         String r = next;
209                         next = null;
210                         return r;
211                     }
212 
213                     public void remove() {
214                         throw new UnsupportedOperationException();
215                     }
216                 };
217             }
218         };
219     }
220 
221     /***
222      * Gets the whole line.
223      */
224     public String getLine() {
225         return str;
226     }
