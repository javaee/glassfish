37  package com.sun.hk2.component;
38  
39  import org.jvnet.hk2.component.ComponentException;
40  
41  import java.io.BufferedReader;
42  import java.io.IOException;
43  import java.io.InputStream;
44  import java.io.InputStreamReader;
45  import java.util.Iterator;
46  
47  /***
48   * Scans the stream that follows the <tt>/META-INF/inhabitants/*</tt> format.
49   *
50   * <p>
51   * This class implements {@link Iterable} so that it can be used in for-each loop,
52   * but it cannot parse the same stream twice.
53   *
54   * @author Kohsuke Kawaguchi
55   */
56  public class InhabitantsScanner implements Iterable<KeyValuePairParser> {
57      private int lineNumber = 0;
58      private final String systemId;
59      private final BufferedReader r;
60      private final KeyValuePairParser kvpp = new KeyValuePairParser();
61  
62      public InhabitantsScanner(InputStream in, String systemId) throws IOException {
63          r = new BufferedReader(new InputStreamReader(in,"UTF-8"));
64          this.systemId = systemId;
65      }
66  
67      public String getSystemId() {
68          return systemId;
69      }
70  
71      public int getLineNumber() {
72          return lineNumber;
73      }
74  
75      public Iterator<KeyValuePairParser> iterator() {
76          return new Iterator<KeyValuePairParser>() {
77              private String nextLine;
78  
79              public boolean hasNext() {
80                  fetch();
81                  return nextLine!=null;
82              }
83  
84              private void fetch() {
85                  if(nextLine!=null)  return;
86  
87                  try {
88                      while((nextLine=r.readLine())!=null) {
89                          lineNumber++;
90                          if(nextLine.startsWith("#"))
91                              continue;   // comment
92  
93                          return; // found the next line
94                      }
95                  } catch (IOException e) {
96                      throw new ComponentException("Failed to parse line " + lineNumber + " of " + systemId,e);
97                  }
98              }
99  
100             public KeyValuePairParser next() {
101                 fetch();
102                 kvpp.set(nextLine);
103                 nextLine=null;
104                 return kvpp;
105             }
106 
107             public void remove() {
108                 throw new UnsupportedOperationException();
109             }
110         };
111     }
112 }
