126     @Override
127     public final T get() {
128         try {
129             if (TracingUtilities.isEnabled()) {
130                 TracingThreadLocal.get().push(this);
131             }
132             try {
133                 return get(this);
134             } catch (RunLevelException e) {
135                 // we are a holder, so we need to allow for {@link RunLevelService} constraints
136                 // not properly being met --- in such cases return null
137                 logger.log(Level.FINER, "swallowing error", e);
138                 return null;
139             }
140         } finally {
141             if (TracingUtilities.isEnabled()) {
142                 TracingThreadLocal.get().pop();
143             }
144         }
145     }

147     @Override
148     public <T> T getSerializedMetadata(final Class<T> type, String key) {
149         String v = metadata().getOne(key);
150         if (v==null) {
151             return null;
152         }
153 
154         try {
155             ObjectInputStream is = new ObjectInputStream(new ByteArrayInputStream(new BASE64Decoder().decodeBuffer(v))) {
156                 final ClassLoader cl = type.getClassLoader();
157 
158                 /**
159                  * Use ClassLoader of the given type. Otherwise by default we end up using the classloader
160                  * that loaded HK2, which won't be able to see most of the user classes.
161                  */
162                 protected Class<?> resolveClass(ObjectStreamClass desc) throws IOException, ClassNotFoundException {
163                     String name = desc.getName();
164                     try {
165                         return Class.forName(name,false,cl);
166                     } catch (ClassNotFoundException ex) {
167                         return super.resolveClass(desc);
168                     }
169                 }
170             };
171 
172             return type.cast(is.readObject());
173         } catch (ClassNotFoundException e) {
174             throw new Error(e);
175         } catch (IOException e) {
176             throw new Error(e);
177         }
178     }

294     /**
295      * FOR INTERNAL USE TO HK2
296      */
297     public static <V extends Annotation> V getAnnotation(Class<?> annotated,
298         Class<V> annotation,
299         boolean walkParentChain) {
300       V v = annotated.getAnnotation(annotation);
301       if (null != v) {
302           return v;
303       }
304       
305       for (Annotation a : annotated.getAnnotations()) {
306           v = a.annotationType().getAnnotation(annotation);
307           if (null != v) {
308               return v;
309           }
310       }
311       
312       if (walkParentChain) {
313         annotated = annotated.getSuperclass();
314         if (null != annotated) {
315           return getAnnotation(annotated, annotation, true);
316         }
317       }
318       
319       return null;
320     }
