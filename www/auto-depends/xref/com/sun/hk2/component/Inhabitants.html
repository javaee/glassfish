71    @SuppressWarnings("unchecked")
72    public static Inhabitant<?> createInhabitant(Habitat habitat,
73        Holder<ClassLoader> classLoader,
74        String typeName,
75        MultiMap<String, String> metadata,
76        Inhabitant<?> lead,
77        Set<String> indicies) {
78      AbstractInhabitantImpl<?> i = new LazyInhabitant(habitat, classLoader, typeName, metadata, lead);
79      if (contains(indicies, RunLevel.class.getName())) {
80        // this is a RunLevel service, we need to load type in order to get
81        // more type information about it, namely it's environment and actual RunLevel id
82        RunLevel rl = i.getAnnotation(RunLevel.class);
83        assert(null != rl) : typeName + " is a problem; " + i + " has no RunLevel annotation";
84        assert(!i.isInstantiated()) : "inhabitant should not be active: " + i;
85  
86        // get the appropriate RLS for this RunLevel
87        RunLevelService<?> rls = runLevelServices.get(habitat, rl);
88        InhabitantListener listener = InhabitantListener.class.isInstance(rls) ?
89            InhabitantListener.class.cast(rls) : null;
90  
91        // wrap the inhabitant with a RunLevelInhabitant
92        int runLevel = rl.value();
93        
94        // construct the runLevel inhabitant
95        i = new RunLevelInhabitant(i, runLevel, rls.getState(), listener);
96      }
97      
98      return i;
99    }

101   private static boolean contains(Set<String> indicies, String name) {
102     if (null != indicies) {
103       if (indicies.contains(name)) {
104         return true;
105       }
106       
107       // it could be a named inhabitant, in which case we need to iterate
108       name += ":";
109       for (String index : indicies) {
110         if (index.startsWith(name)) {
111           return true;
112         }
113       }
114     }
115     return false;
116   }

132   /**
133    * Creates a {@link Inhabitant} by wrapping {@link Womb} to handle scoping right.
134    */
135   public static <T> Inhabitant<T> wrapByScope(Class<T> c, Womb<T> womb, Habitat habitat) {
136       Scoped scoped = c.getAnnotation(Scoped.class);
137       if(scoped==null)
138           return new SingletonInhabitant<T>(womb); // treated as singleton
139 
140       Class<? extends Scope> scopeClass = scoped.value();
141 
142       return wrapByScope(womb, habitat, scopeClass);
143   }
144 
145   public static <T> Inhabitant<T> wrapByScope(Womb<T> womb, Habitat habitat,
146       Class<? extends Scope> scopeClass) {
147       // those two scopes are so common and different that they deserve
148       // specialized code optimized for them.
149       if(scopeClass== PerLookup.class)
150           return womb;
151       if(scopeClass== Singleton.class)
152           return new SingletonInhabitant<T>(womb);
153   
154       // other general case
155       Scope scope = habitat.getByType(scopeClass);
156       if (scope==null)
157           throw new ComponentException("Failed to look up %s", scopeClass);
158   
159       return new ScopedInhabitant<T>(womb,scope);
160   }

178   /**
179    * Performs basic validation of the injection.
180    * 
181    * @param target
182    * @param injectedVal
183    */
184   static void validate(Object target, Object injectedVal) {
185       if (null != injectedVal) {
186           RunLevel targetRL = AbstractInhabitantImpl.getAnnotation(target.getClass(), RunLevel.class, false);
187           RunLevel injectedValRL = AbstractInhabitantImpl.getAnnotation(injectedVal.getClass(), RunLevel.class, false);
188           if (null == targetRL && null != injectedValRL) {
189               throw new ComponentException("invalid dependency from a non-RunLevel instance " +
190                   targetRL + " to a RunLevel instance " + injectedVal);
191         }
192     }
