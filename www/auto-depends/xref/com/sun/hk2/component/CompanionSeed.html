58  /***
59   * Marks the companion relationship with another component.
60   *
61   * <p>
62   * Implementations of this contract are generated for each component
63   * with the {@link CompanionOf} annotation.  
64   *
65   * @author Kohsuke Kawaguchi
66   */
67  @Contract
68  @InhabitantAnnotation("default")
69  @CagedBy(CompanionSeed.Registerer.class)
70  public @interface CompanionSeed {
71      /***
72       * The type of the lead class.
73       */
74      @Index
75      @InhabitantMetadata("lead")
76      Class<?> lead();
77  
78      /***
79       * Other metadata to capture the companion class.
80       */
81      @InhabitantMetadata("companionClass")
82      Class<?> companion();
83  
84      /***
85       * When {@link CompanionSeed} enters habitat, look
86       * for existing inhabitants and make sure all of them
87       * get its corresponding companions from this new seed.
88       *
89       * <p>
90       * Normal {@link CageBuilder}s register themselves as components,
91       * but this one is defined in HK2 so we have to get in by a special means.
92       */
93      public static final class Registerer implements CageBuilder {
94          private final Habitat habitat;
95  
96          public Registerer(Habitat habitat) {
97              this.habitat = habitat;
98          }
99  
100         public void onEntered(Inhabitant<?> i) {
101             // and if a companion seed is added to something else, make sure
102             // existing lead inhabitants will get this as a companion
103             assert i.metadata()!=null;
104             for (Inhabitant lead : habitat.getInhabitantsByType(i.metadata().getOne("lead")))
105                 lead.setCompanions(cons(lead.companions(),createCompanion(habitat,lead,i)));
106         }
107 
108         /***
109          * Allocates a new read-only list by adding one more element.
110          */
111         private <T> List<T> cons(Collection<T> list, T oneMore) {
112             int sz = list.size();
113             Object[] a = list.toArray(new Object[sz+1]);
114             a[sz]=oneMore;
115             return (List) Arrays.asList(a);
116         }
117 
118         /***
119          * Creates a companion inhabitant from the inhabitant of a {@link CompanionSeed},
120          * to be associated with a lead component.
121          */
122         public static LazyInhabitant createCompanion(Habitat habitat, final Inhabitant<?> lead, final Inhabitant<?> seed) {
123             Holder<ClassLoader> cl = new Holder<ClassLoader>() {
124                     public ClassLoader get() {
125                         return seed.type().getClassLoader();
126                     }
127                 };
128 
129             // this is the name of the class with @CompanionOf
130             String fqcn = seed.metadata().getOne(COMPANION_CLASS_KEY);
131 
132             String metadataLine = seed.metadata().getOne(COMPANION_CLASS_METADATA_KEY);
133             MultiMap<String,String> metadata;
134             if(metadataLine==null)
135                 metadata = MultiMap.emptyMap();
136             else
137                 metadata = InhabitantsParser.buildMetadata(new KeyValuePairParser(metadataLine));
138 
139             LazyInhabitant ci = new LazyInhabitant(habitat, cl, fqcn, metadata) {
140                 public Inhabitant lead() {
141                     return lead;
142                 }
143             };
144             habitat.add(ci);
145             return ci;
146         }
147     }
