75      /**
76       * Obtains the value to inject, based on the type and {@link Inject} annotation.
77       */
78      @Override
79      public <V> V getValue(final Object component,
80                            final Inhabitant<?> onBehalfOf,
81                            final AnnotatedElement target,
82                            final Type genericType,
83                            final Class<V> type) throws ComponentException {
84          final Inject inject = target.getAnnotation(Inject.class);
85          final Callable<V> callable = new Callable<V>() {
86              @Override
87              public V call() throws ComponentException {
88                  V result=null;
89  
90                  if (type.isArray()) {
91                      result = getArrayInjectValue(habitat, component, onBehalfOf, target, genericType, type);
92                  } else {
93                      if (Types.isSubClassOf(type, org.glassfish.hk2.Factory.class)) {
94                          result = getHolderInjectValue(habitat, component, onBehalfOf, target, genericType, type, inject);
95                      } else {
96                          if (genericType instanceof TypeVariable) {
97                              // ok this is a bit more complicated, the user wants us to inject a parameterized type
98                              // so we need to look at the class declaration to find out in the index of the
99                              // declaration of that parameterized type and reconcile it with our inhabitant metadata.
100                             TypeVariable[] typeVariables = component.getClass().getTypeParameters();
101                             for (int i=0;i<typeVariables.length;i++) {
102                                 if (typeVariables[i].getName().equals(((TypeVariable) genericType).getName())) {
103                                     // that's the parameterized type we are looking for.
104                                     String parameterizedType = onBehalfOf.metadata().get(InhabitantsFile.PARAMETERIZED_TYPE).get(i);
105                                     try {
106                                         Class<?> clazz = component.getClass().getClassLoader().loadClass(parameterizedType);
107                                         ContractLocatorImpl<V> contractLocator = new ContractLocatorImpl<V>(habitat, clazz, habitat.isContract(clazz));
108                                         populateContractLocator(contractLocator, target, inject);
109                                         result = contractLocator.get();
110                                     } catch(ClassNotFoundException e) {
111                                         Logger.getAnonymousLogger().warning("Cannot load class " + parameterizedType);
112                                         return null;
113                                     }
114                                 }
115                             }
116                         } else {
117                             ContractLocatorImpl<V> contractLocator = new ContractLocatorImpl<V>(habitat, genericType, habitat.isContractExt(genericType));
118                             populateContractLocator(contractLocator, target, inject);
119                             result = contractLocator.get();
120                         }
121                     }
122                 }
123 
124                 return result==null?null:validate(component, onBehalfOf, result);
125             }
126         };
127 
128         try {
129             if (habitat.isContextualFactoriesPresent()) {
130                 InjectionPoint ip = new InjectionPointImpl(component, target, type, inject, onBehalfOf);
131                 return Hk2ThreadContext.captureIPandRun(ip, callable);
132             } else {
133                 return callable.call();
134             }
135         } catch (Exception e) {
136             if (e instanceof ComponentException) throw ComponentException.class.cast(e);
137             throw new ComponentException(e);
138         }
139     }

141     protected <V> V getArrayInjectValue(Habitat habitat,
142                                         Object component,
143                                         Inhabitant<?> onBehalfOf,
144                                         AnnotatedElement target,
145                                         Type genericType,
146                                         Class<V> type) {
147         V result;
148         Class<?> ct = type.getComponentType();
149 
150         Collection<?> instances;
151         if (habitat.isContract(ct)) {
152             instances = getAllByContract(onBehalfOf, habitat, ct);
153         } else {
154             instances = getAllByType(onBehalfOf, habitat, ct);
155         }
156 
157         result = type.cast(instances.toArray((Object[]) Array.newInstance(ct, instances.size())));
158         // TODO: validate() here too
159         return result;
160     }

162     protected <V> V getHolderInjectValue(final Habitat habitat,
163                                          final Object component,
164                                          final Inhabitant<?> onBehalfOf,
165                                          final AnnotatedElement target,
166                                          final Type genericType,
167                                          final Class<V> type,
168                                          final Inject inject) throws ComponentException {
169 
170         final Type t = Types.getTypeArgument(genericType, 0);
171 
172         if (habitat.isContractExt(t)) {
173             ContractLocatorImpl<V> contractLocator = new ContractLocatorImpl<V>(habitat, t, true);
174             populateContractLocator(contractLocator, target, inject);
175             return type.cast(contractLocator.getProvider());
176         }
177 
178         // the receiver maybe requesting the inhabitant pointing to itself to have
179         // access to its own metadata.
180         final Class<?> finalType = Types.erasure(t);
181         try {
182             if (finalType.cast(component) != null) {
183                 return type.cast(onBehalfOf);
184             }
185         } catch (ClassCastException e) {
186             // ignore
187         }
188 
189         ContractLocatorImpl<V> contractLocator = new ContractLocatorImpl<V>(habitat, finalType, false);
190         return type.cast(contractLocator.getProvider());
191     }

236     @SuppressWarnings("unchecked")
237     protected <V> Collection<V> manage(Inhabitant<?> onBehalfOf, Iterable<?> inhabitants) {
238         if (null == inhabitants) {
239             return null;
240         }
241 
242         final ArrayList<V> managed = new ArrayList<V>();
243         for (Object iObj : inhabitants) {
244             Inhabitant<V> i = (Inhabitant<V>) iObj;
245             managed.add((V) manage(onBehalfOf, i).get());
246         }
247 
248         return managed;
249     }

262     void populateContractLocator(ContractLocatorImpl<?> contractLocator, AnnotatedElement target, Inject inject) {
263         Named named = target.getAnnotation(Named.class);
264         String name = inject.name();
265         if (named!=null && !inject.name().isEmpty()) {
266             throw new RuntimeException("Field or method [" + target.toString() + "] is annotated with both a @Named" +
267                     " annotation as well as a @Inject.name() value, please choose");
268         }
269         if (named!=null) {
270             name = named.value();
271         }
272         contractLocator.named(name);
273 
274         // now we need to obtain all qualifiers on the injection target so we narrow the search.
275         Annotation[] targetAnnotations = target.getAnnotations();
276 
277         // if there is only one annotation, it's @Inject, we can ignore qualifiers narrowing
278         if (name.isEmpty() && targetAnnotations.length > 1) {
279             for (Annotation annotation : target.getAnnotations()) {
280                 for (Annotation annotationAnnotation : annotation.annotationType().getAnnotations()) {
281                     if (annotationAnnotation.annotationType()==Qualifier.class) {
282                         contractLocator.annotatedWith(annotation.annotationType());
283                     }
284                 }
285             }
286         }
