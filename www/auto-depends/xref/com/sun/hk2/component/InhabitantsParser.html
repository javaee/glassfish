52  /**
53   * Parses <tt>/META-INF/inhabitants</tt> and populate {@link Habitat}.
54   *
55   * <p>
56   * This class can be subclasses to customize the parsing behavior, which is useful
57   * for ignoring some components.
58   *
59   * @author Kohsuke Kawaguchi
60   * @author Jerome Dochez
61   */
62  public class InhabitantsParser {
63    
64      private final Logger logger = Logger.getLogger(InhabitantsParser.class.getCanonicalName());
65    
66      public final Habitat habitat;
67  
68      /**
69       * Fully qualified class names of components to be replaced, to {@link Class} that replaces them.
70       * If the value is null, that means just drops it without a replacement.
71       *
72       * @see #drop(Class)
73       * @see #replace(Class, Class)
74       */
75      // Whether this feature should belong to this base class is arguable --- perhaps a better
76      // approach is to create a sub class that does it?
77      private final Map<String,Class<?>> replacements = new HashMap<String,Class<?>>();
78  
79      public InhabitantsParser(Habitat habitat) {
80          this.habitat = habitat;
81      }
82  
83      /**
84       * Tells {@link InhabitantsParser} that if it encounters the specified component
85       * while parsing inhabitants file,
86       * simply drop it and pretend that such an inhabitant had never existed.
87       *
88       * <p>
89       * This is useful when the application that's hosting an HK2 environment
90       * wants to tweak the inhabitant population at sub-module level.
91       */
92      public void drop(Class<?> component) {
93          drop(component.getName());
94      }
95  
96      public void drop(String fullyQualifiedClassName) {
97          replace(fullyQualifiedClassName,null);
98      }
99  
100     /**
101      * Tells {@link InhabitantsParser} that if it encounters the specified component
102      * while parsing inhabitants file,
103      * ignore the one in the inhabitants file and instead insert the specified 'new' component.
104      *
105      * <p>
106      * This is useful when the application that's hosting an HK2 environment
107      * wants to tweak the inhabitant population at sub-module level.
108      */
109     public void replace(Class<?> oldComponent, Class<?> newComponent) {
110         replace(oldComponent.getName(),newComponent);
111     }
112 
113     public void replace(String oldComponentFullyQualifiedClassName, Class<?> newComponent) {
114         replacements.put(oldComponentFullyQualifiedClassName,newComponent);
115     }
116 
117     /**
118      * Parses the inhabitants file (which is represented by {@link InhabitantsScanner}.
119      *
120      * <p>
121      * All the earlier drop/replace commands will be honored during this process.
122      */
123     @SuppressWarnings("unchecked")
124     public void parse(Iterable<InhabitantParser> scanner, Holder<ClassLoader> classLoader) throws IOException {
125         if (scanner==null)
126             return;
127         
128         for( InhabitantParser inhabitantParser : scanner) {
129             if (isFilteredInhabitant(inhabitantParser)) {
130                 continue;    
131             }
132 
133             String typeName = inhabitantParser.getImplName();
134             if (isFilteredInhabitant(typeName)) {
135                 continue;
136             }
137             
138             if (replacements.containsKey(typeName)) {
139                 // create a replacement instead
140                 Class<?> target = replacements.get(typeName);
141                 if(target!=null) {
142                     inhabitantParser.setImplName(target.getName());
143                     Inhabitant i = null;
144                     try {
145                       i = Inhabitants.create(target,habitat,inhabitantParser.getMetaData());
146                     } catch (Exception e) {
147                       log(typeName, e);
148                     }
149                     if (null != i) {
150                       add(i, inhabitantParser);
151                       // add index so that the new component can be looked up by the name of the old component.
152                       addIndex(i, typeName, null);
153                     }
154                 }
155             } else {
156                 Set<String> indicies = new HashSet<String>();
157                 Iterator<String> iter = inhabitantParser.getIndexes().iterator();
158                 while (iter.hasNext()) {
159                   indicies.add(iter.next());
160                 }
161                 Inhabitant<?> i = null;
162                 try {
163                   i = com.sun.hk2.component.Inhabitants.
164                     createInhabitant(habitat, classLoader, typeName,
165                         inhabitantParser.getMetaData(), null,
166                         Collections.unmodifiableSet(indicies));
167                 } catch (Exception e) {
168                   log(typeName, e);
169                 }
170                 if (null != i) {
171                   add(i, inhabitantParser);
172                 }
173             }
174         }
175     }
176 
177     private void log(String typeName, Exception e) {
178       logger.log(Level.FINE, "Warning: unable to create inhabitant for {0} - and therefore ignoring it; check classpath; re: {1}", 
179           new Object[] {typeName, e.getMessage()});
180       logger.log(Level.FINER, "", e);
181     }

183     /**
184      * Returns true if this inhabitant should be ignored.
185      *  
186      * @param inhabitantParser
187      * @return
188      */
189     protected boolean isFilteredInhabitant(InhabitantParser inhabitantParser) {
190         return false;
191     }
192 
193     /**
194      * Returns true if this inhabitant should be ignored.
195      *  
196      * @param typeName
197      * @return
198      */
199     protected boolean isFilteredInhabitant(String typeName) {
200         return false;
201     }
202     
203     /**
204      * Adds the given inhabitant to the habitat, with all its indices.
205      */
206     protected void add(Inhabitant<?> i, InhabitantParser parser) {
207         add(i);
208 
209         for (String v : parser.getIndexes()) {
210             // register inhabitant to the index
211             int idx = v.indexOf(':');
212             if(idx==-1) {
213                 // no name
214                 addIndex(i,v,null);
215             } else {
216                 // v=contract:name
217                 String contract = v.substring(0, idx);
218                 String name = v.substring(idx + 1);
219                 addIndex(i, contract, name);
220             }
221         }
222     }
223 
224     /**
225      * Adds the given inhabitant to the habitat
226      * @param i
227      */
228     protected void add(Inhabitant<?> i) {
229       logger.log(Level.FINE, "adding inhabitant: {0} to habitat {1}", 
230           new Object[] {i, habitat});
231       habitat.add(i);
232     }
233 
234     /**
235      * Adds the given inhabitant index to the habitat
236      */
237     protected void addIndex(Inhabitant<?> i, String typeName, String name) {
238       logger.log(Level.FINE, "adding index for inhabitant: {0} with typeName {1} and name {2} to habitat {3}",
239           new Object[] {i, typeName, name, habitat});
240       habitat.addIndex(i, typeName, name);
241     }
