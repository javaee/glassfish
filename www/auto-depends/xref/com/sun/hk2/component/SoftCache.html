40  package com.sun.hk2.component;

42  import java.lang.ref.SoftReference;
43  import java.util.concurrent.Callable;
44  import java.util.logging.Level;
45  import java.util.logging.Logger;
46  
47  /**
48   * A very primitive cache of at most one entry, that uses both SoftReference(s) to
49   * the key as well as the value it caches.  Useful to provide minimal caching
50   * without imposing on heap.
51   * 
52   * @author Jeff Trent
53   */
54  public class SoftCache<K, V> {
55  
56    private Logger logger = Logger.getLogger(SoftCache.class.getName());
57    
58    protected SoftReference<K> key;
59    protected SoftReference<V> value;
60    
61    /**
62     * Returns either the cached value if hasn't been reclaimed by GC.  Otherwise invokes
63     * the callable to refresh the cached value, and caching it again in the process.
64     * 
65     * @param key the key; must be non-null
66     * @param callable the callable in case the cache is empty; must be non-null
67     * @return the value (either from cache or populated by the callable as a last resort)
68     */
69    public synchronized V get(K k, Callable<V> callable) {
70      assert(null != callable);
71      K thisK = (null == key) ? null : key.get();
72      if (k.equals(thisK)) {
73        if (null == value) {
74          return usingCacheValue(null);
75        } else {
76          V thisV = value.get();
77          if (null != thisV) {
78            return usingCacheValue(thisV);
79          }
80        }
81      }
82  
83      key = new SoftReference<K>(k);
84      try {
85        V thisV = callable.call();
86        return cacheValue(value, thisV);
87      } catch (Exception e) {
88        throw new RuntimeException(e);
89      }
90    }
91  
92    protected  V cacheValue(SoftReference<V> value2, V thisV) {
93      if (null == thisV) {
94        value = null;
95      } else {
96        value = new SoftReference<V>(thisV);
97      }
98      
99      logger.log(Level.FINEST, "caching value");
100     return thisV;
101   }
102 
103   protected V usingCacheValue(V thisV) {
104     logger.log(Level.FINEST, "using cached value");
105     return thisV;
106   }
107 
108   /**
109    * Clears the cache
110    */
111   public void clear() {
112     logger.log(Level.FINEST, "clearing cached value");
113     key = null;
114     value = null;
115   }
