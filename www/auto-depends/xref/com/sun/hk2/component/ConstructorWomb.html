55  /***
56   * Creates an object from its constructor.
57   * @author Kohsuke Kawaguchi
58   */
59  public class ConstructorWomb<T> extends AbstractWombImpl<T> {
60      private final Habitat habitat;
61      private final ScopeInstance singletonScope;
62  
63  
64      public ConstructorWomb(Class<T> type, Habitat habitat, MultiMap<String,String> metadata) {
65          super(type,metadata);
66          this.habitat = habitat;
67          singletonScope = habitat.singletonScope;
68      }

70      public T create(Inhabitant onBehalfOf) throws ComponentException {
71          try {
72              return type.newInstance();
73          } catch (InstantiationException e) {
74              throw new ComponentException("Failed to create "+type,e);
75          } catch (IllegalAccessException e) {
76              throw new ComponentException("Failed to create "+type,e);
77          } catch (LinkageError e) {
78              throw new ComponentException("Failed to create "+type,e);
79          }
80      }

83      public void initialize(final T t, final Inhabitant onBehalfOf) throws ComponentException {
84  
85          Scoped scoped = t.getClass().getAnnotation(Scoped.class);
86          ScopeInstance si = (scoped == null ? singletonScope : getScope(scoped));
87          AccessController.doPrivileged(new PrivilegedAction() {
88              //doprivileged required for running with SecurityManager ON
89              public java.lang.Object run() {
90                  inject(habitat, t, onBehalfOf);
91                  return null;
92              }
93          });
94  
95          if(si!=null)
96              // extraction amounts to no-op if this is prototype scope. so skip that.
97              extract(t, si);
98  
99      }

101     /***
102      * Extracts resources identified with {@link org.jvnet.hk2.annotations.Extract} annotations
103      * into the given scope.
104      *
105      * <p>
106      * This method is for the use within HK2, and not really meant
107      * to be used by application code.
108      *
109      * @param component the component we should extract resources from
110      * @throws ComponentException if the resource extract fail
111      */
112     public void extract(Object component, ScopeInstance si) throws ComponentException {
113         Class<?> currentClass = component.getClass();
114         while (!currentClass.equals(Object.class)) {
115             for (Field field : currentClass.getDeclaredFields()) {
116                 Extract extract = field.getAnnotation(Extract.class);
117                 if (extract == null)    continue;
118 
119                 try {
120                     field.setAccessible(true);
121                     Object value = field.get(component);
122                     Class<?> type = field.getType();
123 
124 //                    if (LOGGER.isLoggable(Level.FINER)) {
125 //                        LOGGER.log(Level.FINER, "Extracting resource " + value + " returned from " + field);
126 //                    }
127                     if (value!=null) {
128                         extractValue(value, si, extract.name(), type);
129                     } else {
130 //                        if (LOGGER.isLoggable(Level.FINE)) {
131 //                            LOGGER.log(Level.FINE, "Resource returned from " + field + " is null");
132 //                        }
133                     }
134                 } catch (IllegalArgumentException ex) {
135                     throw new ComponentException("Extraction failed on " + field, ex);
136 
137                 } catch (IllegalAccessException ex) {
138                     throw new ComponentException("Extraction failed on " + field, ex);
139                 }
140             }
141             for (Method method : currentClass.getDeclaredMethods()) {
142                 Extract extract = method.getAnnotation(Extract.class);
143                 if (extract == null)    continue;
144 
145                 Class<?> type = method.getReturnType();
146                 if (type == null) {
147                     throw new ComponentException("Extraction failed : %s has a void return type",method);
148                 }
149                 if (method.getParameterTypes().length > 0) {
150                     throw new ComponentException("Extraction failed : %s takes parameters, it should not",method);
151                 }
152                 try {
153                     method.setAccessible(true);
154                     Object value = method.invoke(component);
155 //                    if (LOGGER.isLoggable(Level.FINER)) {
156 //                        LOGGER.log(Level.FINER, "Extracting resource " + value + " returned from " + method);
157 //                    }
158                     if (value!=null) {
159                         extractValue(value, si, extract.name(), type);
160                     } else {
161 //                        if (LOGGER.isLoggable(Level.FINE)) {
162 //                            LOGGER.log(Level.FINE, "Resource returned from " + method + " is null");
163 //                        }
164                     }
165                 } catch (IllegalArgumentException ex) {
166                     throw new ComponentException("Extraction failed on " + method.toGenericString(), ex);
167                 } catch (InvocationTargetException ex) {
168                     throw new ComponentException("Extraction failed on " + method.toGenericString(), ex);
169                 } catch (IllegalAccessException ex) {
170                     throw new ComponentException("Extraction failed on " + method.toGenericString(), ex);
171                 }
172             }
173             currentClass = currentClass.getSuperclass();
174         }
175     }

177     @SuppressWarnings("unchecked")
178     private void extractValue(Object value, ScopeInstance si, String name,  Class type) {
179         if(value instanceof Iterable) {
180             for (Object o : (Iterable)value) {
181                 extractSingleValue(o,name, type, si);
182             }
183         } else
184         if (type.isArray()) {
185             Object[] values = (Object[]) value;
186             for (Object o : values) {
187                 extractSingleValue(o,name, type, si);
188             }
189         } else {
190             extractSingleValue(value, name, type, si);
191         }
192     }
193 
194     private <T> void extractSingleValue(T o, String name, Class type, final ScopeInstance si) {
195         // TODO: name support. Wouldn't it be nice if Map<String,Object> extracts to named objects?
196         // or recognize the "Named" interface?
197         Inhabitant<T> i = new AbstractWombImpl<T>((Class<T>)o.getClass(),MultiMap.<String,String>emptyMap()) {
198             public T create(Inhabitant onBehalfOf) throws ComponentException {
199                 // only look at objects already available in the scope.
200                 // TODO: this obviously can return null. think about how to reconcile this semantics with the rest
201                 return si.get(this);
202             }
203         };
204         if (name==null) {
205             habitat.add(i);
206         } else {
207             habitat.addIndex(i, type.getName(), name);
208         }
209         si.put(i,o);
210     }

212     /***
213      * Determines the {@link ScopeInstance} that stores the component.
214      *
215      * @return
216      *      null for prototype scope. (Note that in {@link Scope#current()}
217      *      null return value is an error.)
218      */
219     private ScopeInstance getScope(Scoped svc) throws ComponentException {
220         Class<? extends Scope> s = svc.value();
221         // for performance reason and to avoid infinite recursion,
222         // recognize these two fundamental built-in scopes and process them differently.
223         if(s==Singleton.class)
224             return singletonScope;
225         if(s==PerLookup.class)
226             return null;
227 
228         // for all the other scopes, including user-defined ones.
229         Scope scope = habitat.getByType(s);
230         ScopeInstance si = scope.current();
231         if(si==null) // scope is an extension point, so beware for broken implementations
232             throw new ComponentException(scope+" returned null from the current() method");
233         return si;
234     }
