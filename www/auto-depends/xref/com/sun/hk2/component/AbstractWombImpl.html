58  public abstract class AbstractWombImpl<T> extends AbstractInhabitantImpl<T> implements Womb<T> {
59      protected final Class<T> type;
60      private final MultiMap<String,String> metadata;
61  
62      public AbstractWombImpl(Class<T> type, MultiMap<String,String> metadata) {
63          this.type = type;
64          this.metadata = metadata;
65      }
66  
67      public final String typeName() {
68          return type.getName();
69      }
70  
71      public final Class<T> type() {
72          return type;
73      }
74  
75      public final T get(Inhabitant onBehalfOf) throws ComponentException {
76          T o = create(onBehalfOf);
77          initialize(o,onBehalfOf);
78          return o;
79      }
80  
81      public void  initialize(T t, Inhabitant onBehalfOf) throws ComponentException {
82          // default is no-op
83      }

85      public void release() {
86          // Womb creates a new instance every time,
87          // so there's nothing to release here.
88      }

91          return metadata;
92      }

94      /***
95       * Performs resource injection on the given instance from the given habitat.
96       *
97       * <p>
98       * This method is an utility method for subclasses for performing injection.
99       */
100     protected void inject(final Habitat habitat, T t, final Inhabitant onBehalfOf) {
101         // TODO: make it a field of Habitat to avoid unnecessary InjectionManager allocations
102         (new InjectionManager<Inject>() {
103             public boolean isOptional(Inject annotation) {
104                 return annotation.optional();
105             }
106 
107             /***
108              * Obtains the value to inject, based on the type and {@link Inject} annotation.
109              */
110             @SuppressWarnings("unchecked")
111             protected Object getValue(Object component, AnnotatedElement target, Class type) throws ComponentException {
112                 if (type.isArray()) {
113                     Class<?> ct = type.getComponentType();
114 
115                     Collection instances;
116                     if(habitat.isContract(ct))
117                         instances = habitat.getAllByContract(ct);
118                     else
119                         instances = habitat.getAllByType(ct);
120                     return instances.toArray((Object[]) Array.newInstance(ct, instances.size()));
121                 } else if (Types.isSubClassOf(type, Holder.class)){
122                     Type t = Types.getTypeArgument(((java.lang.reflect.Field) target).getGenericType(), 0);
123                     Class finalType = Types.erasure(t);
124                     if (habitat.isContract(finalType)) {
125                         return habitat.getInhabitants(finalType, target.getAnnotation(Inject.class).name());
126                     }
127                     return habitat.getInhabitantByType(finalType);
128 
129                 } else {
130                     if(habitat.isContract(type))
131                         // service lookup injection
132                         return habitat.getComponent(type, target.getAnnotation(Inject.class).name());
133 
134                     // ideally we should check if type has @Service or @Configured
135 
136                     // component injection
137                     return habitat.getByType(type);
138                 }
139             }
140         }).inject(t, Inject.class);
141 
142         (new InjectionManager<Lead>() {
143             protected Object getValue(Object component, AnnotatedElement target, Class type) throws ComponentException {
144                 Inhabitant lead = onBehalfOf.lead();
145                 if(lead==null)
146                     // TODO: we should be able to check this error at APT, too.
147                     throw new ComponentException(component.getClass()+" requested @Lead injection but this is not a companion");
148 
149                 if(type==Inhabitant.class) {
150                     return lead;
151                 }
152 
153                 // otherwise inject the target object
154                 return lead.get();
155             }
156         }).inject(t, Lead.class);
157 
158         // postContruct call if any
159         if(t instanceof PostConstruct)
160             ((PostConstruct)t).postConstruct();
161     }
