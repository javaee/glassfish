157   protected boolean getTheAndSetFor(Set<String> result, String propName, boolean removeIt, Node node) {
158     boolean hadMatch = false;
159     if (node.type == BASE_TYPE) {
160       if (node.leftHandSide.equals(propName)) {
161         result.add(node.rightHandSide);
162         hadMatch = true;
163         if (removeIt) {
164           node.leftHandSide="";
165         }
166       }
167     } else if (node.type == AND_TYPE) {
168       Iterator<Node> iter = node.children.iterator();
169       while (iter.hasNext()) {
170         Node andNode = iter.next();
171         boolean matched = getTheAndSetFor(result, propName, removeIt, andNode);
172         hadMatch |= matched;
173         if (matched && removeIt) {
174           iter.remove();
175         }
176       }
177     }
178     return hadMatch;
179   }

182   private Map<String, Set<String>> getMatchingMap(MultiMap<String, String> props) {
183     Map<String, Set<String>> result = new HashMap<String, Set<String>>();
184     for (Entry<String, List<String>> entry : props.entrySet()) {
185       String key = entry.getKey().toLowerCase();
186       Set<String> set = result.get(key);
187       if (null == set) {
188         set = new HashSet<String>();
189         result.put(entry.getKey().toLowerCase(), set);
190       }
191       set.addAll(entry.getValue());
192     }
193     return result;
194   }

205   /* package */static boolean filterMatch(String filter,
206       Map<String, Object> properties) {
207     if (filter == null || properties == null) {
208       throw new IllegalArgumentException();
209     }

211     ActiveString workString = new ActiveString(filter);

218   /* package */static Map<String, Set<String>> getMatchingMap(
219       Map<String, Object> paramProperties) {
220     HashMap<String, Set<String>> retVal = new HashMap<String, Set<String>>();
221 
222     // Get the "fixed" map AFTER adding the system generated properties
223     for (Map.Entry<String, Object> entry : paramProperties.entrySet()) {
224       String key = entry.getKey();
225       if (key == null) {
226         throw new RuntimeException("Null key in property map");
227       }
228 
229       Object value = entry.getValue();
230       if (value == null) {
231         throw new RuntimeException("Null value in property map for " + key);
232       }
233 
234       String toLowerKey = key.toLowerCase();
235       Set<String> valueSet = new HashSet<String>();
236 
237       Class<?> valueClass = value.getClass();
238       if (valueClass.isArray()) {
239         int valueLength = Array.getLength(value);
240         for (int lcv = 0; lcv < valueLength; lcv++) {
241           Object arrayValue = Array.get(value, lcv);
242           if (arrayValue == null) {
243             throw new RuntimeException("Null value in array for " + key
244                 + " at index " + lcv);
245           }
246 
247           String arrayValueString = arrayValue.toString();
248           valueSet.add(arrayValueString);
249         }
250       } else {
251         String valueString = value.toString();
252         valueSet.add(valueString);
253       }
254 
255       retVal.put(toLowerKey, valueSet);
256     }
257 
258     return retVal;
259   }

261   private static Node parse(ActiveString workString) {
262     Node retVal = new Node();
263 
264     // I am looking for the start of an expression
265     workString.toStatementStart();
266 
267     int type = workString.getStatementType();
268     switch (type) {
269     case BASE_TYPE:
270       retVal.setType(BASE_TYPE);
271       String lhs = workString.getLeftHandSide();
272       boolean isEquals = workString.getEqualsOrNotEquals();
273       String rhs = workString.getRightHandSide();
274 
275       retVal.setLeftHandSide(lhs);
276       retVal.setIsEquals(isEquals);
277       retVal.setRightHandSide(rhs);
278       break;
279     case AND_TYPE:
280     case OR_TYPE:
281       retVal.setType(type);
282       while (workString.finishedOrNewStatement()) {
283         Node child = parse(workString);
284         retVal.addChild(child);
285       }
286 
287       if (retVal.getNumChildren() <= 1) {
288         throw new RuntimeException(
289             "There was zero or one children of an AND or OR construct");
290       }
291 
292       break;
293     case NEST_TYPE:
294       // This type was NOT incremented past the '('
295       Node lastNode = parse(workString);
296       retVal.copyNode(lastNode);
297       break;
298     case FALSE_TYPE: // getStatementType cannot return FALSE_TYPE
299     default:
300       throw new AssertionError("Unknown node type " + type);
301     }
302 
303     return retVal;
304   }

307   private static class Node {
308 
309     private int type = UNKNOWN_TYPE;
310 
311     // For base type
312     private String leftHandSide;
313 
314     private boolean equals;
315 
316     private String rightHandSide;
317 
318     // For and and or types
319     private final List<Node> children = new LinkedList<Node>();
320 
321     private Node() {
322       this.type = AND_TYPE;
323     }
324 
325     private Node(String lhs, boolean equals, String rhs) {
326       this.type = BASE_TYPE;
327       this.leftHandSide = lhs;
328       this.equals = equals;
329       this.rightHandSide = rhs;
330     }
331 
332     private void setType(int paramType) {
333       type = paramType;
334     }
335 
336     private void setLeftHandSide(String lhs) {
337       leftHandSide = lhs.toLowerCase();
338     }
339 
340     private void setRightHandSide(String rhs) {
341       rightHandSide = rhs;
342     }
343 
344     private void setIsEquals(boolean isEquals) {
345       equals = isEquals;
346     }
347 
348     private void addChild(Node child) {
349       children.add(child);
350     }
351 
352     private int getNumChildren() {
353       return children.size();
354     }
355 
356     private void copyNode(Node copyNode) {
357       type = copyNode.type;
358 
359       leftHandSide = copyNode.leftHandSide;
360       equals = copyNode.equals;
361       rightHandSide = copyNode.rightHandSide;
362 
363       children.clear();
364       children.addAll(copyNode.children);
365     }
366 
367     private boolean doesMatch(Map<String, Set<String>> toMatch) {
368       if (type == BASE_TYPE) {
369         Set<String> valueSet = toMatch.get(leftHandSide);
370         if (valueSet == null) {
371           if (equals) {
372             return (null == leftHandSide || leftHandSide.length() == 0);
373           }
374 
375           return true;
376         }
377 
378         WildcardMatcher matcher = new WildcardMatcher(rightHandSide);
379         for (String val : valueSet) {
380           if (matcher.matches(val)) {
381             return equals;
382           }
383         }
384 
385         // valueSet does *not* contain the value
386         if (equals) {
387           return false;
388         }
389 
390         return true;
391       }
392 
393       if (type == AND_TYPE) {
394         for (Node andNode : children) {
395           if (!andNode.doesMatch(toMatch)) {
396             return false;
397           }
398         }
399 
400         return true;
401       }
402 
403       if (type == OR_TYPE) {
404         for (Node orNode : children) {
405           if (orNode.doesMatch(toMatch)) {
406             return true;
407           }
408         }
409 
410         return false;
411       }
412 
413       if (type == FALSE_TYPE) {
414         return false;
415       }
416 
417       throw new AssertionError("Unknown node type " + type);
418     }

422       String typeString;
423 
424       switch (type) {
425       case BASE_TYPE:
426         typeString = "BASE_TYPE";
427         break;
428       case AND_TYPE:
429         typeString = "AND_TYPE";
430         break;
431       case OR_TYPE:
432         typeString = "OR_TYPE";
433         break;
434       case FALSE_TYPE:

437       default:

442           + "equals=" + equals + "," + "rhs=" + rightHandSide + ","
443           + System.identityHashCode(this) + ")";
444     }

447   private static class ActiveString {
448     final String original;
449 
450     int dot = 0;
451 
452     char data[];
453 
454     private ActiveString(String fullString) {
455       original = fullString;
456       data = fullString.toCharArray();
457     }
458 
459     /**
460      * Increments dot safely
461      * 
462      * @param doThrow
463      *          Tells this method to throw if it reaches the end of the string,
464      *          the state machine knows that this should not reach the end
465      * @return true if there are more characters, false if this is the end
466      */
467     private boolean incrementDot(boolean doThrow) {
468       if (dot + 1 >= data.length) {
469         // finished
470         if (doThrow) {
471           throw new RuntimeException("Invalid end of string reached in "
472               + original + " at character " + dot);
473         }
474 
475         return false;
476       }
477 
478       dot++;
479       return true;
480     }
481 
482     /**
483      * Skips white space
484      * 
485      * @param doThrow
486      *          if true throw an exception if the end is reached
487      * @return true if there are more characters, false if this is the end
488      */
489     private boolean skipWhiteSpace(boolean doThrow) {
490       boolean retVal = false;
491 
492       while (Character.isWhitespace(data[dot])) {
493         retVal = incrementDot(doThrow);
494       }
495 
496       return retVal;
497     }
498 
499     private void toStatementStart() {
500       skipWhiteSpace(true);
501 
502       if (!(data[dot] == LEFT)) {
503         throw new RuntimeException("Statement did not start with '(', found "
504             + data[dot] + " at character " + dot);
505       }
506 
507       // Get past the left parenthesis
508       incrementDot(true);
509     }
510 
511     private int getStatementType() {
512       skipWhiteSpace(true);
513 
514       if (data[dot] == AND) {
515         incrementDot(true);
516         return AND_TYPE;
517       }
518 
519       if (data[dot] == OR) {
520         incrementDot(true);
521         return OR_TYPE;
522       }
523 
524       if (data[dot] == LEFT) {
525         // Do not increment dot
526         return NEST_TYPE;
527       }
528 
529       // Do not increment dot
530       return BASE_TYPE;
531     }
532 
533     /**
534      * 
535      * @return false if this is finished (increments past the final ')' and true
536      *         if there is another statement (does NOT increment past the '(')
537      */
538     private boolean finishedOrNewStatement() {
539       skipWhiteSpace(true);
540 
541       if (data[dot] == RIGHT) {
542         incrementDot(false);
543         return false;
544       }
545 
546       // Do not increment past the dot
547       return true;
548     }
549 
550     private String getLeftHandSide() {
551       int startIndex = dot;
552       int count = 0;
553 
554       while (data[dot] != EQ) {
555         if ((data[dot] == NOT) && ((dot + 1) < data.length)
556             && (data[dot + 1] == EQ)) {
557           // Found a not equals
558           return new String(data, startIndex, count);
559         }
560 
561         count++;
562         incrementDot(true);
563       }
564 
565       return new String(data, startIndex, count);
566     }
567 
568     private boolean getEqualsOrNotEquals() {
569       if (data[dot] == EQ) {
570         incrementDot(true);
571         return true;
572       }
573 
574       if (data[dot] == NOT) {
575         incrementDot(true);
576         if (!(data[dot] == EQ)) {
577           throw new AssertionError("Invalid not equals at character " + dot);
578         }
579 
580         incrementDot(true);
581         return false;
582       }
583 
584       throw new AssertionError(
585           "Was looking for equals or not equals but found " + data[dot]
586               + " at character " + dot);
587     }
588 
589     private String getRightHandSide() {
590       int startIndex = dot;
591       int count = 0;
592 
593       while (data[dot] != RIGHT) {
594         count++;
595         incrementDot(true);
596       }
597 
598       // This might be the end, but need to get past the )
599       incrementDot(false);
600       return new String(data, startIndex, count);
601     }
602 
603     private void finish() {
604       if (skipWhiteSpace(false) == true) {
605         throw new RuntimeException(
606             "Found extra characters at end of LDAP string " + original
607                 + " at character " + dot);
608       }
609     }
610 
611     public String toString() {
612       char dataDot = (dot < data.length) ? data[dot] : ' ';
613 
614       return "ActiveString(" + original + "," + dot + "," + dataDot + ")";
615     }
616   }
