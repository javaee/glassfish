47  /***
48   * ComponentManager is responsible for instantiating components, as well as injecting
49   * and extracting resources from components. Components are defined with an interface they
50   * implement. Usually a component will implement the {@link Service} interface so the
51   * corresponding META-INF/services entries are created automatically.
52   *
53   * The component manager is a thread safe, it keeps no context between invocations.
54   *
55   * @author Jerome Dochez
56   * @author Kohsuke Kawaguchi
57   */
58  @Service(scope=Singleton.class)
59  public class ComponentManager extends InjectionManager<Inject> {

100     /***
101      * Obtains a reference to the component inside the manager.
102      *
103      * <p>
104      * This is the "new Foo()" equivalent in the IoC world.
105      *
106      * <p> 
107      * Depending on the {@link Scope} of the component, a new instance
108      * might be created, or an existing instance might be returned.
109      *
110      * @return
111      *      non-null.
112      * @throws ComponentException
113      *      If failed to obtain a requested instance.
114      *      In practice, failure only happens when we try to create a
115      *      new instance of the component.
116      */
117     public <T> T getComponent(Class<T> clazz) throws ComponentException {
118         Service svc = clazz.getAnnotation(Service.class);
119         if(svc==null)
120             throw new ComponentException(clazz+" is not a service");
121         try {
122             ScopeInstance si = getScope(svc);
123             if(si==null)    // prototype scope?
124                 return instantiate(clazz,svc,null,null);
125 
126             // TODO: ResourceLocator needs to get contract class, not service class.
127             ResourceLocator<T> loc = new ResourceLocator<T>(svc, clazz);
128 
129             T c = si.store.lookup(loc);
130             if(c!=null) return c; // component already exists
131 
132             // so far the execution is concurrent, so to avoid creating
133             // the same component twice, enter a lock
134             final Object lock = instanciationLock[(loc.hashCode()&0x7FFFFFFF)%instanciationLock.length];
135             synchronized(lock) {
136                 c = si.store.lookup(loc);
137                 if(c!=null)
138                     return c; // glad we locked!
139 
140                 // really create a new instance now
141                 return instantiate(clazz,svc,si,loc);
142             }
143         } catch (ComponentException e) {
144             throw new ComponentException("Failed to obtain "+clazz+" component",e);
145         }
146     }

148     /***
149      * Add an already instantiated component to this manager. The component has
150      * been instantiated by external code, however dependency injection, PostConstruct
151      * invocation and dependency extraction will be performed on this instance before
152      * it is store in the relevant scope's resource manager.
153      *
154      * @param name name of the component, could be default name
155      * @param component component instance
156      * @throws ComponentException if the passed object is not an HK2 component or
157      * injection/extraction failed.
158      */
159     public <T> void addComponent(String name, T component)
160         throws ComponentException {
161 
162         Service svc = component.getClass().getAnnotation(Service.class);
163         if (svc==null) {
164             throw new ComponentException(component.getClass() + " is not annotated with @Service");
165         }
166         // perform injection
167         inject(component);
168 
169         // postContruct call if any
170         if(component instanceof PostConstruct)
171             ((PostConstruct)component).postConstruct();
172 
173         ScopeInstance si = getScope(svc);
174         if(si!=null)
175             // extraction amounts to no-op if this is prototype scope. so skip that.
176             extract(component, si);
177 
178         // return component for future use.
179         si.store.add(new ResourceLocator(name, component.getClass()), component);
180     }

182     private static final Object[] instanciationLock = new Object[32];
183     static {
184         for( int i=0; i<instanciationLock.length; i++ )
185             instanciationLock[i] = new Object();
186     }

188     /***
189      * Determines the {@link ScopeInstance} that stores the component.
190      *
191      * @return
192      *      null for prototype scope. (Note that in {@link Scope#current()}
193      *      null return value is an error.)
194      */
195     private ScopeInstance getScope(Service svc) throws ComponentException {
196         Class<? extends Scope> s = svc.scope();
197         // for performance reason and to avoid infinite recursion,
198         // recognize these two fundamental built-in scopes and process them differently.
199         if(s==Singleton.class)
200             return singletonScope;
201         if(s==PerLookup.class)
202             return null;
203 
204         // for all the other scopes, including user-defined ones.
205         Scope scope = getComponent(s);
206         ScopeInstance si = scope.current();
207         if(si==null) // scope is an extension point, so beware for broken implementations
208             throw new ComponentException(scope+" returned null from the current() method");
209         return si;
210     }

212     /***
213      * Instantiate a new component instance and performs injection.
214      *
215      * @param impl
216      *      Implementation class to be instanciated.
217      * @param svc
218      *      Service annotation on the class.
219      * @param si
220      *      the instanciated component will be put into this scope.
221      *      this is used to store all {@link Extract extracted} resources.
222      * @param loc
223      *      the loaded component will be stored with this name.
224      * @return a new instance of a service implementing the contract interface
225      */
226     private <T> T instantiate(Class<T> impl, Service svc, ScopeInstance si, ResourceLocator<T> loc) throws ComponentException {
227 
228         T component;
229         try {
230             if(svc.factory()!=Factory.class) {
231                 // create a component via a factory
232                 Factory factory = getComponent(svc.factory());
233                 Object o = factory.getObject();
234                 try {
235                     component = impl.cast(o);
236                 } catch (ClassCastException e) {
237                     throw new ComponentException(svc.factory()+" returned "+o+" but expected "+impl, e);
238                 }
239             } else {
240                 component = impl.newInstance();
241             }
242         } catch (InstantiationException e) {
243             throw new ComponentException("Exception while instantiating component of type " + impl, e);
244         } catch (IllegalAccessException e) {
245             throw new ComponentException("Exception while instantiating component of type " + impl, e);
246         } catch (NoClassDefFoundError e) {
247             // when dependency of 'impl' fails to load, we get NCDFE.
248             // wrap the exception to assist trouble-shooting
249             throw new ComponentException("Exception while instantiating component of type " + impl, e);
250         }
251 
252         // store the component. The component is not fully constructed yet,
253         // but when the injection and postConstruct triggers cascading
254         // effects and that causes a cyclic reference to this component,
255         // this early-storing allows them to find the right component.
256         //
257         // partially constructed components are not exposed to other threads
258         // since this part of the code executes in synchronized block.
259         if(loc!=null)
260             si.store.add(loc,component);
261 
262         // perform injection
263         inject(component);
264 
265         // postContruct call if any
266         if(component instanceof PostConstruct)
267             ((PostConstruct)component).postConstruct();
268 
269         if(si!=null)
270             // extraction amounts to no-op if this is prototype scope. so skip that.
271             extract(component, si);
272 
273         // return component for future use.
274         return component;
275     }

289     /***
290      * Extracts resources identified with {@link Extract} annotations.
291      *
292      * TODO: it's not clear what type under which such components should be stored.
293      *
294      * @param component the component we should extract resources from
295      * @throws ComponentException if the resource extract fail 
296      */
297     private void extract(Object component, ScopeInstance si) throws ComponentException {
298 
299         Class<?> currentClass = component.getClass();
300         while (!currentClass.equals(Object.class)) {
301             for (Field field : currentClass.getDeclaredFields()) {
302                 Extract extract = field.getAnnotation(Extract.class);
303                 if (extract == null)    continue;
304 
305                 try {
306                     field.setAccessible(true);
307                     Object value = field.get(component);
308                     Class<?> type = field.getType();
309 
310 //                    if (LOGGER.isLoggable(Level.FINER)) {
311 //                        LOGGER.log(Level.FINER, "Extracting resource " + value + " returned from " + field);
312 //                    }
313                     if (value!=null) {
314                         extractValue(value, si, type, extract);
315                     } else {
316 //                        if (LOGGER.isLoggable(Level.FINE)) {
317 //                            LOGGER.log(Level.FINE, "Resource returned from " + field + " is null");
318 //                        }
319                     }
320                 } catch (IllegalArgumentException ex) {
321                     throw new ComponentException("Extraction failed on " + field, ex);
322 
323                 } catch (IllegalAccessException ex) {
324                     throw new ComponentException("Extraction failed on " + field, ex);
325                 }
326             }
327             for (Method method : currentClass.getDeclaredMethods()) {
328                 Extract extract = method.getAnnotation(Extract.class);
329                 if (extract == null)    continue;
330 
331                 Class<?> type = method.getReturnType();
332                 if (type == null) {
333                     throw new ComponentException("Extraction failed : " + method + " has a void return type");
334                 }
335                 if (method.getParameterTypes().length > 0) {
336                     throw new ComponentException("Extraction failed : " + method + " takes parameters, it should not");
337                 }
338                 try {
339                     method.setAccessible(true);
340                     Object value = method.invoke(component);
341 //                    if (LOGGER.isLoggable(Level.FINER)) {
342 //                        LOGGER.log(Level.FINER, "Extracting resource " + value + " returned from " + method);
343 //                    }
344                     if (value!=null) {
345                         extractValue(value, si, type, extract);
346                     } else {
347 //                        if (LOGGER.isLoggable(Level.FINE)) {
348 //                            LOGGER.log(Level.FINE, "Resource returned from " + method + " is null");
349 //                        }
350                     }
351                 } catch (IllegalArgumentException ex) {
352                     throw new ComponentException("Extraction failed on " + method.toGenericString(), ex);
353                 } catch (InvocationTargetException ex) {
354                     throw new ComponentException("Extraction failed on " + method.toGenericString(), ex);
355                 } catch (IllegalAccessException ex) {
356                     throw new ComponentException("Extraction failed on " + method.toGenericString(), ex);
357                 }
358             }
359             currentClass = currentClass.getSuperclass();
360         }
361 
362     }

364     @SuppressWarnings("unchecked")
365     private void extractValue(Object value, ScopeInstance si, Class type, Extract extract) {
366         if(value instanceof Iterable) {
367             for (Object o : Iterable.class.cast(value)) {
368                 si.store.add(null, o);
369             }
370         } else
371         if (type.isArray()) {
372             Object[] values = (Object[]) value;
373             for (Object o : values) {
374                 si.store.add(null, o);
375             }
376         } else {
377             si.store.add(new ResourceLocator(extract, type), value);
378         }
379     }

382     /***
383      * Obtains all the components that implement the given contract.
384      *
385      * <p>
386      * If necessary, the components are created.
387      *
388      * @param contract type of service to load/intitialized
389      * @return
390      *      can be empty but never null.
391      */
392     public <T> Collection<T> getComponents(Class<T> contract)  {
393 
394         Set<T> providers = new HashSet<T>();
395         for (Class<? extends T> providerClass : serviceLookup.getProvidersClass(contract)) {
396 
397             try {
398                 Service service = providerClass.getAnnotation(Service.class);
399                 assert service!=null;
400 
401                 Configurable configurable = providerClass.getAnnotation(Configurable.class);
402                 if (configurable!=null) {
403                     // get them from the scope.
404                     ScopeInstance scope = getScope(service);
405                     providers.addAll(scope.store.lookupAll(providerClass));
406                 } else {
407                     providers.add(getComponent(providerClass));
408                 }
409             } catch(ComponentException e) {
410                 LOGGER.log(Level.SEVERE,"Cannot instantiate service " + providerClass,e);
411             }
412         }
413         return providers;
414         
415     }

427     /***
428      * Loads a component that implements the given contract.
429      *
430      * <p>
431      * If this method is used with a non-null name, then a service
432      * that implements the given contract AND have the specified name
433      * will be returned.
434      *
435      * <p>
436      * If this method is used with null name, then one service
437      * that implements the given contract will be returned, regardless of its name.
438      * If there are multiple service classes, which one gets returned is unspecified.
439      *
440      * @return

445         if(provider==null)
446             return null;    // not found
447         return getComponent(provider);
448     }

450     /***
451      * Finds the class of the component that implements the given contract
452      * and has the specified name.
453      *
454      * @see #getComponent(ResourceLocator)
455      */
456     public <T> Class<? extends T> getComponentClass(ResourceLocator<T> locator) throws ComponentException {
457         // TODO: more efficient implementation needed, but for now to get things moving
458         Class<? extends T> cachedProvider = cachedProviders.get(locator);
459         if (cachedProvider!=null)   return cachedProvider;
460 
461         Class<T> t = locator.getType();
462         for (Class<? extends T>  providerClass : serviceLookup.getProvidersClass(t)) {
463             // let's cache this for future usage
464             ResourceLocator cacheID = new ResourceLocator<T>(providerClass.getAnnotation(Service.class),
465                 t);
466             cachedProviders.put(cacheID, providerClass);
467 
468             String name = locator.getName();
469             if(name!=null) {
470                 String providerName;
471                 if(Annotation.class.isAssignableFrom(t)) {
472                     // this is a look up from the contract annotation.
473                     try {
474                         providerName = (String)t.getMethod("name").invoke(providerClass.getAnnotation((Class)t));
475                     } catch (IllegalAccessException e) {
476                         IllegalAccessError iae = new IllegalAccessError();
477                         iae.initCause(e);
478                         throw iae;
479                     } catch (InvocationTargetException e) {
480                         throw new Error(e); // impossible
481                     } catch (NoSuchMethodException e) {
482                         NoSuchMethodError nsme = new NoSuchMethodError();
483                         nsme.initCause(e);
484                         throw nsme;
485                     }
486                 } else {
487                     providerName = providerClass.getAnnotation(Service.class).name();
488                 }
489                 
490                 if(!providerName.equals(name))
491                     continue;   // name doesn't match
492             }
493 
494             return providerClass;
495         }
496 
497         return null; // not found
498     }

500     /***
501      * Destroys the component instance explicitly.
502      * <p>
503      * Normally, a component is destroyed when the scope it's in is destroyed,
504      * but you can call this method to force the early destruction.
505      * <p>
506      * Note that as soon as someone else tries to {@link #getComponent(Class)}
507      * a component will be reinstanciated.
508      */
509     public void releaseComponent(Object component) throws ComponentException {
510         Service svc = component.getClass().getAnnotation(Service.class);
511         if(svc==null)
512             throw new ComponentException("Unable to release "+component+"; no @Service annotation");
513         ScopeInstance si = getScope(svc);
514         if(si==null)
515             // if si==null, then this component is prototype scope, so no there's store.
516             // as per the prototype scope contract, there's no @PreDestroy
517             return;
518 
519         // call the preDestroy if defined on the service
520         if(component instanceof PreDestroy)
521             ((PreDestroy)component).preDestroy();
522         
523         si.store.remove(component);
524     }

541     /***
542      * Obtains the value to inject, based on the type and {@link Inject} annotation.
543      */
544     @SuppressWarnings("unchecked")
545     protected Object getValue(AnnotatedElement target, Class type) throws ComponentException {
546 
547 
548         Inject inject = target.getAnnotation(Inject.class);
549         if (type.isArray()) {
550             Class<?> ct = type.getComponentType();
551 
552             Contract ctr = ct.getAnnotation(Contract.class);
553             if(ctr!=null) {
554                 Collection instances = getComponents(ct);
555                 return instances.toArray((Object[])Array.newInstance(ct, instances.size()));
556             }
557         } else {
558             Annotation svc = type.getAnnotation(Service.class);
559             if(svc!=null)
560                 // component injection
561                 return getComponent(type);
562 
563             Annotation ctr = type.getAnnotation(Contract.class);
564             if(ctr!=null)
565                 // service lookup injection
566                 return getComponent(new ResourceLocator(inject, type));
567         }
568         throw new ComponentException(type+" cannot be injected: it's neither a contract nor a service");
569     }
