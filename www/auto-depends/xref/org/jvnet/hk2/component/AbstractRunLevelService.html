40  package org.jvnet.hk2.component;
41  
42  import org.glassfish.hk2.RunLevelDefaultScope;
43  import org.jvnet.hk2.annotations.Inject;
44  import org.jvnet.hk2.annotations.RunLevel;
45  import org.jvnet.hk2.component.internal.runlevel.DefaultRunLevelService;
46  
47  /**
48   * A default implementation for the {@link RunLevelService} that component
49   * provides can derive from to provide their own component-related lifecycle.
50   * 
51   * <p/>
52   * The targetScope is a placeholder type to represent a sub-component
53   * set of {@link RunLevel}-gated services.  It can be anything of the caller's
54   * choosing as long as it is kept unique and does not clash with the
55   * default internal run level service.
56   * 
57   * @author Jeff Trent
58   */
59  public abstract class AbstractRunLevelService<T>
60          implements RunLevelService<T>, InhabitantListener {
61  
62      @Inject
63      private Habitat habitat;
64  
65      private final Class<?> targetScope;
66      
67      private volatile DefaultRunLevelService delegate;
68      
69  
70      protected AbstractRunLevelService(Class<?> targetScope) {
71          if (null == targetScope || RunLevelDefaultScope.class == targetScope) {
72              throw new IllegalStateException("invalid target scope");
73          }
74          
75          this.targetScope = targetScope;
76      }
77  
78      @Override
79      public String toString() {
80          return getClass().getName() + "-" + System.identityHashCode(this) + "(" + delegate + ")";
81      }
82      
83      @SuppressWarnings("unchecked")
84      protected RunLevelService<T> getDelegate() {
85          if (null == delegate) {
86              if (null == habitat) {
87                  throw new IllegalStateException("habitat was not initialized");
88              }
89  
90              synchronized (this) {
91                  if (null == delegate) {
92                      delegate = new DefaultRunLevelService(habitat, 
93                              DefaultRunLevelService.DEFAULT_ASYNC_ENABLED,
94                              targetScope.getName(), targetScope, null);
95                      delegate.setParent(this);
96                  }
97              }
98          }
99          
100         return (RunLevelService<T>) delegate;
101     }
102     
103     protected DefaultRunLevelService getDefault() {
104         RunLevelService<T> rls = getDelegate();
105         if (!DefaultRunLevelService.class.isInstance(rls)) {
106             throw new UnsupportedOperationException();
107         }
108 
109         return DefaultRunLevelService.class.cast(rls);
110     }
111     
112     /**
113      * Overrides the default behavior of getting all listeners from the habitat
114      * to use a stand-in listener instead. If set to null, the default behavior
115      * will be restored.
116      * 
117      * @param listener the alternative, stand-in listener
118      */
119     public void setListener(RunLevelListener listener) {
120         getDefault().setListener(listener);
121     }
122     
123     /**
124      * Overrides the default behavior of getting the inhabitant sorter from the habitat
125      * to use a stand-in sorter instead. If set to null, the default behavior
126      * will be restored.
127      * 
128      * @param sorter the alternative, stand-in sorter
129      */
130     public void setInhabitantSorter(InhabitantSorter sorter) {
131         getDefault().setInhabitantSorter(sorter);
132     }
133 
134     /**
135      * Overrides the default behavior of getting the inhabitant activator from the habitat
136      * to use a stand-in activator instead. If set to null, the default behavior
137      * will be restored.
138      * 
139      * @param activator the alternative, stand-in activator
140      */
141     public void setInhabitantActivator(InhabitantActivator activator) {
142         getDefault().setInhabitantActivator(activator);
143     }
144     
145     @Override
146     public RunLevelState<T> getState() {
147         return getDelegate().getState();
148     }
149 
150     @Override
151     public void proceedTo(int runLevel) {
152         getDelegate().proceedTo(runLevel);
153     }
154 
155     @Override
156     public void interrupt() {
157         getDelegate().interrupt();
158     }
159 
160     @Override
161     public void interrupt(int runLevel) {
162         getDelegate().interrupt(runLevel);
163     }
164 
165     @Override
166     public boolean inhabitantChanged(EventType eventType, Inhabitant<?> inhabitant) {
167         return getDefault().inhabitantChanged(eventType, inhabitant);
168     }
