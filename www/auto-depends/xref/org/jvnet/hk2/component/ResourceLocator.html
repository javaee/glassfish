24  package org.jvnet.hk2.component;
25  
26  import org.jvnet.hk2.annotations.Inject;
27  import org.jvnet.hk2.annotations.Service;
28  import org.jvnet.hk2.annotations.Extract;
29  
30  /***
31   * Used to located resource in our resource manager.
32   * Resources are indexed with their types and can be further
33   * indexed using a name.
34   *
35   * @author Jerome Dochez
36   */
37  public final class ResourceLocator<T> {
38  
39      final static String DEFAULT_NAME="";
40  
41      final private Class<T> type;
42      final private String name;
43  
44      /***
45       * Creates a new ResourceLocator instance using the resource meta-data
46       * from the @{link Inject} annotation instance
47       * @param instance the annoation instance
48       * @param contract if the type of the resource
49       */
50      public ResourceLocator(Inject instance, Class<T> contract) {
51          this(instance!=null?instance.name():DEFAULT_NAME,contract);
52      }
53  
54      /***
55       * Creates a new ResourceLocator instance using the resource meta-data
56       * from the @{link Service} annotation instance
57       * @param instance the annoation instance
58       * @param contract if the type of the resource
59       */
60      public ResourceLocator(Service instance,  Class<T> contract) {
61          this(instance!=null?instance.name():DEFAULT_NAME,contract);
62      }
63  
64      /***
65       * Creates a new ResourceLocator instance using the resource meta-data
66       * from the @{link Extract} annotation instance
67       * @param instance the annoation instance
68       * @param contract if the type of the resource
69       */
70      public ResourceLocator(Extract instance,  Class<T> contract) {
71          this(instance!=null?instance.name():DEFAULT_NAME,contract);
72      }
73  
74      public ResourceLocator(String name,  Class<T> contract) {
75          assert contract !=null;
76          if(name==null)  name=DEFAULT_NAME;
77          this.name = name;
78          this.type = contract;
79      }
80  
81      public ResourceLocator(Class<T> contract) {
82          this(DEFAULT_NAME,contract);
83      }
84  
85      /***
86       * Returns the name of the resource if any
87       * @return the name of the resource if any
88       */
89      public String getName() {
90          return name;
91      }
92  
93      public Class<T> getType() {
94          return type;
95      }
96  
97      /***
98       * Returns true if the resource meta-data is the same
99       * @param o the other resource locator to compare with
100      * @return true if the resource meta-data is the same
101      */
102     @Override
103     public boolean equals(Object o) {
104         if(this.getClass()!=o.getClass())
105             return false;
106 
107         ResourceLocator that = (ResourceLocator) o;
108         if (name!=null) {
109             return name.equals(that.name) && type==that.type;
110         } else {
111             return that.name==null && type==that.type;
112         }
113     }
114 
115     public int hashCode() {
116         return type.hashCode()*31 ^ (name==null?0:name.hashCode());
117     }
118 }
