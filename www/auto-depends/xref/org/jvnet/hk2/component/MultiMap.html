39  import java.util.ArrayList;
40  import java.util.Collections;
41  import java.util.LinkedHashMap;
42  import java.util.List;
43  import java.util.Map;
44  import java.util.Map.Entry;
45  import java.util.concurrent.CopyOnWriteArrayList;
46  import java.util.Set;
47  import java.util.Collection;
48  import java.io.Serializable;

50  /**
51   * Map from a key to multiple values.
52   * Order is significant among values, and null values are allowed, although null keys are not.
53   * 
54   * @author Kohsuke Kawaguchi
55   * @author Jerome Dochez
56   */
57  public class MultiMap<K,V> implements Serializable {
58      private final Map<K,List<V>> store;
59      private final boolean concurrencyControls;
60  
61      /**
62       * Creates an empty multi-map with default concurrency controls
63       */
64      public MultiMap() {
65         this(new LinkedHashMap<K, List<V>>(), Habitat.CONCURRENCY_CONTROLS_DEFAULT);
66      }
67  
68      /**
69       * Creates an empty multi-map with option to use concurrency controls.
70       * Concurrency controls applies to the inner List collection held per each key.
71       * There are currently no concurrency controls around the Map portion of the data
72       * structure.
73       */
74      MultiMap(Map<K,List<V>> store) {
75        this(store, Habitat.CONCURRENCY_CONTROLS_DEFAULT);
76      }
77  
78      /**
79       * Creates an empty multi-map with option to use concurrency controls
80       */
81      MultiMap(boolean concurrencyControls) {
82        this(new LinkedHashMap<K, List<V>>(), concurrencyControls);
83      }
84      
85      
86      /**
87       * Creates a multi-map backed by the given store.
88       * @param store map to copy
89       */
90      protected MultiMap(Map<K,List<V>> store, boolean concurrencyControls) {
91          this.store = store;
92          this.concurrencyControls = concurrencyControls;
93      }
94  
95      /**
96       * Copy constructor.
97       * @param base map to copy
98       */
99      public MultiMap(MultiMap<K,V> base) {
100         this();
101         for (Entry<K, List<V>> e : base.entrySet())
102             store.put(e.getKey(), newList(e.getValue()));
103     }
104 
105     @Override
106     public final String toString() {
107         final StringBuilder builder = new StringBuilder();
108         final String newline = System.getProperty( "line.separator" );
109         builder.append("{");
110         for ( final K key : store.keySet() ) {
111             builder.append(key).append(": ");
112             builder.append(store.get(key));
113             builder.append(newline);
114         }
115         builder.append("}");
116         return builder.toString();
117     }
118     
119     /**
120      * Creates an optionally populated list to be used as an entry in the map.
121      * @param initialVal
122      * @return
123      */
124     protected List<V> newList(Collection<? extends V> initialVals) {
125       if (concurrencyControls) {
126         if (null == initialVals) {
127           return new CopyOnWriteArrayList<V>();
128         } else {
129           return new CopyOnWriteArrayList<V>(initialVals);
130         }
131       } else {
132         if (null == initialVals) {
133           return new ArrayList<V>(1);
134         } else {
135           return new ArrayList<V>(initialVals);
136         }
137       }
138     }
139 
140     public Set<K> keySet() {
141       return store.keySet();
142     }
143     
144     /**
145      * Adds one more key-value pair.
146      * @param k key to store the entry under
147      * @param v value to store in the k's values.
148      */
149     public final void add(K k,V v) {
150         List<V> l = store.get(k);
151         if (l==null) {
152             l = newList(null);
153             store.put(k,l);
154         }
155         l.add(v);
156     }
157 
158     /**
159      * Replaces all the existing values associated with the key
160      * by the given value.
161      *
162      * @param k key for the values
163      * @param v
164      *      Can be null or empty.
165      */
166     public void set(K k, Collection<? extends V> v) {
167         store.put(k, newList(v));
168     }
169 
170     /**
171      * Replaces all the existing values associated with the key
172      * by the given single value.
173      *
174      * @param k key for the values
175      * @param v singleton value for k key
176      * <p>
177      * This is short for <tt>set(k,Collections.singleton(v))</tt>
178      */
179     public void set(K k, V v) {
180         List<V> vlist = newList(null);
181         vlist.add(v);
182         store.put(k, vlist);
183     }
184 
185     /**
186      * Returns the elements indexed by the provided key
187      * @param k key for the values
188      * @return
189      *      Can be empty but never null. Read-only.
190      */
191     public final List<V> get(K k) {
192         List<V> l = store.get(k);
193         if (l==null) return Collections.emptyList();
194         return l;
195     }
196 
197     /**
198      * Package private (for getting the raw map for direct manipulation by the habitat)
199      * @param k the key
200      * @return
201      */
202     final List<V> _get(K k) {
203       List<V> l = store.get(k);
204       if (l==null) return Collections.emptyList();
205       return l;
206     }
207 
208     /**
209      * Checks if the map contains the given key.
210      * @param k key to test
211      * @return true if the map contains at least one element for this key
212      */
213     public boolean containsKey(K k) {
214         return !get(k).isEmpty();
215     }
216 
217     /**
218      * Checks if the map contains the given key(s), also extending the search
219      * to including the sub collection.
220      * 
221      * @param k1 key from top collection
222      * @param k2 key (value) from inner collection
223      * 
224      * @return true if the map contains at least one element for these keys
225      */
226     public boolean contains(K k1, V k2) {
227       List<V> list = _get(k1);
228       return list.contains(k2);
229     }
230     
231     /**
232      * Removes an key value from the map
233      * @param key key to be removed
234      * @return the value stored under this key or null if there was none
235      */
236     public List<V> remove(K key) {
237         return store.remove(key);
238     }
239 
240     /**
241      * Removes an key value pair from the map
242      * @param key key to be removed
243      * @param entry the entry to be removed from the key'ed list
244      * @return true if there was none that was deleted
245      */
246     public boolean remove(K key, V entry) {
247       List<V> list = store.get(key);
248       return (null == list) ? false : list.remove(entry);
249     }
250 
251     /**
252      * Gets the first value if any, or null.
253      * <p>
254      * This is useful when you know the given key only has one value and you'd like
255      * to get to that value.
256      *
257      * @param k key for the values
258      * @return
259      *      null if the key has no values or it has a value but the value is null.
260      */
261     public final V getOne(K k) {
262         List<V> lst = store.get(k);
263         if(lst ==null)  return null;
264         if(lst.isEmpty())   return null;
265         return lst.get(0);
266     }
267 
268     /**
269      * Lists up all entries.
270      * @return a {@link java.util.Set} of {@link java.util.Map.Entry} of entries
271      */
272     public Set<Entry<K,List<V>>> entrySet() {
273         return store.entrySet();
274     }
275 
276     /**
277      * @return the map as "key=value1,key=value2,...."
278      */
279     public String toCommaSeparatedString() {
280         StringBuilder buf = new StringBuilder();
281         for (Entry<K,List<V>> e : entrySet()) {
282             for (V v : e.getValue()) {
283                 if(buf.length()>0)  buf.append(',');
284                 buf.append(e.getKey()).append('=').append(v);
285             }
286         }
287         return buf.toString();
288     }

290     /**
291      * Creates a copy of the map that contains the exact same key and value set.
292      * Keys and values won't cloned.
293      */
294     public MultiMap<K,V> clone() {
295         return new MultiMap<K,V>(this);
296     }

298     /**
299      * Returns the size of the map
300      * @return integer or 0 if the map is empty
301      */
302     public int size() {
303         return store.size();
304     }

306     private static final MultiMap EMPTY = new MultiMap(Collections.emptyMap());

308     /**
309      * Gets the singleton read-only empty multi-map.

311      * @return an empty map

315         return EMPTY;
