64  /**
65   * Responsible for generating the collection of inhabitants, decoupling from
66   * implementation detail for the caller.
67   * <p/>
68   * The caller is expected to continually build up the InhabitantsGenerator
69   * context by calling add*(), followed by calling getModelInhabitants() to
70   * obtain the progenitors of the inhabitants.
71   * <p/>
72   * There are two ways to close this instance, either through {@link #getContext()}
73   * or through calling {@link #close()} directly.
74   *
75   * @author Jerome Dochez
76   * @author Jeff Trent
77   * @since 3.1
78   */
79  public abstract class InhabitantsParsingContextGenerator implements Closeable {
80  
81      private static final Logger logger = Logger
82              .getLogger(InhabitantsParsingContextGenerator.class.getName());
83  
84      private final Parser parser;
85      private final ParsingContext context;
86  
87      private final LinkedHashMap<String, InhabitantsScanner> metaInfScanners = 
88              new LinkedHashMap<String, InhabitantsScanner>();
89  
90      /**
91       * Factory for the {@link InhabitantsParsingContextGenerator}
92       *
93       * @param h habitat not currently used; reserved for future use
94       * @return an empty context InhabitantsGenerator
95       */
96      public static InhabitantsParsingContextGenerator create(Habitat h) {
97          return new InhabitantsParsingContextGenerator(null, null) {};
98      }
99  
100     /**
101      * Factory for the {@link InhabitantsParsingContextGenerator}
102      *
103      * @param h habitat not currently used; reserved for future use
104      * @param es the executor to use for any async processing (e.g., parsing)
105      * @param inhabitantsClassPath the fully qualified classpath in order to resolve class-model
106      * 
107      * @return an empty context InhabitantsGenerator
108      */
109     public static InhabitantsParsingContextGenerator create(Habitat h, 
110             ExecutorService es,
111             ClassPath inhabitantsClassPath) {
112         return new InhabitantsParsingContextGenerator(es, inhabitantsClassPath) {};
113     }
114     
115     protected InhabitantsParsingContextGenerator(ExecutorService es,
116             final ClassPath inhabitantsClassPath) {
117         // setup the parser
118         ParsingContext.Builder builder = new ParsingContext.Builder();
119         final Set<String> annotations = new HashSet<String>();
120         annotations.add(Contract.class.getCanonicalName());
121         annotations.add(Service.class.getCanonicalName());
122         annotations.add("org.jvnet.hk2.config.Configured");
123 
124         builder.config(new ParsingConfig() {
125             final Set<String> empty = Collections.emptySet();
126 
127             public Set<String> getAnnotationsOfInterest() {
128                 return empty;
129             }
130 
131             public Set<String> getTypesOfInterest() {
132                 return annotations;
133             }
134 
135         });
136 
137         // optionally provide an executor
138         builder.executorService(es);
139         
140         // optionally provide an inhabitants locator
141         if (null != inhabitantsClassPath) {
142           builder.locator(new Locator(inhabitantsClassPath));
143         }
144         
145         context = builder.build();
146         parser = new Parser(context);
147     }
148 
149     /**
150      * Add the collection of files to the current InhabitantsGenerator context.
151      *
152      * @param files the files to parse.
153      * @throws IOException
154      */
155     public void parse(Collection<File> files) throws IOException {
156         for (File file : files) {
157             parse(file);
158         }
159     }
160 
161     /**
162      * Retrieves the parsing context that can be used for model generation elsewhere.
163      * <b>Note that this can be called at most once and then this instance is implicitly
164      * closed.</b>
165      *
166      * @return the parsing context given the code sources provided
167      */
168     public ParsingContext getContext() {
169         try {
170             parser.awaitTermination();
171         } catch (InterruptedException e) {
172             close();
173             throw new RuntimeException(e);
174         }
175         return context;
176     }
177 
178     /**
179      * @return the collection of {@link InhabitantsScanner}s being maintained
180      */
181     public Collection<InhabitantsScanner> getInhabitantsScanners() {
182         return Collections.unmodifiableCollection(metaInfScanners.values());
183     }
184 
185     protected void addInhabitantsScanner(String name, InhabitantsScanner is) {
186         synchronized (metaInfScanners) {
187             if (!metaInfScanners.containsKey(name)) {
188                 metaInfScanners.put(name, is);
189             }
190         }
191     }
192 
193     /**
194      * Eventually we can perform optimizations here instead of a pass-thru to parseAlways.
195      */
196     public void parse(final File f) throws IOException {
197         parseAlways(parser, f);
198     }
199 
200     protected void parseAlways(Parser parser, final File f) throws IOException {
201       try {
202         parser.parse(f, new Runnable() {
203             public void run() {
204                 logger.log(Level.FINER, "Finished introspecting {0}", f.getName());
205             }
206         });
207       } catch (IOException e) {
208         logger.log(Level.FINE, "problem during parsing - closing prematurely", e);
209         close();
210       }
211     }
212 
213     @Override
214     public void close() {
215       if (null != parser) {
216         parser.close();
217       }
218     }

221     private static class Locator implements ResourceLocator {
222       private final ClassLoader resourceLoader;
223       
224       public Locator(ClassPath inhabitantsClassPath) {
225         try {
226           resourceLoader = new URLClassLoader(inhabitantsClassPath.getRawURLs(), null);
227           
228           if (logger.isLoggable(Level.FINE)) {
229             logger.log(Level.FINER, "resourceLoader is {0}", Arrays.asList(inhabitantsClassPath.getRawURLs()).toString());
230           }
231         } catch (IOException e) {
232           throw new RuntimeException(e);
233         }
234       }
235 
236       @Override
237       public URL getResource(String name) {
238         if (name.startsWith("java/")) {
239           logger.log(Level.FINE, "skipping {0}", name);
240           return null;  // wasteful to parse these
241         }
242         
243         URL resource = resourceLoader.getResource(name);
244         logger.log(Level.FINE, "resource {0} resolved to {1}", new Object[] {name, resource});
245         
246         return resource;
247       }
248     }
