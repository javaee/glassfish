119     Habitat(ExecutorService exec, Boolean concurrency_controls) {
120         this.concurrencyControls = 
121             (null == concurrency_controls) 
122                 ? CONCURRENCY_CONTROLS_DEFAULT : concurrency_controls;
123         this.byContract = new MultiMap<String,NamedInhabitant>(this.concurrencyControls);
124         this.byType = new MultiMap<String,Inhabitant>(this.concurrencyControls);
125         this.singletonScope = new ScopeInstance("singleton", new HashMap());
126 
127         if (null == exec) {
128           if (ASYNC_EXECUTOR) {
129             exec = Executors.newCachedThreadPool(new ThreadFactory() {
130               @Override
131               public Thread newThread(Runnable runnable) {
132                 Thread t = Executors.defaultThreadFactory().newThread(runnable);
133                 t.setDaemon(true);
134                 return t;
135               }
136             });
137           } else {
138             exec = new SameThreadExecutor();
139           }
140         }
141         this.exec = exec;
142         
143         // make the listeners available very early in lifecycle
144         addHabitatListener(new SelfListener());
145 
146         // add the set of injection resolvers
147         add(new ExistingSingletonInhabitant<InjectionResolver>(InjectionResolver.class,
148                 new InjectInjectionResolver(this)));
149         add(new ExistingSingletonInhabitant<InjectionResolver>(InjectionResolver.class,
150                 new LeadInjectionResolver(this)));
151         
152         // make the habitat itself available
153         add(new ExistingSingletonInhabitant<Habitat>(Habitat.class,this));
154 
155         add(new ExistingSingletonInhabitant<CompanionSeed.Registerer>(CompanionSeed.Registerer.class,
156                 new CompanionSeed.Registerer(this)));
157         add(new ExistingSingletonInhabitant<CageBuilder.Registerer>(CageBuilder.Registerer.class,
158                 new CageBuilder.Registerer(this)));
159 
160         // the default RunLevelService
161         DefaultRunLevelService rls = new DefaultRunLevelService(this);
162         ExistingSingletonInhabitant<RunLevelService> rlsI = 
163           new ExistingSingletonInhabitant<RunLevelService>(RunLevelService.class, rls);
164         add(rlsI);
165         addIndex(rlsI, RunLevelService.class.getName(), "default");
166     }

197      * @param typeNames

202     public void addHabitatListener(HabitatListener listener, String... typeNames) {
203       if (null == typeNames || 0 == typeNames.length) throw new IllegalArgumentException();
204       addHabitatListener(listener, new HashSet<String>(Arrays.asList(typeNames)));

207     protected void addHabitatListener(HabitatListener listener, Set<String> typeNames) {
208         ExistingSingletonInhabitant<HabitatListener> inhabitant = 
209           new ExistingSingletonInhabitant<HabitatListener>(HabitatListener.class,
210               listener, metaData(typeNames));
211         add(inhabitant);
212   
213         // track listeners by type - support of filtered notification
214         for (String contract : typeNames) {
215           ListenersByTypeInhabitant sameListeners; 
216           synchronized (byContract) {
217               sameListeners = (ListenersByTypeInhabitant)
218                       getInhabitantByContract(ListenersByTypeInhabitant.class.getName(),
219                       contract);
220               if (null == sameListeners) {
221                   sameListeners = new ListenersByTypeInhabitant(contract);
222                   addIndex(sameListeners, ListenersByTypeInhabitant.class.getName(),
223                       contract,
224                       false); // don't send notifications for this type (no particular reason really)
225               }
226           }
227           sameListeners.add(listener);
228         }
229     }

231     /***
232      * Remove a habitat listener.
233      * 
234      * @param listener
235      *      The habitat Listener to be removed
236      * @param typeNames
237      *      The contracts to filter on
238      * 
239      * @return true; if the listener was indeed removed
240      * 
241      * @since 3.1
242      */
243     public boolean removeHabitatListener(HabitatListener listener) {
244       List<Inhabitant> list = byType._get(HabitatListener.class.getName());
245       List<Inhabitant> releaseList = new ArrayList<Inhabitant>();
246       Iterator<Inhabitant> iter = list.iterator();
247       while (iter.hasNext()) {
248           Inhabitant existing = iter.next();
249           if (existing.get() == listener) {
250               releaseList.add(existing);
251               if (concurrencyControls) {
252                   list.remove(existing);
253               } else {
254                   iter.remove();
255               }
256           }
257       }
258 
259       // need to release the per-listener-contract entries too
260       for (Inhabitant released : releaseList) {
261           MultiMap<String, String> metadata = released.metadata();
262           if (null != metadata) {
263               List<String> filters = metadata.get(Constants.OBJECTCLASS);
264               for (String contract : filters) {
265                   ListenersByTypeInhabitant sameListeners; 
266                   synchronized (byContract) {
267                       sameListeners = (ListenersByTypeInhabitant)
268                               getInhabitantByContract(ListenersByTypeInhabitant.class.getName(),
269                               contract);
270                   }
271                   sameListeners.remove(listener);
272               }                    
273           }
274           
275           released.release();
276           notify(released, EventType.INHABITANT_REMOVED, null, released);
277       }
278       
279       return !releaseList.isEmpty();
280     }

282     /***
283      * Convert a set of type names into OBJECTCLASS keys within metadata
284      */
285     private MultiMap<String, String> metaData(Set<String> typeNames) {
286         if (null == typeNames) {
287           return null;
288         }
289         
290         MultiMap<String, String> metadata = new MultiMap<String, String>();
291         for (String typeName : typeNames) {
292           metadata.add(Constants.OBJECTCLASS, typeName);
293         }
294         
295         return metadata;
296     }

355     /***
356      * Removes all inhabitants for a particular type
357      *
358      * @param type of the component
359      * 
360      * @return true if any inhabitants were removed
361      */
362     public boolean removeAllByType(Class<?> type) {
363         boolean removed = false;
364         String name = type.getName();
365 
366         // remove all existing inhabitants by type
367         List<Inhabitant> list = new ArrayList<Inhabitant>(byType.get(name));
368         for (Inhabitant existing : list) {
369             removed |= remove(existing);
370         }
371         
372         return removed;
373     }

375     /***
376      * Adds a new inhabitant.
377      *
378      * <p>
379      * See {@link Inhabitants} for typical ways to create {@link Inhabitant}s.
380      */
381     public void add(final Inhabitant<?> i) {
382         String name = i.typeName();
383         byType.add(name,i);
384 
385         // for each companion, create an inhabitant that goes with the lead and hook them up
386         List<Inhabitant> companions=null;
387         for(Inhabitant<?> c : getInhabitantsByAnnotation(CompanionSeed.class,name)) {
388             if(companions==null)
389                 companions = new ArrayList<Inhabitant>();
390             companions.add(createCompanion(this,i,c));
391         }
392         i.setCompanions(companions);
393 
394         String cageBuilderName = i.metadata().getOne(CAGE_BUILDER_KEY);
395         if(cageBuilderName!=null) {
396             Inhabitant cageBuilder = byType.getOne(cageBuilderName);
397             if(cageBuilder!=null)
398                 ((CageBuilder)cageBuilder.get()).onEntered(i);
399             // if cageBuilder==null, we can't cage this component now, but
400             // we'll do that when cageBuilder comes into the habitat.
401             // that happens because every CageBuilder implementations are caged by
402             // CageBuilder.Registerer.
403         }
404         
405         notify(i, EventType.INHABITANT_ADDED, null, null);
406     }

435         String name = inhabitant.typeName();
436         if (byType.remove(name, inhabitant)) {
437             notify(inhabitant, EventType.INHABITANT_REMOVED, null, null);
438             inhabitant.release();
439             return true;
440         }
441         return false;
442     }

444     /***
445      * Removes a NamedInhabitant for a specific contract
446      *
447      * @param index contract name
448      * @param name instance name
449      * @return true if the removal was successful
450      */
451     public boolean removeIndex(String index, String name) {
452         boolean removed = false;
453         if (byContract.containsKey(index)) {
454             List<NamedInhabitant> contracted = byContract._get(index);
455             Iterator<NamedInhabitant> iter = contracted.iterator();
456             while (iter.hasNext()) {
457                 NamedInhabitant i = iter.next();
458                 
459                 if ((i.name == null && name == null) ||
460                         (i.name != null && i.name.equals(name))) {
461                     if (concurrencyControls) {
462                         removed = contracted.remove(i);
463                         assert(removed);
464                     } else {
465                         iter.remove();
466                     }
467                     removed = true;
468                     notify(i.inhabitant, EventType.INHABITANT_INDEX_REMOVED,
469                         index, name, null, null);
470 
471                     // remember to remove the components stored under its type
472                     remove(i.inhabitant);
473                 }
474             }
475         }
476 
477         return removed;
478     }

480     /***
481      *  Removes a Contracted service
482      *
483      * @param index the contract name
484      * @param serviceOrInhabitant the service instance, or an Inhabitant instance
485      */
486     public boolean removeIndex(String index, Object serviceOrInhabitant) {
487         boolean removed = false;
488         if (byContract.containsKey(index)) {
489              List<NamedInhabitant> contracted = byContract._get(index);
490              Iterator<NamedInhabitant> iter = contracted.iterator();
491              while (iter.hasNext()) {
492                  NamedInhabitant i = iter.next();
493                  if (matches(i.inhabitant, serviceOrInhabitant)) {
494                      if (concurrencyControls) {
495                          removed = contracted.remove(i);
496                          assert(removed);
497                      } else {
498                          iter.remove();
499                      }
500                      removed = true;
501                      notify(i.inhabitant, EventType.INHABITANT_INDEX_REMOVED,
502                          index, null, service(serviceOrInhabitant), null);
503 
504                      // remember to remove the components stored under its type
505                      remove(i.inhabitant);
506                  }
507              }
508          }
509         
510          return removed;
511     }

513     protected boolean matches(Inhabitant<?> inhabitant, Object serviceOrInhabitant) {
514       boolean matches;
515       if (serviceOrInhabitant instanceof Inhabitant) {
516         matches = (serviceOrInhabitant == inhabitant);
517       } else {
518         // call to isInstantiated is necessary to avoid loading of services that are not yet loaded.
519         matches = inhabitant.isInstantiated() && inhabitant.get()==serviceOrInhabitant;
520       }
521       return matches;
522     }
523     
524     protected Object service(Object serviceOrInhabitant) {
525       return (serviceOrInhabitant instanceof Inhabitant) 
526         ? ((Inhabitant)serviceOrInhabitant).get() : serviceOrInhabitant;
527     }

534      * Trigger a notification that an inhabitant has changed.
535      * 
536      * @param inhabitant the inhabitant that has changed
537      * @param contracts the contracts associated with the inhabitant
538      */
539     public void notifyInhabitantChanged(Inhabitant<?> inhabitant,
540           String... contracts) {
541       if (null == contracts || 0 == contracts.length) {
542         // generic inhabitant modification (only general listeners will get notification)
543         notify(inhabitant, EventType.INHABITANT_MODIFIED, null, null);
544       } else {
545         // all scoped-listeners will get modification
546         for (String contract : contracts) {
547           notify(inhabitant, EventType.INHABITANT_MODIFIED, contract, null);
548         }
549       }
550     }

592     protected void notify(final NotifyCall innerCall,
593         final Inhabitant<?> inhabitant,
594         final EventType event,
595         final String index,
596         final Inhabitant<HabitatListener> extraListenerToBeNotified) {
597       // do scoped listeners first
598       if (null != index) {
599         doNotify(innerCall, inhabitant, event, index, null);
600       }
601       
602       // do general listeners next
603       doNotify(innerCall, inhabitant, event, DEFAULT_NAME, extraListenerToBeNotified);
604     }
605       
606     private void doNotify(final NotifyCall innerCall,
607         final Inhabitant<?> inhabitant,
608         final EventType event,
609         final String index,
610         final Inhabitant<HabitatListener> extraListenerToBeNotified) {
611       final ListenersByTypeInhabitant sameListeners = 
612           (ListenersByTypeInhabitant)getInhabitantByContract(
613               ListenersByTypeInhabitant.class.getName(),
614               index);
615       if (null != sameListeners && !sameListeners.listeners.isEmpty()) {
616         exec.execute(new Runnable() {
617           public void run() {
618             Iterator<HabitatListener> iter = sameListeners.listeners.iterator();
619             while (iter.hasNext()) {
620               Object entry = iter.next();
621               HabitatListener listener = 
622                 (entry instanceof HabitatListener) ? 
623                     (HabitatListener)entry :
624                     (HabitatListener)((Inhabitant)entry).get();
625               try {
626                 boolean keepMe = innerCall.inhabitantChanged(listener);
627                 if (!keepMe) {
628                   removeHabitatListener(listener);
629                 }
630               } catch (Exception e){
631                 // don't percolate the exception since it may negatively impact processing
632                 Logger.getLogger(Habitat.class.getName()).
633                   log(Level.WARNING, "exception caught from listener: ", e);
634               }
635             }
636             
637             // we take the extraListenerToBeNotified because
638             //  (a) we want to have all notifications in the executor, and
639             //  (b) it might trigger an exception that we want to trap
640             if (null != extraListenerToBeNotified) {
641               extraListenerToBeNotified.get().inhabitantChanged(event, Habitat.this, extraListenerToBeNotified);
642             }
643           }
644         });
645       }
646     }

649     /***
650      * Checks if the given type is a contract interface that has some implementations in this {@link Habitat}.
651      *
652      * <p>
653      * There are two ways for a type to be marked as a contract.
654      * Either it has {@link Contract}, or it's marked by {@link ContractProvided} from the implementation.
655      *
656      * <p>
657      * Note that just having {@link Contract} is not enough to make this method return true.
658      * It can still return false if the contract has no implementation in this habitat.
659      *
660      * <p>
661      * This method is useful during the injection to determine what lookup to perform,
