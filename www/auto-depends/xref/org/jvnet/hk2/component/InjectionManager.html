105     /***
106       * Initializes the component by performing injection.
107       *
108       * @param component component instance to inject
109       * @param onBehalfOf the inhabitant to do injection on behalf of
110       * @param type component class
111       * @param targets the injection resolvers to resolve all injection points
112       * @throws ComponentException
113       *      if injection failed for some reason.
114       */
115     public void inject(Object component,
116                 Inhabitant<?> onBehalfOf,
117                 Class type,
118                 InjectionResolver... targets) {
119 
120         try {
121             assert component!=null;
122 
123             // TODO: faster implementation needed.
124 
125             Class currentClass = type;
126             while (currentClass!=null && !currentClass.equals(Object.class)) {
127                 // get the list of the instances variable
128                 for (Field field : currentClass.getDeclaredFields()) {
129 
130                     for (InjectionResolver target : targets) {
131                         Annotation inject = field.getAnnotation(target.type);
132                         if (inject == null)     continue;
133 
134                         Class fieldType = field.getType();
135                         try {
136                             Object value = target.getValue(component, onBehalfOf, field, fieldType);
137                             if (value != null) {
138                                 field.setAccessible(true);
139                                 field.set(component, value);
140                                 Injectable injectable;
141                                 try {
142                                     injectable = Injectable.class.cast(value);
143                                     if (injectable!=null) {
144                                         injectable.injectedInto(component);
145                                     }
146                                 } catch (Exception e) {
147                                 }
148 
149                             } else {
150                                 if(!target.isOptional(field, inject)) {
151                                     Logger.getAnonymousLogger().info("Cannot inject " + field + " into component " + component);
152                                     throw new UnsatisfiedDependencyException(field);
153                                 }
154                             }
155                         } catch (ComponentException e) {
156                             if (!target.isOptional(field, inject)) {
157                                 throw new UnsatisfiedDependencyException(field,e);
158                             }
159                         } catch (IllegalAccessException e) {
160                             throw new ComponentException("Injection failed on " + field.toGenericString(), e);
161                         } catch (RuntimeException e) {
162                             throw new ComponentException("Injection failed on " + field.toGenericString(), e);
163                         }
164                     }
165                 }
166                 for (Method method : currentClass.getDeclaredMethods()) {
167 
168                     for (InjectionResolver target : targets) {
169 
170                         Annotation inject = method.getAnnotation(target.type);
171                         if (inject == null)     continue;
172 
173                         Method setter = target.getSetterMethod(method, inject);
174 
175                         if (setter.getReturnType() != void.class) {
176                             if (Collection.class.isAssignableFrom(setter.getReturnType())) {
177                                 injectCollection(component, setter, 
178                                     target.getValue(component, onBehalfOf, method, setter.getReturnType()));
179                                 continue;
180                             }
181                             throw new ComponentException("Injection failed on %s : setter method is not declared with a void return type",method.toGenericString());
182                         }
183 
184                         Class<?>[] paramTypes = setter.getParameterTypes();
185 
186                         if (paramTypes.length > 1) {
187                             throw new ComponentException("injection failed on %s : setter method takes more than 1 parameter",method.toGenericString());
188                         }
189                         if (paramTypes.length == 0) {
190                             throw new ComponentException("injection failed on %s : setter method does not take a parameter",method.toGenericString());
191                         }
192 
193                         try {
194                             Object value = target.getValue(component, onBehalfOf, method, paramTypes[0]);
195                             if (value != null) {
196                                 setter.setAccessible(true);
197                                 setter.invoke(component, value);
198                                 try {
199                                     Injectable injectable = Injectable.class.cast(value);
200                                     if (injectable!=null) {
201                                         injectable.injectedInto(component);
202                                     }
203                                 } catch (Exception e) {
204                                 }
205                             } else {
206                                 if (!target.isOptional(method, inject))
207                                     throw new UnsatisfiedDependencyException(method);
208                             }
209                         } catch (IllegalAccessException e) {
210                             throw new ComponentException("Injection failed on " + setter.toGenericString(), e);
211                         } catch (InvocationTargetException e) {
212                             throw new ComponentException("Injection failed on " + setter.toGenericString(), e);
213                         } catch (RuntimeException e) {
214                             throw new ComponentException("Injection failed on " + setter.toGenericString(), e);
215                         }
216                     }
217                 }
218                 currentClass = currentClass.getSuperclass();
219             }
220         } catch (LinkageError e) {
221             // reflection could trigger additional classloading and resolution, so it can cause linkage error.
222             // report more information to assist diagnosis.
223             // can't trust component.toString() as the object could be in an inconsistent state.
224             Class<?> cls = type;
225             LinkageError x = new LinkageError("Failed to inject " + cls +" from "+cls.getClassLoader());
226             x.initCause(e);
227             throw x;
228         }
229

231     }

233     private void injectCollection(Object component, Method method, Object value) {
234         if (value==null) {
235             return;
236         }
237         Collection c = Collection.class.cast(value);
238         Collection target = null;
239         try {
240             target = Collection.class.cast(method.invoke(component));
241         } catch (IllegalAccessException e) {
242             e.printStackTrace();  //To change body of catch statement use File | Settings | File Templates.
243             return;
244         } catch (InvocationTargetException e) {
245             e.printStackTrace();  //To change body of catch statement use File | Settings | File Templates.
246             return;
247         }
248         target.addAll(c);
249     }
