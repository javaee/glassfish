38  package org.jvnet.hk2.component;
39  
40  import java.lang.reflect.Field;
41  import java.lang.reflect.Method;
42  import java.lang.reflect.InvocationTargetException;
43  import java.lang.reflect.AnnotatedElement;
44  import java.lang.annotation.Annotation;
45  import java.util.logging.Logger;
46  
47  /***
48   * InjectionManager is responsible for injecting resources into a component.
49   * Resources tagged for injection are identified via an annotation which is
50   * a parameter type of this class.
51   *
52   * @author Jerome Dochez
53   */
54  public abstract class InjectionManager<T extends Annotation> {
55  
56      /***
57       * Initializes the component by performing injection.
58       *
59       * @param component component instance to inject
60       * @throws ComponentException
61       *      if injection failed for some reason.
62       */
63      public void inject(Object component, Class<T> type) throws ComponentException {
64          try {
65              assert component!=null;
66  
67              // TODO: faster implementation needed.
68  
69              Class currentClass = component.getClass();
70              while (!currentClass.equals(Object.class)) {
71                  // get the list of the instances variable
72                  for (Field field : currentClass.getDeclaredFields()) {
73  
74                      T inject = field.getAnnotation(type);
75                      if (inject == null)     continue;
76  
77                      Class fieldType = field.getType();
78                      try {
79                          Object value = getValue(component, field, fieldType);
80                          if (value != null) {
81                              field.setAccessible(true);
82                              field.set(component, value);
83                              Injectable injectable;
84                              try {
85                                  injectable = Injectable.class.cast(value);
86                                  if (injectable!=null) {
87                                      injectable.injectedInto(component);
88                                  }
89                              } catch (Exception e) {
90                              }
91  
92                          } else {
93                              if(!isOptional(inject)) {
94                                  Logger.getAnonymousLogger().info("Cannot inject " + field + " in component" + component);   
95                                  throw new UnsatisfiedDepedencyException(field);
96                              }
97                          }
98                      } catch (ComponentException e) {
99                          if (!isOptional(inject)) {
100                             throw new UnsatisfiedDepedencyException(field,e);
101                         }
102                     } catch (IllegalAccessException e) {
103                         throw new ComponentException("Injection failed on " + field.toGenericString(), e);
104                     } catch (RuntimeException e) {
105                         throw new ComponentException("Injection failed on " + field.toGenericString(), e);                        
106                     }
107                 }
108                 for (Method method : currentClass.getDeclaredMethods()) {
109                     T inject = method.getAnnotation(type);
110                     if (inject == null)     continue;
111 
112                     if (method.getReturnType() != void.class) {
113                         throw new ComponentException("Injection failed on %s : setter method is not declared with a void return type",method.toGenericString());
114                     }
115 
116                     Class<?>[] paramTypes = method.getParameterTypes();
117 
118                     if (paramTypes.length > 1) {
119                         throw new ComponentException("injection failed on %s : setter method takes more than 1 parameter",method.toGenericString());
120                     }
121                     if (paramTypes.length == 0) {
122                         throw new ComponentException("injection failed on %s : setter method does not take a parameter",method.toGenericString());
123                     }
124 
125                     try {
126                         Object value = getValue(component, method, paramTypes[0]);
127                         if (value != null) {
128                             method.setAccessible(true);
129                             method.invoke(component, value);
130                             try {
131                                 Injectable injectable = Injectable.class.cast(value);
132                                 if (injectable!=null) {
133                                     injectable.injectedInto(component);
134                                 }
135                             } catch (Exception e) {
136                             }
137                         } else {
138                             if (!isOptional(inject))
139                                 throw new UnsatisfiedDepedencyException(method);
140                         }
141                     } catch (IllegalAccessException e) {
142                         throw new ComponentException("Injection failed on " + method.toGenericString(), e);
143                     } catch (InvocationTargetException e) {
144                         throw new ComponentException("Injection failed on " + method.toGenericString(), e);
145                     } catch (RuntimeException e) {
146                         throw new ComponentException("Injection failed on " + method.toGenericString(), e);
147                     }
148                 }
149                 currentClass = currentClass.getSuperclass();
150             }
151         } catch (LinkageError e) {
152             // reflection could trigger additional classloading and resolution, so it can cause linkage error.
153             // report more information to assist diagnosis.
154             // can't trust component.toString() as the object could be in an inconsistent state.
155             Class<?> cls = component.getClass();
156             LinkageError x = new LinkageError("Failed to inject " + cls +" from "+cls.getClassLoader());
157             x.initCause(e);
158             throw x;
159         }
160     }
161 
162     protected boolean isOptional(T annotation) {
163         return false;
164     }
165 
166     protected abstract Object getValue(Object component, AnnotatedElement annotated, Class type) throws ComponentException;
167 }
