39  import com.sun.hk2.component.Holder;
40  import com.sun.hk2.component.ScopeInstance;
41  import org.jvnet.hk2.annotations.Service;

43  import java.util.Collection;

45  /***
46   * Represents a component in the world of {@link Habitat}.
47   *
48   * <p>
49   * {@link Inhabitant} extends from {@link Holder}, as one of its
50   * purposes is to encapsulate how we obtain an instance of a component.
51   * On topf of that, {@link Inhabitant} enhances {@link Holder} by
52   * adding more metadata that {@link Habitat} uses for finding
53   * components and hooking them up together.
54   *
55   * <p>
56   * All the methods exept {@link #get()} are immutable, meaning
57   * they never change the value they return.
58   *
59   * <p>
60   * See {@link Inhabitants} for several factory methods for typical
61   * {@link Inhabitant} constructions.
62   * 
63   * @author Kohsuke Kawaguchi
64   * @see Inhabitants
65   */
66  public interface Inhabitant<T> extends Holder<T> {
67      /***
68       * The short-cut for {@code type().getName()}
69       * but this allows us to defer loading the actual types.
70       */
71      String typeName();
72  
73      /***
74       * Type of the inhabitant.
75       *
76       * <p>
77       * The only binding contract that needs to be honored is that the {@link #get()}
78       * method returns an instance assignable to this type. That is,
79       * {@code get().getClass()==type()} doesn't necessarily have to hold,
80       * but {@code type().isInstance(get())} must.
81       *
82       * <p>
83       * This is particularly true when {@link Factory} is involved, as in such
84       * case HK2 has no way of knowing the actual type.
85       *
86       * That said, this method is not designed for the semantics of
87       * contract/implementation split --- implementations of a contract
88       * should return the concrete type from this method, and use
89       * {@link Habitat#addIndex(Inhabitant, String, String) habitat index}
90       * to support look-up by contract. 
91       *
92       * @return
93       *      Always non-null, same value.
94       */
95      Class<T> type();
96  
97      /***
98       * Returns the instance of this inhabitant.
99       *
100      * <p>
101      * Some {@link Inhabitant}s return the same instance for multiple
102      * invocations (AKA singleton), but
103      * the method may return different instances to invocations from different
104      * context (AKA scope.) The extreme case is where the each invocation
105      * returns a different object.
106      */
107     T get();
108 
109     /***
110      * Returns the instance of this inhabitant.
111      *
112      * <p>
113      * <b>THIS METHOD SHOULD BE ONLY USED BY HK2 IMPLEMENTATION</b>.
114      *
115      * <p>
116      * {@link Inhabitant}s are often used with the decorator pattern
117      * (see {@link com.sun.hk2.component.AbstractWombInhabitantImpl} for example), yet during
118      * the object initializtion inside the {@link #get()} method, we often
119      * need the reference to the outer-most {@link Inhabitant} registered to
120      * the {@link Habitat} (for example so that we can request the injection
121      * of {link Inhabita} that represents itself, or to inject companions.)
122      *
123      * <p>
124      * So this overloaded version of the get method takes the outer-most
125      * {@link Inhabitant}. This method is only invoked from within HK2
126      * where the decorator pattern is used.
127      */
128     // TODO: this and the lead/companions method make you wonder whether we should
129     // define Inhabitant as an abstract class.
130     T get(Inhabitant onBehalfOf);
131 
132     /***
133      * Gets the metadata associated with this inhabitant.
134      *
135      * <p>
136      * This data is usually used by a sub-system of HK2, and not really meant to
137      * be used by applications. (At least for now.)
138      * The main benefit of metadata is that it's available right away
139      * as soon as the {@link Habitat} is properly initialized, even before
140      * component classes are loaded. In contrast, accessing annotations would require
141      * classes to be loaded and resolved.
142      *
143      * @return
144      *      can be empty but never null. The values are read-only.
145      * @see Service#metadata() 
146      */
147     MultiMap<String,String> metadata();
148 
149     /***
150      * Obtains the serialized metadata.
151      *
152      * <p>
153      * This method is a wrapper around {@link #metadata()} and useful for
154      * defining a highly structured metadata that doesn't easily fit
155      * a simple string representation.
156      *
157      * <p>
158      * The implementation of this method is to obtain the value associated with
159      * this key as {@code metadata().getOne(key)}, and if that exists, treat
160      * the value as base64-encoded binary, and deserializes it and returns the object.
161      *
162      * <p>
163      * The classes used in the serialization need to be available during the build time
164      * (normally during the HK2 compile mojo runs) so that the metadata can be
165      * serialized. The evolution of these classes need to be careful done, otherwise
166      * the deserialization of the metadata may fail unexpectedly.
167      *
168      * @throws Error
169      *      If the deserialization fails. This can be for example because of
170      *      the incompatible class change, or failure to resolve the classes.
171      *      Sine these problems can only happen in a critical situation,
172      *      this method throws unchecked error.
173      *      TODO: switch this to IOError when we can depend on JDK6.
174      *
175      * @return
176      *      the deserialized object.
177      */
178     <T> T getSerializedMetadata(Class<T> type, String key);
179 
180     /***
181      * Obtains the metadata serialized into String.
182      *
183      * <p>
184      * This is a convenient short-cut that does {@code getSerializedMetadata(type,type.getName())}
185      */
186     <T> T getSerializedMetadata(Class<T> type);
187 
188     /***
189      * Called to orderly shutdown {@link Habitat}.
190      * <p>
191      * The expected behavior is for objects to get its {@link PreDestroy}
192      * callback invoked, and its reference released. For singleton
193      * objects, this method is expected to dispose that object.
194      * <p>
195      * For scoped objects, those are released when {@link ScopeInstance#release()}
196      * is invoked.
197      */
198     void release();
199 
200     /***
201      * Returns true if the underlying object has been instantiated.
202      *
203      * @return true if the instance of this imhabitant has been created, false
204      * otherwise
205      */
206     boolean isInstantiated();
207 
208 //
209 // methods below here are more or less used for book-keeping purpose by Habitat,
210 // and implementations of Inhabitant should implement them just by using
211 // AbstractInhabitantImpl
212 //
213 
214     /***
215      * If this inhabitant is a companion to another inhabitant (called "lead"),
216      * This method returns that inhabitant. Otherwise null.
217      */
218     Inhabitant lead();
219 
220     /***
221      * Returns the companion inhabitants associated with this inhabitant.
222      *
223      * <p>
224      * This method works with the {@link #lead()} method in pairs, such
225      * that the following condition always holds:
226      *
227      * <pre>x.companions().contains(y) &lt;-> y.lead()==x</pre>
228      *
229      * @return
230      *      Can be empty but never null.
231      */
232     Collection<Inhabitant> companions();
233 
234     /***
235      * This method is only meant to be invoked by {@link Habitat}.
236      */
237     void setCompanions(Collection<Inhabitant> companions);
238 }
