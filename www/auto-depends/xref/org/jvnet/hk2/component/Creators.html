63      @SuppressWarnings({ "unchecked", "rawtypes" })
64      public static <T> Creator<T> create(Class<T> c, Habitat habitat, MultiMap<String,String> metadata) {
65          Factory f = c.getAnnotation(Factory.class);
66          if (f != null) {
67              return new FactoryCreator<T>(c,f.value(),habitat,metadata);
68          }
69  
70          Inhabitant factory = habitat.getInhabitantByAnnotation(FactoryFor.class, c.getName());
71          if (factory != null) {
72              return new FactoryCreator<T>(c,factory,habitat,metadata);
73          }
74  
75          Reference<Constructor<T>> defaultCtorRef = new Reference();
76          Reference<Constructor<T>> noArgCtorRef = new Reference();
77          qualifyingConstructors(c, null, getInjectionAnnotations(habitat), defaultCtorRef, noArgCtorRef);
78          if (defaultCtorRef.get() != null) {
79              return new InjectableParametizedConstructorCreator<T>(c, defaultCtorRef.get(), habitat, metadata);
80          }
81          if (noArgCtorRef.get() != null) {
82              return new ConstructorCreator<T>(c,habitat,metadata);
83          }
84  
85          return null;
86      }

88      /**
89       * Returns all currently available injection annotations
90       *
91       */
92      private static Set<Class<? extends Annotation>> getInjectionAnnotations(Habitat habitat) {
93  
94          Set<Class<? extends Annotation>> annotations = new HashSet<Class<? extends Annotation>>();
95          for (Inhabitant<? extends InjectionResolver> resolver : getAllInjectionResolvers(habitat)) {
96              annotations.add(resolver.get().type);
97          }
98          return annotations;
99      }

101     /**
102      * Returns all currently available injection annotations
103      * @param habitat the service registry
104      * @return the list of injection resolvers registered in this service registry in its parent.
105      *
106      */
107     public static List<Inhabitant<? extends InjectionResolver>> getAllInjectionResolvers(Habitat habitat) {
108 
109         List<Inhabitant<? extends InjectionResolver>> resolvers = new ArrayList<Inhabitant<? extends InjectionResolver>>();
110 
111         // we treat the @Inject injection resolver specially as the other injectors might be
112         // using default HK2 injection themselves.
113         Inhabitant<InjectionResolver> specialResolver = habitat.getInhabitant(InjectionResolver.class, "Inject");
114         specialResolver.get(); // force initialization
115         resolvers.add(specialResolver);
116         for (Inhabitant<? extends InjectionResolver> injectionResolver : habitat.getInhabitants(InjectionResolver.class)) {
117             if (injectionResolver!=specialResolver)
118                 resolvers.add(injectionResolver);
119         }
120 
121         if (habitat.getDefault()!=habitat) {
122             for (Provider<? extends InjectionResolver> injectionResolver : habitat.getDefault().forContract(InjectionResolver.class).all()) {
123                 resolvers.add((Inhabitant<? extends InjectionResolver>) injectionResolver);
124             }
125         }
126         return resolvers;
127     }

129     /**
130      * Returns all "eligible" constructors for injection. Eligibility is
131      * determined by:
132      * 
133      * (a) the no-arg constructor, or (b) @Inject annotated constructor.
134      * 
135      * </p>
136      * Note, however, that this does not verify that the arguments on the
137      * constructor are indeed injectable out of the habitat.
138      * 
139      * @param c
140      *      The class to find the constructors for
141      * @param defaultCtor
142      *      (optional) The "favored" default constructor reference; meaning that all arguments can be injected based on the contents of the habitat
143      * @param noArgCtor
144      *      (optional The no-arg constructor reference
145      * @return
146      *      The list of all eligible constructors
147      */
148     @SuppressWarnings("unchecked")
149     public static <T> void qualifyingConstructors(Class<T> c,
150             List<Constructor<?>> allList,
151             Set<Class<? extends Annotation>> injectionAnnotations,
152             Reference<Constructor<T>> defaultCtor,
153             Reference<Constructor<T>> noArgCtor) {
154         Constructor<T>[] ctors;
155         try {
156             ctors = (Constructor<T>[]) c.getDeclaredConstructors();
157         } catch(Exception e) {
158             ctors = (Constructor<T>[]) c.getConstructors();
159         }
160         
161         for (Constructor<T> ctor : ctors) {
162             if (0 == ctor.getParameterTypes().length) {
163                 if (null != allList) {
164                     allList.add(ctor);
165                 }
166                 if (null != noArgCtor) {
167                     noArgCtor.set(ctor);
168                 }
169             } else if (isInjectable(injectionAnnotations, ctor)) {
170                 if (null != allList) {
171                     allList.add(ctor);
172                 }
173                 if (null != defaultCtor && null == defaultCtor.get()) {
174                     defaultCtor.set(ctor);
175                 }
176             } else {
177                 boolean allInjectable = true;
178                 for (Annotation[] a : ctor.getParameterAnnotations()) {
179                     if (!isInject(injectionAnnotations, a)) {
180                         allInjectable = false;
181                         break;
182                     }
183                 }
184                 if (allInjectable) {
185                     if (null != allList) {
186                         allList.add(ctor);
187                     }
188                     if (null != defaultCtor && null == defaultCtor.get()) {
189                         defaultCtor.set(ctor);
190                     }
191                 }
192             }
193         }
194     }

196     private static boolean isInject(Set<Class<? extends Annotation>> injectionAnnotations, Annotation... a) {
197         for (Annotation ae : a) {
198             if (injectionAnnotations.contains(ae.annotationType()) ||
199                     ae.annotationType().equals(javax.inject.Inject.class)) {
200                 return true;
201             }
202         }
203         
204         return false;
205     }
206 
207     public static boolean isInjectable(Set<Class<? extends Annotation>> injectionAnnotations,AnnotatedElement ae) {
208         for (Class<? extends Annotation> injectionAnnotation : injectionAnnotations) {
209             if (ae.getAnnotation(injectionAnnotation)!=null) return true;
210         }
211         return (null != ae.getAnnotation(Inject.class) ||
212                 null != ae.getAnnotation(javax.inject.Inject.class));
213     }
