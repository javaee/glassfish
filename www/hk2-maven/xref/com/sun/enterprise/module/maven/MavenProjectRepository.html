49      public MavenProjectRepository(MavenProject project, ArtifactResolver artifactResolver, ArtifactRepository localRepository, ArtifactFactory artifactFactory) throws IOException {
50          super(project.getName(),project.getFile().toURI());
51          this.project = project;
52          this.artifactResolver = artifactResolver;
53          this.localRepository = localRepository;
54          this.artifactFactory = artifactFactory;
55  
56          // adds the ones that we already know of.
57          Artifact ma = project.getArtifact();
58          if(!project.getPackaging().equals("pom") && ma.getFile()==null) {
59              // allow people to run "hk2:run" without executing the compilie phase.
60              // so pick up the default
61              File outdir = new File(project.getBuild().getOutputDirectory());
62              if(!outdir.exists())
63                  logger.warning("No output directory "+outdir);
64              else
65                  ma.setFile(outdir);
66          }
67  
68          if(ma.getFile()!=null) {
69              // if the 'ma' is the distribution module, it won't have its own output.
70              artifacts.put(ma.getId(),ma);
71              if(ma.getFile().isDirectory()) {
72                  // if the main artifact is from target/classes, create META-INF.MF
73                  new Packager().writeManifest(project,ma.getFile());
74              }
75          }
76  
77          for (Artifact a : (Set<Artifact>) project.getArtifacts())
78              artifacts.put(a.getId(),a);
79      }

81      @Override
82      protected Map<String, ModuleDefinition> loadModuleDefs() throws IOException {
83          Map<String, ModuleDefinition> moduleDefs = new HashMap<String, ModuleDefinition>();
84  
85          logger.info("Loading modules list from "+project.getFile());
86  
87          buildModule(project.getArtifact(),moduleDefs);
88  
89          for (Artifact a : (List<Artifact>) project.getAttachedArtifacts()) {
90              buildModule(a, moduleDefs);
91          }
92  
93          for (Artifact a : (Set<Artifact>) project.getArtifacts()) {
94              buildModule(a, moduleDefs);
95          }
96  
97          if(moduleDefs.isEmpty())
98              throw new Error("No modules found");// should this error check be done by the caller of loadModuleDefs?
99  
100         return moduleDefs;
101     }

103     private void buildModule(Artifact a, Map<String, ModuleDefinition> moduleDefs) throws IOException {
104         File jarFile = a.getFile();
105         if(jarFile==null || (!jarFile.getName().endsWith(".jar") && !jarFile.isDirectory()))
106             // between the compile phase and the package phase, the main artifact is
107             // set to the target/classes. allow that to be used as a jar.
108             return;
109 
110         ModuleDefinition moduleDef = loadJar(jarFile);
111         if(moduleDef.getManifest().getMainAttributes().getValue(ManifestConstants.BUNDLE_NAME)==null)
112             // project.getArtifacts() pick up all the transitive dependencies,
113             // including to the normal jar files through modules.
114             // we don't want to create modules from those jar files, so we need to ignore non-modules.
115             return;
116 
117         if(logger.isLoggable(Level.CONFIG))
118             logger.config("Adding module "+a.getId()+" trail: "+a.getDependencyTrail());
119 
120         moduleDefs.put(moduleDef.getName(), moduleDef);
121     }

123     /*package*/ File resolveArtifact(String id) throws IOException {
124         Artifact artifact = artifacts.get(id);
125         if(artifact!=null)  return artifact.getFile();
126         
127         try {
128             Matcher m = ID_PATTERN.matcher(id);
129             if(!m.matches())
130                 throw new IllegalArgumentException("Wrong ID: "+id);
131             Artifact a = artifactFactory.createArtifactWithClassifier(
132                 m.group(1), m.group(2), m.group(5), m.group(3), m.group(4)
133             );
134 
135             artifactResolver.resolve(a, project.getRemoteArtifactRepositories(), localRepository );
136             artifacts.put(a.getId(),a);
137             return a.getFile();
138         } catch (ArtifactResolutionException e) {
139             throw new IOException2("Failed to resolve "+id,e);
140         } catch (ArtifactNotFoundException e) {
141             throw new IOException2("Failed to resolve "+id,e);
142         }
143     }
