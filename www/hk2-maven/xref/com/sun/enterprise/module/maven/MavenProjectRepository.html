92      public MavenProjectRepository(MavenProject project, ArtifactResolver artifactResolver, ArtifactRepository localRepository, ArtifactFactory artifactFactory) {
93          super(project.getName(),project.getFile().toURI());
94          this.project = project;
95          this.artifactResolver = artifactResolver;
96          this.localRepository = localRepository;
97          this.artifactFactory = artifactFactory;
98  
99          // adds the ones that we already know of.
100         Artifact ma = project.getArtifact();
101         if(ma.getFile()!=null) {
102             // if the 'ma' is the distribution module, it won't have its own output.
103             artifacts.put(ma.getId(),ma);
104         }
105 
106         for (Artifact a : (Set<Artifact>) project.getArtifacts())
107             artifacts.put(a.getId(),a);
108     }

110     /**
111      * When creating {@link MavenProjectRepository} from the current project (which is used
112      * to launch mvn), and if the compile phase has run yet, then the main artifact is
113      * still null.
114      *
115      * <p>
116      * However, it's often convenient to pick up the files that were left in the file system
117      * from the previous execution. This method checks this situation and updates {@link MavenProject}
118      * accordingly, so that it can be then passed to the constructor of {@link MavenProjectRepository}.
119      *
120      * <p>
121      * Think of this as a pre-processing phase to compensate for the lack of the compile phase
122      * invocation.
123      */
124     public static void prepareProject(MavenProject project) throws IOException {
125         Artifact ma = project.getArtifact();
126         if(!project.getPackaging().equals("pom") && ma.getFile()==null) {
127             File outdir = new File(project.getBuild().getOutputDirectory());
128             if(!outdir.exists())
129                 logger.warning("No output directory "+outdir);
130             else
131                 ma.setFile(outdir);
132         }
133 
134         if(ma.getFile()!=null) {
135             // if the 'ma' is the distribution module, it won't have its own output.
136             if(ma.getFile().isDirectory()) {
137                 // if the main artifact is from target/classes, create META-INF.MF
138                 new Packager().writeManifest(project,ma.getFile());
139             }
140         }
141     }

143     @Override
144     protected void loadModuleDefs(Map<ModuleId, ModuleDefinition> moduleDefs, List<URI> libraries) throws IOException {
145 
146         logger.info("Loading modules list from "+project.getFile());
147 
148         MavenModuleDefinition main = buildModule(project.getArtifact(), moduleDefs, libraries);
149         if(main!=null) {
150             // artifact from the main project, in case those are not compiled yet
151             main.addClasspath(new File(project.getBuild().getOutputDirectory()));
152             for (Resource res : (List<Resource>)project.getBuild().getResources())
153                 main.addClasspath(new File(res.getDirectory()));
154         }
155 
156         for (Artifact a : (List<Artifact>) project.getAttachedArtifacts()) {
157             buildModule(a, moduleDefs, libraries);
158         }
159 
160         for (Artifact a : (Set<Artifact>) project.getArtifacts()) {
161             buildModule(a, moduleDefs, libraries);
162         }
163 
164         if(moduleDefs.isEmpty()) {
165 //            throw new Error("No modules found");// should this error check be done by the caller of loadModuleDefs?
166             logger.warning("No modules found");            
167         }
168 
169     }

171     private MavenModuleDefinition buildModule(Artifact a, Map<ModuleId, ModuleDefinition> moduleDefs, List<URI> libraries)
172             throws IOException {
173 
174         File jarFile = a.getFile();
175         if(jarFile==null || (!jarFile.getName().endsWith(".jar") && !jarFile.isDirectory()))
176             // between the compile phase and the package phase, the main artifact is
177             // set to the target/classes. allow that to be used as a jar.
178             return null;
179 
180         MavenModuleDefinition moduleDef = loadJar(jarFile);
181         if(moduleDef.getManifest().getMainAttributes().getValue(ManifestConstants.BUNDLE_NAME)==null) {
182             // project.getArtifacts() pick up all the transitive dependencies,
183             // including to the normal jar files through modules.
184             libraries.add(jarFile.toURI());
185             return null;
186         }
187 
188         if(logger.isLoggable(Level.CONFIG))
189             logger.config("Adding module "+a.getId()+" trail: "+a.getDependencyTrail());
190 
191         moduleDefs.put(AbstractFactory.getInstance().createModuleId(moduleDef), moduleDef);
192         return moduleDef;
193     }

195     /*package*/ File resolveArtifact(String id) throws IOException {
196         Artifact artifact = artifacts.get(id);
197         if(artifact!=null)  return artifact.getFile();
198         
199         try {
200             Matcher m = ID_PATTERN.matcher(id);
201             if(!m.matches())
202                 throw new IllegalArgumentException("Wrong ID: "+id);
203             Artifact a = artifactFactory.createArtifactWithClassifier(
204                 m.group(1), m.group(2), m.group(5), m.group(3), m.group(4)
205             );
206 
207             artifactResolver.resolve(a, project.getRemoteArtifactRepositories(), localRepository );
208             artifacts.put(a.getId(),a);
209             return a.getFile();
210         } catch (ArtifactResolutionException e) {
211             throw new IOException2("Failed to resolve "+id,e);
212         } catch (ArtifactNotFoundException e) {
213             throw new IOException2("Failed to resolve "+id,e);
214         }
215     }

217     // the format is 'groupId:artifactId:type(:classifier)?:version
218     private static final String TOKEN = "([^:]+)";
219     private static final Pattern ID_PATTERN = Pattern.compile(MessageFormat.format("{0}:{0}:{0}(?:\\:{0})?:{0}",TOKEN));

223         return new MavenModuleDefinition(this,jar);
