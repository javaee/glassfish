24  /***
25   * Prepares manifest entries in {@link MavenArchiveConfiguration}.
26   *
27   * <p>
28   * This class is moved outside {@code PackageMojo} and doesn't rely
29   * on Mojo resource injection to allow other plugins to reuse
30   * this capability without hitting http://jira.codehaus.org/browse/MNG-2789.
31   *
32   * @author Kohsuke Kawaguchi

34  public class Packager {
35      public void writeManifest(MavenProject pom, File classesDirectory) throws IOException {
36          Manifest mf = new Manifest();
37          for( Map.Entry<String,String> e : configureManifet(pom,null,classesDirectory).entrySet()) {
38              mf.getMainAttributes().put(
39                  new Attributes.Name(e.getKey()),e.getValue());
40          }
41          mf.getMainAttributes().put(Attributes.Name.MANIFEST_VERSION,"1.0");
42          
43          File manifest = new File(classesDirectory, "META-INF/MANIFEST.MF");
44          manifest.getParentFile().mkdir();
45          FileOutputStream os = new FileOutputStream(manifest);
46          try {
47              mf.write(os);
48          } finally{
49              os.close();
50          }
51      }
52      public Map<String,String> configureManifet(MavenProject pom, MavenArchiveConfiguration archive, File classesDirectory) throws IOException {
53          Map<String,String> entries;
54          if(archive!=null)
55              entries = archive.getManifestEntries();
56          else
57              entries = new HashMap<String,String>();
58  
59          entries.put(ManifestConstants.BUNDLE_NAME, pom.getGroupId()+':'+pom.getArtifactId() );
60  
61          // check META-INF/services/xxx.ImportPolicy to fill in Import-Policy
62          configureImportPolicy(classesDirectory, entries, ImportPolicy.class, ManifestConstants.IMPORT_POLICY);
63          configureImportPolicy(classesDirectory, entries, LifecyclePolicy.class, ManifestConstants.LIFECYLE_POLICY);
64  
65  
66          // check direct dependencies to find out dependency modules.
67          // we don't need to list transitive dependencies here, so use getDependencyArtifacts().
68  
69          TokenListBuilder dependencyModuleNames = new TokenListBuilder();
70          Set<String> dependencyModules = new HashSet<String>();  // used to find transitive dependencies through other modules.
71          for (Artifact a : (Set<Artifact>)pom.getDependencyArtifacts()) {
72              Manifest manifest = Jar.create(a.getFile()).getManifest();
73              String name = null;
74              if (manifest!=null) {
75                  Attributes attributes = manifest.getMainAttributes();
76  
77                  name = attributes.getValue(ManifestConstants.BUNDLE_NAME);
78              }
79              if (name != null) {
80                  // this is a hk2 module
81                  if (!a.isOptional())
82                      dependencyModuleNames.add(name);
83  
84                  // even optional modules need to be listed here
85                  dependencyModules.add(a.getGroupId() + ':' + a.getArtifactId() + ':');
86              }
87          }
88          if(!dependencyModuleNames.isEmpty()) {
89              entries.put(ManifestConstants.BUNDLE_IMPORT_NAME,dependencyModuleNames.toString());
90          }
91  
92          // find jar files to be listed in Class-Path. This needs to include transitive
93          // dependencies, except when the path involves a hk2 module.
94          TokenListBuilder classPathNames = new TokenListBuilder(" ");
95          TokenListBuilder classPathIds = new TokenListBuilder(" ");
96          for (Artifact a : (Set<Artifact>)pom.getArtifacts()) {
97              // check the trail. does that include hk2 module in the path?
98              boolean throughModule = false;
99              for (String module : dependencyModules)
100                 throughModule |= a.getDependencyTrail().get(1).toString().startsWith(module);
101             if(throughModule)
102                 continue;   // yep
103 
104             if(a.getScope().equals("system") || a.getScope().equals("provided"))
105                 continue;   // ignore tools.jar and such dependencies.
106 
107             if(a.isOptional())
108                 continue;   // optional dependency
109             
110             classPathNames.add(a.getFile().getName());
111             classPathIds.add(a.getId());
112         }
113         if(!classPathNames.isEmpty()) {
114             String existingClassPath = entries.get(ManifestConstants.CLASS_PATH);
115             if(existingClassPath!=null)
116                 entries.put(ManifestConstants.CLASS_PATH,existingClassPath+" "+classPathNames);
117             else
118                 entries.put(ManifestConstants.CLASS_PATH,classPathNames.toString());
119 
120             entries.put(ManifestConstants.CLASS_PATH_ID,classPathIds.toString());
121         }
122 
123         return entries;
124     }

126     private void configureImportPolicy(File classesDirectory, Map<String, String> entries, Class<?> clazz, String entryName) throws IOException {
127         File importPolicy = new File(classesDirectory, "META-INF/services/" + clazz.getName());
128         if(importPolicy.exists()) {
129             BufferedReader in = new BufferedReader(new InputStreamReader(new FileInputStream(importPolicy), "UTF-8"));
130             entries.put(entryName,in.readLine());
131             in.close();
132         }
133     }
