148     public void inhabitantMetaDataIncludesRunLevel() throws Exception {
149 
150         ServiceLocator locator = ServiceLocatorFactory.getInstance().create("inhabitantMetaDataIncludesRunLevel");
151 
152         configureLocator(locator, new Class[]{
153                 RunLevelControllerNegOne.class,
154                 RunLevelControllerOne.class});
155 
156         RunLevelController rlc = locator.getService(RunLevelController.class);
157 
158         final Filter filter = new Filter() {
159             @Override
160             public boolean matches(Descriptor d) {
161                 return RunLevel.class.getName().equals(d.getScope());
162             }
163         };
164 
165         List<ActiveDescriptor<?>> descriptors = locator.getDescriptors(filter);
166 
167         assertNotNull(descriptors);
168         int count = 0;
169         for (ActiveDescriptor<?> descriptor : descriptors) {
170             count++;
171             assertNotNull(descriptor.getMetadata());
172             String val = descriptor.getMetadata().get(RunLevel.RUNLEVEL_VAL_META_TAG).get(0);
173             assertNotNull(descriptor.toString(), val);
174             assertTrue(descriptor + " runLevel val=" + val, Integer.valueOf(val) >= RunLevel.RUNLEVEL_VAL_IMMEDIATE);
175         }
176         assertTrue(String.valueOf(count), count == 2);
177     }

678         public void configure(DynamicConfiguration configurator) {
679 
680             // Bind the run level service context
681             configurator.bind(BuilderHelper.link(RunLevelContext.class).
682                     to(Context.class).to(RunLevelContext.class).in(Singleton.class.getName()).build());
683 
684             // Bind the run level service
685             configurator.bind(BuilderHelper.link(RunLevelControllerImpl.class).
686                     to(RunLevelController.class).in(Singleton.class.getName()).build());
687 
688             bindService(configurator, TestListener.class, Singleton.class);

691             for (int i = 0; i < services.length; i++) {
692                 bindService(configurator, services[i], null);
693             }
694         }

696         private void bindService(DynamicConfiguration configurator, Class<?> service, Class<? extends Annotation> scope) {
697             final DescriptorBuilder descriptorBuilder = BuilderHelper.link(service);
698 
699             final RunLevel rla = service.getAnnotation(RunLevel.class);
700             if (rla != null) {
701                 descriptorBuilder.//to(RunLevel.class).
702                         has(RunLevel.RUNLEVEL_VAL_META_TAG, Collections.singletonList(((Integer) rla.value()).toString())).
703                         has(RunLevel.RUNLEVEL_MODE_META_TAG, Collections.singletonList(((Integer) rla.mode()).toString()));
704 
705                 descriptorBuilder.in(RunLevel.class);
706             }
707             Class clazz = service;
708             while (clazz != null) {
709                 Class<?>[] interfaces = clazz.getInterfaces();
710                 for (int j = 0; j < interfaces.length; j++) {
711                     descriptorBuilder.to(interfaces[j]);
712                 }
713                 clazz = clazz.getSuperclass();
714             }
715 
716             final RunLevelControllerIndicator runLevelControllerIndicator = service.getAnnotation(RunLevelControllerIndicator.class);
717             if (runLevelControllerIndicator != null) {
718                 descriptorBuilder.has(RunLevelControllerIndicator.RUNLEVEL_CONTROLLER_NAME_META_TAG, runLevelControllerIndicator.value());
719             }
720 
721             final Named named = service.getAnnotation(Named.class);
722             if (named != null) {
723                 descriptorBuilder.named(named.value());
724             }
725             
726             if (scope != null) {
727                 descriptorBuilder.in(scope);
728             }
729 
730             configurator.bind(descriptorBuilder.build());
731         }
