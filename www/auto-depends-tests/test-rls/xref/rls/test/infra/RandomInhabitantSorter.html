40  package rls.test.infra;

42  import java.io.BufferedReader;
43  import java.io.File;
44  import java.io.FileReader;
45  import java.io.IOException;
46  import java.io.PrintWriter;
47  import java.util.LinkedList;
48  import java.util.List;
49  import java.util.Random;
50  
51  import org.jvnet.hk2.annotations.Service;
52  import org.jvnet.hk2.component.Inhabitant;
53  import org.jvnet.hk2.component.InhabitantSorter;

55  @Service
56  public class RandomInhabitantSorter implements InhabitantSorter {
57    private static boolean called;
58    
59    private final static String DO_RANDOM = "com.oracle.hk2.debug.RandomizeServerServices";
60    private final static String DO_RANDOM_SEED_FILE = "com.oracle.hk2.debug.RandomizeServerServices.filename";
61    private final static String DO_RANDOM_SEED_DEFAULT_FILE = "target/test-classes/RandomServerServiceSeed";
62    private final static String DO_RANDOM_SEED = "com.oracle.hk2.debug.RandomizeServerServices.seed";
63    private final static int HIGH_ORDER_BITMASK = 0x7fffffff;
64    
65    private final boolean doRandom =
66      System.getProperty(DO_RANDOM, Boolean.TRUE.toString()).
67      equalsIgnoreCase(Boolean.TRUE.toString());
68    
69    private final Random random;
70    
71    public RandomInhabitantSorter() {
72      if (!doRandom) {
73        random = null;
74        return;
75      }
76      
77      String randomFileString =
78        System.getProperty(DO_RANDOM_SEED_FILE, DO_RANDOM_SEED_DEFAULT_FILE);
79      File randomFile = new File(randomFileString);
80      
81      long randomSeed;
82      
83      boolean writeOutputFile = true;
84      String userPropertySeed = System.getProperty(DO_RANDOM_SEED);
85      if (userPropertySeed != null) {
86        // The user property seed trumps all
87        try {
88          randomSeed = Long.parseLong(userPropertySeed);
89        }
90        catch (NumberFormatException nfe) {
91          randomSeed = System.currentTimeMillis();
92          
93          System.out.println(
94              "Could not parse the user random seed property whose value is " +
95              userPropertySeed + ".  Using seed " + randomSeed);
96        }
97      }
98      else if (randomFile.exists()) {
99        try {
100         randomSeed = readRandomFile(randomFile);
101         writeOutputFile = false;
102       }
103       catch (IOException ioe) {
104         randomSeed = System.currentTimeMillis();
105         
106         System.err.println("Could not read server service randomizer file " +
107           randomFile.getAbsolutePath() + ".  The seed used in this run is " +
108           randomSeed);
109       }
110     }
111     else {
112       // No random file or property, generate a "uniquish" random number
113       randomSeed = System.currentTimeMillis();
114     }
115     
116     random = new Random(randomSeed);
117     
118     System.out.println("Warning:  The order in which the server services are started has been randomized " +
119         "with seed " + randomSeed + ".  The seed can be found in file " + randomFile.getAbsolutePath());
120     
121     if (!writeOutputFile) return;  // Same file, no need to re-write it
122     
123     try {
124       writeNewRandomFile(randomFile, randomSeed);
125     }
126     catch (IOException ioe) {
127       System.err.println("Could not write server service randomizer file " +
128         randomFile.getAbsolutePath() + ".  The seed used in this run is " +
129         randomSeed);
130     }
131   }

133   /**
134    * Reads the randomizer file and returns the number found within
135    * 
136    * @param randomFile The file to read
137    * @return The number found in the file
138    * @throws IOException If the file could not be read for some reason
139    */
140   private static long readRandomFile(File randomFile) throws IOException {
141     long retVal;
142     
143     if (!randomFile.canRead()) {
144       retVal = System.currentTimeMillis();
145       
146       System.err.println("Could not read server service randomizer file " +
147           randomFile.getAbsolutePath() + ".  The seed used in this run is " +
148           retVal);
149       
150       return retVal;
151     }
152     
153     BufferedReader reader = new BufferedReader(new FileReader(randomFile));
154     try {
155       String line = reader.readLine();
156     
157       try {
158         retVal = Long.parseLong(line);
159       }
160       catch (NumberFormatException nfe) {
161         retVal = System.currentTimeMillis();
162       
163         System.err.println("Could not parse the data server service randomizer file " +
164           randomFile.getAbsolutePath() + ".  The seed used in this run is " +
165           retVal);
166       }
167     }
168     finally {
169       reader.close();
170     }
171     
172     return retVal;
173   }

175   /**
176    * Writes out the seed used in this run to a file
177    * @param randomFile The file to write the seed out to
178    * @param seed The seed used in this run
179    * @throws IOException If we could not write the file
180    */
181   private static void writeNewRandomFile(File randomFile, long seed) throws IOException {
182     if (randomFile.exists()) {
183       randomFile.delete();
184     }
185     
186     randomFile.createNewFile();
187     
188     PrintWriter writer = null;
189     try {
190       writer = new PrintWriter(randomFile);
191       writer.println("" + seed);
192     }
193     finally {
194       if (writer != null) writer.close();
195     }
196   }

198   /**
199    * @see org.jvnet.hk2.component.InhabitantSorter#sort(java.util.List)
200    */
201   @Override
202   public List<Inhabitant<?>> sort(List<Inhabitant<?>> arg0) {
203     setCalled();
204 
205     if (!doRandom) return arg0;
206     if (arg0 == null || arg0.size() <= 0) return arg0;
207     
208     List<Inhabitant<?>> original = new LinkedList<Inhabitant<?>>(arg0);
209     List<Inhabitant<?>> retVal = new LinkedList<Inhabitant<?>>();
210     
211     while (original.size() > 0) {
212       int removeMe = (random.nextInt() & HIGH_ORDER_BITMASK) % original.size();
213       
214       retVal.add(original.remove(removeMe));
215     }
216     
217     return retVal;
218   }

220   private static void setCalled() {
221       called = true;
222   }

224   public static boolean wasCalled() {
225       return called;
226   }
