71      @Element("*")
72      public List<Object> all = new ArrayList<Object>();

74      public FooBean() {
75          e = new Exception();
76      }
77

80          this.bar = bar;
81      }
82  
83      @Element
84      public void setJvmOptions(List<String> opts) {
85          this.jvmOptions.addAll(opts);
86      }
87  
88      public <T> T find(Class<T> t) {
89          for (Object o : all) {
90              if(t.isInstance(o))
91                  return t.cast(o);
92          }
93          return null;
94      }
