55      /***
56       * Finds and returns a <code>DefaultModuleDefinition</code> instance
57       * for a module given the name and version constraints.
58       * @param name the requested module name
59       * @param version the requestion module version
60       * @return a <code>DefaultModuleDefinition</code> or null if not found
61       * in this repository.
62       */
63      public ModuleDefinition find(String name, String version) {
64          for (ModuleDefinition moduleDef : findAll()) {
65              // so far we ignore the module version...
66              if (moduleDef.getName().equals(name)) {
67                  return moduleDef;
68              }
69          }
70          return null;
71      }

75       * @return a list of available modules

77      public abstract List<ModuleDefinition> findAll();

79      /***
80       * Finds and returns a list of all the available versions of a 
81       * module given its name.
82       * @param name the requested module name
83       */
84      public List<ModuleDefinition> findAll(String name) {
85          List<ModuleDefinition> matchingDefs = new ArrayList<ModuleDefinition>();
86          for (ModuleDefinition moduleDef : findAll()) {
87              if (moduleDef.getName().equals(name)) {
88                  matchingDefs.add(moduleDef);
89              }
90          }
91          return matchingDefs;
92      }
