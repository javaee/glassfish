43  public class ModuleClassLoader extends URLClassLoader {
44      
45      private final List<ClassLoader> surrogates = Collections.synchronizedList(new ArrayList<ClassLoader>());
46      private final List<ClassLoaderFacade> facadeSurrogates = Collections.synchronizedList(new ArrayList<ClassLoaderFacade>());
47      private final Module module;
48  
49      /***
50       * Module will be initialized when this classloader is consulted for the first time.
51       */
52      private volatile boolean initialized = false;
53      private StackTraceElement[] initializerThread;
54      private String initializerClassName;
55  
56      /*** Creates a new instance of ClassLoader */
57      public ModuleClassLoader(Module owner, URL[] shared, ClassLoader parent) {
58          super(shared, parent);
59          this.module = owner;
60      }
61      
62      protected void finalize() throws Throwable {
63          super.finalize();
64          Utils.getDefaultLogger().info("ModuleClassLoader gc'ed " + module.getModuleDefinition().getName());
65          stop();
66      }
67  
68  
69      protected synchronized Class<?> loadClass(String name, boolean resolve) throws ClassNotFoundException {
70          initialize(name);
71          return super.loadClass(name, resolve);
72      }
73  
74      public URL getResource(String name) {
75          initialize(name);
76          return super.getResource(name);
77      }
78  
79      public Enumeration<URL> getResources(String name) throws IOException {
80          initialize(name);
81          return super.getResources(name);
82      }
83  
84      /***
85       * Lazily initialize this module if not done so yet.
86       */
87      private void initialize(String name) {
88          if (initialized)    return;
89  
90          synchronized(this) {
91              if(!initialized) {
92                  // if we are validating, we should just not initiate initialization. 
93                  if (module.getState().equals(ModuleState.VALIDATING)) {
94                      return;
95                  }
96                  initialized = true;
97  
98                  module.start();
99                  // remember who started us to assist debugging.
100                 initializerThread = Thread.currentThread().getStackTrace();
101                 initializerClassName = name;
102             }
103         }
104     }
105 
106     /*package*/ void dumpState(PrintStream writer) {
107         if (initializerThread!=null) {
108             writer.println("Initialized when " + initializerClassName + " was requested by :");
109             for (StackTraceElement e : initializerThread) {
110                 writer.println(e.toString());
111             }
112         }
113     }

116     /***
117      * called by the facade class loader when it is garbage collected. 
118      * this is a good time to see if this module should be unloaded.
119      */
120     public void stop() {
121         
122         // we should only detach if the sticky flag is not set
123         if (!module.isSticky()) {
124             
125             Utils.getDefaultLogger().info("ModuleClassLoader stopped " + module.getModuleDefinition().getName());
126             surrogates.clear();
127             facadeSurrogates.clear();
128             module.stop();
129         }
130     }
131         
132     public Module getOwner() {
133         return module;
134     }

137     protected Class<?> findClass(String name) throws ClassNotFoundException {
138        
139         Class c = null;
140         synchronized(facadeSurrogates) {
141             for (ClassLoaderFacade classLoader : facadeSurrogates) {
142                 try {
143                     c = classLoader.getClass(name);
144                 } catch(ClassNotFoundException e) {
145                     // ignored.
146                 }
147                 if (c!=null) {
148                     return c;
149                 }
150             }
151         }
152         synchronized(surrogates) {
153             for (ClassLoader classLoader : surrogates) {
154                 try {
155                         c = classLoader.loadClass(name);
156                 } catch(ClassNotFoundException e) {
157                     // ignored.
158                 }
159                 if (c!=null) {
160                     return c;
161                 }
162             }
163         }
164         return super.findClass(name);
165     }

167     public URL findResource(String name) {
168         synchronized(facadeSurrogates) {
169             for (ClassLoaderFacade classLoader : facadeSurrogates) {
170 
171                 URL url = classLoader.findResource(name);
172                 if (url!=null) {
173                     return url;
174                 }
175             }
176         }
177         synchronized(surrogates) {
178             for (ClassLoader classLoader : surrogates) {
179 
180                 URL url = classLoader.getResource(name);
181                 if (url!=null) {
182                     return url;
183                 }
184             }
185         }
186         return super.findResource(name);
187         
188     }
189     
190     public Enumeration<URL> findResources(String name) throws IOException {
191 
192         for (ClassLoaderFacade classLoader : facadeSurrogates) {
193             
194             Enumeration<URL> enumerat = classLoader.getResources(name);
195             if (enumerat!=null && enumerat.hasMoreElements()) {
196                 return enumerat;
197             }
198         }
199         for (ClassLoader classLoader : surrogates) {
200             Enumeration<URL> enumerat = classLoader.getResources(name);
201             if (enumerat!=null && enumerat.hasMoreElements()) {
202                 return enumerat;
203             }
204             
205         }
206         return super.findResources(name);
207     }

209     public void addDelegate(ClassLoader cl) {
210         if (cl instanceof ClassLoaderFacade) {
211             facadeSurrogates.add((ClassLoaderFacade) cl);
212         } else {
213             surrogates.add(cl);
214         }                    
215     }
216                                                                             
217     public void removeDelegate(ClassLoader cl) {
218         if (cl instanceof ClassLoaderFacade) {
219             facadeSurrogates.remove(cl);
220         } else {
221             surrogates.remove(cl);
222         }
223     }
224     
225     public Collection<ClassLoader> getDelegates() {
226         return new ArrayList<ClassLoader>(surrogates);
227     }
228     
229     public String toString() {
230         StringBuffer s= new StringBuffer(); 
231         s.append(super.toString()).append("(name=").append(module.getName());
232         s.append(",init=").append(initialized);
233         s.append(",URls[]=");
234         for (URL url : getURLs()) {
235             s.append(url).append(",");
236         }
237         s.append(")");
238                
239         for (ClassLoader surrogate : surrogates) {
240             s.append("\n ref : ").append(surrogate.toString());
241         }
242         return s.toString();
243     }
