19   * Abstraction of {@link JarFile} so that we can handle
20   * both a jar file and a directory image transparently.

28       * See {@link JarFile#getManifest()} for the contract.

33       * Loads all <tt>META-INF/services</tt> entries and store them to the list.
34       */

38       * Gets the base name of the jar.
39       *
40       * <p>
41       * For example, "bar" for "bar.jar".

43      public abstract String getBaseName();

45      public static Jar create(File file) throws IOException {
46          if(file.isDirectory())
47              return new Directory(file);
48          else
49              return new Archive(file);
50      }

52      /***
53       * Loads a single service file.
54       */
55      protected final void load(InputStream is, String serviceName, List<ServiceProviderInfo> result) throws IOException {
56          try {
57              Scanner scanner = new Scanner(is);
58              while (scanner.hasNext()) {
59                  ServiceProviderInfo info = new ServiceProviderInfo(serviceName, scanner.next());
60                  result.add(info);
61              }
62          } finally {
63              is.close();
64          }
65      }

68      private static final class Directory extends Jar {
69          private final File dir;
70  
71          public Directory(File dir) {
72              this.dir = dir;
73          }
74  
75          public Manifest getManifest() throws IOException {
76              File mf = new File(dir,JarFile.MANIFEST_NAME);
77              if(mf.exists()) {
78                  FileInputStream in = new FileInputStream(mf);
79                  try {
80                      return new Manifest(in);
81                  } finally {
82                      in.close();
83                  }
84              } else {
85                  return null;
86              }
87          }
88  
89          public void getServiceProviders(List<ServiceProviderInfo> result) {
90              File[] services = new File(dir,SERVICE_LOCATION).listFiles();
91              if(services==null)  return;
92  
93              for( File svc : services ) {
94                  if(svc.isDirectory())
95                      continue;
96  
97                  try {
98                      load(new FileInputStream(svc), svc.getName(), result);
99                  } catch(IOException e) {
100                     Utils.getDefaultLogger().log(Level.SEVERE, "Error reading service provider in " + svc, e);
101                 }
102             }
103         }

105         public String getBaseName() {
106             return dir.getName();
107         }

110     private static final class Archive extends Jar {
111         private final JarFile jar;
112         private final File file;
113 
114         public Archive(File jar) throws IOException {
115             try {
116                 this.jar = new JarFile(jar);
117                 this.file = jar;
118             } catch (ZipException e) {
119                 // ZipException doesn't include this crucial information, so rewrap
120                 IOException x = new IOException("Failed to open " + jar);
121                 x.initCause(e);
122                 throw x;
123             }
124         }
125 
126         public Manifest getManifest() throws IOException {
127             return jar.getManifest();
128         }
129 
130         public void getServiceProviders(List<ServiceProviderInfo> result) {
131             Enumeration<JarEntry> entries = jar.entries();
132             while (entries.hasMoreElements()) {
133                 JarEntry entry = entries.nextElement();
134                 if (entry.getName().startsWith(SERVICE_LOCATION)) {
135                     String serviceName = entry.getName().substring(SERVICE_LOCATION.length()+1);
136 
137                     try {
138                         load(jar.getInputStream(entry), serviceName, result);
139                     } catch(IOException e) {
140                         Utils.getDefaultLogger().log(Level.SEVERE, "Error reading service provider in " + jar.getName(), e);
141                     }
142                 }
143             }
144         }
145 
146         public String getBaseName() {
147             String name = file.getName();
148             int idx = name.lastIndexOf('.');
149             if(idx>=0)
150                 name = name.substring(0,idx);
151             return name;
152         }
