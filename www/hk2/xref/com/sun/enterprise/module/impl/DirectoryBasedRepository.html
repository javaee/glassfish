30  /***
31   * This class is a directory based repository implementation. This mean that all jar
32   * file residing a particular directory will be elligible to be added to this 
33   * repository instance. Jar File will need to be compliant with the module definition
34   * spec which mean have a manifest file with the correct elements OR a separate 
35   * manifest file with the same file name with a .mf extension.
36   *
37   * @author Jerome Dochez
38   */
39  public class DirectoryBasedRepository extends AbstractRepositoryImpl {
40      
41      private final File repository;
42  
43      /*** Creates a new instance of DirectoryBasedRepository */
44      public DirectoryBasedRepository(String name, File repository) throws FileNotFoundException {
45          super(name,repository.toURI());
46          this.repository = repository;
47      }
48  
49      @Override
50      protected Map<String, ModuleDefinition> loadModuleDefs() throws IOException {
51          if (!repository.exists()) {
52              throw new FileNotFoundException(repository.getAbsolutePath());
53          }
54  
55          Map<String, ModuleDefinition> moduleDefs = new HashMap<String, ModuleDefinition>();
56  
57          try {
58              File[] files = repository.listFiles();
59              for (File aFile : files) {
60                  if (aFile.getName().endsWith(".jar")) {
61                      ModuleDefinition moduleDef = loadJar(aFile);
62                      if (moduleDef!=null) {
63                          moduleDefs.put(moduleDef.getName(), moduleDef);
64                      }
65                  }
66              }
67  
68              return moduleDefs;
69          } catch (IOException e) {
70              IOException x = new IOException("Failed to load modules from " + repository);
71              x.initCause(e);
72              throw x;
73          }
74      }
75  
76      /***
77       * This module adds a new Module Definition to the repository.
78       *
79       * @param definition is the module definition
80       * @return true if the addition was successful
81       */
82      public void add(ModuleDefinition definition) throws IOException {
83  
84          // we need to copy all locations as found in the definition to our repository.
85          for (URI location : definition.getLocations()) {
86              InputStream is=null;
87              FileOutputStream os=null;
88              try {
89                  is = location.toURL().openStream();
90                  ReadableByteChannel channel = Channels.newChannel(is);
91                  // this naming scheme may need to be reworked...
92                  File outFile = new File(repository, definition.getName() + ".jar");
93                  os = new FileOutputStream(outFile);
94                  FileChannel outChannel = os.getChannel();
95                  long bytes;
96                  long transferedBytes=0;
97                  do {
98                      bytes = outChannel.transferFrom(channel,transferedBytes , 4096);
99                      transferedBytes+=bytes;
100                 } while (bytes==4096);
101             } finally {
102                 try {
103                     if (is!=null)
104                         is.close();
105                 } catch(IOException e) {
106                     // ignore
107                 }
108                 try {
109                     if (os!=null)
110                         os.close();
111                 } catch(IOException e) {
112                     // ignore
113                 }                
114             }
115         }
116 
117         // we also need to save the definition as an external manifest file since
118         // the defintion may have been expressed with different means than our
119         // manifest entries.
120         Manifest manifest = definition.getManifest();
121         if (manifest!=null) {
122             if (manifest.getMainAttributes().getValue(ManifestConstants.BUNDLE_NAME)!=null) {
123                 return;
124             }
125         }
126         // write the external manifest file
127         // TODO : have some plugability here where we separate the Manifest IO from
128         // the definition and the respository
129         File manifestFile = new File(repository, definition.getName() + ".mf");
130         manifest = new Manifest();
131         Attributes attr = manifest.getMainAttributes();
132         // name
133         attr.put(ManifestConstants.BUNDLE_NAME, definition.getName());
134         // list of imported bundle
135         StringBuffer buffer = new StringBuffer();
136         for (URI location : definition.getLocations()) {
137             if (buffer.length()>0) {
138                 buffer.append(", ");
139             }
140             String fileName = location.getPath().substring(location.getPath().lastIndexOf('/'));
141             buffer.append(fileName);
142         }
143         attr.put(ManifestConstants.BUNDLE_IMPORT_NAME, buffer.toString());
144         FileOutputStream os = null;
145         try {
146             os = new FileOutputStream(manifestFile);
147             manifest.write(new BufferedOutputStream(os));
148         } finally {
149             if (os!=null) {
150                 os.close();
151             }
152         }
153     }
