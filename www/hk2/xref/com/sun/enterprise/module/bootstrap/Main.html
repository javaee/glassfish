76      /***
77       *  We need to determine which jar file has been used to load this class
78       *  Using the getResourceURL we can get this information, after that, it
79       *  is just a bit of detective work to get the file path for the jar file.
80       *
81       * @return
82       *      the path to the jar file containing this class.
83       *      always returns non-null.
84       *
85       * @throws BootException
86       *      If failed to determine the bootstrap file name.
87       */
88      protected File getBootstrapFile() throws BootException {
89          String resourceName = getClass().getName().replace(".","/")+".class";
90          URL resource = getClass().getClassLoader().getResource(resourceName);
91          if (resource==null) {
92              throw new BootException("Cannot get bootstrap path from "
93                      + resourceName + " class location, aborting");
94          }
95  
96          if (resource.getProtocol().equals("jar")) {
97              try {
98                  JarURLConnection c = (JarURLConnection) resource.openConnection();
99                  URL jarFile = c.getJarFileURL();
100                 File f = new File(jarFile.toURI());
101                 return f;
102             } catch (IOException e) {
103                 throw new BootException("Cannot open bootstrap jar file", e);
104             } catch (URISyntaxException e) {
105                 throw new BootException("Incorrect bootstrap class URI", e);
106             }
107         } else
108             throw new BootException("Don't support packaging "+resource+" , please contribute !");
109     }

111     /***
112      * Start the server from the command line
113      * @param args the command line arguments
114      */
115     public void start(String[] args) throws BootException {
116         
117         File bootstrap = this.getBootstrapFile();
118         File root = bootstrap.getAbsoluteFile().getParentFile();
119 
120         // root is the directory in which this bootstrap.jar is located
121         // For most cases, this is the lib directory although this is completely
122         // dependent on the usage of this facility.
123         if (root==null) {
124             throw new BootException("Cannot find root installation from "+bootstrap);
125         }
126 
127         String targetModule = findMainModuleName(bootstrap);
128 
129         // get the ModuleStartup implementation.
130         ModulesRegistry mr = ModulesRegistry.createRegistry();
131         createRepository(root,mr);
132 
133         launch(mr, targetModule, root, args);
134     }

140     protected void createRepository(File root, ModulesRegistry mr) throws BootException {
141         try {
142             DirectoryBasedRepository lib = new DirectoryBasedRepository("lib", root);
143             lib.initialize();
144             mr.addRepository(lib);
145             mr.setParentClassLoader(this.getClass().getClassLoader());
146         } catch(IOException ioe) {
147             throw new BootException("Error while initializing lib repository at : "+root, ioe);
148         }

151     /***
152      * Launches the module system and hand over the execution to the {@link ModuleStartup}
153      * implementation of the main module.
154      *
155      * <p>
156      * This version of the method auto-discoveres the main module.
157      * If there's more than one {@link ModuleStartup} implementation, it is an error. 
158      *
159      * @param root
160      *      This becomes {@link StartupContext#getRootDirectory()}
161      * @param args
162      *      This becomes {@link StartupContext#getArguments()}
163      *
164      */
165     public void launch(ModulesRegistry registry, File root, String[] args) throws BootException {
166         Collection<ModuleStartup> startups = new ComponentManager(registry).getComponents(ModuleStartup.class);
167         if(startups.isEmpty())
168             throw new BootException("No module has ModuleStartup");
169         if(startups.size()>1) {
170             Iterator<ModuleStartup> itr = startups.iterator();
171             throw new BootException("Multiple ModuleStartup found: "+itr.next()+" and "+itr.next());
172         }
173 
174         ModuleStartup ms = startups.iterator().next();
175         Module mainModule = Module.find(ms.getClass());
176         StartupContext context = new StartupContext(root, mainModule, args);
177         launch(ms,context, mainModule);
178     }

180     /***
181      * Launches the module system and hand over the execution to the {@link ModuleStartup}
182      * implementation of the main module.
183      *
184      * @param root
185      *      This becomes {@link StartupContext#getRootDirectory()}
186      * @param args
187      *      This becomes {@link StartupContext#getArguments()}
188      *
189      */
190     public void launch(ModulesRegistry registry, String mainModuleName, File root, String[] args) throws BootException {
191         // instantiate the main module, this is the entry point of the application
192         // code. it is supposed to have 1 ModuleStartup implementation.
193         final Module mainModule = registry.makeModuleFor(mainModuleName, null);
194         if (mainModule == null) {
195             if(registry.getModules().isEmpty())
196                 throw new BootException("Registry has no module at all");
197             else
198                 throw new BootException("Cannot find main module " + mainModuleName+" : no such module");
199         }
200 
201         String targetClassName = findModuleStartup(mainModule);
202         if (targetClassName==null) {
203             throw new BootException("Cannot find a ModuleStartup implementation in the META-INF/services/com.sun.enterprise.v3.ModuleStartup file, aborting");
204         }
205 
206         mainModule.setSticky(true);
207 
208         Class<? extends ModuleStartup> targetClass=null;
209         ModuleStartup startupCode;
210         ClassLoader currentCL = Thread.currentThread().getContextClassLoader();
211         try {
212             Thread.currentThread().setContextClassLoader(mainModule.getClassLoader());
213             try {
214                 targetClass = mainModule.getClassLoader().loadClass(targetClassName).asSubclass(ModuleStartup.class);
215                 startupCode = new ComponentManager(registry).getComponent(targetClass);
216             } catch (ClassNotFoundException e) {
217                 throw new BootException("Unable to load "+targetClassName,e);
218             } catch (ComponentException e) {
219                 throw new BootException("Unable to load "+targetClass,e);                
220             }
221             StartupContext context = new StartupContext(root, mainModule, args);
222 
223             launch(startupCode, context, mainModule);
224         } finally {
225             Thread.currentThread().setContextClassLoader(currentCL);
226         }
227 
228     }

230     protected String findMainModuleName(File bootstrap) throws BootException {
231         String targetModule;
232         try {
233             JarFile jarFile = new JarFile(bootstrap);
234             Manifest manifest = jarFile.getManifest();
235 
236             Attributes attr = manifest.getMainAttributes();
237             targetModule = attr.getValue(ManifestConstants.BUNDLE_IMPORT_NAME);
238             if (targetModule==null) {
239                 throw new BootException("No Import-Bundles module found in manifest of " + bootstrap.getAbsoluteFile());
240             }
241         } catch(IOException ioe) {
242             throw new BootException("Cannot get manifest from " + bootstrap.getAbsolutePath(), ioe);
243         }
244         return targetModule;
245     }

268     /***
269      * Finds {@link ModuleStartup} implementation class name to perform the launch.
270      *
271      * <p>
272      * This implementation does so by looking it up from services.
273      */
274     protected String findModuleStartup(Module mainModule) {
275         // so far, I only take the first implementation...
276         ClassLoader mainModuleClassLoader = mainModule.getClassLoader();
277 
278         LineNumberReader fr=null;
279         InputStream is=null;
280         final String serviceIntf = ModuleStartup.class.getName();
281         try {
282             is = mainModuleClassLoader.getResourceAsStream("META-INF/services/"+serviceIntf);
283             if (is==null) {
284                 System.err.println("no META-INF/services/"+ serviceIntf + " file found in " + mainModule);
285                 return null;
286             }
287             fr = new LineNumberReader(new InputStreamReader(is));
288             return fr.readLine();
289         } catch (IOException e) {
290             System.err.println("Cannot read the META-INF/services/"+serviceIntf+" file : " +  e);
291             return null;
292         } finally {
293             if (fr!=null) {
294                 try {
295                     fr.close();
296                 } catch(Exception e) {
297                     System.err.println("Cannot close file reader "  + e);
298 
299                 }
300             }
301             if (is!=null) {
302                 try {
303                     is.close();
304                 } catch(Exception e) {
305                     System.err.println( "Cannot close input stream " + e);
306                 }
307             }
308         }
