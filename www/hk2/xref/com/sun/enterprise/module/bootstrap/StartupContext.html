10  package com.sun.enterprise.module.bootstrap;
11  
12  import com.sun.enterprise.module.Module;
13  
14  import java.io.File;
15  import java.util.HashMap;
16  import java.util.Map;
17  
18  /***
19   * This class contains important information about the startup process
20   *
21   * @author dochez
22   */
23  public class StartupContext {
24      
25      final File root;
26      final Module mainModule;
27      final Map<String, String> args;
28      final ClassLoader mainModuleClassLoader;
29      
30      /*** Creates a new instance of StartupContext */
31      public StartupContext(File root, Module mainModule, String[] args) {
32          this.root = root;
33          this.mainModule = mainModule;
34          // I am saving the main module class loader as it cannot be garbage collected
35          // during the startup sequence...
36          this.mainModuleClassLoader = mainModule.getClassLoader();
37          this.args = new HashMap<String, String>();
38          for (int i=0;i<args.length;i++) {
39              if (args[i].startsWith("-")) {
40                  if (i+1<args.length) {
41                      this.args.put(args[i], args[i+1]);
42                      i++;
43                  }
44              } else {
45                  this.args.put("default", args[i]);
46              }
47          }
48      }
49  
50      /***
51       * Gets the "root" directory where the data files are stored.
52       *
53       * TODO: in case of Glassfish, this is the domain directory?
54       */
55      public File getRootDirectory() {
56          return root;
57      }
58      
59      public Module getMainModule() {
60          return mainModule;
61      }
62      
63      public Map<String, String> getArguments() {
64          return args;
65      }
66  }
