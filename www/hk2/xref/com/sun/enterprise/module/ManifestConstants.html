24  package com.sun.enterprise.module;
25  
26  import org.jvnet.hk2.annotations.Service;
27  
28  /***
29   * Constants used in the module manifest.
30   * These values define a module.
31   *
32   * @author dochez
33   * @author Kohsuke Kawaguchi
34   */
35  public class ManifestConstants {
36      // No instanciation allowed.
37      private ManifestConstants() {
38      }
39  
40      /***
41       * Module name.
42       * <p>
43       * This uniquely identifies the module within modules,
44       * but this shouldn't include the version number.
45       * <p>
46       * Maven's <tt>hk2-module</tt> packaging type uses GROUPID:ARTIFACTID for this.
47       */
48      public static final String BUNDLE_NAME = "Bundle-Name";
49      /***
50       * Exported package list.
51       * <p>
52       * The value is a comma separated list of packages that are
53       * visible to other modules that depend on this module.
54       * For example, "com.sun.foobar.spi,com.sun.foobar.abc"
55       * <p>
56       * If this manifest entry doesn't exist, all classes will be exported.
57       *
58       * <p>
59       * TODO: expand Maven's <tt>hk2-module</tt> packaging to generate this.
60       */
61      public static final String PKG_EXPORT_NAME = "Export-Package";
62  
63      /***
64       * List of modules that this module depends on.
65       * <p>
66       * The value is a comma separated list of {@link #BUNDLE_NAME modules names}
67       * that this module depends on. Whitespaces are allowed before and after
68       * commas.
69       * <p>
70       *
71       * <p>
72       * Maven's <tt>hk2-module</tt> packaging uses dependency list in POM
73       * to fill in this entry.
74       *
75       * TODO: expand this field to support version specifier.
76       */
77      public static final String BUNDLE_IMPORT_NAME = "Import-Bundles";
78  
79      /***
80       * List of other jar files in this module.
81       * <p>
82       * This identifies other jar files in this module, not other modules
83       * that this module depends on.
84       * <p>
85       * See <a href="http://java.sun.com/j2se/1.5.0/docs/guide/jar/jar.html#Main%20Attributes">
86       * jar file specification</a> for the format of the value.
87       *
88       * <p>
89       * Maven's <tt>hk2-module</tt> packaging uses dependency list in POM
90       * to fill in this entry.
91       *
92       * @see #CLASS_PATH_ID
93       */
94      public static final String CLASS_PATH = "Class-Path";
95  
96      /***
97       * List of other jar files in this module.
98       * <p>
99       * The value is a whitespace separated list of Maven ID of all the dependencies,
100      * which is of the form "GROUPID:ARTIFACTID:TYPE[:CLASSIFIER]:VERSION".
101      * <p>
102      * This entry contains essentially the same information as the {@link #CLASS_PATH}
103      * entry, but this enables HK2 to assemble all the required jars by using Maven.
104      *
105      * <p>
106      * Maven's <tt>hk2-module</tt> packaging uses dependency list in POM
107      * to fill in this entry.
108      *
109      * @see #CLASS_PATH
110      */
111     public static final String CLASS_PATH_ID = "Class-Path-Id";
112 
113     /***
114      * See {@link ImportPolicy}.
115      *
116      * <p>
117      * The value is the fully-qualified class name of a class that implements
118      * {@link ImportPolicy}.
119      * <p>
120      * Maven's <tt>hk2-module</tt> packaging automatically finds such a class
121      * and puts its name, provided that the class has @{@link Service} annotation.
122      */
123     public static final String IMPORT_POLICY = "Module-Import-Policy";
124 
125     /***
126      * See {@link LifecyclePolicy}.
127      *
128      * <p>
129      * The value is the fully-qualified class name of a class that implements
130      * {@link LifecyclePolicy}.
131      * <p>
132      * Maven's <tt>hk2-module</tt> packaging automatically finds such a class
133      * and puts its name, provided that the class has @{@link Service} annotation.
134      */
135     public static final String LIFECYLE_POLICY = "Module-Lifecycle-Policy";
136 }
