48  /***
49   * A module represents a set of resources accessible to third party modules. 
50   * Each module has a module definition which defines its name, its list of 
51   * exported resources and its dependencies to other modules. 
52   * A module instance stores references to the class loader instances giving
53   * access to the module's implementation. 
54   * A module instance belongs to a <code>ModuleRegistry</code> which can be used
55   * to get the list of available modules and/or get particular module
56   * implementation. 
57   * Modules can only satisfy their dependencies within the <code>ModuleRegistry
58   * </code> instance they are registered in.
59   *
60   * @author Jerome Dochez
61   */

87      /*** Creates a new instance of Module */
88      public Module(ModulesRegistry registry, ModuleDefinition info) {
89          assert registry!=null && info!=null;
90          this.registry = registry;
91          moduleDef = info;
92          for (URI lib : info.getLocations()) {
93              File f = new File(lib);
94              if (f.exists()) {
95                  lastModifieds.put(f.getAbsolutePath(), f.lastModified());
96              }

98          }
99          state = ModuleState.NEW;
100     }

103      * Return the <code>ClassLoader</code>  instance associated with this module.
104      * Only designated public interfaces will be loaded and returned by 
105      * this classloader

108     public ClassLoader getClassLoader() {
109         ClassLoader r=null;
110         if (publicCL!=null)
111             r = publicCL.get();
112         if (r!=null)
113             return r;
114 
115         ClassLoaderFacade facade = AccessController.doPrivileged(new PrivilegedAction<ClassLoaderFacade>() {
116             public ClassLoaderFacade run() {
117                 return new ClassLoaderFacade(getPrivateClassLoader());
118             }
119         });
120         facade.setPublicPkgs(moduleDef.getPublicInterfaces());
121         publicCL = new WeakReference<ClassLoader>(facade);
122         return facade;
123     }

125     /***
126      * Return the private class loader for this module. This class loader will
127      * be loading all the classes which are not explicitely exported in the 
128      * module definition
129      * @return the private <code>ClassLoader</code> instance
130      */
131     private ModuleClassLoader getPrivateClassLoader() {
132         if (privateCL==null) {
133             synchronized(this) {
134                 if(privateCL==null) {
135                     URI[] locations = moduleDef.getLocations();
136                     URL[] urlLocations = new URL[locations.length];
137                     for (int i=0;i<locations.length;i++) {
138                         try {
139                             urlLocations[i] = locations[i].toURL();
140                         } catch(MalformedURLException e) {
141                             e.printStackTrace();
142                             return null;
143                         }
144                     }
145                     final URL[] urls = urlLocations.clone();
146                     privateCL = AccessController.doPrivileged(new PrivilegedAction<ModuleClassLoader>() {
147                         public ModuleClassLoader run() {
148                             return new ModuleClassLoader(
149                                 Module.this,
150                                 urls,
151                                 registry.getParentClassLoader());
152                         }
153                     });
154                 }
155             }
156         }
157         return privateCL;
158     }

223     /***
224      * Trigger manual refresh mechanism, the module will check all its 
225      * URLs and generate change events if any of them has changed. This 
226      * will allow the owning registry to force a module upgrade at next 
227      * module request.
228      */
229     public void refresh() {
230         URI[] urls = moduleDef.getLocations();
231         for (URI lib : urls) {
232             File f = new File(lib);
233             if (f.exists() && lastModifieds.containsKey(f.getAbsolutePath())) {
234                 if (lastModifieds.get(f.getAbsolutePath()) !=f.lastModified()) {
235                     //Utils.getDefaultLogger().info("Changed : " + this);
236                     fireChangeEvent();                       
237                     // detach for now...
238                     detach();
239                     return;
240                 }
241             }
242         }
243     }

258     /*** 
259      * Ensure that this module is {@link ModuleState#RESOLVED resolved}.
260      *
261      * <p>
262      * If the module is already resolved, this method does nothing.
263      * Otherwise, iterate over all declared ModuleDependency instances and use the
264      * associated <code>ModuleRegistry</code> to resolve it. After successful
265      * completion of this method, the module state is 
266      * {@link ModuleState#RESOLVED}.
267      *
268      * @throws <code>ResolveError</code> if any of the declared dependency of this module 
269      * cannot be satisfied
270      */
271     public synchronized void resolve() throws ResolveError {
272         
273         // already resolved ?
274         if (state==ModuleState.ERROR)
275             throw new ResolveError("Module " + getName() + " is in ERROR state");
276         if (state.compareTo(ModuleState.RESOLVED)>=0)
277             return;
278 
279         state = ModuleState.VALIDATING;
280         
281         if (moduleDef.getImportPolicyClassName()!=null) {
282             try {
283                 Class<ImportPolicy> importPolicyClass = (Class<ImportPolicy>) getPrivateClassLoader().loadClass(moduleDef.getImportPolicyClassName());
284                 ImportPolicy importPolicy = importPolicyClass.newInstance();
285                 importPolicy.prepare(this);
286             } catch(ClassNotFoundException e) {
287                 throw new ResolveError(e);
288             } catch(java.lang.InstantiationException e) {
289                 throw new ResolveError(e);
290             } catch(IllegalAccessException e) {
291                 throw new ResolveError(e);
292             }
293         }
294         for (ModuleDependency dependency : moduleDef.getDependencies()) {
295             
296             Module depModule = registry.makeModuleFor(dependency.getName(), null);
297             if (depModule==null) {
298                 throw new ResolveError(dependency + " referenced from " 
299                         + moduleDef.getName() + " is not resolved");
300             }
301             addImport(depModule);
302         }
303         state = ModuleState.RESOLVED;
304         
305         // time to initialize the lifecycle instance
306         if (moduleDef.getLifecyclePolicyClassName()!=null) {
307             try {
308                 Class<LifecyclePolicy> lifecyclePolicyClass = (Class<LifecyclePolicy>) getPrivateClassLoader().loadClass(moduleDef.getLifecyclePolicyClassName());
309                 lifecyclePolicy = lifecyclePolicyClass.newInstance();
310             } catch(ClassNotFoundException e) {
311                 throw new ResolveError("ClassNotFound : " + e.getMessage(), e);
312             } catch(java.lang.InstantiationException e) {
313                 throw new ResolveError(e);
314             } catch(IllegalAccessException e) {
315                 throw new ResolveError(e);
316             }
317             lifecyclePolicy.load(this);
318         }
319         //Logger.global.info("Module " + getName() + " resolved");
320 
321         // module resolution complete. notify listeners
322         for (ModuleLifecycleListener listener : registry.lifecycleListeners)
323             listener.moduleStarted(this);
324     }

326     /***
327      * Forces module startup. In most cases, the runtime will take care 
328      * of starting modules when they are first used. There could be cases where
329      * code need to manually start a sub module. Invoking this method will 
330      * move the module to the {@link ModuleState#READY ModuleState.READY}, the 
331      * {@link LifecyclePolicy#start Lifecycle.start} method will be invoked.
332      */
333     public void start() throws ResolveError {
334         
335         if (state==ModuleState.READY)
336             return;
337         
338         // ensure RESOLVED state
339         resolve();      
340         
341         state = ModuleState.READY;
342         try {
343             for (Module subModules : dependencies) {
344                 subModules.start();
345             }
346         } catch(ResolveError e) {
347             state = ModuleState.RESOLVED;
348             throw e;
349         }
350         
351         if (lifecyclePolicy!=null) {
352             lifecyclePolicy.start(this);
353         }
354         //Logger.global.info("Module " + getName() + " started");
355     }

357     /*** 
358      * Forces module stop. In most cases, the runtime will take care of stopping
359      * modules when the last module user released its interest. However, in 
360      * certain cases, it may be interesting to manually stop the module. 
361      * Stopping the module means that the module is removed from the registry, 
362      * the class loader references are released (note : the class loaders will
363      * only be released if all instances of any class loaded by them are gc'ed). 
364      * If a <code>LifecyclePolicy</code> for this module is defined, the 
365      * {@link LifecyclePolicy#stop(Module) Lifecycle.stop(Module)}
366      * method will be called and finally the module state will be 
367      * returned to {@link ModuleState#NEW ModuleState.NEW}.
368      *
369      * @return true if unloading was successful
370      */
371     public boolean stop() {
372 
373         if (sticky) {
374             return false;
375         } 
376         
377         if (lifecyclePolicy!=null) {
378             lifecyclePolicy.stop(this);
379             lifecyclePolicy=null;
380         }
381         
382         detach();
383         
384         // we do NOT stop our sub modules which are shared...
385         for (Module subModule : dependencies) {
386             if (!subModule.isShared()) {
387                 subModule.stop();
388             }
389         }
390         // release all sub modules class loaders
391         privateCL = null;
392         publicCL = null;
393         dependencies.clear();
394         
395         state = ModuleState.NEW;
396         return true;
397     }

412     public Module addImport(ModuleDependency dependency) {
413 
414         Module newModule;
415         if (dependency.isShared()) {
416             newModule = registry.makeModuleFor(dependency.getName(), dependency.getVersion());
417         } else {
418             newModule = registry.newPrivateModuleFor(dependency.getName(), dependency.getVersion());
419         }
420         addImport(newModule);
421         return newModule;

496     public <T> Iterable<Class<? extends T>> getProvidersClass(Class<T> serviceClass) {
497         String name = serviceClass.getName();
498 
499         List r = serviceClasses.get(name);
500         if(r!=null) return r;
501 
502         // the worst case scenario in the race situation is we end up creating the same list twice,
503         // which is not a big deal.
504 
505         for( ServiceProviderInfo spi : getServiceProviders()) {
506             if(!spi.getServiceName().equals(name))
507                 continue;
508 
509             if(r==null)
510                 r = new ArrayList<Class>();
511             try {
512                 r.add(getPrivateClassLoader().loadClass(spi.getProviderName()));
513             } catch (ClassNotFoundException e) {
514                 Utils.getDefaultLogger().log(Level.SEVERE, "Failed to load "+spi.getProviderName()+" from "+getName(),e);
515             }
516         }
517 
518         if(r==null)
519             r = Collections.emptyList();
520 
521         serviceClasses.put(name, r);
522         return r;
523     }

525     /***
526      * Returns true if this module has any provider for the given service class.
527      */
528     public boolean hasProvider(Class serviceClass) {
529         String name = serviceClass.getName();
530 
531         List<Class> v = serviceClasses.get(name);
532         if(v!=null && !v.isEmpty())    return true;
533 
534         for( ServiceProviderInfo spi : getServiceProviders())
535             if(spi.getServiceName().equals(name))
536                 return true;
537 
538         return false;
539     }

542         
543         writer.println("Module " + getName() + " Dump");
544         writer.println("State " + getState());

554     /***

560     public static Module find(Class clazz) {
561         ClassLoader cl = clazz.getClassLoader();
562         if(cl==null)    return null;
