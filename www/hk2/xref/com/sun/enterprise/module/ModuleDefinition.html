24  package com.sun.enterprise.module;
25  
26  import java.net.URI;
27  import java.util.Collection;
28  import java.util.jar.Manifest;
29  
30  /***
31   * A module definition contains all information about a module
32   * existence, its dependencies and its exported interfaces.
33   *
34   * This module meta information can be obtained from different
35   * sources and format. For instance OSGi modules use the manifest
36   * file and so is the glassfish application server. Others can
37   * use api or xml file.
38   *
39   * @author Jerome Dochez
40   */
41  public interface ModuleDefinition {
42  
43      /***
44       *  Returns the module name, usually this is the same name as
45       *  the jar file name containing the module's implementation.
46  
47       * @return module name
48       */
49      String getName();
50  
51      /***
52       *  Returns a list of public interfaces for this module.
53       *  Public interface can be packages, interfaces, or classes
54       *
55       * @return a array of public interfaces
56       */
57      String[] getPublicInterfaces();
58  
59      /***
60       * Returns the list of this module's dependencies. Each dependency
61       * must be satisfied at run time for this module to function
62       * properly.
63       *
64       * @return list of dependencies
65       */
66      ModuleDependency[] getDependencies();
67  
68      /***
69       * A Module is implemented by one to many jar files. This method returns
70       * the list of jar files implementing the module
71       *
72       * @return the module's list of implementation jars
73       */
74      URI[] getLocations();
75  
76      /***
77       * Returns the version of this module's implementation
78       *
79       * @return a version number
80       */
81      String getVersion();
82  
83      /***
84       * Returns the import policy class name. Although the implementation of
85       * this policy does not necessary have to implement the ImportPolicy, but
86       * could use another interface, it is the responsibility of the associated
87       * Repository to invoke that interface when the module is started.
88       *
89       * @return the ImportPolicy or equivalent interface implementation
90       */
91      String getImportPolicyClassName();
92  
93      /***
94       * Returns the lifecycle policy class name. Although the implementation of
95       * this policy does not necessary have to implement the LifecyclePolicy, but
96       * could use another interface, it is the responsibility of the associated
97       * Repository to invoke that interface when the module is started.
98       *
99       * @return the LifecyclePolicy or equivalent interface implementation
100      */
101     String getLifecyclePolicyClassName();
102 
103     /***
104      * Returns the manifest file for the main module jar file
105      * 
106      * @return the manifest file
107      */
108     Manifest getManifest();
109 
110     /***
111      * Requests the list of Services this module provide (generally, module
112      * provide one service but it's not limited). The Service is described
113      * by the module implementation using the META_INF/services format
114      * described in the JDK jar file format description.
115      *
116      * @return
117      *      Can be empty but never be null.
118      */
119     Collection<ServiceProviderInfo> getServiceProviders();
120 }
