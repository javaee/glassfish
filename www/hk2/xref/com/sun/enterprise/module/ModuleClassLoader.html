75      public URL getResource(String name) {
76          initialize(name);
77          if(name.startsWith(META_INF_SERVICES)) {
78              // punch in. find the service loader from any module
79              String serviceName = name.substring(META_INF_SERVICES.length());
80  
81              ModulesRegistry reg = module.getRegistry();
82              for( Module m : reg.getModules() ) {
83                  List<URL> list = m.getServiceProviders().getDescriptors(serviceName);
84                  if(!list.isEmpty())     return list.get(0);
85              }
86  
87              // no such resource
88              return null;
89          } else {
90              // normal service look up
91              URL url = super.getResource(name);
92              if(url!=null)
93                  return url;
94  
95              // commons-logging looks for a class file resource for providers,
96              // so check for those
97              if(name.endsWith(".class")) {
98                  String className = name.replace('/', '.').substring(0, name.length() - 6);
99                  Module m = module.getRegistry().getProvidingModule(className);
100                 if(m!=null)
101                     return m.getPrivateClassLoader().getResource(name);
102             }
103 
104             return null;
105         }
106     }

108     public Enumeration<URL> getResources(String name) throws IOException {
109         initialize(name);
110         if(name.startsWith(META_INF_SERVICES)) {
111             // punch in. find the service loader from any module
112             String serviceName = name.substring(META_INF_SERVICES.length());
113 
114             Vector<URL> urls = new Vector<URL>();
115 
116             ModulesRegistry reg = module.getRegistry();
117             for( Module m : reg.getModules() )
118                 urls.addAll(m.getServiceProviders().getDescriptors(serviceName));
119 
120             return urls.elements();
121         } else {
122             // normal look up
123             return super.getResources(name);
124         }
125     }

127     /***
128      * Lazily initialize this module if not done so yet.
129      */
130     private void initialize(String name) {
131         if (initialized)    return;
132 
133         synchronized(this) {
134             if(!initialized) {
135                 // if we are validating, we should just not initiate initialization. 
136                 if (module.getState().equals(ModuleState.VALIDATING)) {
137                     return;
138                 }
139                 initialized = true;
140 
141                 module.start();
142                 // remember who started us to assist debugging.
143                 initializerThread = Thread.currentThread().getStackTrace();
144                 initializerClassName = name;
145             }
146         }
147     }

167         
168         // we should only detach if the sticky flag is not set
169         if (!module.isSticky()) {

183     protected Class<?> findClass(String name) throws ClassNotFoundException {
184         try {
185             Class c=null;
186             synchronized(facadeSurrogates) {
187                 for (ClassLoaderFacade classLoader : facadeSurrogates) {
188                     try {
189                         c = classLoader.getClass(name);
190                     } catch(ClassNotFoundException e) {
191                         // ignored.
192                     }
193                     if (c!=null) {
194                         return c;
195                     }
196                 }
197             }
198             synchronized(surrogates) {
199                 for (ClassLoader classLoader : surrogates) {
200                     try {
201                         c = classLoader.loadClass(name);
202                     } catch(ClassNotFoundException e) {
203                         // ignored.
204                     }
205                     if (c!=null) {
206                         return c;
207                     }
208                 }
209             }
210             return findClassDirect(name);
211         } catch (ClassNotFoundException e) {
212             // check if this for service loader punch-in, and if so, let it through
213             Module m = getOwner().getRegistry().getProvidingModule(name);
214             if(m!=null)
215                 return m.getPrivateClassLoader().findClassDirect(name);
216             throw e;
217         }
218     }

220     /***
221      * {@link #findClass(String)} except the classloader punch-in hack.
222      */
223     private Class findClassDirect(String name) throws ClassNotFoundException {
224         Class c = findLoadedClass(name);
225         if(c!=null) return c;
226         return super.findClass(name);
227     }

229     public URL findResource(String name) {
230         synchronized(facadeSurrogates) {
231             for (ClassLoaderFacade classLoader : facadeSurrogates) {
232 
233                 URL url = classLoader.findResource(name);
234                 if (url!=null) {
235                     return url;
236                 }
237             }
238         }
239         synchronized(surrogates) {
240             for (ClassLoader classLoader : surrogates) {
241 
242                 URL url = classLoader.getResource(name);
243                 if (url!=null) {
244                     return url;
245                 }
246             }
247         }
248         return super.findResource(name);
249         
250     }

252     public Enumeration<URL> findResources(String name) throws IOException {
253 
254         for (ClassLoaderFacade classLoader : facadeSurrogates) {
255             
256             Enumeration<URL> enumerat = classLoader.getResources(name);
257             if (enumerat!=null && enumerat.hasMoreElements()) {
258                 return enumerat;
259             }
260         }
261         for (ClassLoader classLoader : surrogates) {
262             Enumeration<URL> enumerat = classLoader.getResources(name);
263             if (enumerat!=null && enumerat.hasMoreElements()) {
264                 return enumerat;
265             }
266             
267         }
268         return super.findResources(name);
269     }

271     public void addDelegate(ClassLoader cl) {
272         if (cl instanceof ClassLoaderFacade) {
273             facadeSurrogates.add((ClassLoaderFacade) cl);
274         } else {
275             surrogates.add(cl);
276         }                    
277     }

279     public void removeDelegate(ClassLoader cl) {
280         if (cl instanceof ClassLoaderFacade) {
281             facadeSurrogates.remove(cl);
282         } else {
283             surrogates.remove(cl);
284         }
285     }

291     public String toString() {
292         StringBuffer s= new StringBuffer(); 
293         s.append(super.toString()).append("(name=").append(module.getName());
294         s.append(",init=").append(initialized);
295         s.append(",URls[]=");
296         for (URL url : getURLs()) {
297             s.append(url).append(",");
298         }
299         s.append(")");
300                
301         for (ClassLoader surrogate : surrogates) {
302             s.append("\n ref : ").append(surrogate.toString());
303         }
304         return s.toString();
305     }
