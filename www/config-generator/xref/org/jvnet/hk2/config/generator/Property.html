55  /***
56   * Represents configurable property of the component.
57   *
58   * @author Kohsuke Kawaguchi
59   */
60  abstract class Property {
61      /***
62       * Field/method declaration of this property.
63       * Used to read annotations on this property.
64       */
65      abstract MemberDeclaration decl();
66  
67      /***
68       * Name used as a seed for the XML name of this property.
69       * This is the property name / field name.
70       */
71      String seedName() {
72          return decl().getSimpleName();
73      };
74  
75      /***
76       * The type of the property.
77       */
78      abstract TypeMirror type();
79  
80      abstract void assign(JVar $target, JBlock block, JExpression rhs);
81  
82      <A extends Annotation> A getAnnotation(Class<A> a) {
83          return decl().getAnnotation(a);
84      }
85  
86      private Boolean isKey=null;
87      
88      /***
89       * Does this property have {@link Attribute#key()} or {@link Element#key()}.
90       */
91      final boolean isKey() {
92          if(isKey==null)
93              isKey = _isKey();
94          return isKey;
95      }
96  
97      private boolean _isKey() {
98          Element e = getAnnotation(Element.class);
99          if(e!=null && e.key())  return true;
100 
101         Attribute a = getAnnotation(Attribute.class);
102         if(a!=null && a.key())  return true;
103 
104         return false;
105     }
106 
107     String inferName(String name) {
108         if(name.length()==0)
109             name = Dom.convertName(seedName());
110         return name;
111     }
112 
113     /***
114      * Property that consists of a set/add/get method.
115      */
116     static final class Method extends Property {
117         final MethodDeclaration decl;
118         /***
119          * True if this property is based on the getter method. False if the setter/adder.
120          */
121         final boolean getter;
122 
123         public Method(MethodDeclaration decl) {
124             this.decl = decl;
125             getter = !decl.getReturnType().toString().equals("void");
126         }
127 
128         MemberDeclaration decl() {
129             return decl;
130         }
131 
132         TypeMirror type() {
133             if(getter)
134                 return decl.getReturnType();
135             else
136                 return decl.getParameters().iterator().next().getType();
137         }
138 
139         void assign(JVar $target, JBlock block, JExpression rhs) {
140             block.invoke($target,decl.getSimpleName()).arg(rhs);
141         }
142     }
143 
144     /***
145      * Property that consists of a field.
146      */
147     static final class Field extends Property {
148         final FieldDeclaration decl;
149 
150         public Field(FieldDeclaration decl) {
151             this.decl = decl;
152         }
153 
154         MemberDeclaration decl() {
155             return decl;
156         }
157 
158         TypeMirror type() {
159             return decl.getType();
160         }
161 
162         void assign(JVar $target, JBlock block, JExpression rhs) {
163             block.assign($target.ref(decl.getSimpleName()),rhs);
164         }
165     }
