60  public class ConfigGeneratorPlugin extends AbstractMojo {
61      private final static String GENERATED_SOURCES = "generated-sources/hk2-config-generator/src";
62      private final static String MAIN_NAME = "main";
63      private final static String TEST_NAME = "test";
64      private final static String JAVA_NAME = "java";

71      protected MavenProject project;

86      private File testSourceDirectory;

91      private File outputDirectory;

111     private String includes;

116     private String excludes;

118     /* (non-Javadoc)
119      * @see org.apache.maven.plugin.Mojo#execute()
120      */
121     private void internalExecute() throws Throwable {
122         List<String> projectTypes = Arrays.asList(supportedProjectTypes.split(","));
123         File srcDir = (test) ? testSourceDirectory : sourceDirectory;
124         if(!projectTypes.contains(project.getPackaging())
125                 || !srcDir.exists() 
126                 || !srcDir.isDirectory()){
127             return;
128         }
129         
130         // prepare the generated source directory
131         File generatedSources = new File(buildDirectory, GENERATED_SOURCES);
132         File phaseGeneratedFile = (test) ? new File(generatedSources, TEST_NAME) : new File(generatedSources, MAIN_NAME);
133         File javaGeneratedFile = new File(phaseGeneratedFile, JAVA_NAME);
134         javaGeneratedFile.mkdirs();
135         
136         // prepare command line arguments
137         LinkedList<String> args = new LinkedList<String>();
138         args.add("-nocompile");
139         args.add("-s");
140         args.add(javaGeneratedFile.getAbsolutePath());
141         args.add("-cp");
142         args.add(getBuildClasspath());
143         args.addAll(FileUtils.getFileNames(srcDir, includes, excludes,true));
144         
145         String[] cmdLine = args.toArray(new String[args.size()]);
146         if(verbose){
147             getLog().info("");
148             getLog().info("-- Apt Command Line --");
149             getLog().info("");
150             getLog().info(Arrays.toString(cmdLine));
151             getLog().info("");
152         }
153         Main.process(new AnnotationProcessorFactoryImpl(), cmdLine);
154         
155         // make the generated source directory visible for compilation
156         if(test){
157             project.addTestCompileSourceRoot(javaGeneratedFile.getAbsolutePath());
158             if (getLog().isInfoEnabled()) {
159                 getLog().info("Test Source directory: " + javaGeneratedFile + " added.");
160             }            
161         } else {
162             project.addCompileSourceRoot(javaGeneratedFile.getAbsolutePath());
163             if (getLog().isInfoEnabled()) {
164                 getLog().info("Source directory: " + javaGeneratedFile + " added.");
165             }
166         }
167     }
168 
169     /* (non-Javadoc)
170      * @see org.apache.maven.plugin.Mojo#execute()
171      */
172     @Override
173     public void execute() throws MojoExecutionException, MojoFailureException {
174         try {
175           internalExecute();
176         }
177         catch (Throwable th) {
178             if (th instanceof MojoExecutionException) {
179                 throw (MojoExecutionException) th;
180             }
181             if (th instanceof MojoFailureException) {
182                 throw (MojoFailureException) th;
183             }
184             
185             Throwable cause = th;
186             int lcv = 0;
187             while (cause != null) {
188                 getLog().error("Exception from hk2-config-generator[" + lcv++ + "]=" + cause.getMessage());
189                 cause.printStackTrace();
190                 
191                 cause = cause.getCause();
192             }
193             
194             throw new MojoExecutionException(th.getMessage(), th);
195         }
196     }

198     private String getBuildClasspath() {
199         StringBuilder sb = new StringBuilder();
200         // Make sure to add in the directory that has been built
201         if (test) {
202             sb.append(outputDirectory.getAbsolutePath());
203             sb.append(File.pathSeparator);
204         }        
205         
206         List<Artifact> artList = new ArrayList<Artifact>(project.getArtifacts());
207         Iterator<Artifact> i = artList.iterator();
208         
209         if (i.hasNext()) {
210             sb.append(i.next().getFile().getPath());
211 
212             while (i.hasNext()) {
213                 sb.append(File.pathSeparator);
214                 sb.append(i.next().getFile().getPath());
215             }
216         }
217         
218         String classpath = sb.toString();
219         if(verbose){
220             getLog().info("");
221             getLog().info("-- Classpath --");
222             getLog().info("");
223             getLog().info(classpath);
224             getLog().info("");
225         }
226         return classpath;
227     }
