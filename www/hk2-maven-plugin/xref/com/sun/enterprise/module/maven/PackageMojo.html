16  package com.sun.enterprise.module.maven;
17  
18  import org.apache.maven.archiver.MavenArchiveConfiguration;
19  import org.apache.maven.archiver.MavenArchiver;
20  import org.apache.maven.plugin.AbstractMojo;
21  import org.apache.maven.plugin.MojoExecutionException;
22  import org.apache.maven.project.MavenProject;
23  import org.apache.maven.project.MavenProjectHelper;
24  import org.codehaus.plexus.archiver.jar.JarArchiver;
25  
26  import java.io.File;
27  
28  /***
29   * Creates a jar with a special manifest entry.
30   *
31   * <p>
32   * Mostly copied from JarMojo. See http://jira.codehaus.org/browse/MNG-2789
33   * why we can't rely on it.
34   *
35   * @author Kohsuke Kawaguchi
36   * @goal package
37   * @phase package
38   * @requiresProject
39   */
40  public class PackageMojo extends AbstractMojo {
41      private static final String[] DEFAULT_EXCLUDES = new String[]{"**/package.html"};
42  
43      private static final String[] DEFAULT_INCLUDES = new String[]{"**/**"};
44  
45      /***
46       * Directory containing the generated JAR.
47       *
48       * @parameter expression="${project.build.directory}"
49       * @required
50       */
51      protected File outputDirectory;
52  
53      /***
54       * Name of the generated JAR.
55       *
56       * @parameter alias="jarName" expression="${project.build.finalName}"
57       * @required
58       */
59      protected String finalName;
60  
61      /***
62       * The Jar archiver.
63       *
64       * @parameter expression="${component.org.codehaus.plexus.archiver.Archiver#jar}"
65       * @required
66       */
67      protected JarArchiver jarArchiver;
68  
69      /***
70       * The maven project.
71       *
72       * @parameter expression="${project}"
73       * @required
74       * @readonly
75       */
76      protected MavenProject project;
77  
78      /***
79       * The maven archive configuration to use.
80       * <p/>
81       * See <a href="http://maven.apache.org/ref/current/maven-archiver/apidocs/org/apache/maven/archiver/MavenArchiveConfiguration.html">the Javadocs for MavenArchiveConfiguration</a>.
82       *
83       * @parameter
84       */
85      protected MavenArchiveConfiguration archive = new MavenArchiveConfiguration();
86  
87      /***
88       * @component
89       */
90      protected MavenProjectHelper projectHelper;
91  
92      /***
93       * Directory containing the classes.
94       *
95       * @parameter expression="${project.build.outputDirectory}"
96       * @required
97       */
98      protected File classesDirectory;
99  
100     protected final MavenProject getProject() {
101         return project;
102     }
103 
104     protected static File getJarFile(File basedir, String finalName) {
105         return new File(basedir, finalName + ".jar");
106     }
107 
108     /***
109      * Generates the JAR.
110      *
111      * @todo Add license files in META-INF directory.
112      */
113     public File createArchive()
114         throws MojoExecutionException {
115         File jarFile = getJarFile(outputDirectory, finalName);
116 
117         MavenArchiver archiver = new MavenArchiver();
118 
119         archiver.setArchiver(jarArchiver);
120 
121         archiver.setOutputFile(jarFile);
122 
123         try {
124             new Packager().configureManifet(project,archive,classesDirectory);
125 
126             File contentDirectory = classesDirectory;
127             if (!contentDirectory.exists()) {
128                 getLog().warn("JAR will be empty - no content was marked for inclusion!");
129             } else {
130                 archiver.getArchiver().addDirectory(contentDirectory, DEFAULT_INCLUDES, DEFAULT_EXCLUDES);
131             }
132 
133             archiver.createArchive(project, archive);
134 
135             return jarFile;
136         } catch (Exception e) {
137             // TODO: improve error handling
138             throw new MojoExecutionException("Error assembling JAR", e);
139         }
140     }
141 
142     /***
143      * Generates the JAR.
144      *
145      * @todo Add license files in META-INF directory.
146      */
147     public void execute() throws MojoExecutionException {
148         File jarFile = createArchive();
149 
150         getProject().getArtifact().setFile(jarFile);
151     }
152 }
