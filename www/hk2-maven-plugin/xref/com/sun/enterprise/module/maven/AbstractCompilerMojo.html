244     public void execute()
245         throws MojoExecutionException, CompilationFailureException
246     {
247         // ----------------------------------------------------------------------
248         // Look up the compiler. This is done before other code than can
249         // cause the mojo to return before the lookup is done possibly resulting
250         // in misconfigured POMs still building.
251         // ----------------------------------------------------------------------
252 
253         Compiler compiler;
254 
255         getLog().debug( "Using compiler '" + compilerId + "'." );
256 
257         try
258         {
259             compiler = compilerManager.getCompiler( compilerId );
260         }
261         catch ( NoSuchCompilerException e )
262         {
263             throw new MojoExecutionException( "No such compiler '" + e.getCompilerId() + "'." );
264         }
265 
266         // ----------------------------------------------------------------------
267         //
268         // ----------------------------------------------------------------------
269 
270         List compileSourceRoots = removeEmptyCompileSourceRoots( getCompileSourceRoots() );
271 
272         if ( compileSourceRoots.isEmpty() )
273         {
274             getLog().info( "No sources to compile" );
275 
276             return;
277         }
278 
279         if ( getLog().isDebugEnabled() )
280         {
281             getLog().debug( "Source directories: " + compileSourceRoots.toString().replace( ',', '\n' ) );
282             getLog().debug( "Classpath: " + getClasspathElements().toString().replace( ',', '\n' ) );
283             getLog().debug( "Output directory: " + getOutputDirectory() );
284         }
285 
286         // ----------------------------------------------------------------------
287         // Create the compiler configuration
288         // ----------------------------------------------------------------------
289 
290         CompilerConfiguration compilerConfiguration = new CompilerConfiguration();
291 
292         compilerConfiguration.setOutputLocation( getOutputDirectory().getAbsolutePath() );
293 
294         compilerConfiguration.setClasspathEntries( getClasspathElements() );
295 
296         compilerConfiguration.setSourceLocations( compileSourceRoots );
297 
298         compilerConfiguration.setOptimize( optimize );
299 
300         compilerConfiguration.setDebug( debug );
301 
302         compilerConfiguration.setVerbose( verbose );
303 
304         compilerConfiguration.setShowWarnings( showWarnings );
305 
306         compilerConfiguration.setShowDeprecation( showDeprecation );
307 
308         compilerConfiguration.setSourceVersion( source );
309 
310         compilerConfiguration.setTargetVersion( target );
311 
312         compilerConfiguration.setSourceEncoding( encoding );
313 
314         if (( compilerArguments != null ) || ( compilerArgument != null ))
315         {
316             LinkedHashMap cplrArgsCopy = new LinkedHashMap();
317             if ( compilerArguments != null )
318             {
319                 for ( Iterator i = compilerArguments.entrySet().iterator(); i.hasNext(); )
320                 {
321                     Map.Entry me = (Map.Entry) i.next();
322                     String key = (String) me.getKey();
323                     String value = (String) me.getValue();
324                     if ( !key.startsWith( "-" ))
325                     {
326                         key = "-" + key;
327                     }
328                     cplrArgsCopy.put( key, value );
329                 }
330             }
331             if ( !StringUtils.isEmpty( compilerArgument) )
332             {
333                 cplrArgsCopy.put( compilerArgument, null );
334             }
335             compilerConfiguration.setCustomCompilerArguments( cplrArgsCopy );
336         }
337 
338         compilerConfiguration.setFork( fork );
339 
340         if( fork )
341         {
342             if ( !StringUtils.isEmpty( meminitial ) )
343             {
344                 String value = getMemoryValue( meminitial );
345 
346                 if ( value != null )
347                 {
348                     compilerConfiguration.setMeminitial( value );
349                 }
350                 else
351                 {
352                     getLog().info( "Invalid value for meminitial '" + meminitial + "'. Ignoring this option." );
353                 }
354             }
355 
356             if ( !StringUtils.isEmpty( maxmem ) )
357             {
358                 String value = getMemoryValue( maxmem );
359 
360                 if ( value != null )
361                 {
362                     compilerConfiguration.setMaxmem( value );
363                 }
364                 else
365                 {
366                     getLog().info( "Invalid value for maxmem '" + maxmem + "'. Ignoring this option." );
367                 }
368             }
369         }
370 
371         compilerConfiguration.setExecutable( executable );
372 
373         compilerConfiguration.setWorkingDirectory( basedir );
374 
375         compilerConfiguration.setCompilerVersion( compilerVersion );
376 
377         compilerConfiguration.setBuildDirectory( buildDirectory );
378 
379         compilerConfiguration.setOutputFileName( outputFileName );
380 
381         // TODO: have an option to always compile (without need to clean)
382         Set staleSources;
383 
384         boolean canUpdateTarget;
385 
386         try
387         {
388             staleSources =
389                 computeStaleSources( compilerConfiguration, compiler, getSourceInclusionScanner( staleMillis ) );
390 
391             canUpdateTarget = compiler.canUpdateTarget( compilerConfiguration );
392 
393             if ( compiler.getCompilerOutputStyle().equals( CompilerOutputStyle.ONE_OUTPUT_FILE_FOR_ALL_INPUT_FILES ) &&
394                 !canUpdateTarget )
395             {
396                 getLog().info( "RESCANNING!" );
397                 // TODO: This second scan for source files is sub-optimal
398                 String inputFileEnding = compiler.getInputFileEnding( compilerConfiguration );
399 
400                 Set sources = computeStaleSources( compilerConfiguration, compiler,
401                                                    getSourceInclusionScanner( inputFileEnding ) );
402 
403                 compilerConfiguration.setSourceFiles( sources );
404             }
405             else
406             {
407                 compilerConfiguration.setSourceFiles( staleSources );
408             }
409         }
410         catch ( CompilerException e )
411         {
412             throw new MojoExecutionException( "Error while computing stale sources.", e );
413         }
414 
415         if ( staleSources.isEmpty() )
416         {
417             getLog().info( "Nothing to compile - all classes are up to date" );
418 
419             return;
420         }
421 
422         // ----------------------------------------------------------------------
423         // Dump configuration
424         // ----------------------------------------------------------------------
425 
426         if ( getLog().isDebugEnabled() )
427         {
428             getLog().debug( "Classpath:" );
429 
430             for ( Iterator it = getClasspathElements().iterator(); it.hasNext(); )
431             {
432                 String s = (String) it.next();
433 
434                 getLog().debug( " " + s );
435             }
436 
437             getLog().debug( "Source roots:" );
438 
439             for ( Iterator it = getCompileSourceRoots().iterator(); it.hasNext(); )
440             {
441                 String root = (String) it.next();
442 
443                 getLog().debug( " " + root );
444             }
445 
446             if ( fork )
447             {
448                 try
449                 {
450                     if ( compilerConfiguration.getExecutable() != null )
451                     {
452                         getLog().debug( "Excutable: " );
453                         getLog().debug( " " + compilerConfiguration.getExecutable() );
454                     }
455 
456                     String[] cl = compiler.createCommandLine( compilerConfiguration );
457                     if ( cl != null && cl.length > 0 )
458                     {
459                         StringBuffer sb = new StringBuffer();
460                         sb.append( cl[0] );
461                         for ( int i = 1; i < cl.length; i++ )
462                         {
463                             sb.append( " " );
464                             sb.append( cl[i] );
465                         }
466                         getLog().debug( "Command line options:" );
467                         getLog().debug( sb );
468                     }
469                 }
470                 catch ( CompilerException ce )
471                 {
472                     getLog().debug( ce );
473                 }
474             }
475         }
476 
477         // ----------------------------------------------------------------------
478         // Compile!
479         // ----------------------------------------------------------------------
480 
481         List messages;
482 
483         try
484         {
485             messages = compiler.compile( compilerConfiguration );
486         }
487         catch ( Exception e )
488         {
489             // TODO: don't catch Exception
490             throw new MojoExecutionException( "Fatal error compiling", e );
491         }
492 
493         boolean compilationError = false;
494 
495         for ( Iterator i = messages.iterator(); i.hasNext(); )
496         {
497             CompilerError message = (CompilerError) i.next();
498 
499             if ( message.isError() )
500             {
501                 compilationError = true;
502             }
503         }
504 
505         if ( compilationError )
506         {
507             throw new CompilationFailureException( messages );
508         }
509         else
510         {
511             for ( Iterator i = messages.iterator(); i.hasNext(); )
512             {
513                 CompilerError message = (CompilerError) i.next();
514 
515                 getLog().warn( message.toString() );
516             }
517         }
518     }

520     private String getMemoryValue( String setting )
521     {
522         String value = null;
523 
524         // Allow '128' or '128m'
525         if ( isDigits( setting ) )
526         {
527             value = setting + "m";
528         }
529         else
530         {
531             if ( ( isDigits( setting.substring( 0, setting.length() - 1 ) ) ) &&
532                 ( setting.toLowerCase().endsWith( "m" ) ) )
533             {
534                 value = setting;
535             }
536         }
537         return value;
538     }

540     private boolean isDigits( String string )
541     {
542         for ( int i = 0; i < string.length(); i++ )
543         {
544             if ( !Character.isDigit( string.charAt( i ) ) )
545             {
546                 return false;
547             }
548         }
549         return true;
550     }

552     private Set computeStaleSources( CompilerConfiguration compilerConfiguration, Compiler compiler,
553                                      SourceInclusionScanner scanner )
554         throws MojoExecutionException, CompilerException
555     {
556         CompilerOutputStyle outputStyle = compiler.getCompilerOutputStyle();
557 
558         SourceMapping mapping;
559 
560         File outputDirectory;
561 
562         if ( outputStyle == CompilerOutputStyle.ONE_OUTPUT_FILE_PER_INPUT_FILE )
563         {
564             mapping = new SuffixMapping( compiler.getInputFileEnding( compilerConfiguration ), compiler
565                 .getOutputFileEnding( compilerConfiguration ) );
566 
567             outputDirectory = getOutputDirectory();
568         }
569         else if ( outputStyle == CompilerOutputStyle.ONE_OUTPUT_FILE_FOR_ALL_INPUT_FILES )
570         {
571             mapping = new SingleTargetSourceMapping( compiler.getInputFileEnding( compilerConfiguration ), compiler
572                 .getOutputFile( compilerConfiguration ) );
573 
574             outputDirectory = buildDirectory;
575         }
576         else
577         {
578             throw new MojoExecutionException( "Unknown compiler output style: '" + outputStyle + "'." );
579         }
580 
581         scanner.addSourceMapping( mapping );
582 
583         Set staleSources = new HashSet();
584 
585         for ( Iterator it = getCompileSourceRoots().iterator(); it.hasNext(); )
586         {
587             String sourceRoot = (String) it.next();
588 
589             File rootFile = new File( sourceRoot );
590 
591             if ( !rootFile.isDirectory() )
592             {
593                 continue;
594             }
595 
596             try
597             {
598                 staleSources.addAll( scanner.getIncludedSources( rootFile, outputDirectory ) );
599             }
600             catch ( InclusionScanException e )
601             {
602                 throw new MojoExecutionException(
603                     "Error scanning source root: \'" + sourceRoot + "\' " + "for stale files to recompile.", e );
604             }
605         }
606 
607         return staleSources;
608     }

610     /***
611      * @todo also in ant plugin. This should be resolved at some point so that it does not need to
612      * be calculated continuously - or should the plugins accept empty source roots as is?
613      */
614     private static List removeEmptyCompileSourceRoots( List compileSourceRootsList )
615     {
616         List newCompileSourceRootsList = new ArrayList();
617         if ( compileSourceRootsList != null )
618         {
619             // copy as I may be modifying it
620             for ( Iterator i = compileSourceRootsList.iterator(); i.hasNext(); )
621             {
622                 String srcDir = (String) i.next();
623                 if ( !newCompileSourceRootsList.contains( srcDir ) && new File( srcDir ).exists() )
624                 {
625                     newCompileSourceRootsList.add( srcDir );
626                 }
627             }
628         }
629         return newCompileSourceRootsList;
630     }
