72      public void execute() throws MojoExecutionException, MojoFailureException {
73  
74          // sanity check
75          if(project.getPackaging().equals("hk2-jar") && project.getArtifact()==null) {
76              getLog().warn("This project isn't compiled yet. Perhaps you meant 'mvn compile hk2:run'?");
77          }
78  
79          try {
80              new Main().launch(createModuleRegistry(),rootDir,args);
81          } catch (BootException e) {
82              throw new MojoExecutionException("Failed to boot up the module system",e);
83          } catch (IOException e) {
84              throw new MojoExecutionException("Failed to boot up the module system",e);
85          }
86      }
87  
88      /***
89       * Creates a fully configured module registry.
90       */
91      protected ModulesRegistry createModuleRegistry() throws IOException {
92          ModulesRegistry r = ModulesRegistry.createRegistry();
93          r.setParentClassLoader(this.getClass().getClassLoader());
94          MavenProjectRepository lib = new MavenProjectRepository(project,artifactResolver,localRepository,artifactFactory);
95          r.addRepository(lib);
96          lib.initialize();
97          return r;
98      }
