73  /**
74   * Replacement of the default javac mojo that uses APT, so that we can not only
75   * compile source code but also generate HK2 related metadata at the same time.
76   *
77   * @goal hk2-compile
78   * @phase compile
79   * @author Kohsuke Kawaguchi
80   */
81  public class HK2CompileMojo extends CompilerMojo {
82      /**
83       * @parameter expression="${project}"
84       * @required
85       * @readonly
86       */
87      private MavenProject project;
88  
89      /**
90       * Additional jars that contain annotation processors.
91       *
92       * @parameter
93       */
94      public Processor[] processors;
95  
96      /**
97       * @component
98       */
99      private ArtifactFactory artifactFactory;
100 
101     /**
102      * @component
103      */
104     private ArtifactResolver artifactResolver;
105 
106     /**
107      * @parameter expression="${localRepository}"
108      * @required
109      */
110     private ArtifactRepository localRepository;
111 
112     /**
113      * @component
114      */
115     private ArtifactMetadataSource metadataSource;
116 
117     public void execute() throws MojoExecutionException, CompilationFailureException {
118         // overwrite the compilerId value. This seems to be the only way to
119         //do so without touching the copied files.
120         setField("compilerId", "hk2-apt");
121 
122         // invoke APT with the known set of annotation processors that we care about.
123         AptInvoker old = AptInvoker.replace(new AptInvoker() {
124             List<CompilerError> compileInProcess(String[] args) throws CompilerException {
125                 com.sun.tools.apt.Main aptTool = new com.sun.tools.apt.Main();
126                 int r = aptTool.process(
127                     new CompositeAnnotationProcessorFactory(
128                         new MetainfServiceGenerator(),
129                         new AnnotationProcessorFactoryImpl(),
130                         loadAdditionalAnnotationProcessors()
131                     ), new PrintWriter(System.out,true),args);
132                 if(r!=0)
133                     throw new CompilerException("APT failed: "+r);
134 
135                 // TODO: should I try to parse the output?
136                 return Collections.emptyList();
137             }
138         });
139         try {
140             super.execute();
141         } finally {
142             AptInvoker.replace(old);
143         }
144 
145         // TODO: ideally we should do this in the AptCompiler class, but I don't know
146         // how to get MavenProject injected there
147         project.getCompileSourceRoots().add(new File(project.getBasedir(),"target/apt-generated-sources").getAbsolutePath());
148     }
149 
150     /**
151      * Loads additional annotation processors specified in &lt;processor> configuration element.
152      *
153      * <p>
154      * We need to resolve their transitive dependencies as well, because otherwise they won't load.
155      */
156     private AnnotationProcessorFactory loadAdditionalAnnotationProcessors() throws CompilerException {
157         if(processors==null)
158             return new CompositeAnnotationProcessorFactory();   // none
159 
160         Set<Artifact> processorArtifacts = new HashSet<Artifact>();
161         for (Processor p : processors) {
162             processorArtifacts.add(p.createArtifact(artifactFactory));
163         }
164 
165         // TODO: perhaps we should create one URLClassLoader per each annotation processor to
166         // ensure sufficient isolation?
167         Set<URL> classpaths = new LinkedHashSet<URL>();
168         try {
169             ArtifactResolutionResult result = artifactResolver.resolveTransitively(
170                     processorArtifacts, project.getArtifact(),
171                     localRepository,
172                     this.project.getRemoteArtifactRepositories(),
173                     metadataSource, new ScopeArtifactFilter("runtime"));
174             for( Artifact a : (Set<Artifact>)result.getArtifacts()) {
175                 classpaths.add(a.getFile().toURL());
176             }
177         } catch (AbstractArtifactResolutionException e) {
178             throw new CompilerException("Failed to resolve annotation processors",e);
179         } catch (MalformedURLException e) {
180             throw new CompilerException("Failed to resolve annotation processors",e);
181         }
182 
183         // load them into the classloader
184         ClassLoader cl = new URLClassLoader(classpaths.toArray(new URL[classpaths.size()]),getClass().getClassLoader());
185 
186         List<AnnotationProcessorFactory> factories = new ArrayList<AnnotationProcessorFactory>();
187         for(AnnotationProcessorFactory apf : ServiceFinder.find(AnnotationProcessorFactory.class,cl)) {
188             factories.add(apf);
189             getLog().info("Picked up annotation processor "+apf.getClass().getName());
190         }
191 
192         return new CompositeAnnotationProcessorFactory(factories);
193     }

195     private void setField(String name, String value) {
196         try {
197             Field field = AbstractCompilerMojo.class.getDeclaredField(name);
198             field.setAccessible(true);
199             field.set(this, value);
200         } catch (NoSuchFieldException e) {
201             throw new AssertionError(e); // impossible
202         } catch (IllegalAccessException e) {
203             throw new AssertionError(e); // impossible
204         }
205     }
