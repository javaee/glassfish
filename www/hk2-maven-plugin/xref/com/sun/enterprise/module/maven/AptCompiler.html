37  package com.sun.enterprise.module.maven;
38  
39  import org.codehaus.plexus.compiler.CompilerConfiguration;
40  import org.codehaus.plexus.compiler.CompilerException;
41  import org.codehaus.plexus.compiler.javac.JavacCompiler;
42  
43  import java.io.File;
44  import java.util.Collections;
45  import java.util.List;
46  
47  /***
48   * {@link Compiler} for APT.
49   *
50   * <p>
51   * In Maven, {@link Compiler} handles the actual compiler invocation.
52   *
53   * @author Kohsuke Kawaguchi
54   */
55  public class AptCompiler extends JavacCompiler {
56  
57      public List compile( CompilerConfiguration config ) throws CompilerException {
58          // force 1.5
59          config.setTargetVersion("1.5");
60          config.setSourceVersion("1.5");
61  
62  
63          File destinationDir = new File( config.getOutputLocation() );
64  
65          if ( !destinationDir.exists() )
66          {
67              destinationDir.mkdirs();
68          }
69  
70          String[] sourceFiles = getSourceFiles( config );
71  
72          if ( sourceFiles.length == 0 )
73          {
74              return Collections.EMPTY_LIST;
75          }
76  
77          getLogger().info( "Compiling " + sourceFiles.length + " " +
78                            "source file" + ( sourceFiles.length == 1 ? "" : "s" ) +
79                            " to " + destinationDir.getAbsolutePath() );
80  
81          /*
82              Source path pitfall.
83  
84              When APT goes multiple rounds, the files compiled in the previous round
85              needs to be made available to the next round. APT does this internally
86              by adding '-sourcepath' option, but since Maven only passes in the list
87              of individual files to be compiled, this simply doesn't work.
88  
89              So here I'm manually constructing -sourcepath option and pass that on to APT.
90           */
91          String sourcePath = "";
92          for (Object o : config.getSourceLocations()) {
93              sourcePath += o+File.pathSeparator;
94          }
95          config.addCompilerCustomArgument("-sourcepath",sourcePath);
96  
97          // keep the generated source files on the side so that they can be bundled into sources-jar.
98          File sourceDest = new File(config.getBuildDirectory(), "apt-generated-sources");
99          sourceDest.mkdirs();
100         config.addCompilerCustomArgument("-s", sourceDest.getAbsolutePath());
101         String[] args = buildCompilerArguments( config, sourceFiles );
102 
103         return AptInvoker.get().compileInProcess(args);
104     }
105 }
