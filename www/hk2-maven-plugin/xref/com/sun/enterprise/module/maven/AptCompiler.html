40  package com.sun.enterprise.module.maven;
41  
42  import org.codehaus.plexus.compiler.CompilerConfiguration;
43  import org.codehaus.plexus.compiler.CompilerException;
44  import org.codehaus.plexus.compiler.javac.JavacCompiler;
45  
46  import java.io.File;
47  import java.util.Collections;
48  import java.util.List;
49  
50  /**
51   * {@link Compiler} for APT.
52   *
53   * <p>
54   * In Maven, {@link Compiler} handles the actual compiler invocation.
55   *
56   * @author Kohsuke Kawaguchi
57   */
58  public class AptCompiler extends JavacCompiler {
59  
60      public List compile( CompilerConfiguration config ) throws CompilerException {
61          // force 1.5
62          config.setTargetVersion("1.5");
63          config.setSourceVersion("1.5");
64  
65  
66          File destinationDir = new File( config.getOutputLocation() );
67  
68          if ( !destinationDir.exists() )
69          {
70              destinationDir.mkdirs();
71          }
72  
73          String[] sourceFiles = getSourceFiles( config );
74  
75          if ( sourceFiles.length == 0 )
76          {
77              return Collections.EMPTY_LIST;
78          }
79  
80          getLogger().info( "Compiling " + sourceFiles.length + " " +
81                            "source file" + ( sourceFiles.length == 1 ? "" : "s" ) +
82                            " to " + destinationDir.getAbsolutePath() );
83  
84          /*
85              Source path pitfall.
86  
87              When APT goes multiple rounds, the files compiled in the previous round
88              needs to be made available to the next round. APT does this internally
89              by adding '-sourcepath' option, but since Maven only passes in the list
90              of individual files to be compiled, this simply doesn't work.
91  
92              So here I'm manually constructing -sourcepath option and pass that on to APT.
93           */
94          String sourcePath = "";
95          for (Object o : config.getSourceLocations()) {
96              sourcePath += o+File.pathSeparator;
97          }
98          config.addCompilerCustomArgument("-sourcepath",sourcePath);
99  
100         // keep the generated source files on the side so that they can be bundled into sources-jar.
101         File sourceDest = new File(config.getBuildDirectory(), "apt-generated-sources");
102         sourceDest.mkdirs();
103         config.addCompilerCustomArgument("-s", sourceDest.getAbsolutePath());
104         String[] args = buildCompilerArguments( config, sourceFiles );
105 
106         return AptInvoker.get().compileInProcess(args);
107     }
108 }
