40  package com.sun.enterprise.module;
41  
42  import java.io.PrintStream;
43  import java.util.List;
44  
45  /**
46   * @author Sanjeeb.Sahoo@Sun.COM
47   */
48  public interface Module {
49      /**
50       * Returns the module definition for this module instance
51       * @return the module definition
52       */
53      ModuleDefinition getModuleDefinition();
54  
55      /**
56       * Short-cut for {@code getModuleDefinition().getName()}.
57       */
58      String getName();
59  
60      /**
61       * Returns the registry owning this module
62       * @return the registry owning the module
63       */
64      ModulesRegistry getRegistry();
65  
66      /**
67       * Returns the module's state
68       * @return the module's state
69       */
70      ModuleState getState();
71  
72      /**
73       * Ensure that this module is {@link ModuleState#RESOLVED resolved}.
74       *
75       * <p>
76       * If the module is already resolved, this method does nothing.
77       * Otherwise, iterate over all declared ModuleDependency instances and use the
78       * associated <code>ModuleRegistry</code> to resolve it. After successful
79       * completion of this method, the module state is
80       * {@link ModuleState#RESOLVED}.
81       *
82       * @throws ResolveError if any of the declared dependency of this module
83       * cannot be satisfied
84       */
85      void resolve() throws ResolveError;
86  
87      /**
88       * Forces module startup. In most cases, the runtime will take care
89       * of starting modules when they are first used. There could be cases where
90       * code need to manually start a sub module. Invoking this method will
91       * move the module to the {@link ModuleState#READY ModuleState.READY}, the
92       * {@link LifecyclePolicy#start Lifecycle.start} method will be invoked.
93       */
94      void start() throws ResolveError;
95  
96      /**
97       * Forces module stop. In most cases, the runtime will take care of stopping
98       * modules when the last module user released its interest. However, in
99       * certain cases, it may be interesting to manually stop the module.
100      * Stopping the module means that the module is removed from the registry,
101      * the class loader references are released (note : the class loaders will
102      * only be released if all instances of any class loaded by them are gc'ed).
103      * If a <code>LifecyclePolicy</code> for this module is defined, the
104      * {@link LifecyclePolicy#stop(Module) Lifecycle.stop(Module)}
105      * method will be called and finally the module state will be
106      * returned to {@link ModuleState#NEW ModuleState.NEW}.
107      *
108      * @return true if unloading was successful
109      */
110     boolean stop();
111 
112     /**
113      * Detach this module from its registry. This does not free any of the
114      * loaded resources. Only proper release of all references to the public
115      * class loader will ensure module being garbage collected.
116      * Detached modules are orphan and will be garbage collected if resources
117      * are properly disposed.
118      */
119     void detach();
120 
121     /**
122      * Trigger manual refresh mechanism, the module will check all its
123      * URLs and generate change events if any of them has changed. This
124      * will allow the owning registry to force a module upgrade at next
125      * module request.
126      */
127     void refresh();
128 
129     /**
130      * Gets the metadata of this module.
131      */
132     ModuleMetadata getMetadata();
133 
134     /**
135      * Add a new module change listener
136      * @param listener the listener
137      */
138     void addListener(ModuleChangeListener listener);
139 
140     /**
141      * Unregister a module change listener
142      * @param listener the listener to unregister
143      */
144     void removeListener(ModuleChangeListener listener);
145 
146     /**
147      * Return the <code>ClassLoader</code>  instance associated with this module.
148      * Only designated public interfaces will be loaded and returned by
149      * this classloader
150      * @return the public <code>ClassLoader</code>
151      */
152     ClassLoader getClassLoader();
153 
154     /**
155      * Returns the list of imported modules.
156      *
157      * <p>
158      * This is the module version of {@link ModuleDefinition#getDependencies()},
159      * but after fully resolved.
160      *
161      * <p>
162      * To enforce the stable class visibility, once {@link Module} is
163      * created, dependencies cannot be changed &mdash; that is, we
164      * don't want "a.b.C" to suddenly mean something different once
165      * the code starts running.
166      *
167      * @return
168      *      Can be empty but never null. Read-only.
169      */
170     List<Module> getImports();
171 
172     void addImport(Module module);
173 
174     /**
175      * Create and add a new module to this module's list of
176      * imports.
177      * @param dependency new module's definition
178      */
179     Module addImport(ModuleDependency dependency);
180 
181     /**
182      * Returns true if this module is sharable. A sharable module means that
183      * onlu one instance of the module classloader will be used by all users.
184      *
185      * @return true if this module is sharable.
186      */
187     boolean isShared();
188 
189     /**
190      * Returns true if the module is sticky. A sticky module cannot be stopped or
191      * unloaded. Once a sticky module is loaded or started, it will stay in the
192      * JVM until it exists.
193      * @return true is the module is sticky
194      */
195     boolean isSticky();
196 
197     /**
198      * Sets the sticky flag.
199      * @param sticky true if the module should stick around
200      */
201     void setSticky(boolean sticky);
202 
203     <T> Iterable<Class<? extends T>> getProvidersClass(Class<T> serviceClass);
204 
205     Iterable<Class> getProvidersClass(String name);
206 
207     /**
208      * Returns true if this module has any provider for the given service class.
209      */
210     boolean hasProvider(Class serviceClass);
211 
212     void dumpState(PrintStream writer);
213 
214     /**
215      * Removes the module from the registry backing store, the module will need
216      * be reinstalled to be loaded.
217      */
218     void uninstall();
219 }
