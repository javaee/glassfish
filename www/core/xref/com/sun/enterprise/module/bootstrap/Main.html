131 	/**
132 	 * Start the server from the command line
133 	 * 
134 	 * @param args
135 	 *            the command line arguments
136 	 */
137 	public void start(String[] args) throws BootException {
138 		
139 		File bootstrap = this.getBootstrapFile();
140 		File root = bootstrap.getAbsoluteFile().getParentFile();
141 
142 		// root is the directory in which this bootstrap.jar is located
143 		// For most cases, this is the lib directory although this is completely
144 		// dependent on the usage of this facility.
145 		if (root == null) {
146 			throw new BootException("Cannot find root installation from "
147 					+ bootstrap);
148 		}
149 
150         // get the ModuleStartup implementation.
151         ModulesRegistry mr = AbstractFactory.getInstance().createModulesRegistry();
152 		StartupContext context = new StartupContext(
153 				ArgumentManager.argsToMap(args));
154 		launch(mr, context.getPlatformMainServiceName(), context);
155 	}

233 	public ServiceLocator createServiceLocator(ModulesRegistry mr,
234                                                StartupContext context,
235                                                List<PopulatorPostProcessor> postProcessors,
236                                                DescriptorFileFinder descriptorFileFinder)
237 			throws BootException {
238         // Why is this method not just delagting to ModulesRegistry.createServiceLocator(...)?
239         // That will require changes to ModulesRegistry.createServiceLocator() to accept
240         // postProcessors and desriptorFileFinder, but that's OK IMO.
241 
242 		// set the parent class loader before we start loading modules
243 		defineParentClassLoader();
244 
245         ServiceLocator serviceLocator = mr.newServiceLocator();
246 		
247 		addDescriptorFileFinder(serviceLocator, descriptorFileFinder);
248 		
249 		DynamicConfigurationService dcs = serviceLocator
250 				.getService(DynamicConfigurationService.class);
251 		
252 		DynamicConfiguration config = dcs.createDynamicConfiguration();
253 		
254 		config.addActiveDescriptor(BuilderHelper
255 				.createConstantDescriptor(context));
256 
257 		config.commit();
258 
259 		final ClassLoader oldCL = AccessController
260 				.doPrivileged(new PrivilegedAction<ClassLoader>() {
261 					@Override
262 					public ClassLoader run() {
263 						ClassLoader cl = Thread.currentThread()
264 								.getContextClassLoader();
265 						Thread.currentThread().setContextClassLoader(
266 								getClass().getClassLoader());
267 						return cl;
268 					}
269 				});
270 
271 		try {
272             mr.populateServiceLocator(DEFAULT_NAME, serviceLocator, postProcessors);
273             mr.populateConfig(serviceLocator);
274 		} finally {
275 			AccessController.doPrivileged(new PrivilegedAction<Object>() {
276 				@Override
277 				public Object run() {
278 					Thread.currentThread().setContextClassLoader(oldCL);
279 					return null;
280 				}
281 			});
282 		}
283 		return serviceLocator;
284 	}

296 	private void addDescriptorFileFinder(
297             ServiceLocator serviceLocator, DescriptorFileFinder descriptorFileFinder) {
298 		
299 		if (descriptorFileFinder != null) {
300 			DynamicConfigurationService dcs = serviceLocator
301 					.getService(DynamicConfigurationService.class);
302 			DynamicConfiguration config = dcs.createDynamicConfiguration();
303 
304 			config.addActiveDescriptor(BuilderHelper
305 					.createConstantDescriptor(descriptorFileFinder));
306 
307 			config.commit();
308 		}

311 	/**
312 	 * This filter matches against the name, including only
313 	 * matching a ModuleStartup with no name if name is
314 	 * null (unlike a normal "null" returned from name, which
315 	 * acts as a wildcard for the name)
316 	 * 
317 	 * @author jwells
318 	 *
319 	 */
320 	private static class MainFilter implements IndexedFilter {
321 	    private final String name;
322 	    
323 	    /**
324 	     * The name given here will match the ModuleStartup
325 	     * 
326 	     * @param name The name of the ModuleStartup to find.  If
327 	     * null this Filter will only match ModuleStartups with
328 	     * no name
329 	     */
330 	    private MainFilter(String name) {
331 	        this.name = name;
332 	    }
333 
334         /* (non-Javadoc)
335          * @see org.glassfish.hk2.api.Filter#matches(org.glassfish.hk2.api.Descriptor)
336          */
337         @Override
338         public boolean matches(Descriptor d) {
339             if (name == null) {
340                 if (d.getName() == null) return true;
341                 
342                 return false;
343             }
344             
345             return true;
346         }
347 
348         /* (non-Javadoc)
349          * @see org.glassfish.hk2.api.IndexedFilter#getAdvertisedContract()
350          */
351         @Override
352         public String getAdvertisedContract() {
353             return ModuleStartup.class.getName();
354         }
355 
356         /* (non-Javadoc)
357          * @see org.glassfish.hk2.api.IndexedFilter#getName()
358          */
359         @Override
360         public String getName() {
361             return name;
362         }
