157         config.bind(BuilderHelper.createConstantDescriptor(this));

168     /**
169      * Creates a {@link ServiceLocator} from all the modules in this registry
170      *
171      * @param name
172      *      Determines which descriptors are loaded.
173      * @param postProcessors
174      */
175      public void populateServiceLocator(String name, ServiceLocator serviceLocator, List<PopulatorPostProcessor> postProcessors) throws MultiException {
176          try {
177              for (final Module module : getModules()) { 
178             	   // TODO: should get the inhabitantsParser out of Main instead since
179                  // this could have been overridden
180              	List<ActiveDescriptor> allDescriptors =
181              	        parseInhabitants(module, name, serviceLocator, postProcessors);
182              	if (allDescriptors == null) continue;
183              	if (allDescriptors.isEmpty()) continue;
184              	
185              	Map<ServiceLocator, List<ActiveDescriptor>> descriptorByServiceLocator = moduleDescriptors.get(module);
186              	if (descriptorByServiceLocator == null) {
187              	    descriptorByServiceLocator = new HashMap<ServiceLocator, List<ActiveDescriptor>>();
188              	    
189              	   moduleDescriptors.put(module, descriptorByServiceLocator);
190              	}
191              	
192              	List<ActiveDescriptor> foundDs = descriptorByServiceLocator.get(serviceLocator);
193              	if (foundDs == null) {
194              	    foundDs = new LinkedList<ActiveDescriptor>();
195              	    
196              	    descriptorByServiceLocator.put(serviceLocator, foundDs);
197              	}
198              	
199              	foundDs.addAll(allDescriptors);
200              }
201          } catch (Exception e) {
202              throw new MultiException(e);
203          }
204          // From now on, we will keep this service registry up-to-date with module system state
205          habitats.put(serviceLocator, name);
206      }

235     /**
236      * Add a new <code>Repository</code> to this registry. From now on
237      * the repository will be used to procure requested module not yet registered
238      * in this registry instance. Repository can be searched in a particular 
239      * order (to accomodate performance requirements like looking at local 
240      * repositories first), a search order (1 to 100) can be specified when 
241      * adding a repository to the registry (1 is highest priority). 
242      * @param repository new repository to attach to this registry
243      * @param weight int value from 1 to 100 to specify the search order
244      */
245     public synchronized void addRepository(Repository repository, int weight) {
246         // check that we don't already have this repository
247         for (Repository repo : repositories.values()) {
248             if (repo.getLocation().equals(repository.getLocation())) {
249                 throw new RuntimeException("repository at " + repository.getLocation() + " already registered");
250             }
251         }
252         while (repositories.containsKey(weight)) {
253             weight++;
254         }
255         repositories.put(weight, repository);
256     }

314     public Module makeModuleFor(String name, String version, boolean resolve) throws ResolveError {
315         Module module;
316         Logger.getAnonymousLogger().fine("this.makeModuleFor("+name+ ", " +
317                 version + ", " + resolve + ") called.");
318         if(parent!=null) {
319             module = parent.makeModuleFor(name,version, resolve);
320             if(module!=null)        return module;
321         }
322         
323         module = modules.get(AbstractFactory.getInstance().createModuleId(name, version));
324         if (module == null && version == null) {
325             Collection<Module> matchingModules = getModules(name);
326             if (!matchingModules.isEmpty()) {
327                 module = matchingModules.iterator().next();
328             }
329         }
330         if (module==null) {
331             module = loadFromRepository(name, version);
332             if (module!=null) {
333                 add(module);
334             }
335         }
336         if (module!=null && resolve) {
337             try {
338                 module.resolve();
339             } catch(Throwable e) {
340                 module.uninstall();
341                 throw new ResolveError(e);
342             }
343         }
344         Logger.getAnonymousLogger().info("this.makeModuleFor("+name+ ", " + version + ") returned " + module);
345         return module;
346     }

348     /**
349      * Find and return a loaded Module that has the package name in its list
350      * of exported interfaces.
351      *
352      * @param packageName the requested implementation package name.
353      * @return the <code>Module</code> instance implementing the package
354      * name or null if not found.
355      * @throws ResolveError if the module dependencies cannot be resolved
356      */
357     public Module makeModuleFor(String packageName) throws ResolveError {
358         if(parent!=null) {
359             Module m = parent.makeModuleFor(packageName);
360             if(m!=null)     return m;
361         }
362 
363         for (Module module : modules.values()) {
364             String[] exportedPkgs = module.getModuleDefinition().getPublicInterfaces();
365             for (String exportedPkg : exportedPkgs) {
366                 if (exportedPkg.equals(packageName)) {
367                     module.resolve();
368                     return module;
369                 }
370             }
371         }
372         return null;
373     }

375     protected Module loadFromRepository(String name, String version) {
376         Set<Integer> keys = repositories.keySet();
377         TreeSet<Integer> sortedKeys = new TreeSet<Integer>();
378         sortedKeys.addAll(keys);
379         for (Integer key : sortedKeys) {
380             Repository repo = repositories.get(key);
381             ModuleDefinition moduleDef = repo.find(name, version);
382             if (moduleDef!=null) {
383                 return newModule(moduleDef);
384             }
385         }
386         
387         return null;
388     }

397     /**
398      * Add a new module to this registry. Once added, the module will be 
399      * available through one of the getServiceImplementor methods.
400      * @param newModule the new module
401      */
402     protected void add(Module newModule) {
403     	//if (Utils.isLoggable(Level.INFO)) {
404         //    Utils.getDefaultLogger().info("New module " + newModule);
405         //}
406         assert newModule.getRegistry()==this;
407         // see if this module is already added. This can happen
408         // in case of OSGiModulesRegistryImpl which uses SynchronousBundleListener.
409         ModuleId id = AbstractFactory.getInstance().createModuleId(
410                 newModule.getModuleDefinition());
411         if (modules.get(id) != null) return;
412         modules.put(id, newModule);
413 
414         // pick up providers from this module
415         for( ModuleMetadata.Entry spi : newModule.getMetadata().getEntries() ) {
416             for( String name : spi.providerNames )
417                 providers.put(name,newModule);
418         }
419             
420         for (Map.Entry<ServiceLocator, String> entry : habitats.entrySet()) {
421             String name = entry.getValue();
422             ServiceLocator serviceLocator = entry.getKey();
423             try
424             {
425                 parseInhabitants(newModule, name, serviceLocator, new ArrayList<PopulatorPostProcessor>());
426             }
427             catch (Exception e)
428             {
429                 throw new RuntimeException("Not able to parse inhabitants information");
430             }
431           
432         }
433     }

435     private void removeShutdownLocators() {
436         for (Map<ServiceLocator, List<ActiveDescriptor>> descriptorsByServiceLocator : moduleDescriptors.values()) {
437             Set<ServiceLocator> keys = new HashSet<ServiceLocator>(descriptorsByServiceLocator.keySet());
438             
439             for (ServiceLocator key : keys) {
440                 if (!ServiceLocatorState.SHUTDOWN.equals(key.getState())) continue;
441                 
442                 descriptorsByServiceLocator.remove(key);
443                 
444                 habitats.remove(key);
445             }
446         }
447    
448         
449     }

451     /**
452      * Removes a module from the registry. The module will not be accessible 
453      * from this registry after this method returns.
454      */
455 	public void remove(Module module) {
456 		// if (Utils.isLoggable(Level.INFO)) {
457 		// Utils.getDefaultLogger().info("Removed module " + module);
458 		// }
459 		assert module.getRegistry() == this;
460 		modules.remove(AbstractFactory.getInstance().createModuleId(
461 				module.getModuleDefinition()));
462 		
463 		// Removes any shutdown locators before we operate on them
464 		removeShutdownLocators();
465 
466 		// TODO: modules comes right back when getModules() is called.
467 		// the modeling is incorrect
468 
469 		Map<ServiceLocator, List<ActiveDescriptor>> descriptorsByServiceLocator = moduleDescriptors
470 				.get(module);
471 
472 		if (descriptorsByServiceLocator != null) {
473 			for (Entry<ServiceLocator, List<ActiveDescriptor>> e : descriptorsByServiceLocator
474 					.entrySet()) {
475 				ServiceLocator sl = e.getKey();
476 				if (!sl.getState().equals(ServiceLocatorState.RUNNING)) continue;
477 				
478 				List<ActiveDescriptor> descriptors = e.getValue();
479 
480 				for (Descriptor descriptor : descriptors) {
481 					ServiceLocatorUtilities.removeOneDescriptor(sl, descriptor);
482 				}
483 			}
484 			moduleDescriptors.remove(module);
485 		}
486 	}

494     /** 
495      * Returns the list of shared Modules registered in this instance.
496      *
497      * <p>
498      * The returned list will not include the modules defined in the ancestor
499      * {@link AbstractModulesRegistryImpl}s.
500      *
501      * @return an umodifiable list of loaded modules
502      */
503     public Collection<Module> getModules() {
504 
505         // make a copy to avoid synchronizing since this API can be called while
506         // modules are added or removed by other threads.
507         Map<Integer, Repository> repos = new TreeMap<Integer, Repository>();
508         repos.putAll(repositories);
509 
510         // force repository extraction
511         Set<Integer> keys = repos.keySet();
512         TreeSet<Integer> sortedKeys = new TreeSet<Integer>();
513         sortedKeys.addAll(keys);
514         for (Integer key : sortedKeys) {
515             Repository repo = repos.get(key);
516             for (ModuleDefinition moduleDef : repo.findAll()) {
517                 if (modules.get(AbstractFactory.getInstance().createModuleId(moduleDef)) == null) {
518                     Module newModule = newModule(moduleDef);
519                     if (newModule != null) {
520                         // When some module can't get installed,
521                         // don't halt proceeding, instead continue
522                         add(newModule);
523                         // don't resolve such modules, we just want to know about them
524                     }
525                 }
526             }
527         }
528         return modules.values();
529     }

629     /**
630      * Registers a running service, this is useful when other components need
631      * to have access to a provider of a service without having to create
632      * a new instance and initialize it.
633      * @param serviceClass the service interface
634      * @param provider the provider of that service.
635      */
636     public <T> void registerRunningService(Class<T> serviceClass, T provider) {
637         CopyOnWriteArrayList rs = runningServices.get(serviceClass);
638         if (rs==null) {
639             rs = new CopyOnWriteArrayList<T>();
640             CopyOnWriteArrayList existing = runningServices.putIfAbsent(serviceClass, rs);
641             if(existing!=null)
642                 rs = existing;
643         }
644         rs.add(provider);
645     }

647     /**
648      * Removes a running service, this is useful when a service instance is no longer
649      * available as a provider of a service.
650      */
651     public <T> boolean unregisterRunningService(Class<T> serviceClass, T provider) {
652         CopyOnWriteArrayList rs = runningServices.get(serviceClass);
653         if (rs==null) {
654             return false;
655         }
656         return rs.remove(provider);
