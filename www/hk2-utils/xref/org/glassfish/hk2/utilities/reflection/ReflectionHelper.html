93      /**
94       * Given the type parameter gets the raw type represented
95       * by the type, or null if this has no associated raw class
96       * @param type The type to find the raw class on
97       * @return The raw class associated with this type
98       */
99      public static Class<?> getRawClass(Type type) {
100         if (type == null) return null;
101         
102         if (type instanceof GenericArrayType) {
103             Type componentType = ((GenericArrayType) type).getGenericComponentType();
104             
105             if (!(componentType instanceof ParameterizedType)) {
106                 // type variable is not supported
107                 return null;
108             }
109             
110             Class<?> rawComponentClass = getRawClass(componentType);
111             
112             String forNameName = "[L" + rawComponentClass.getName() + ";";
113             try {
114                 return Class.forName(forNameName);
115             }
116             catch (Throwable th) {
117                 // ignore, but return null
118                 return null;
119             }
120         }
121         
122         if (type instanceof Class) {
123             return (Class<?>) type;
124         }
125         
126         if (type instanceof ParameterizedType) {
127             Type rawType = ((ParameterizedType) type).getRawType();
128             if (rawType instanceof Class) {
129                 return (Class<?>) rawType;
130             }
131         }
132         
133         return null;
134     }

136     private static String getNamedName(Named named, Class<?> implClass) {
137         String name = named.value();
138         if (name != null && !name.equals("")) return name;
139         
140         String cn = implClass.getName();
141             
142         int index = cn.lastIndexOf(".");
143         if (index < 0) return cn;
144         
145         return cn.substring(index + 1);
146     }

164     /**
165      * Gets all the interfaces on this particular class (but not any
166      * superclasses of this class).
167      */
168     private static void addAllGenericInterfaces(Class<?> rawClass ,
169                                                 Type type,
170                                                 Set<Type> closures) {
171 
172         Map<String, Type> typeArgumentsMap = null;
173 
174         for (Type currentType : rawClass.getGenericInterfaces()) {
175 
176             if (type instanceof ParameterizedType &&
177                     currentType instanceof ParameterizedType) {
178 
179                 if (typeArgumentsMap == null ) {
180                     typeArgumentsMap = getTypeArguments(rawClass, (ParameterizedType) type);
181                 }
182 
183                 currentType = fixTypeVariables((ParameterizedType) currentType, typeArgumentsMap);
184             }
185 
186             closures.add(currentType);
187 
188             rawClass = ReflectionHelper.getRawClass(currentType);
189             if (rawClass != null) {
190                 addAllGenericInterfaces(rawClass, currentType, closures);
191             }
192         }
193     }

195     /**
196      * Replace any TypeVariables in the given type's arguments with
197      * the actual argument types.  Return the given type if no replacing
198      * is required.
199      */
200     private static Type fixTypeVariables(ParameterizedType type,
201                                          Map<String, Type> typeArgumentsMap) {
202 
203         Type[] newTypeArguments = getNewTypeArguments(type, typeArgumentsMap);
204 
205         if (newTypeArguments != null) {
206             type = new ParameterizedTypeImpl(type.getRawType(), newTypeArguments);
207         }
208         return type;
209     }

211     /**
212      * Get a new array of type arguments for the given ParameterizedType, replacing any TypeVariables with
213      * actual types.  The types should be found in the given arguments map, keyed by variable name.  Return
214      * null if no arguments needed to be replaced.
215      */
216     private static Type[] getNewTypeArguments(final ParameterizedType type,
217                                               final Map<String, Type> typeArgumentsMap) {
218 
219         Type[]  typeArguments    = type.getActualTypeArguments();
220         Type[]  newTypeArguments = new Type[typeArguments.length];
221         boolean newArgsNeeded    = false;
222 
223         int i = 0;
224         for (Type argType : typeArguments) {
225             if (argType instanceof TypeVariable) {
226                 newTypeArguments[i++] = typeArgumentsMap.get(((TypeVariable<?>) argType).getName());
227                 newArgsNeeded = true;
228             } else {
229                 newTypeArguments[i++] = argType;
230             }
231         }
232         return newArgsNeeded ? newTypeArguments : null;
233     }

235     /**
236      * Gets a mapping of type variable names of the raw class to type arguments of the
237      * parameterized type.
238      */
239     private static Map<String, Type> getTypeArguments(Class<?> rawClass,
240                                                       ParameterizedType type) {
241 
242         Map<String, Type> typeMap       = new HashMap<String, Type>();
243         Type[]            typeArguments = type.getActualTypeArguments();
244 
245         int i = 0;
246         for (TypeVariable<?> typeVariable : rawClass.getTypeParameters() ) {
247             typeMap.put(typeVariable.getName(), typeArguments[i++]);
248         }
249         return typeMap;
250     }

252     /**
253      * Returns the type closure of the given class
254      * 
255      * @param ofType The full type closure of the given class
256      * with nothing omitted (normal case).  May not be null
257      * @return The non-null (and never empty) set of classes
258      * that this class can be assigned to
259      */
260     private static Set<Type> getTypeClosure(Type ofType) {
261         Set<Type> retVal   = new HashSet<Type>();
262         Class<?>  rawClass = ReflectionHelper.getRawClass(ofType);
263 
264         if (rawClass != null) {
265             Map<String, Type> typeArgumentsMap = null;
266             Type              currentType      = ofType;
267 
268             while (currentType != null && rawClass != null) {
269 
270                 retVal.add(currentType);
271 
272                 addAllGenericInterfaces(rawClass, currentType, retVal);
273 
274                 if (typeArgumentsMap == null && currentType instanceof ParameterizedType){
275                     typeArgumentsMap = getTypeArguments(rawClass, (ParameterizedType) currentType);
276                 }
277 
278                 currentType = rawClass.getGenericSuperclass();
279                 if (currentType != null) {
280                     rawClass = ReflectionHelper.getRawClass(currentType);
281 
282                     if (typeArgumentsMap != null && currentType instanceof ParameterizedType){
283                         currentType = fixTypeVariables((ParameterizedType) currentType, typeArgumentsMap);
284                     }
285                 }
286             }
287         }
288         return retVal;
289     }

291     /**
292      * Returns the type closure, as restricted by the classes listed in the
293      * set of contracts implemented
294      * 
295      * @param ofType The type to check
296      * @param contracts The contracts this type is allowed to handle
297      * @return The type closure restricted to the contracts
298      */
299     public static Set<Type> getTypeClosure(Type ofType, Set<String> contracts) {
300         Set<Type> closure = getTypeClosure(ofType);
301         
302         HashSet<Type> retVal = new HashSet<Type>();
303         for (Type t : closure) {
304             Class<?> rawClass = ReflectionHelper.getRawClass(t);
305             if (rawClass == null) continue;
306             
307             if (contracts.contains(rawClass.getName())) {
308                 retVal.add(t);
309             }
310         }
311         
312         return retVal;
313     }

315     /**
316      * Returns the set of types this class advertises
317      * @param type The outer type to analyze
318      * @param markerAnnotation The annotation to use to discover the advertised types
319      * @return The type itself and the contracts it implements
320      */
321     public static Set<Type> getAdvertisedTypesFromClass(Type type, Class<? extends Annotation> markerAnnotation) {
322         Set<Type> retVal = new LinkedHashSet<Type>();
323         if (type == null) return retVal;
324         
325         retVal.add(type);
326         
327         Class<?> originalRawClass = getRawClass(type);
328         if (originalRawClass == null) return retVal;
329         
330         Type genericSuperclass = originalRawClass.getGenericSuperclass();
331         while (genericSuperclass != null) {
332             Class<?> rawClass = getRawClass(genericSuperclass);
333             if (rawClass == null) break;
334             
335             if (rawClass.isAnnotationPresent(markerAnnotation)) {
336                 retVal.add(genericSuperclass);
337             }
338             
339             genericSuperclass = rawClass.getGenericSuperclass();
340         }
341         
342         Set<Class<?>> alreadyHandled = new HashSet<Class<?>>();
343         while (originalRawClass != null) {
344             getAllContractsFromInterfaces(originalRawClass,
345                 markerAnnotation,
346                 retVal,
347                 alreadyHandled);
348             
349             originalRawClass = originalRawClass.getSuperclass();
350         }
351         
352         return retVal;
353     }

355     private static void getAllContractsFromInterfaces(Class<?> clazzOrInterface,
356             Class<? extends Annotation> markerAnnotation,
357             Set<Type> addToMe,
358             Set<Class<?>> alreadyHandled) {
359         Type interfacesAsType[] = clazzOrInterface.getGenericInterfaces();
360         
361         for (Type interfaceAsType : interfacesAsType) {
362             Class<?> interfaceAsClass = getRawClass(interfaceAsType);
363             if (interfaceAsClass == null) continue;
364             if (alreadyHandled.contains(interfaceAsClass)) continue;
365             alreadyHandled.add(interfaceAsClass);
366             
367             if (interfaceAsClass.isAnnotationPresent(markerAnnotation)) {
368                 addToMe.add(interfaceAsType);
369             }
370             
371             getAllContractsFromInterfaces(interfaceAsClass, markerAnnotation, addToMe, alreadyHandled);
372         }
373     }

387     /**
388      * Returns the set of types this class advertises
389      * @param clazz the class we are analyzing
390      * @param markerAnnotation The annotation to use to discover annotated types
391      * @return The type itself and the contracts it implements
392      */
393     public static Set<String> getContractsFromClass(Class<?> clazz, Class<? extends Annotation> markerAnnotation) {
394         Set<String> retVal = new LinkedHashSet<String>();
395         if (clazz == null) return retVal;
396         
397         retVal.add(clazz.getName());
398         
399         Class<?> extendsClasses = clazz.getSuperclass();
400         while (extendsClasses != null) {
401             if (extendsClasses.isAnnotationPresent(markerAnnotation)) {
402                 retVal.add(extendsClasses.getName());
403             }
404             
405             extendsClasses = extendsClasses.getSuperclass();
406         }
407         
408         while (clazz != null) {
409             Class<?> interfaces[] = clazz.getInterfaces();
410             for (Class<?> iFace : interfaces) {
411                 if (iFace.isAnnotationPresent(markerAnnotation)) {
412                     retVal.add(iFace.getName());
413                 }
414             }
415             
416             clazz = clazz.getSuperclass();
417         }
418         
419         return retVal;
420     }

533             }

535             clazz = clazz.getSuperclass();
536         }

541     private static Set<Annotation> internalGetQualifierAnnotations(AnnotatedElement annotatedGuy) {
542         Set<Annotation> retVal = new LinkedHashSet<Annotation>();
543         if (annotatedGuy == null) return retVal;
544         
545         for (Annotation annotation : annotatedGuy.getAnnotations()) {
546             if (isAnnotationAQualifier(annotation)) {
547                 retVal.add(annotation);
548             }
549         }
550         
551         if (!(annotatedGuy instanceof Class)) return retVal;
552         
553         Class<?> clazz = (Class<?>) annotatedGuy;
554         while (clazz != null) {
555             for (Class<?> iFace : clazz.getInterfaces()) {
556                 for (Annotation annotation : iFace.getAnnotations()) {
557                     if (isAnnotationAQualifier(annotation)) {
558                         retVal.add(annotation);
559                     }
560                 }
561             }
562             
563             clazz = clazz.getSuperclass();
564         }
565         
566         return retVal;
567         
568     }

591      * 
592      * @param set The set to write
593      * @return a representation of a list
594      */
595     public static String writeSet(Set<?> set) {
596         if (set == null) return "{}";
597         
598         StringBuffer sb = new StringBuffer("{");
599         
600         boolean first = true;
601         for (Object writeMe : set) {
602             if (first) {
603                 first = false;
604                 sb.append(escapeString(writeMe.toString()));
605             }
606             else {
607                 sb.append("," + escapeString(writeMe.toString()));
608             }
609         }
610         
611         sb.append("}");
612         
613         return sb.toString();
614     }

631     /**
632      * Writes a set in a way that can be read from an input stream as well.  The values in
633      * the set may not contain the characters "{},"
634      * 
635      * @param asChars The line to read
636      * @param addToMe The set to add the strings to
637      * @return The number of characters read until the end of the set
638      * @throws IOException On a failure
639      */
640     private static int internalReadSet(char asChars[], int startIndex, Collection<String> addToMe) throws IOException {
641         int dot = startIndex;
642         int startOfSet = -1;
643         while (dot < asChars.length) {
644             if (asChars[dot] == '{') {
645                 startOfSet = dot;
646                 dot++;
647                 break;
648             }
649             dot++;
650         }
651         
652         if (startOfSet == -1) {
653             throw new IOException("Unknown set format, no initial { character : " + new String(asChars));
654         }
655         
656         StringBuffer elementBuffer = new StringBuffer();
657         int endOfSet = -1;
658         while (dot < asChars.length) {
659             char dotChar = asChars[dot];
660             
661             if (dotChar == '}') {
662                 addToMe.add(elementBuffer.toString());
663                 
664                 endOfSet = dot;
665                 break;  // Done!
666             }
667             
668             if (dotChar == ',') {
669                 // Terminating a single element
670                 addToMe.add(elementBuffer.toString());
671                 
672                 elementBuffer = new StringBuffer();
673             }
674             else {
675                 // This character is either an escape character or a real character
676                 if (dotChar != '\\') {
677                     elementBuffer.append(dotChar);
678                 }
679                 else {
680                     // This is an escape character
681                     if (dot + 1 >= asChars.length) {
682                         // This is an error, escape at end of buffer
683                         break;
684                     }
685                     
686                     dot++;  // Moves it forward
687                     dotChar = asChars[dot];
688                     
689                     if (dotChar == 'n') {
690                         elementBuffer.append('\n');
691                     }
692                     else if (dotChar == 'r') {
693                         elementBuffer.append('\r');
694                     }
695                     else {
696                         elementBuffer.append(dotChar);
697                     }
698                 }
699                 
700             }
701             
702             dot++;
703         }
704         
705         if (endOfSet == -1) {
706             throw new IOException("Unknown set format, no ending } character : " + new String(asChars));
707         }
708         
709         return dot - startIndex;
710     }

712     private static int readKeyStringListLine(char asChars[], int startIndex, Map<String, List<String>> addToMe) throws IOException {
713         int dot = startIndex;
714         
715         int equalsIndex = -1;
716         while (dot < asChars.length) {
717             char dotChar = asChars[dot];
718             
719             if (dotChar == '=') {
720                 equalsIndex = dot;
721                 break;
722             }
723             
724             dot++;
725         }
726         
727         if (equalsIndex < 0) {
728             throw new IOException("Unknown key-string list format, no equals: " + new String(asChars));
729         }
730         
731         String key = new String(asChars, startIndex, (equalsIndex - startIndex));  // Does not include the =
732         dot++;  // Move it past the equals
733         
734         if (dot >= asChars.length) {
735             // Key with no values, this is illegal
736             throw new IOException("Found a key with no value, " + key + " in line " + new String(asChars));
737             
738         }
739         
740         LinkedList<String> listValues = new LinkedList<String>();
741         
742         int addOn = internalReadSet(asChars, dot, listValues);
743         if (!listValues.isEmpty()) {
744             addToMe.put(key, listValues);
745         }
746         
747         dot += addOn + 1;
748         if (dot < asChars.length) {
749             char skipComma = asChars[dot];

751                 dot++;
752             }
753         }
754         
755         return dot - startIndex;  // The +1 gets us to the next character in the stream
756     }

776         char asChars[] = new char[escapeMe.length()];
777         
778         escapeMe.getChars(0, escapeMe.length(), asChars, 0);
779         
780         StringBuffer sb = new StringBuffer();
781         for (int lcv = 0; lcv < asChars.length; lcv++) {
782             char candidateChar = asChars[lcv];
783             
784             if (ESCAPE_CHARACTERS.contains(candidateChar)) {
785                 sb.append('\\');
786                 sb.append(candidateChar);
787             }
788             else if (REPLACE_CHARACTERS.containsKey(candidateChar)) {
789                 char replaceWithMe = REPLACE_CHARACTERS.get(candidateChar);
790                 sb.append('\\');
791                 sb.append(replaceWithMe);
792             }
793             else {
794                 sb.append(candidateChar);
795             }
796         }
797         
798         return sb.toString();
