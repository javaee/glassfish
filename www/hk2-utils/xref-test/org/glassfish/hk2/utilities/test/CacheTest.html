84      /**
85       * If cache is size three, only three values should be stored.  And the one removed
86       * should be the last one
87       */
88      @Test
89      public void testLastIsRemoved() {
90          LRUCache<Integer, Integer> cache = LRUCache.createCache(3);
91          
92          cache.put(1, 1);
93          cache.put(2, 2);
94          cache.put(3, 3);
95          cache.put(4, 4);
96          
97          // 1 is least recently used, so it should be gone
98          Assert.assertNull(cache.get(1));
99          
100         // Also check that the others ARE there
101         Assert.assertSame(4, cache.get(4));
102         Assert.assertSame(2, cache.get(2));
103         Assert.assertSame(3, cache.get(3));
104     }

106     /**
107      * Ensure that get changes the LRU order of the list
108      */
109     @Test
110     public void testGetChangesOrder() {
111         LRUCache<Integer, Integer> cache = LRUCache.createCache(3);
112         
113         cache.put(1, 1);
114         cache.put(2, 2);
115         cache.put(3, 3);
116         
117         // 1 is last now, but if I get it, it should move to the front, and
118         // 2 would be put out on the next add
119         Assert.assertSame(1, cache.get(1));
120         
121         cache.put(4, 4);
122         
123         // 2 was least recently used, so it should be gone
124         Assert.assertNull(cache.get(2));
125         
126         // Also check that the others ARE there
127         Assert.assertSame(4, cache.get(4));
128         Assert.assertSame(1, cache.get(1));
129         Assert.assertSame(3, cache.get(3));
130     }

132     /**
133      * Tests that an entry in the middle of a larger cache can be properly moved
134      */
135     @Test
136     public void testMakeChangesToLargerCache() {
137         LRUCache<Integer, Integer> cache = LRUCache.createCache(5);
138         
139         cache.put(1, 1);
140         cache.put(2, 2);
141         cache.put(3, 3);
142         cache.put(4, 4);
143         cache.put(5, 5);
144         
145         // Three would be in the middle of the queue
146         Assert.assertSame(3, cache.get(3));
147         
148         cache.put(6, 6);
149         
150         // 1 was least recently used, should be gone
151         Assert.assertNull(cache.get(1));
152         
153         // Also check that the others ARE there
154         Assert.assertSame(4, cache.get(4));
155         Assert.assertSame(2, cache.get(2));
156         Assert.assertSame(3, cache.get(3));
157         Assert.assertSame(5, cache.get(5));
158     }
