64  /**
65   * reads a wires XML file generated by the dependency-verifier and outputs a dot
66   * file (Use <a href="http://www.graphviz.org/">GraphViz</a> to create images
67   * from the dot file) that helps visualizing the wiring dependencies between
68   * packages. The tool also supports viewing the dependencies of only a subset of
69   * the packages.
70   * 
71   * @author Sivakumar Thyagarajan
72   */
73  public class DotGenerator {
74  	private final static boolean DEBUG = false;
75  
76  	PrintStream wireOut = null;
77  	GeneratorOptions options = null;
78  
79  	/**
80  	 * An internal class holding all the command line options
81  	 * supported by this DotGenerator
82  	 * 
83  	 * @author Sivakumar Thyagarajan
84  	 */
85  	static class GeneratorOptions {
86  		// Accept XML files generated by HK2 dependency-verifier
87  		@Option(name = "-i", usage = "Input Wires XML file")
88  		public String input = "wires.xml";
89  
90  		@Option(name = "-o", usage = "Output DOT file", required = true)
91  		public String output;
92  
93  		@Option(name = "-m", usage = "Show only packages that contains the specified substring")
94  		public String match = "";// By default, match all.
95  
96  		// receives other command line parameters than options
97  		@Argument
98  		public List<String> arguments = new ArrayList<String>();
99  	}

101 	/**
102 	 * A simple class representing all the information about a
103 	 * package that can be derived from the wires.xml
104 	 * 
105 	 * @author Sivakumar Thyagarajan
106 	 */
107 	class PackageInfo {
108 		String packageName, exportedBy = null;
109 		String[] importedBy = null;
110 
111 		public PackageInfo(String packageName, String exportedBy,
112 				String[] importedBy) {
113 			this.packageName = packageName;
114 			this.exportedBy = exportedBy;
115 			this.importedBy = importedBy;
116 		}
117 	}

119 	public DotGenerator(GeneratorOptions go) throws Exception {
120 		this.options = go;
121 		wireOut = new PrintStream(new FileOutputStream(this.options.output));
122 		initXML();
123 		generate();
124 	}

126 	private Document doc = null;
127 
128 	private void initXML() throws Exception {
129 		File file = new File(this.options.input);
130 		debug("file " + file);
131 		DocumentBuilderFactory dbf = DocumentBuilderFactory.newInstance();
132 		DocumentBuilder db = dbf.newDocumentBuilder();
133 		doc = db.parse(file);
134 		doc.getDocumentElement().normalize();
135 	}

137 	private PackageInfo[] getPackageXML() throws IOException {
138 		NodeList pkgLst = doc.getElementsByTagName("Package");
139 		debug("Package count:" + pkgLst.getLength());
140 		List<PackageInfo> pkgInfos = new ArrayList<PackageInfo>();
141 		for (int i = 0; i < pkgLst.getLength(); i++) {
142 			Node pkgNode = pkgLst.item(i);
143 			if (pkgNode.getNodeType() == Node.ELEMENT_NODE) {
144 				Element packageElement = (Element) pkgNode;
145 				// exports
146 				NodeList ExportersList = packageElement
147 						.getElementsByTagName("Exporters");
148 				Element exporterElt = (Element) ExportersList.item(0);
149 				String exporter = ((Node) exporterElt.getChildNodes().item(0))
150 						.getNodeValue().trim();
151 				debug("Exporter : " + exporter);
152 
153 				// importers
154 				NodeList importersList = packageElement
155 						.getElementsByTagName("Importers");
156 				Element importerElt = (Element) importersList.item(0);
157 				String importers = ((Node) importerElt.getChildNodes().item(0))
158 						.getNodeValue().trim();
159 				debug("Importers : " + importers);
160 
161 				// Get package name and return PackageInfos
162 				String pkgName = packageElement.getAttribute("name").trim();
163 				debug("Package Name : " + pkgName);
164 				PackageInfo pkgInfo = new PackageInfo(pkgName, exporter,
165 						split(importers));
166 				pkgInfos.add(pkgInfo);
167 			}
168 		}
169 		return pkgInfos.toArray(new PackageInfo[] {});
170 	}
171 
172 	private void generate() throws Exception {
173 		generateDotStart();
174 		PackageInfo[] pkgInfos = getPackageXML();
175 		for (PackageInfo pkgInfo : pkgInfos) {
176 			// Match if needed
177 			String matchString = this.options.match.trim();
178 			boolean matchNeeded = !matchString.isEmpty();
179 			if (matchNeeded) {
180 				if (pkgInfo.exportedBy.contains(matchString)) {
181 					generateDotEdge(pkgInfo.importedBy, pkgInfo.exportedBy,
182 							pkgInfo.packageName);
183 				}
184 			} else {
185 				generateDotEdge(pkgInfo.importedBy, pkgInfo.exportedBy,
186 						pkgInfo.packageName);
187 			}
188 		}
189 		generateDotEnd();
190 	}

209 	public static void main(String[] args) throws Exception {
210 		DotGenerator.GeneratorOptions options = new DotGenerator.GeneratorOptions();
211 		CmdLineParser parser = new CmdLineParser(options);
212 		try {
213 			parser.parseArgument(args);
214 		} catch (CmdLineException e) {
215 			System.err.println(e.getMessage());
216 			System.err.println("java -jar program-name.jar [options...] arguments...");
217 			parser.printUsage(System.err);
218 			return;
219 		}
220 		new DotGenerator(options);
221 	}

223 	// Generate the beginning of the dot file
224 	private void generateDotStart() {
225 		this.wireOut.println("digraph  wiring {");
226 		this.wireOut.println("node [color=grey, style=filled];");
227 		this.wireOut.println("node [fontname=\"Verdana\", size=\"30,30\"];");
228 		String date = DateFormat.getDateTimeInstance(DateFormat.MEDIUM,
229 				DateFormat.SHORT).format(new Date());
230 		StringBuffer footer = new StringBuffer();
231 		footer.append("graph [ fontname = \"Arial\", fontsize = 26,style = \"bold\", ");
232 		footer.append("label = \"\\nGlassFish v3 OSGi bundle wiring relationship diagram");
233 		if (!this.options.match.trim().isEmpty()) {
234 			footer.append("\\n Filter: " + this.options.match.trim()
235 					+ " bundles");
236 		}
237 		footer.append("\\nSun Microsystems");
238 		footer.append("\\n\\nDate: " + date + "\\n\", "
239 				+ "ssize = \"30,60\" ];");
240 		this.wireOut.println(footer.toString());
241 	}

243 	// Generate a Dot representation for each edge in the graph
244 	private void generateDotEdge(String[] importedBy, String exportedBy,
245 			String pkg) {
246 		if (importedBy.length == 0)
247 			return;
248 		for (String s : importedBy) {
249 			if (!s.equals(exportedBy)) { // remove self-loops for readability
250 				this.wireOut.println("\"" + s + "\" -> \"" + exportedBy
251 						+ "\" [label =\"" + pkg + "\"" + "]");
252 			}
253 		}
254 	}
255 
256 	// End the dot file generation
257 	private void generateDotEnd() {
258 		this.wireOut.println("}");
259 	}

261 	// Utility class to split the importers representation (space separated) 
262 	//in wires.xml
263 	private String[] split(String s) {
264 		StringTokenizer st = new StringTokenizer(s);
265 		List<String> l = new ArrayList<String>();
266 		while (st.hasMoreTokens()) {
267 			l.add(st.nextToken());
268 		}
269 		return l.toArray(new String[] {});
270 	}
