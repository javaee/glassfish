13  /**
14   * Normal modules registry with configuration handling backed up
15   * by a single class loader. There is one virtual module available in the modules
16   * registry and that module's class loader is the single class loader used to
17   * load all artifacts.
18   *
19   * @author Jerome Dochez
20   */
21  public class SingleModulesRegistry  extends ModulesRegistryImpl {
22  
23      final ClassLoader singleClassLoader;
24      final Module[] proxyMod = new Module[1];
25  
26      public SingleModulesRegistry(ClassLoader singleCL) {
27          this(singleCL, null);
28      }

30  
31      public SingleModulesRegistry(ClassLoader singleCL, List<ManifestProxy.SeparatorMappings> mappings) {
32  
33          super(null);
34          this.singleClassLoader = singleCL;
35          setParentClassLoader(singleClassLoader);
36  
37          ModuleDefinition moduleDef = null;
38          try {
39              moduleDef = new ProxyModuleDefinition(singleClassLoader, mappings);
40          } catch (IOException e) {
41              e.printStackTrace();  //To change body of catch statement use File | Settings | File Templates.
42          }
43  
44          proxyMod[0] = new ProxyModule(this, moduleDef, singleClassLoader);
45          add(moduleDef);
46      }
47  
48      @Override
49      public Module find(Class clazz) {
50          Module m = super.find(clazz);
51          if (m == null)
52              return proxyMod[0];
53          return m;
54      }
55  
56      @Override
57      public Collection<Module> getModules(String moduleName) {
58          // I could not care less about the modules names
59          return getModules();
60      }
61  
62      @Override
63      public Collection<Module> getModules() {
64          ArrayList<Module> list = new ArrayList<Module>();
65          list.add(proxyMod[0]);
66          return list;
67      }
68  
69      @Override
70      public Module makeModuleFor(String name, String version, boolean resolve) throws ResolveError {
71          return proxyMod[0];
72      }
73  
74      @Override
75      public void parseInhabitants(Module module, String name, InhabitantsParser inhabitantsParser)
76              throws IOException {
77  
78          Holder<ClassLoader> holder = new Holder<ClassLoader>() {
79              public ClassLoader get() {
80                  return proxyMod[0].getClassLoader();
81              }
82          };
83  
84          for (InhabitantsDescriptor d : proxyMod[0].getMetadata().getHabitats(name))
85              inhabitantsParser.parse(d.createScanner(), holder);
86      }
