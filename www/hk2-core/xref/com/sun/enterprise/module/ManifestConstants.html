40  package com.sun.enterprise.module;
41  
42  import org.jvnet.hk2.annotations.Service;
43  import com.sun.enterprise.module.bootstrap.ModuleStartup;
44  
45  /**
46   * Constants used in the module manifest.
47   * These values define a module.
48   *
49   * @author dochez
50   * @author Kohsuke Kawaguchi
51   */
52  public class ManifestConstants {
53      /*
54       * All headers are prefixed by HK2 to avoid name collision. By Sahoo
55       */
56  
57      // No instanciation allowed.
58      private ManifestConstants() {
59      }
60  
61      /**
62       * Module name.
63       * <p>
64       * This uniquely identifies the module within modules,
65       * but this shouldn't include the version number.
66       * <p>
67       * <tt>hk2-module</tt> packaging type uses GROUPID.ARTIFACTID for this.
68       */
69      public static final String BUNDLE_NAME = "Bundle-SymbolicName";
70  
71      /**
72       * Bundle Version
73       */
74      public static final String BUNDLE_VERSION = "Bundle-Version";
75  
76      /**
77       * Module name that we'll launch.
78       * <p>
79       * This is equivalent of "Main-Class" for us. We'll find {@link ModuleStartup}
80       * from this module.
81       */
82      public static final String MAIN_BUNDLE = "HK2-Main-Bundle";
83  
84      /**
85       * Exported package list.
86       * <p>
87       * The value is a comma separated list of packages that are
88       * visible to other modules that depend on this module.
89       * For example, "com.sun.foobar.spi,com.sun.foobar.abc"
90       * <p>
91       * If this manifest entry doesn't exist, all classes will be exported.
92       *
93       * <p>
94       * TODO: expand Maven's <tt>hk2-module</tt> packaging to generate this.
95       */
96      public static final String PKG_EXPORT_NAME = "HK2-Export-Package";
97  
98      /**
99       * List of modules that this module depends on.
100      * <p>
101      * The value is a comma separated list of {@link #BUNDLE_NAME modules names}
102      * that this module depends on. Whitespaces are allowed before and after
103      * commas.
104      * <p>
105      *
106      * <p>
107      * Maven's <tt>hk2-module</tt> packaging uses dependency list in POM
108      * to fill in this entry.
109      *
110      * TODO: expand this field to support version specifier.
111      */
112     public static final String BUNDLE_IMPORT_NAME = "HK2-Import-Bundles";
113 
114     /**
115      * List of other jar files in this module.
116      * <p>
117      * This identifies other jar files in this module, not other modules
118      * that this module depends on.
119      * <p>
120      * See <a href="http://java.sun.com/j2se/1.5.0/docs/guide/jar/jar.html#Main%20Attributes">
121      * jar file specification</a> for the format of the value.
122      *
123      * <p>
124      * Maven's <tt>hk2-module</tt> packaging uses dependency list in POM
125      * to fill in this entry.
126      *
127      * @see #CLASS_PATH_ID
128      */
129     public static final String CLASS_PATH = "Class-Path";
130 
131     /**
132      * List of other jar files in this module.
133      * <p>
134      * The value is a whitespace separated list of Maven ID of all the dependencies,
135      * which is of the form "GROUPID:ARTIFACTID:TYPE[:CLASSIFIER]:VERSION".
136      * <p>
137      * This entry contains essentially the same information as the {@link #CLASS_PATH}
138      * entry, but this enables HK2 to assemble all the required jars by using Maven.
139      *
140      * <p>
141      * Maven's <tt>hk2-module</tt> packaging uses dependency list in POM
142      * to fill in this entry.
143      *
144      * @see #CLASS_PATH
145      */
146     public static final String CLASS_PATH_ID = "HK2-Class-Path-Id";
147 
148     /**
149      * See {@link ImportPolicy}.
150      *
151      * <p>
152      * The value is the fully-qualified class name of a class that implements
153      * {@link ImportPolicy}.
154      * <p>
155      * Maven's <tt>hk2-module</tt> packaging automatically finds such a class
156      * and puts its name, provided that the class has @{@link Service} annotation.
157      */
158     public static final String IMPORT_POLICY = "HK2-Module-Import-Policy";
159 
160     /**
161      * See {@link LifecyclePolicy}.
162      *
163      * <p>
164      * The value is the fully-qualified class name of a class that implements
165      * {@link LifecyclePolicy}.
166      * <p>
167      * Maven's <tt>hk2-module</tt> packaging automatically finds such a class
168      * and puts its name, provided that the class has @{@link Service} annotation.
169      */
170     public static final String LIFECYLE_POLICY = "HK2-Module-Lifecycle-Policy";
171 
172     /**
173      * <p>
174      * List of repositories that should be set up before launching the main module
175      * </p>
176      */
177     public static final String REPOSITORIES = "HK2-Repositories";
178 }
