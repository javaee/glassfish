40  package com.sun.enterprise.module;
41  
42  /**
43   * A ModuleDependency instance holds all information necessary to identify 
44   * a dependency between modules. Modules can declare their dependency on a 
45   * separate module using the name, the version and whether they accept the 
46   * sub module implementation to be shared. They can also specify whether or
47   * not they want to re-export the sub module public interfaces. Re-exporting
48   * means that the sub-module's public interfaces will also be published as 
49   * a public interface of the enclosing module.
50   *
51   * @author Jerome Dochez
52   */
53  public class ModuleDependency {
54      
55      final private String name;
56      final private String version;
57      final private boolean shared;
58      final private boolean reexport;
59      
60      /**
61       * Create a new instance of ModuleDependency, where the sub module is 
62       * idenfied by its name and version. The sub module implementation should 
63       * be shared among users of that module
64       * @param name the module name
65       * @param version the module version
66       */
67      public ModuleDependency(String name, String version) {
68          this.name = name;
69          this.version = version;
70          this.shared = true;
71          this.reexport = false;
72      }
73  
74      /**
75       * Create a new instance of ModuleDependency, where the sub module is 
76       * idenfied by its name and version and wheter the containing module 
77       * requires a private copy or not
78       * @param name the module name
79       * @param version the module version
80       * @param shared true if the containing module accept a shared copy 
81       */
82      public ModuleDependency(String name, String version, 
83              boolean shared, boolean reexport) {
84          this.name = name;
85          this.version = version;
86          this.shared = shared;
87          this.reexport = reexport;
88      }
89      
90      
91      /**
92       * Returns the module name
93       * @return the module name 
94       */
95      public String getName() {
96          return name;
97      }
98      
99      /**
100      * Returns the module version
101      * @return the module version
102      */
103     public String getVersion() {
104         return version;
105     }
106     
107     /**
108      * Returns true if the containing module accept a shared implementation
109      * of the sub module
110      * @return true if shared implementation is acceptable
111      */
112     public boolean isShared() {
113         return shared;
114     }
115     
116     /**
117      * Returns true if the containing module is reexporting the public 
118      * interfaces of the sub module
119      * @return true if reexporting the sub module public interface
120      */
121     public boolean isReexporting() {
122         return reexport;
123     }
124     
125     /**
126      * Returns a string representation
127      * @return a printable string about myself
128      */
129     public String toString() {
130        return "Module Dependency : " + getName() + ":" + getVersion();
131     }
132 }
