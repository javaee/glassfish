80  /**
81   * A module represents a set of resources accessible to third party modules. 
82   * Each module has a module definition which defines its name, its list of 
83   * exported resources and its dependencies to other modules. 
84   * A module instance stores references to the class loader instances giving
85   * access to the module's implementation. 
86   * A module instance belongs to a <code>ModuleRegistry</code> which can be used
87   * to get the list of available modules and/or get particular module
88   * implementation. 
89   * Modules can only satisfy their dependencies within the <code>ModuleRegistry
90   * </code> instance they are registered in.
91   *
92   * @author Jerome Dochez
93   */

120     /* package */ ModuleImpl(ModulesRegistryImpl registry, ModuleDefinition info) {
121         assert registry!=null && info!=null;
122         this.registry = registry;
123         moduleDef = info;
124         for (URI lib : info.getLocations()) {
125             File f = new File(lib);
126             if (f.exists()) {
127                 lastModifieds.put(f.getAbsolutePath(), f.lastModified());
128             }

130         }
131         state = ModuleState.NEW;

141     public ClassLoaderFacade getClassLoader() {
142         ClassLoaderFacade r=null;
143         if (publicCL!=null)
144             r = publicCL.get();
145         if (r!=null)
146             return r;
147 
148         ClassLoaderFacade facade = AccessController.doPrivileged(new PrivilegedAction<ClassLoaderFacade>() {
149             public ClassLoaderFacade run() {
150                 return new ClassLoaderFacade(getPrivateClassLoader());
151             }
152         });
153         facade.setPublicPkgs(moduleDef.getPublicInterfaces());
154         publicCL = new WeakReference<ClassLoaderFacade>(facade);
155         return facade;
156     }

158     /**
159      * Return the private class loader for this module. This class loader will
160      * be loading all the classes which are not explicitely exported in the 
161      * module definition
162      * @return the private <code>ClassLoader</code> instance
163      */
164     /*package*/ ModuleClassLoader getPrivateClassLoader() {
165         if (privateCL==null) {
166             synchronized(this) {
167                 if(privateCL==null) {
168                     URI[] locations = moduleDef.getLocations();
169                     URL[] urlLocations = new URL[locations.length];
170                     for (int i=0;i<locations.length;i++) {
171                         try {
172                             urlLocations[i] = locations[i].toURL();
173                         } catch(MalformedURLException e) {
174                             e.printStackTrace();
175                             return null;
176                         }
177                     }
178                     final URL[] urls = urlLocations.clone();
179                     privateCL = AccessController.doPrivileged(new PrivilegedAction<ModuleClassLoader>() {
180                         public ModuleClassLoader run() {
181                             return new ModuleClassLoader(
182                                 ModuleImpl.this,
183                                 urls,
184                                 registry.getParentClassLoader());
185                         }
186                     });
187                 }
188             }
189         }
190         return privateCL;
191     }

256     /**
257      * Trigger manual refresh mechanism, the module will check all its 
258      * URLs and generate change events if any of them has changed. This 
259      * will allow the owning registry to force a module upgrade at next 
260      * module request.
261      */
262     public void refresh() {
263         URI[] urls = moduleDef.getLocations();
264         boolean notify = false;
265         for (URI lib : urls) {
266             File f = new File(lib);
267             if (f.exists() && lastModifieds.containsKey(f.getAbsolutePath())) {
268                 if (lastModifieds.get(f.getAbsolutePath()) !=f.lastModified()) {
269                     //Utils.getDefaultLogger().info("Changed : " + this);
270                     notify = true;
271                     break;
272                 }
273             }
274         }
275         if(notify) {
276             fireChangeEvent();
277         }
278     }

301     /**
302      * Ensure that this module is {@link ModuleState#RESOLVED resolved}.
303      *
304      * <p>
305      * If the module is already resolved, this method does nothing.
306      * Otherwise, iterate over all declared ModuleDependency instances and use the
307      * associated <code>ModuleRegistry</code> to resolve it. After successful
308      * completion of this method, the module state is 
309      * {@link ModuleState#RESOLVED}.
310      *
311      * @throws com.sun.enterprise.module.ResolveError if any of the declared dependency of this module
312      * cannot be satisfied
313      */
314     public synchronized void resolve() throws ResolveError {
315         
316         // already resolved ?
317         if (state==ModuleState.ERROR)
318             throw new ResolveError("Module " + getName() + " is in ERROR state");
319         if (state.compareTo(ModuleState.RESOLVED)>=0)
320             return;
321 
322         if (state==ModuleState.PREPARING) {
323             Utils.identifyCyclicDependency(this, Logger.getAnonymousLogger());
324             throw new ResolveError("Cyclic dependency with " + getName());
325         }
326         state = ModuleState.PREPARING;
327         
328         if (moduleDef.getImportPolicyClassName()!=null) {
329             try {
330                 Class<ImportPolicy> importPolicyClass = (Class<ImportPolicy>) getPrivateClassLoader().loadClass(moduleDef.getImportPolicyClassName());
331                 ImportPolicy importPolicy = importPolicyClass.newInstance();
332                 importPolicy.prepare(this);
333             } catch(ClassNotFoundException e) {
334                 state = ModuleState.ERROR;
335                 throw new ResolveError(e);
336             } catch(java.lang.InstantiationException e) {
337                 state = ModuleState.ERROR;
338                 throw new ResolveError(e);
339             } catch(IllegalAccessException e) {
340                 state = ModuleState.ERROR;
341                 throw new ResolveError(e);
342             }
343         }
344         for (ModuleDependency dependency : moduleDef.getDependencies()) {
345             ModuleImpl depModule = (ModuleImpl)registry.makeModuleFor(dependency.getName(), dependency.getVersion());
346             if (depModule==null) {
347                 state = ModuleState.ERROR;                
348                 throw new ResolveError(dependency + " referenced from " 
349                         + moduleDef.getName() + " is not resolved");
350             }
351 
352             //if (Utils.isLoggable(Level.INFO)) {
353             //    Utils.getDefaultLogger().info("For module" + getName() + " adding new dependent " + module.getName());
354             //}
355             dependencies.add(depModule);
356         }
357 
358         // once we have proper import/export filtering for modules, we can
359         // build a look-up table to improve performance
360 
361         // build up the complete list of transitive dependency modules, without any duplication,
362         // in a breadth-first fashion. The reason we do this in breadth-first is to reduce
363         // the search time based on the assumption that classes tend to be discovered in close dependencies. 
364         List<ModuleImpl> transitiveDependencies = new ArrayList<ModuleImpl>();
365         Set<ModuleImpl> transitiveDependenciesSet = new HashSet<ModuleImpl>();
366         LinkedList<ModuleImpl> q = new LinkedList<ModuleImpl>();
367         q.addAll(dependencies);
368         while(!q.isEmpty()) {
369             ModuleImpl m = q.removeFirst();
370             if(transitiveDependenciesSet.add(m)) {
371                 // first time visited
372                 transitiveDependencies.add(m);
373                 m.resolve();
374                 q.addAll(m.dependencies);
375             }
376         }
377 
378         for (ModuleImpl m : transitiveDependencies) {
379             getPrivateClassLoader().addDelegate(m.getClassLoader());
380         }
381 
382         //Logger.global.info("Module " + getName() + " resolved");
383         state = ModuleState.RESOLVED;
384         for (ModuleLifecycleListener l : registry.getLifecycleListeners()) {
385             l.moduleResolved(this);
386         }
387     }

389     /**
390      * Forces module startup. In most cases, the runtime will take care 
391      * of starting modules when they are first used. There could be cases where
392      * code need to manually start a sub module. Invoking this method will 
393      * move the module to the {@link ModuleState#READY ModuleState.READY}, the 
394      * {@link LifecyclePolicy#start Lifecycle.start} method will be invoked.
395      */
396     public void start() throws ResolveError {
397         
398         if (state==ModuleState.READY)
399             return;
400         
401         // ensure RESOLVED state
402         resolve();      
403         
404         for (Module subModules : dependencies) {
405             subModules.start();
406         }
407 
408         // time to initialize the lifecycle instance
409         if (moduleDef.getLifecyclePolicyClassName()!=null) {
410             try {
411                 Class<LifecyclePolicy> lifecyclePolicyClass = (Class<LifecyclePolicy>) getPrivateClassLoader().loadClass(moduleDef.getLifecyclePolicyClassName());
412                 lifecyclePolicy = lifecyclePolicyClass.newInstance();
413             } catch(ClassNotFoundException e) {
414                 state = ModuleState.ERROR;
415                 throw new ResolveError("ClassNotFound : " + e.getMessage(), e);
416             } catch(java.lang.InstantiationException e) {
417                 state = ModuleState.ERROR;
418                 throw new ResolveError(e);
419             } catch(IllegalAccessException e) {
420                 state = ModuleState.ERROR;
421                 throw new ResolveError(e);
422             }
423         }
424         if (lifecyclePolicy!=null) {
425             lifecyclePolicy.start(this);
426         }
427         state = ModuleState.READY;
428 
429         //Logger.global.info("Module " + getName() + " started");
430 
431         // module started. notify listeners
432         for (ModuleLifecycleListener listener : registry.getLifecycleListeners())
433             listener.moduleStarted(this);
434     }

436     /** 
437      * Forces module stop. In most cases, the runtime will take care of stopping
438      * modules when the last module user released its interest. However, in 
439      * certain cases, it may be interesting to manually stop the module. 
440      * Stopping the module means that the module is removed from the registry, 
441      * the class loader references are released (note : the class loaders will
442      * only be released if all instances of any class loaded by them are gc'ed). 
443      * If a <code>LifecyclePolicy</code> for this module is defined, the 
444      * {@link LifecyclePolicy#stop(Module) Lifecycle.stop(Module)}
445      * method will be called and finally the module state will be 
446      * returned to {@link ModuleState#NEW ModuleState.NEW}.
447      *
448      * @return true if unloading was successful
449      */
450     public boolean stop() {
451 
452         if (sticky) {
453             return false;
454         } 
455         
456         if (lifecyclePolicy!=null) {
457             lifecyclePolicy.stop(this);
458             lifecyclePolicy=null;
459         }
460         
461         detach();
462         
463         // we do NOT stop our sub modules which are shared...
464         for (ModuleImpl subModule : dependencies) {
465             if (!subModule.isShared()) {
466                 subModule.stop();
467             }
468         }
469         // release all sub modules class loaders
470         privateCL = null;
471         publicCL = null;
472         dependencies.clear();
473         
474         state = ModuleState.NEW;
475 
476         // notify interested listeners that I have stopped.
477         for (ModuleLifecycleListener listener : registry.getLifecycleListeners()) {
478             listener.moduleStopped(this);
479         }
480         return true;
481     }

535             getPrivateClassLoader().removeDelegate(module.getClassLoader());

591     public Iterable<Class> getProvidersClass(String name) {
592         List<Class> r = serviceClasses.get(name);
593         if(r!=null) return r;
594 
595         // the worst case scenario in the race situation is we end up creating the same list twice,
596         // which is not a big deal.
597 
598         for( String provider : getMetadata().getEntry(name).providerNames) {
599             if(r==null)
600                 r = new ArrayList<Class>();
601             try {
602                 r.add(getPrivateClassLoader().loadClass(provider));
603             } catch (ClassNotFoundException e) {
604                 LogHelper.getDefaultLogger().log(Level.SEVERE, "Failed to load "+provider+" from "+getName(),e);
605             }
606         }
607 
608         if(r==null)
609             r = Collections.emptyList();
610 
611         serviceClasses.put(name, r);
612         return r;
613     }
