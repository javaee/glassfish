66      public static boolean isEnabled() {
67          return org.jvnet.hk2.tracing.TracingUtilities.isEnabled();
68      }

70      public static File getLocation() {
71          String location = System.getProperty("hk2.module.loglocation");
72          if (location==null) {
73              location = System.getProperty("user.dir");
74          }
75          File f = new File(location);
76          if (f.isAbsolute()) {
77              return f;
78          } else {
79              return new File(System.getProperty("user.dir"), location);
80          }
81      }

83      public static void traceResolution(ModulesRegistry registry, long bundleId, String bundleName, Loader loader) {
84          traceState(registry, "resolved", bundleId, bundleName, loader);

91      public static void traceState(ModulesRegistry registry, String state, long bundleId, String bundleName, Loader loader) {
92          File out = new File(getLocation(), state + "-" + bundleId+".log");
93          Writer w = null;
94          try {
95              w = new FileWriter(out);
96              w.append("\n");
97              w.append("Module ["+ bundleId + "] " + state + " " + bundleName+"\n");
98              String prefix="-";
99              StackTraceElement[] stack = Thread.currentThread().getStackTrace();
100             InhabitantTracing tracing = TracingThreadLocal.get();
101             Iterator<Inhabitant> itr = tracing.inOrder();
102 
103             w.append("\n");
104             w.append("-----------------------------------\n");
105             w.append("Inhabitants / stack combination\n");
106             w.append("-----------------------------------\n");
107 
108             String currentBundleName = bundleName;
109             
110             for (int i=0;i<stack.length;i++) {
111                 StackTraceElement element = stack[i];
112                 if (element.getClassName().equals(AbstractInhabitantImpl.class.getName()) &&
113                         element.getMethodName().equals("get"))  {
114                     // now let's find out the first non hk2 class asking for this...
115                     int j=i+1;
116                     for (;j<stack.length;j++) {
117                         StackTraceElement caller = stack[j];
118                         if (!caller.getClassName().contains("hk2")) {
119                             break;
120                         }
121                     }
122                     if (itr.hasNext()) {
123                         Inhabitant reason = itr.next();
124                         StackTraceElement caller = stack[j];
125                         Module m = null;
126                         try {
127                             if (reason.isActive()) {
128                                 m = registry.find(loader.loadClass(reason.typeName()));
129                             }
130                         } catch (ClassNotFoundException e) {
131                             m = null;
132                         }
133                         if (m!=null && !m.getModuleDefinition().getName().equals(currentBundleName)) {
134                             w.append("Looks like module " + currentBundleName + " was " + state + " because "+
135                                     m.getName() + " was " + state + "\nSince " + m.getName() + " contains ");
136                         } else {
137                             w.append("Requested by ");
138                         }
139                         if (m!=null)
140                             currentBundleName = m.getName();
141 
142                         w.append(reason + " called from " + caller.getClassName()
143                                 + "." + caller.getMethodName() + ":" + caller.getLineNumber()+" metadata \n" + reason.metadata()+"\n");
144                         w.append("\n");
145 
146                     }
147 
148                 }
149 
150             }
151             w.append("\n");
152 
153             w.append("-----------------------------------\n");
154             w.append("Inhabitants initialization stack\n");
155             w.append("-----------------------------------\n");
156             itr = tracing.inOrder();
157             while (itr.hasNext()) {
158                 Inhabitant i = itr.next();
159                 w.append(prefix +"> requested from "+ i+"\n");
160                 prefix=prefix+"-";
161             }
162 
163             w.append("\n");
164 
165             w.append("---------------------------\n");
166             w.append("Complete thread stack Trace\n");
167             w.append("---------------------------\n");
168 
169             for (StackTraceElement element : stack) {
170                 w.append(element.toString()+"\n");
171             }
172 
173 
174         } catch (IOException e) {
175             e.printStackTrace();  //To change body of catch statement use File | Settings | File Templates.
176         } finally {
177             if (w!=null) {
178                 try {
179                     w.close();
180                 } catch (IOException e) {
181                     e.printStackTrace();  //To change body of catch statement use File | Settings | File Templates.
182                 }
183             }
184         }
185 
186     }
