59  /**
60   * Abstraction of {@link JarFile} so that we can handle
61   * both a jar file and a directory image transparently.
62   *
63   * @author Kohsuke Kawaguchi
64   */
65  public abstract class Jar {
66      protected Jar() {}
67  
68      /**
69       * See {@link JarFile#getManifest()} for the contract.
70       */
71      public abstract Manifest getManifest() throws IOException;
72  
73      /**
74       * Loads all <tt>META-INF/habitats</tt> entries and store them to the list.
75       */
76      public abstract void loadMetadata(ModuleMetadata result);
77  
78      /**
79       * Gets the base name of the jar.
80       *
81       * <p>
82       * For example, "bar" for "bar.jar".
83       */
84      public abstract String getBaseName();
85  
86      public static Jar create(File file) throws IOException {
87          if(file.isDirectory())
88              return new Directory(file);
89          else
90              return new Archive(file);
91      }

93      private static final class Directory extends Jar {
94          private final File dir;
95  
96          public Directory(File dir) {
97              this.dir = dir;
98          }
99  
100         public Manifest getManifest() throws IOException {
101             File mf = new File(dir,JarFile.MANIFEST_NAME);
102             if(mf.exists()) {
103                 FileInputStream in = new FileInputStream(mf);
104                 try {
105                     return new Manifest(in);
106                 } finally {
107                     in.close();
108                 }
109             } else {
110                 return null;
111             }
112         }
113 
114         private File[] fixNull(File[] f) {
115             if(f==null) return new File[0];
116             else        return f;
117         }
118 
119         public void loadMetadata(ModuleMetadata result) {
120             File inhabitantFileLocation = new File(dir, InhabitantsFile.PATH);
121             if (!inhabitantFileLocation.exists()) {
122                 // rely on introspection...
123                 result.addHabitat("default", new InhabitantsDescriptor() {
124                     public String getSystemId() {
125                         return null;  //To change body of implemented methods use File | Settings | File Templates.
126                     }
127 
128 
129 
130                     public Iterable<InhabitantParser> createScanner() throws IOException {
131                         ParsingContext context = (new ParsingContext.Builder()).build();
132                         Parser parser = new Parser(context);
133                         parser.parse(dir, null);
134                         try {
135                             parser.awaitTermination();
136                         } catch (InterruptedException e) {
137                             throw new IOException(e);
138                         }
139 
140                         return new InhabitantIntrospectionScanner(context);
141                     }                    
142                 });
143             }
144             for( File svc : fixNull(new File(dir, InhabitantsFile.PATH).listFiles())) {
145                 if(svc.isDirectory())
146                     continue;
147 
148                 try {
149                     result.addHabitat(svc.getName(),
150                         new ByteArrayInhabitantsDescriptor(svc.getPath(), readFully(svc))
151                     );
152                 } catch(IOException e) {
153                     LogHelper.getDefaultLogger().log(Level.SEVERE, "Error reading habitats file from " + svc, e);
154                 }
155             }
156 
157             for( File svc : fixNull(new File(dir, SERVICE_LOCATION).listFiles()) ) {
158                 if(svc.isDirectory())
159                     continue;
160 
161                 try {
162                     result.load( svc.toURL(), svc.getName() );
163                 } catch(IOException e) {
164                     LogHelper.getDefaultLogger().log(Level.SEVERE, "Error reading service provider from " + svc, e);
165                 }
166             }
167         }
168 
169         public String getBaseName() {
170             return dir.getName();
171         }
172 
173         private byte[] readFully(File f) throws IOException {
174             byte[] buf = new byte[(int)f.length()];
175             DataInputStream in = new DataInputStream(new FileInputStream(f));
176             try {
177                 in.readFully(buf);
178                 return buf;
179             } finally {
180                 in.close();
181             }
182         }
183     }

185     private static final class Archive extends Jar {
186         private final JarFile jar;
187         private final File file;
188 
189         public Archive(File jar) throws IOException {
190             try {
191                 this.jar = new JarFile(jar);
192                 this.file = jar;
193             } catch (ZipException e) {
194                 // ZipException doesn't include this crucial information, so rewrap
195                 IOException x = new IOException("Failed to open " + jar);
196                 x.initCause(e);
197                 throw x;
198             }
199         }
200 
201         public Manifest getManifest() throws IOException {
202             return jar.getManifest();
203         }
204 
205         public void loadMetadata(ModuleMetadata result) {
206 
207             if (jar.getJarEntry(InhabitantsFile.PATH)==null
208                     && jar.getJarEntry(SERVICE_LOCATION)==null) {
209                 return;
210             }
211             Enumeration<JarEntry> entries = jar.entries();
212             while (entries.hasMoreElements()) {
213                 JarEntry entry = entries.nextElement();
214                 if (!entry.isDirectory() && entry.getName().startsWith(InhabitantsFile.PATH)) {
215                     String habitatName = entry.getName().substring(InhabitantsFile.PATH.length()+1);
216 
217                     try {
218                         result.addHabitat(habitatName,new ByteArrayInhabitantsDescriptor(
219                             "jar:"+file.toURL()+"!/"+entry.getName(),
220                             loadFully(entry)
221                         ));
222                     } catch(IOException e) {
223                         LogHelper.getDefaultLogger().log(Level.SEVERE, "Error reading inhabitants list in " + jar.getName(), e);
224                     }
225                 } else {
226                     if (!entry.isDirectory() && entry.getName().startsWith(SERVICE_LOCATION)) {
227                         String serviceName = entry.getName().substring(SERVICE_LOCATION.length()+1);
228 
229                         try {
230                             result.load( new URL("jar:"+file.toURL()+"!/"+entry.getName()), serviceName, jar.getInputStream(entry));
231                         } catch(IOException e) {
232                             LogHelper.getDefaultLogger().log(Level.SEVERE, "Error reading service provider in " + jar.getName(), e);
233                         }
234                     }
235                 }
236             }
237         }
238 
239         public String getBaseName() {
240             String name = file.getName();
241             int idx = name.lastIndexOf('.');
242             if(idx>=0)
243                 name = name.substring(0,idx);
244             return name;
245         }
246 
247         private byte[] loadFully(JarEntry e) throws IOException {
248             byte[] buf = new byte[(int)e.getSize()];
249             DataInputStream in = new DataInputStream(jar.getInputStream(e));
250             try {
251                 in.readFully(buf);
252                 return buf;
253             } finally {
254                 in.close();
255             }
256         }
257     }
