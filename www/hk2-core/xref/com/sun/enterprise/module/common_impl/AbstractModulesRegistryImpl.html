162     public Habitat createHabitat(String name, InhabitantsParser parser) throws ComponentException {
163         try {
164             Habitat habitat = parser.habitat;
165 
166             for (final Module module : getModules())
167                 parseInhabitants(module, name,parser);
168 
169             populateConfig(habitat);
170 
171             // default modules registry is the one that created the habitat
172             habitat.addIndex(new ExistingSingletonInhabitant<ModulesRegistry>(this),
173                     ModulesRegistry.class.getName(), null);
174             habitats.put(name, habitat);
175 
176             habitat.initialized();
177             
178             return habitat;
179         } catch (Exception e) {
180             throw new ComponentException("Failed to create a habitat",e);
181         }
182     }

184     protected void populateConfig(Habitat habitat) throws BootException {
185         boolean foundClass = false;
186         try {
187             ConfigParser configParser = new ConfigParser(habitat);
188             foundClass = true;
189             for (Populator p : habitat.getAllByContract(Populator.class)) {
190                 p.run(configParser);
191             }
192         } catch (NoClassDefFoundError e) {
193             if (foundClass) {
194                 // it must have been a different, unexpected problem
195                 throw e;
196             }
197             Logger.getAnonymousLogger().fine("config classes not present - skipping config population.");
198         }
199     }

206     /**
207      * Add a new <code>Repository</code> to this registry. From now on
208      * the repository will be used to procure requested module not yet registered
209      * in this registry instance. Repository can be searched in a particular 
210      * order (to accomodate performance requirements like looking at local 
211      * repositories first), a search order (1 to 100) can be specified when 
212      * adding a repository to the registry (1 is highest priority). 
213      * @param repository new repository to attach to this registry
214      * @param weight int value from 1 to 100 to specify the search order
215      */
216     public synchronized void addRepository(Repository repository, int weight) {
217         // check that we don't already have this repository
218         for (Repository repo : repositories.values()) {
219             if (repo.getLocation().equals(repository.getLocation())) {
220                 throw new RuntimeException("repository at " + repository.getLocation() + " already registered");
221             }
222         }
223         while (repositories.containsKey(weight)) {
224             weight++;
225         }
226         repositories.put(weight, repository);
227     }

285     public Module makeModuleFor(String name, String version, boolean resolve) throws ResolveError {
286         Module module;
287         Logger.getAnonymousLogger().fine("this.makeModuleFor("+name+ ", " +
288                 version + ", " + resolve + ") called.");
289         if(parent!=null) {
290             module = parent.makeModuleFor(name,version, resolve);
291             if(module!=null)        return module;
292         }
293 
294         module = modules.get(AbstractFactory.getInstance().createModuleId(name, version));
295         if (module==null) {
296             module = loadFromRepository(name, version);
297             if (module!=null) {
298                 add(module);
299             }
300         }
301         if (module!=null && resolve) {
302             try {
303                 module.resolve();
304             } catch(Throwable e) {
305                 module.uninstall();
306                 throw new ResolveError(e);
307             }
308         }
309         Logger.getAnonymousLogger().fine("this.makeModuleFor("+name+ ", " + version + ") returned " + module);
310         return module;
311     }

313     /**
314      * Find and return a loaded Module that has the package name in its list
315      * of exported interfaces.
316      *
317      * @param packageName the requested implementation package name.
318      * @return the <code>Module</code> instance implementing the package
319      * name or null if not found.
320      * @throws ResolveError if the module dependencies cannot be resolved
321      */
322     public Module makeModuleFor(String packageName) throws ResolveError {
323         if(parent!=null) {
324             Module m = parent.makeModuleFor(packageName);
325             if(m!=null)     return m;
326         }
327 
328         for (Module module : modules.values()) {
329             String[] exportedPkgs = module.getModuleDefinition().getPublicInterfaces();
330             for (String exportedPkg : exportedPkgs) {
331                 if (exportedPkg.equals(packageName)) {
332                     module.resolve();
333                     return module;
334                 }
335             }
336         }
337         return null;
338     }

340     protected Module loadFromRepository(String name, String version) {
341         Set<Integer> keys = repositories.keySet();
342         TreeSet<Integer> sortedKeys = new TreeSet<Integer>();
343         sortedKeys.addAll(keys);
344         for (Integer key : sortedKeys) {
345             Repository repo = repositories.get(key);
346             ModuleDefinition moduleDef = repo.find(name, version);
347             if (moduleDef!=null) {
348                 return newModule(moduleDef);
349             }
350         }
351         return null;
352     }

361     /**
362      * Add a new module to this registry. Once added, the module will be 
363      * available through one of the getServiceImplementor methods.
364      * @param newModule the new module
365      */
366     protected void add(Module newModule) {
367         //if (Utils.isLoggable(Level.INFO)) {
368         //    Utils.getDefaultLogger().info("New module " + newModule);
369         //}
370         assert newModule.getRegistry()==this;
371         // see if this module is already added. This can happen
372         // in case of OSGiModulesRegistryImpl which uses SynchronousBundleListener.
373         ModuleId id = AbstractFactory.getInstance().createModuleId(
374                 newModule.getModuleDefinition());
375         if (modules.get(id) != null) return;
376         modules.put(id, newModule);
377 
378         // pick up providers from this module
379         for( ModuleMetadata.Entry spi : newModule.getMetadata().getEntries() ) {
380             for( String name : spi.providerNames )
381                 providers.put(name,newModule);
382         }
383         for (Map.Entry<String, Habitat> entry : habitats.entrySet()) {
384             String name = entry.getKey();
385             Habitat h = entry.getValue();
386             try
387             {
388                 // TODO: should get the inhabitantsParser out of Main instead since
389                 // this could have been overridden
390                 parseInhabitants(newModule, name, createInhabitantsParser(h));
391             }
392             catch (IOException e)
393             {
394                 throw new RuntimeException("Not able to parse inhabitants information");
395             }
396         }
397     }

419     /** 
420      * Returns the list of shared Modules registered in this instance.
421      *
422      * <p>
423      * The returned list will not include the modules defined in the ancestor
424      * {@link AbstractModulesRegistryImpl}s.
425      *
426      * @return an umodifiable list of loaded modules
427      */
428     public Collection<Module> getModules() {
429 
430         // make a copy to avoid synchronizing since this API can be called while
431         // modules are added or removed by other threads.
432         Map<Integer,Repository> repos = new TreeMap<Integer,Repository>();
433         repos.putAll(repositories);
434 
435         // force repository extraction
436         Set<Integer> keys = repos.keySet();
437         TreeSet<Integer> sortedKeys = new TreeSet<Integer>();
438         sortedKeys.addAll(keys);
439         for (Integer key : sortedKeys) {
440             Repository repo = repos.get(key);
441             for (ModuleDefinition moduleDef : repo.findAll()) {
442                 if (modules.get(AbstractFactory.getInstance().createModuleId(moduleDef))==null) {
443                     Module newModule = newModule(moduleDef);
444                     if (newModule!=null) {
445                         // When some module can't get installed,
446                         // don't halt proceeding, instead continue
447                         add(newModule);
448                         // don't resolve such modules, we just want to know about them
449                     }
450                 }
451             }
452         }
453         return modules.values();
454     }

554     /**
555      * Registers a running service, this is useful when other components need
556      * to have access to a provider of a service without having to create
557      * a new instance and initialize it.
558      * @param serviceClass the service interface
559      * @param provider the provider of that service.
560      */
561     public <T> void registerRunningService(Class<T> serviceClass, T provider) {
562         CopyOnWriteArrayList rs = runningServices.get(serviceClass);
563         if (rs==null) {
564             rs = new CopyOnWriteArrayList<T>();
565             CopyOnWriteArrayList existing = runningServices.putIfAbsent(serviceClass, rs);
566             if(existing!=null)
567                 rs = existing;
568         }
569         rs.add(provider);
570     }

572     /**
573      * Removes a running service, this is useful when a service instance is no longer
574      * available as a provider of a service.
575      */
576     public <T> boolean unregisterRunningService(Class<T> serviceClass, T provider) {
577         CopyOnWriteArrayList rs = runningServices.get(serviceClass);
578         if (rs==null) {
579             return false;
580         }
581         return rs.remove(provider);
