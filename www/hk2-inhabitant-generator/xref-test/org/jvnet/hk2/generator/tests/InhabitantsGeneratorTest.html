437     private Set<DescriptorImpl> getAllDescriptorsFromInputStream(InputStream is) throws IOException {
438         BufferedReader pr = new BufferedReader(new InputStreamReader(is));
439         
440         Set<DescriptorImpl> retVal = new HashSet<DescriptorImpl>();
441         while (pr.ready()) {
442             DescriptorImpl di = new DescriptorImpl();
443             
444             if (!di.readObject(pr)) {
445                 continue;
446             }
447             
448             retVal.add(di);
449         }
450         
451         return retVal;
452     }

454     private void checkDescriptors(Set<DescriptorImpl> dis) {
455         for (DescriptorImpl di : dis) {
456             Assert.assertTrue("Did not find " + di + " in the expected descriptors <<<" +
457               EXPECTED_DESCRIPTORS + ">>>", EXPECTED_DESCRIPTORS.containsKey(di));
458             
459             // The rank is not part of the calculated equals or hash code (since it can change
460             // over the course of the lifeycle of the object) and hence must be checked
461             // separately from the containsKey above
462             int expectedRank = EXPECTED_DESCRIPTORS.get(di);
463             Assert.assertEquals(expectedRank, di.getRanking());
464         }
465         
466         Assert.assertEquals("Expecting " + EXPECTED_DESCRIPTORS.size() + " descriptors, but only got " + dis.size(),
467                 EXPECTED_DESCRIPTORS.size(), dis.size());
468     }

475     @Test
476     public void testDefaultDirectoryGeneration() throws IOException {
477         String argv[] = new String[2];
478         
479         argv[0] = FILE_ARGUMENT;
480         argv[1] = gendirDirectory.getAbsolutePath();
481         
482         File defaultOutput = new File(inhabitantsDirectory, DEFAULT);
483         if (defaultOutput.exists()) {
484             // Start with a clean plate
485             Assert.assertTrue(defaultOutput.delete());
486         }
487         
488         try {
489             int result = HabitatGenerator.embeddedMain(argv);
490             Assert.assertEquals("Got error code: " + result, 0, result);
491             
492             Assert.assertTrue("did not generate " + defaultOutput.getAbsolutePath(),
493                     defaultOutput.exists());
494             
495             Set<DescriptorImpl> generatedImpls = getAllDescriptorsFromInputStream(
496                     new FileInputStream(defaultOutput));
497             
498             checkDescriptors(generatedImpls);
499         }
500         finally {
501             // The test should be clean
502             defaultOutput.delete();
503         }
