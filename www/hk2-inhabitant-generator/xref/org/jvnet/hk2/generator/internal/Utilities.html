290         this.verbose = verbose;
291         this.searchPath = searchPath;
292         
293         // We can pre-load the cache with some obvious ones and thus reduce searching quite a bit
294         for (KnownClassData kcd : KNOWN_DATA) {
295             ISA_CONTRACT.put(kcd.getClazz(), kcd.isIsa_contract());
296             ISA_SCOPE.put(kcd.getClazz(), kcd.isIsa_scope());
297             ISA_QUALIFIER.put(kcd.getClazz(), kcd.isIsa_qualifier());
298             FOUND_SUPERCLASS.put(kcd.getClazz(), kcd.getSuperclass());
299             FOUND_INTERFACES.put(kcd.getClazz(), kcd.getiFaces());
300             FOUND_GENERATORS.put(kcd.getClazz(), null);
301         }
302     }

304     /**
305      * Will look for a file on a last-ditch sort of effort using the searchHere thing (jar or directory)
306      * and then also in the classpath
307      * 
308      * @param searchHere
309      * @param dotDelimitedName
310      * @param searchClassPath true if the classpath should be searched as well
311      * @return an IOStream if the file could be located
312      * @throws IOException
313      */
314     private InputStream findClass(File searchHere, String dotDelimitedName, boolean searchClassPath, String calledFrom) throws IOException {
315         if (verbose && searchClassPath) {
316             System.out.println("Looking for " + dotDelimitedName + " for discovery of " + calledFrom);
317         }
318         
319         if (searchHere.isDirectory()) {
320             String properPathName = dotDelimitedName.replace('.', File.separatorChar) + DOT_CLASS;
321             
322             File fullFile = new File(searchHere, properPathName);
323             
324             if (fullFile.exists()) {
325                 return new FileInputStream(fullFile);
326             }
327         }
328         else {
329             JarFile jar = new JarFile(searchHere);
330             
331             String entryName = dotDelimitedName.replace('.', '/') + DOT_CLASS;
332             ZipEntry entry = jar.getEntry(entryName);
333             if (entry != null) {
334                 return jar.getInputStream(entry);
335             }
336         }
337         
338         if (!searchClassPath) return null;
339         
340         // Handle classpath
341         String classpath = searchPath;
342         if (classpath == null) return null;
343         
344         StringTokenizer st = new StringTokenizer(classpath, File.pathSeparator);
345         while (st.hasMoreTokens()) {
346             String pathElement = st.nextToken();
347             
348             File nextSearchGuy = new File(pathElement);
349             InputStream is = findClass(nextSearchGuy, dotDelimitedName, false, calledFrom);
350             if (is != null) {
351                 return is;
352             }
353         }
354         
355         return null;
356     }

359         ISA_CONTRACT.put(dotDelimitedName, false);

366     /**
367      * Returns true if the given class is a contract
368      * 
369      * @param searchHere The file or jar to look in
370      * @param dotDelimitedName The fully qualified class name to look for
371      * @return true if this can determine that this is a contract
372      */
373     private boolean isClassAContract(File searchHere, String dotDelimitedName) {
374         if (ISA_CONTRACT.containsKey(dotDelimitedName)) {
375             return ISA_CONTRACT.get(dotDelimitedName);
376         }
377         
378         InputStream is = null;
379         try {
380             is = findClass(searchHere, dotDelimitedName, true, "isaContract");
381             if (is == null) {
382                 nullCaches(dotDelimitedName);
383                 
384                 return false;
385             }
386             
387             ClassReader reader = new ClassReader(is);
388             
389             ContractClassVisitor ccv = new ContractClassVisitor(searchHere, CONTRACT_WITH_SLASHES, dotDelimitedName);
390             
391             reader.accept(ccv, ClassReader.SKIP_CODE | ClassReader.SKIP_DEBUG | ClassReader.SKIP_FRAMES);
392             
393             return ccv.isALookedForThing();
394         }
395         catch (IOException ioe) {
396             nullCaches(dotDelimitedName);
397             
398             return false;
399         }
400         finally {
401             if (is != null) {
402                 try {
403                     is.close();
404                 }
405                 catch (IOException e) {
406                     // Ignore
407                 }
408             }
409         }
410     }

600     private void getAssociatedSuperclassContracts(File searchHere, String dotDelimitedName, Set<String> addToMe) {
601         if (!addToMe.contains(dotDelimitedName) && isClassAContract(searchHere, dotDelimitedName)) {
602             addToMe.add(dotDelimitedName);
603         }
604         
605         String dotDelimitedSuperclass = getSuperclass(searchHere, dotDelimitedName);
606         if (dotDelimitedSuperclass != null) {
607             getAssociatedSuperclassContracts(searchHere, dotDelimitedSuperclass, addToMe);
608         }
609     }

611     /**
612      * Gets the contracts associated with the name passed in
613      * @param searchHere
614      * @param dotDelimitedName
615      * @return The set of contracts associated with this dotDelimited name (ordered iterator)
616      */
617     public Set<String> getAssociatedContracts(File searchHere, String dotDelimitedName) {
618         LinkedHashSet<String> retVal = new LinkedHashSet<String>();
619         retVal.add(dotDelimitedName);
620         
621         getAssociatedSuperclassContracts(searchHere, dotDelimitedName, retVal);
622         
623         while (dotDelimitedName != null) {
624             // getAssociatedSuperclassContracts is guaranteed to fill in the INTERFACES cache
625             Set<String> allInterfaces = FOUND_INTERFACES.get(dotDelimitedName);
626             if (allInterfaces == null) {
627                 dotDelimitedName = getSuperclass(searchHere, dotDelimitedName);
628                 
629                 continue;
630             }
631             
632             for (String dotDelimitedInterface : allInterfaces) {
633                 if (isClassAContract(searchHere, dotDelimitedInterface)) {
634                     retVal.add(dotDelimitedInterface);  
635                 }
636                 
637                 addSubInterface(searchHere, dotDelimitedInterface, retVal);
638             }
639             
640             dotDelimitedName = getSuperclass(searchHere, dotDelimitedName);
641         }
642         
643         return retVal;
644     }
645     
646     private void addSubInterface(File searchHere, String dotDelimitedInterface, LinkedHashSet<String> retVal) {
647         Set<String> subInterfaces = FOUND_INTERFACES.get(dotDelimitedInterface);
648         if (subInterfaces == null) return;
649         
650         for (String dotDelimitedSubInterface : subInterfaces) {
651             if (isClassAContract(searchHere, dotDelimitedSubInterface)) {
652                 retVal.add(dotDelimitedSubInterface);
653             }
654             
655             addSubInterface(searchHere, dotDelimitedSubInterface, retVal);
656         }
657     }

659     private class ContractClassVisitor extends AbstractClassVisitorImpl {
660         private final String cacheKey;
661         private final String lookForMe;
662         private final File searchHere;
663         
664         private final Map<String, String> methodNameToMetadataKey =
665                 new HashMap<String, String>();
666         
667         private boolean isLookedFor = false;
668         
669         private boolean isContract = false;
670         private boolean isScope = false;
671         private boolean isQualifier = false;
672         
673         private String dotDelimitedSuperclass;
674         
675         private ContractClassVisitor(File searchHere, String lookForMe, String cacheKey) {
676             this.searchHere = searchHere;
677             this.lookForMe = lookForMe;
678             this.cacheKey = cacheKey;
679         }
680         
681         /* (non-Javadoc)
682          * @see org.objectweb.asm.ClassVisitor#visit(int, int, java.lang.String, java.lang.String, java.lang.String, java.lang.String[])
683          */
684         @Override
685         public void visit(int version,
686                 int access,
687                 String name,
688                 String signature,
689                 String superName,
690                 String[] interfaces) {
691             if (!FOUND_INTERFACES.containsKey(cacheKey)) {
692                 LinkedHashSet<String> iFaces = new LinkedHashSet<String>();
693                 
694                 for (String iFace : interfaces) {
695                     String iWithDots = iFace.replace('/', '.');
696                     iFaces.add(iWithDots);
697                     
698                     // Fill in the cache with all sub interfaces
699                     getSuperclass(searchHere, iWithDots);
700                 }
701                 
702                 FOUND_INTERFACES.put(cacheKey, iFaces);
703             }
704             
705             if (superName == null) {
706                 FOUND_SUPERCLASS.put(cacheKey, null);
707                 
708                 return;

1050     /* package */ static String getListActualType(String signature) {
1051         if (signature == null) return null;
1052         if (!signature.contains(LIST_WITH_PTYPE)) return null;
1053         
1054         int ltIndex = signature.indexOf('<');
1055         int gtIndex = signature.indexOf('>');
1056         
1057         if (ltIndex < 0 || gtIndex < 0) return null;
1058         
1059         String pType = signature.substring(ltIndex + 1, gtIndex);
1060         if (!pType.startsWith("L")) return null;
1061         
1062         pType = pType.substring(1, pType.length() - 1);  // Remove L and ;
1063         
1064         return pType.replace('/', '.');
1065     }
