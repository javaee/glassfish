55  /**
56   * Note that the ordering of the test methods in this
57   * class is very important.  Therefore we do not rely
58   * on the Junit ordering of &#64;Test methods, instead
59   * having one test that calls the other tests in the
60   * order in which they have to go in order to

66  public class ErrorServiceTest {
67      private final static String TEST_NAME = "ErrorServiceTest";
68      private final static ServiceLocator locator = LocatorHelper.create(TEST_NAME, new ErrorServiceModule());
69      
70      /** The string the exception will throw */
71      public final static String EXCEPTION_STRING = "Expected Exception";
72      
73      /** Another exception string */
74      public final static String EXCEPTION_STRING_DUEX = "Expected Exception Duex";
75      
76      private void testLookupPriorToFixing(ErrorServiceImpl esi) {
77          esi.clear();
78          
79          FaultyClass fc = locator.getService(FaultyClass.class);
80          Assert.assertNull(fc);
81          
82          Descriptor faultyDesc = locator.getBestDescriptor(BuilderHelper.createContractFilter(FaultyClass.class.getName()));
83          Assert.assertNotNull(faultyDesc);
84          
85          ActiveDescriptor<?> fromError = esi.getDescriptor();
86          Assert.assertNotNull(fromError);
87          
88          Assert.assertEquals(faultyDesc, fromError);
89          
90          Assert.assertNull(esi.getInjectee());
91          
92          MultiException me = esi.getMe();
93          Assert.assertNotNull(me);
94          
95          Assert.assertTrue("Expected " + EXCEPTION_STRING + " but got " + me.getMessage(), me.getMessage().contains(EXCEPTION_STRING));
96      }

233     @Test

237         testLookupPriorToFixing(esi);
