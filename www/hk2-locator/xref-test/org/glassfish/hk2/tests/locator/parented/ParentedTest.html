87      /**
88       * Tests three generations of locators
89       */
90      @Test
91      public void testBasicParenting() {
92          ServiceLocator grandpa = factory.create(GRANDPA);
93          ServiceLocator dad = factory.create(DAD, grandpa);
94          ServiceLocator me = factory.create(ME, dad);
95          
96          SimpleService grandPaSimpleService = new SimpleService();
97          SimpleService dadSimpleService = new SimpleService();
98          SimpleService meSimpleService = new SimpleService();
99          
100         ServiceLocatorUtilities.addOneConstant(grandpa, grandPaSimpleService);
101         ServiceLocatorUtilities.addOneConstant(dad, dadSimpleService);
102         ServiceLocatorUtilities.addOneConstant(me, meSimpleService);
103         
104         List<SimpleService> grandpaLocators = grandpa.getAllServices(SimpleService.class);
105         List<SimpleService> dadLocators = dad.getAllServices(SimpleService.class);
106         List<SimpleService> meLocators = me.getAllServices(SimpleService.class);
107         
108         Assert.assertNotNull(grandpaLocators);
109         Assert.assertEquals(1, grandpaLocators.size());
110         Assert.assertEquals(grandPaSimpleService, grandpaLocators.get(0));
111         
112         Assert.assertNotNull(dadLocators);
113         Assert.assertEquals(2, dadLocators.size());
114         Assert.assertEquals(dadSimpleService, dadLocators.get(0));
115         Assert.assertEquals(grandPaSimpleService, dadLocators.get(1));
116         
117         Assert.assertNotNull(meLocators);
118         Assert.assertEquals(3, meLocators.size());
119         Assert.assertEquals(meSimpleService, meLocators.get(0));
120         Assert.assertEquals(dadSimpleService, meLocators.get(1));
121         Assert.assertEquals(grandPaSimpleService, meLocators.get(2));
122         
123         // Make sure the most normal case returns the right guy
124         Assert.assertEquals(meSimpleService, me.getService(SimpleService.class));
125     }

127     /**
128      * Tests that a child getting shutdown will not adversly affect the services
129      * in the parent
130      */
131     @Test
132     public void testShutdownOfChild() {
133         ServiceLocator child1 = factory.create(CHILD1, locator);
134         
135         ServiceLocatorUtilities.bind(child1, new Binder() {
136 
137             @Override
138             public void bind(DynamicConfiguration config) {
139                 config.addActiveDescriptor(SingletonServiceInChild.class);
140                 
141             }
142             
143         });
144         
145         SingletonServiceInParent ssip = child1.getService(SingletonServiceInParent.class);
146         Assert.assertNotNull(ssip);
147         Assert.assertFalse(ssip.isShutdown());
148         
149         SingletonServiceInChild ssic = child1.getService(SingletonServiceInChild.class);
150         Assert.assertNotNull(ssic);
151         Assert.assertFalse(ssic.isShutdown());
152         
153         // Shutting down the child should NOT shutdown the service in the parent
154         factory.destroy(child1);
155         
156         Assert.assertFalse(ssip.isShutdown());
157         Assert.assertTrue(ssic.isShutdown());
158     }
