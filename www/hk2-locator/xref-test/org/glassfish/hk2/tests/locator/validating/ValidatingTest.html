151     /**
152      * This test ensures that validation changes in the parent properly affects children
153      */
154     @Test
155     public void testValidationChangeInParent() {
156         ServiceLocator childLocator = LocatorHelper.create("child", locator, null);
157         
158         DynamicValidator dv = childLocator.getService(DynamicValidator.class);
159         
160         dv.addBadGuy(DynamicServiceImpl1.class.getName());
161         
162         List<DynamicService> services = childLocator.getAllServices(DynamicService.class);
163         
164         // Only one, because the other is not valid in the parent
165         Assert.assertEquals(1, services.size());
166         
167         DynamicService ds2 = services.get(0);
168         
169         Assert.assertEquals(2, ds2.getImplNumber());
170         
171         // Now dynamically change the validation, so that 1 is ok again
172         dv.removeBadGuy(DynamicServiceImpl1.class.getName());
173         
174         services = childLocator.getAllServices(DynamicService.class);
175         
176         // Should now have both services
177         Assert.assertEquals(2, services.size());
178         
179         DynamicService ds1 = services.get(0);
180         ds2 = services.get(1);
181         
182         Assert.assertEquals(1, ds1.getImplNumber());
183         Assert.assertEquals(2, ds2.getImplNumber());
184     }

186     /**
187      * Tests a state based validation using getService
188      */
189     @Test
190     public void testStateBasedValidation() {
191         // Overrides global one
192         ServiceLocator locator = LocatorHelper.create("ValidationTest_2", null);
193         
194         ServiceLocatorUtilities.bind(locator, new Binder() {
195 
196             @Override
197             public void bind(DynamicConfiguration config) {
198                 config.bind(BuilderHelper.link(DynamicServiceImpl1.class.getName()).
199                         to(DynamicService.class.getName()).
200                         in(Singleton.class.getName()).
201                         build());
202                 config.bind(BuilderHelper.link(DynamicServiceImpl2.class.getName()).
203                         to(DynamicService.class.getName()).
204                         in(Singleton.class.getName()).
205                         build());
206                 config.bind(BuilderHelper.link(StateBasedValidationService.class.getName()).
207                         to(ValidationService.class.getName()).
208                         in(Singleton.class.getName()).
209                         build());
210                 
211             }
212             
213         });
214         
215         // First get one
216         StateBasedValidationService sbvs = (StateBasedValidationService)
217                 locator.getService(ValidationService.class);
218         Assert.assertNotNull(sbvs);
219         
220         sbvs.setCurrentState(1);
221         
222         DynamicService d1 = locator.getService(DynamicService.class);
223         Assert.assertNotNull(d1);
224         Assert.assertEquals(1, d1.getImplNumber());
225         
226         // Now switch the state to state 2
227         sbvs.setCurrentState(2);
228         
229         DynamicService d2 = locator.getService(DynamicService.class);
230         Assert.assertNotNull(d2);
231         Assert.assertEquals(2, d2.getImplNumber());
232     }
