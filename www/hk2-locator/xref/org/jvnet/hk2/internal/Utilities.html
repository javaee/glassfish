123      * Checks that the incoming lookup type is not improper in some way
124      * 
125      * @param checkMe  class to check
126      */
127     public static void checkLookupType(Class<?> checkMe) {
128         if (!checkMe.isAnnotation()) return;
129         
130         // If it is in annotation we need to ensure it is either a
131         // scope or a qualifier
132         if (checkMe.isAnnotationPresent(Scope.class)) return;
133         if (checkMe.isAnnotationPresent(Qualifier.class)) return;
134         
135         throw new IllegalArgumentException("Lookup type " + checkMe + " must be a scope or annotation");
136     }

218      * Calls the list of error services for the list of errors
219      * 
220      * @param results    the results
221      * @param callThese  the services to call
222      */
223     public static void handleErrors(NarrowResults results, LinkedList<ErrorService> callThese) {
224         Collector collector = new Collector();
225         for (ErrorResults errorResult : results.getErrors()) {
226             for (ErrorService eService : callThese) {
227                 try {
228                     eService.onFailure(new ErrorInformationImpl(
229                             ErrorType.FAILURE_TO_REIFY,
230                             errorResult.getDescriptor(),
231                             errorResult.getInjectee(),
232                             errorResult.getMe()));
233                 }
234                 catch (MultiException me) {
235                     for (Throwable th : me.getErrors()) {
236                         collector.addThrowable(th);
237                     }
238                 }
239                 catch (Throwable th) {
240                     collector.addThrowable(th);
241                 }
242             }
243         }
244         
245         collector.throwIfErrors();
246     }

248     /**
249      * Loads the class using the loader from the given descriptor or the
250      * classloader of the utilities class otherwise
251      * 
252      * @param loadMe The fully qualified class name
253      * @param fromMe The descriptor to use for the loader
254      * @param collector The error collector to fill in if this returns null
255      * @return null on failure to load (the failure will be added to the collector)
256      */
257     public static Class<?> loadClass(String loadMe, Descriptor fromMe, Collector collector) {
258         HK2Loader loader = fromMe.getLoader();
259         if (loader == null) {
260             ClassLoader cl = Utilities.class.getClassLoader();
261             if (cl == null) {
262                 cl = ClassLoader.getSystemClassLoader();
263             }
264             
265             try {
266                 return cl.loadClass(loadMe);
267             }
268             catch (Throwable th) {
269                 collector.addThrowable(th);
270                 return null;
271             }
272         }
273         
274         try {
275             return loader.loadClass(loadMe);
276         }
277         catch (Throwable th) {
278             if (th instanceof MultiException) {
279                 MultiException me = (MultiException) th;
280                 
281                 for (Throwable th2 : me.getErrors()) {
282                     collector.addThrowable(th2);
283                 }
284             }
285             else {
286                 collector.addThrowable(th);
287             }
288             
289             return null;
290         }
291         
292     }

294     /**
295      * Load the given class for the given injectee.
296      *
297      * @param implementation  the impl class name string
298      * @param injectee        the injectee
299      *
300      * @return The class represented by this implementation and injectee
301      */
302     public static Class<?> loadClass(String implementation, Injectee injectee) {
303         ClassLoader loader;
304         if (injectee != null) {
305             AnnotatedElement parent = injectee.getParent();
306 
307             if (parent instanceof Constructor) {
308                 loader = ((Constructor<?>) parent).getDeclaringClass().getClassLoader();
309             }
310             else if (parent instanceof Method) {
311                 loader = ((Method) parent).getDeclaringClass().getClassLoader();
312             }
313             else if (parent instanceof Field) {
314                 loader = ((Field) parent).getDeclaringClass().getClassLoader();
315             }
316             else {
317                 loader = injectee.getClass().getClassLoader();
318             }
319         }
320         else {
321             loader = Utilities.class.getClassLoader();
322         }
323         
324         try {
325             return loader.loadClass(implementation);
326         }
327         catch (Throwable th) {
328             throw new MultiException(th);
329         }
330     }

332     /**
333      * Will return the class of the injection resolver annotation type, or null if
334      * no injection resolver annotation can be found
335      * 
336      * @param desc The reified descriptor to find the injection resolution on
337      * @return The annotation type for this injection resolver
338      */
339     @SuppressWarnings("unchecked")
340     public static Class<? extends Annotation> getInjectionResolverType(ActiveDescriptor<?> desc) {
341         for (Type advertisedType : desc.getContractTypes()) {
342             Class<?> rawClass = ReflectionHelper.getRawClass(advertisedType);
343             
344             if (!InjectionResolver.class.equals(rawClass)) continue;
345             
346             // Found the InjectionResolver
347             if (!(advertisedType instanceof ParameterizedType)) {
348                 return null;
349             }
350             
351             Type firstType = getFirstTypeArgument(advertisedType);
352             if (!(firstType instanceof Class)) {
353                 return null;
354             }
355             
356             Class<?> retVal = (Class<?>) firstType;
357             
358             if (!Annotation.class.isAssignableFrom(retVal)) {
359                 return null;
360             }
361             
362             return (Class<? extends Annotation>) retVal;
363         }
364         
365         return null;
366     }

369      * This method assumes that this class has already been checked...
370      * 
371      * @param factoryClass The non-null factory class
372      * @return the CLASS version of what the factory produces
373      */
374     private static Class<?> getFactoryProductionClass(Class<?> factoryClass) {
375         for (Type type : factoryClass.getGenericInterfaces()) {
376             Class<?> rawClass = ReflectionHelper.getRawClass(type);
377             if (rawClass == null) continue;
378             
379             if (!Factory.class.equals(rawClass)) continue;

381             Type firstType = getFirstTypeArgument(type);
382             
383             return ReflectionHelper.getRawClass(firstType);
384         }
385         
386         throw new MultiException(new IllegalArgumentException(factoryClass.getName() + " is not a factory"));
387     }

421     private static Set<Type> getAutoAdvertisedTypes(Type t) {
422         LinkedHashSet<Type> retVal = new LinkedHashSet<Type>();
423         retVal.add(t);
424         
425         Class<?> rawClass = ReflectionHelper.getRawClass(t);
426         if (rawClass == null) return retVal;
427         
428         Type genericSuperclass = rawClass.getGenericSuperclass();
429         while (genericSuperclass != null) {
430             Class<?> rawSuperclass = ReflectionHelper.getRawClass(genericSuperclass);
431             if (rawSuperclass == null) break;
432             
433             if (rawSuperclass.isAnnotationPresent(Contract.class)) {
434                 retVal.add(genericSuperclass);
435             }
436             
437             genericSuperclass = rawSuperclass.getGenericSuperclass();
438         }
439         
440         while (rawClass != null) {
441             for (Type iface : rawClass.getGenericInterfaces()) {
442                 Class<?> ifaceClass = ReflectionHelper.getRawClass(iface);
443                 if (ifaceClass.isAnnotationPresent(Contract.class)) { 
444                     retVal.add(iface);
445                 }
446             }
447             
448             rawClass = rawClass.getSuperclass();
449         }
450         
451         return retVal;
452     }

454     /**
455      * Creates a reified automatically generated descriptor
456      * 
457      * @param clazz The class to create the desciptor for
458      * @param locator The service locator for whom we are creating this
459      * @return A reified active descriptor
460      * 
461      * @throws MultiException if there was an error in the class
462      * @throws IllegalArgumentException If the class is null
463      */
464     public static <T> ActiveDescriptor<T> createAutoDescriptor(Class<T> clazz, ServiceLocatorImpl locator)
465             throws MultiException, IllegalArgumentException {
466         if (clazz == null) throw new IllegalArgumentException();
467         
468         Collector collector = new Collector();
469         
470         Creator<T> creator;
471         Set<Annotation> qualifiers;
472         Set<Type> contracts;
473         Class<? extends Annotation> scope;
474         String name;
475         
476         // Qualifiers naming dance
477         qualifiers = ReflectionHelper.getQualifierAnnotations(clazz);
478         name = getNameFromAllQualifiers(qualifiers, clazz);
479         qualifiers = getAllQualifiers(clazz, name, collector);  // Fixes the @Named qualifier if it has no value
480         
481         contracts = getAutoAdvertisedTypes(clazz);
482         ScopeInfo scopeInfo = getScopeInfo(clazz, null, collector);
483         scope = scopeInfo.getAnnoType();
484             
485         creator = new ClazzCreator<T>(locator, clazz, null, collector);
486         
487         collector.throwIfErrors();
488         
489         Map<String, List<String>> metadata = new HashMap<String, List<String>>();
490         if (scopeInfo.getScope() != null) {
491             BuilderHelper.getMetadataValues(scopeInfo.getScope(), metadata);
492         }
493         
494         for (Annotation qualifier : qualifiers) {
495             BuilderHelper.getMetadataValues(qualifier, metadata);
496         }
497         
498         return new AutoActiveDescriptor<T>(
499                 clazz,
500                 creator,
501                 contracts,
502                 scope,
503                 name,
504                 qualifiers,
505                 0,
506                 metadata);
507     }

562     /**
563      * Just creates the thing, doesn't try to do anything else
564      * @param injectMe The object to inject into
565      * @param locator The locator to find the injection points with
566      */
567     public static void justInject(Object injectMe, ServiceLocatorImpl locator) {
568         if (injectMe == null) throw new IllegalArgumentException();
569         
570         Class<?> baseClass = injectMe.getClass();
571         
572         Collector collector = new Collector();
573         
574         Set<Field> fields = findInitializerFields(baseClass, locator, collector);
575         Set<Method> methods = findInitializerMethods(baseClass, locator, collector);
576         
577         collector.throwIfErrors();
578         
579         for (Field field : fields) {
580             InjectionResolver<?> resolver = getInjectionResolver(locator, field);
581             
582             List<Injectee> injecteeFields = Utilities.getFieldInjectees(field);
583             
584             validateSelfInjectees(null, injecteeFields, collector);
585             collector.throwIfErrors();
586             
587             Injectee injectee = injecteeFields.get(0);
588             
589             Object fieldValue = resolver.resolve(injectee, null);
590             
591             setAccessible(field);
592             
593             try {
594                 field.set(injectMe, fieldValue);
595             }
596             catch (IllegalAccessException e) {
597                 throw new MultiException(e);
598             }
599         }
600         
601         for (Method method : methods) {
602             InjectionResolver<?> resolver = getInjectionResolver(locator, method);
603             
604             List<Injectee> injectees = Utilities.getMethodInjectees(method);
605             
606             validateSelfInjectees(null, injectees, collector);
607             collector.throwIfErrors();
608             
609             Object args[] = new Object[injectees.size()];
610             
611             for (Injectee injectee : injectees) {
612                 args[injectee.getPosition()] = resolver.resolve(injectee, null);
613             }
614             
615             setAccessible(method);
616             
617             try {
618                 ReflectionHelper.invoke(injectMe, method, args);
619             }
620             catch (Throwable e) {
621                 throw new MultiException(e);
622             }
623         }
624         
625     }
626     /**
627      * Just creates the thing, doesn't try to do anything else
628      * @param createMe The thing to create
629      * @param locator The locator to find the injection points with
630      * @return The constructed thing, no further injection is performed
631      */
632     @SuppressWarnings("unchecked")
633     public static <T> T justCreate(Class<T> createMe, ServiceLocatorImpl locator) {
634         if (createMe == null) throw new IllegalArgumentException();
635         
636         Collector collector = new Collector();
637

639

643         
644         List<Injectee> injectees = getConstructorInjectees(c);
645         
646         validateSelfInjectees(null, injectees, collector);
647         collector.throwIfErrors();
648

658         }

728     private static void getAllFieldKeys(Class<?> clazz, Set<MemberKey> currentFields, Collector collector) {
729         if (clazz == null) return;
730         
731         // Do superclasses first, so that inherited methods are
732         // overriden in the set
733         getAllFieldKeys(clazz.getSuperclass(), currentFields, collector);
734         
735         try {
736             for (Field field : getDeclaredFields(clazz)) {
737                 currentFields.add(new MemberKey(field));
738             }
739         }
740         catch (Throwable th) {
741             collector.addThrowable(new IllegalStateException("Error while getting fields of class " + clazz.getName()));

807     /**
808      * Validates the constructors of the annotated type and returns the
809      * producer for the annotatedType (if there is no valid producer
810      * constructor then this method returns null)
811      * 
812      * @param annotatedType The type to find the producer constructor
813      * @param locator The service locator to use when analyzing constructors
814      * @param collector The error collector
815      * @return The producer constructor or null if the type has no valid
816      * producer constructor
817      */
818     public static Constructor<?> findProducerConstructor(Class<?> annotatedType, ServiceLocatorImpl locator, Collector collector) {
819         Constructor<?> zeroArgConstructor = null;
820         Constructor<?> aConstructorWithInjectAnnotation = null;
821         
822         Set<Constructor<?>> allConstructors = getAllConstructors(annotatedType);
823         for (Constructor<?> constructor : allConstructors) {
824             
825             Type rawParameters[] = constructor.getGenericParameterTypes();
826             if (rawParameters.length <= 0) {
827                 zeroArgConstructor = constructor;
828             }
829             
830             if (getInjectAnnotation(locator, constructor, true) != null) {
831                 if (aConstructorWithInjectAnnotation != null) {
832                     collector.addThrowable(new IllegalArgumentException("There is more than one constructor on class " +
833                       Pretty.clazz(annotatedType)));
834                     return null;
835                 }
836                 
837                 aConstructorWithInjectAnnotation = constructor;
838             }
839             
840             if (!isProperConstructor(constructor)) {
841                 collector.addThrowable(new IllegalArgumentException("The constructor for " +
842                         Pretty.clazz(annotatedType) + " may not have an annotation as a parameter"));
843                       return null;
844                 
845             }
846             
847         }
848         
849         if (aConstructorWithInjectAnnotation != null) {
850             return aConstructorWithInjectAnnotation;
851         }
852         
853         if (zeroArgConstructor == null) {
854             collector.addThrowable(new IllegalArgumentException("The class "+ Pretty.clazz(annotatedType) +
855                     " has no constructor marked @Inject and no zero argument constructor"));
856             return null;
857         }
858         
859         return zeroArgConstructor;
860     }

1022     /**
1023      * Gets the annotation that was used for the injection
1024      * 
1025      * @param locator The service locator to use (as it will get all
1026      * the annotations that were added on as well as the normal Inject)
1027      * @param annotated  the annotated element
1028      * @param checkParams  check the params if true
1029      * @return The annotation that is the inject annotation, or null
1030      * if no inject annotation was found
1031      */
1032     private static Annotation getInjectAnnotation(ServiceLocatorImpl locator, AnnotatedElement annotated, boolean checkParams) {
1033         for (Annotation anno : annotated.getAnnotations()) {
1034             if (locator.isInjectAnnotation(anno)) return anno;
1035         }
1036         
1037         if (!checkParams) return null;
1038         
1039         boolean isConstructor;
1040         Annotation allAnnotations[][];
1041         if (annotated instanceof Method) {
1042             Method m = (Method) annotated;
1043             
1044             isConstructor = false;
1045             allAnnotations = m.getParameterAnnotations();
1046         }
1047         else if (annotated instanceof Constructor) {
1048             Constructor<?> c = (Constructor<?>) annotated;
1049             
1050             isConstructor = true;
1051             allAnnotations = c.getParameterAnnotations();
1052         }
1053         else {
1054             return null;
1055         }
1056         
1057         for (Annotation allParamAnnotations[] : allAnnotations) {
1058             for (Annotation paramAnno : allParamAnnotations) {
1059                 if (locator.isInjectAnnotation(paramAnno, isConstructor)) return paramAnno;
1060             }
1061         }
1062         
1063         return null;
1064     }

1067         if (ReflectionHelper.isStatic(member)) return false;
1068         if (isAbstract(member)) return false;
1069         for (Class<?> paramClazz : member.getParameterTypes()) {
1070             if (paramClazz.isAnnotation()) {
1071                 return false;
1072             }
1073         }
1074         
1075         return true;
1076     }

1121     @SuppressWarnings("unchecked")
1122     private static ScopeInfo getScopeInfo(
1123             AnnotatedElement annotatedGuy,
1124             Descriptor defaultScope,
1125             Collector collector) {
1126         AnnotatedElement topLevelElement = annotatedGuy;
1127         
1128         Annotation winnerScope = null;
1129         while (annotatedGuy != null) {
1130             Annotation current = internalGetScopeAnnotationType(
1131                     annotatedGuy,
1132                     collector);
1133             if (current != null) {
1134                 if (annotatedGuy.equals(topLevelElement)) {
1135                     // We found a winner, no matter the inherited state
1136                     winnerScope = current;
1137                     break;
1138                 }
1139                 else {
1140                     if (current.annotationType().isAnnotationPresent(Inherited.class)) {
1141                         winnerScope = current;
1142                         break;
1143                     }
1144                     
1145                     // This non-inherited annotation wipes out all scopes above it
1146                     break;
1147                 }
1148             }
1149                 
1150             if (annotatedGuy instanceof Class) {    
1151                 annotatedGuy = ((Class<?>) annotatedGuy).getSuperclass();
1152             }
1153             else {
1154                 Method theMethod = (Method) annotatedGuy;
1155                 Class<?> methodClass = theMethod.getDeclaringClass();
1156                 
1157                 annotatedGuy = null;
1158                 Class<?> methodSuperclass = methodClass.getSuperclass();
1159                 while (methodSuperclass != null) {
1160                     if (Factory.class.isAssignableFrom(methodSuperclass)) {
1161                         annotatedGuy = getFactoryProvideMethod(methodSuperclass);
1162                         break;
1163                     }
1164                     
1165                     methodSuperclass = methodSuperclass.getSuperclass();
1166                 }
1167             }
1168         }
1169         
1170         if (winnerScope != null) {
1171             return new ScopeInfo(winnerScope, winnerScope.annotationType());
1172         }
1173         
1174         
1175         if (topLevelElement.isAnnotationPresent(Service.class)) {
1176             return new ScopeInfo(null, Singleton.class);
1177         }
1178         
1179         if (defaultScope != null && defaultScope.getScope() != null) {
1180             Class<? extends Annotation> descScope = (Class<? extends Annotation>)
1181                     loadClass(defaultScope.getScope(), defaultScope, collector);
1182             if (descScope != null) {
1183                 return new ScopeInfo(null, descScope);
1184             }
1185         }
1186         
1187         return new ScopeInfo(null, PerLookup.class);
1188         
1189     }

1643     public static Set<Annotation> fixAndCheckQualifiers(Annotation qualifiers[], String name) {
1644         Set<Annotation> retVal = new HashSet<Annotation>();
1645         
1646         Set<String> dupChecker = new HashSet<String>();
1647         Named named = null;
1648         for (Annotation qualifier : qualifiers) {

1650             if (dupChecker.contains(annotationType)) {
1651                 throw new IllegalArgumentException(annotationType + " appears more than once in the qualifier list");
1652             }
1653             dupChecker.add(annotationType);
1654             
1655             retVal.add(qualifier);
1656             if (qualifier instanceof Named) {
1657                 named = (Named) qualifier;
1658

1661                 }

1666                 }

1687     private static class MemberKey {
1688         private final Member backingMember;
1689         
1690         private MemberKey(Member method) {
1691             backingMember = method;
1692         }
1693         
1694         private Member getBackingMember() {
1695             return backingMember;
1696         }
1697         
1698         public int hashCode() {
1699             int startCode = 0;
1700             if (backingMember instanceof Method) {
1701                 startCode = 1;
1702             }
1703             else if (backingMember instanceof Constructor) {
1704                 startCode = 2;
1705             }
1706             
1707             startCode ^= backingMember.getName().hashCode();
1708             
1709             Class<?> parameters[];
1710             if (backingMember instanceof Method) {
1711                 parameters = ((Method) backingMember).getParameterTypes();
1712             }
1713             else if (backingMember instanceof Constructor) {
1714                 parameters = ((Constructor<?>) backingMember).getParameterTypes();
1715             }
1716             else {
1717                 parameters = new Class<?>[0];
1718             }
1719             
1720             for (Class<?> param : parameters) {
1721                 startCode ^= param.hashCode();
1722             }
1723             
1724             return startCode;
1725         }
1726         
1727         public boolean equals(Object o) {
1728             if (o == null) return false;
1729             if (!(o instanceof MemberKey)) return false;
1730             
1731             MemberKey omk = (MemberKey) o;
1732             
1733             Member oMember = omk.backingMember;
1734             
1735             if (oMember.equals(backingMember)) {
1736                 // If they are the same, they are the same!
1737                 return true;
1738             }
1739             
1740             if ((backingMember instanceof Field) || (oMember instanceof Field)) {
1741                 // Fields do not inherit
1742                 return false;
1743             }
1744             
1745             if ((backingMember instanceof Method) && !(oMember instanceof Method)) {
1746                 return false;
1747             }
1748             if ((backingMember instanceof Constructor) && !(oMember instanceof Constructor)) {
1749                 return false;
1750             }
1751             
1752             if (!oMember.getName().equals(backingMember.getName())) {
1753                 return false;
1754             }
1755             
1756             if (isPrivate(backingMember) || isPrivate(oMember)) {
1757                 // If either are private, they are not the same
1758                 return false;
1759             }
1760             
1761             Class<?> oParams[];
1762             Class<?> bParams[];
1763             if (backingMember instanceof Method) {
1764                 oParams = ((Method) oMember).getParameterTypes();
1765                 bParams = ((Method) backingMember).getParameterTypes();
1766             }
1767             else if (backingMember instanceof Constructor) {
1768                 oParams = ((Constructor<?>) oMember).getParameterTypes();
1769                 bParams = ((Constructor<?>) backingMember).getParameterTypes();
1770             }
1771             else {
1772                 oParams = new Class<?>[0];
1773                 bParams = new Class<?>[0];
1774             }
1775             
1776             if (oParams.length != bParams.length) return false;
1777             for (int i = 0; i < oParams.length; i++) {
1778                 if (oParams[i] != bParams[i]) return false;
1779             }
1780             
1781             return true;
1782         }
