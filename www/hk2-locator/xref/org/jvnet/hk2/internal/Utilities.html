242      * Loads the class using the loader from the given descriptor or the
243      * classloader of the utilities class otherwise
244      *
245      * @param loadMe The fully qualified class name
246      * @param fromMe The descriptor to use for the loader
247      * @param collector The error collector to fill in if this returns null
248      * @return null on failure to load (the failure will be added to the collector)
249      */
250     public static Class<?> loadClass(String loadMe, Descriptor fromMe, Collector collector) {
251         HK2Loader loader = fromMe.getLoader();
252         if (loader == null) {
253             ClassLoader cl = Utilities.class.getClassLoader();
254             if (cl == null) {
255                 cl = ClassLoader.getSystemClassLoader();
256             }
257 
258             try {
259                 return cl.loadClass(loadMe);
260             } catch (Throwable th) {
261                 collector.addThrowable(th);
262                 return null;
263             }
264         }
265 
266         try {
267             return loader.loadClass(loadMe);
268         } catch (Throwable th) {
269             if (th instanceof MultiException) {
270                 MultiException me = (MultiException) th;
271 
272                 for (Throwable th2 : me.getErrors()) {
273                     collector.addThrowable(th2);
274                 }
275             } else {
276                 collector.addThrowable(th);
277             }
278 
279             return null;
280         }

284     /**
285      * Load the given class for the given injectee.
286      *
287      * @param implementation  the impl class name string
288      * @param injectee        the injectee
289      *
290      * @return The class represented by this implementation and injectee
291      */
292     public static Class<?> loadClass(String implementation, Injectee injectee) {
293         ClassLoader loader;
294         if (injectee != null) {
295             AnnotatedElement parent = injectee.getParent();
296 
297             if (parent instanceof Constructor) {
298                 loader = ((Constructor<?>) parent).getDeclaringClass().getClassLoader();
299             } else if (parent instanceof Method) {
300                 loader = ((Method) parent).getDeclaringClass().getClassLoader();
301             } else if (parent instanceof Field) {
302                 loader = ((Field) parent).getDeclaringClass().getClassLoader();
303             } else {
304                 loader = injectee.getClass().getClassLoader();
305             }
306         } else {
307             loader = Utilities.class.getClassLoader();
308         }
309 
310         try {
311             return loader.loadClass(implementation);
312         } catch (Throwable th) {
313             throw new MultiException(th);
314         }
315     }

317     /**
318      * Will return the class of the injection resolver annotation type, or null if
319      * no injection resolver annotation can be found
320      *
321      * @param desc The reified descriptor to find the injection resolution on
322      * @return The annotation type for this injection resolver
323      */
324     @SuppressWarnings("unchecked")
325     public static Class<? extends Annotation> getInjectionResolverType(ActiveDescriptor<?> desc) {
326         for (Type advertisedType : desc.getContractTypes()) {
327             Class<?> rawClass = ReflectionHelper.getRawClass(advertisedType);
328 
329             if (!InjectionResolver.class.equals(rawClass)) continue;
330 
331             // Found the InjectionResolver
332             if (!(advertisedType instanceof ParameterizedType)) {
333                 return null;
334             }
335 
336             Type firstType = getFirstTypeArgument(advertisedType);
337             if (!(firstType instanceof Class)) {
338                 return null;
339             }
340 
341             Class<?> retVal = (Class<?>) firstType;
342 
343             if (!Annotation.class.isAssignableFrom(retVal)) {
344                 return null;
345             }
346 
347             return (Class<? extends Annotation>) retVal;
348         }
349 
350         return null;
351     }

398     private static Set<Type> getAutoAdvertisedTypes(Type t) {
399         LinkedHashSet<Type> retVal = new LinkedHashSet<Type>();
400         retVal.add(t);
401 
402         Class<?> rawClass = ReflectionHelper.getRawClass(t);
403         if (rawClass == null) return retVal;
404 
405         Type genericSuperclass = rawClass.getGenericSuperclass();
406         while (genericSuperclass != null) {
407             Class<?> rawSuperclass = ReflectionHelper.getRawClass(genericSuperclass);
408             if (rawSuperclass == null) break;
409 
410             if (rawSuperclass.isAnnotationPresent(Contract.class)) {
411                 retVal.add(genericSuperclass);
412             }
413 
414             genericSuperclass = rawSuperclass.getGenericSuperclass();
415         }
416 
417         while (rawClass != null) {
418             for (Type iface : rawClass.getGenericInterfaces()) {
419                 Class<?> ifaceClass = ReflectionHelper.getRawClass(iface);
420                 if (ifaceClass.isAnnotationPresent(Contract.class)) {
421                     retVal.add(iface);
422                 }
423             }
424 
425             rawClass = rawClass.getSuperclass();
426         }
427 
428         return retVal;
429     }

431     /**
432      * Creates a reified automatically generated descriptor
433      *
434      * @param clazz The class to create the desciptor for
435      * @param locator The service locator for whom we are creating this
436      * @return A reified active descriptor
437      *
438      * @throws MultiException if there was an error in the class
439      * @throws IllegalArgumentException If the class is null
440      */
441     public static <T> ActiveDescriptor<T> createAutoDescriptor(Class<T> clazz, ServiceLocatorImpl locator)
442             throws MultiException, IllegalArgumentException {
443         if (clazz == null) throw new IllegalArgumentException();
444 
445         Collector collector = new Collector();
446 
447         Creator<T> creator;
448         Set<Annotation> qualifiers;
449         Set<Type> contracts;
450         Class<? extends Annotation> scope;
451         String name;
452         Boolean proxy = null;
453 
454         // Qualifiers naming dance
455         qualifiers = ReflectionHelper.getQualifierAnnotations(clazz);
456         name = getNameFromAllQualifiers(qualifiers, clazz);
457         qualifiers = getAllQualifiers(clazz, name, collector);  // Fixes the @Named qualifier if it has no value
458 
459         contracts = getAutoAdvertisedTypes(clazz);
460         ScopeInfo scopeInfo = getScopeInfo(clazz, null, collector);
461         scope = scopeInfo.getAnnoType();
462 
463         creator = new ClazzCreator<T>(locator, clazz, null, collector);
464 
465         collector.throwIfErrors();
466 
467         Map<String, List<String>> metadata = new HashMap<String, List<String>>();
468         if (scopeInfo.getScope() != null) {
469             BuilderHelper.getMetadataValues(scopeInfo.getScope(), metadata);
470         }
471 
472         for (Annotation qualifier : qualifiers) {
473             BuilderHelper.getMetadataValues(qualifier, metadata);
474         }
475         
476         UseProxy useProxy = clazz.getAnnotation(UseProxy.class);
477         if (useProxy != null) {
478             proxy = new Boolean(useProxy.value());
479         }
480         
481         DescriptorVisibility visibility = DescriptorVisibility.NORMAL;
482         Visibility vi = clazz.getAnnotation(Visibility.class);
483         if (vi != null) {
484             visibility = vi.value();
485         }
486 
487         return new AutoActiveDescriptor<T>(
488                 clazz,
489                 creator,
490                 contracts,
491                 scope,
492                 name,
493                 qualifiers,
494                 visibility,
495                 0,
496                 proxy,
497                 metadata);
498     }

548     /**
549      * Just creates the thing, doesn't try to do anything else
550      * @param injectMe The object to inject into
551      * @param locator The locator to find the injection points with
552      */
553     public static void justInject(Object injectMe, ServiceLocatorImpl locator) {
554         if (injectMe == null) throw new IllegalArgumentException();
555 
556         Class<?> baseClass = injectMe.getClass();
557 
558         Collector collector = new Collector();
559 
560         Set<Field> fields = findInitializerFields(baseClass, locator, collector);
561         Set<Method> methods = findInitializerMethods(baseClass, locator, collector);
562 
563         collector.throwIfErrors();
564 
565         for (Field field : fields) {
566             InjectionResolver<?> resolver = getInjectionResolver(locator, field);
567 
568             List<Injectee> injecteeFields = Utilities.getFieldInjectees(field);
569 
570             validateSelfInjectees(null, injecteeFields, collector);
571             collector.throwIfErrors();
572 
573             Injectee injectee = injecteeFields.get(0);
574 
575             Object fieldValue = resolver.resolve(injectee, null);
576             
577             try {
578                 ReflectionHelper.setField(field, injectMe, fieldValue);
579             }
580             catch (Throwable th) {
581                 throw new MultiException(th);
582             }
583         }
584 
585         for (Method method : methods) {
586             List<Injectee> injectees = Utilities.getMethodInjectees(method);
587 
588             validateSelfInjectees(null, injectees, collector);
589             collector.throwIfErrors();
590 
591             Object args[] = new Object[injectees.size()];
592 
593             for (Injectee injectee : injectees) {
594                 InjectionResolver<?> resolver = getInjectionResolver(locator, injectee);
595                 args[injectee.getPosition()] = resolver.resolve(injectee, null);
596             }
597 
598             try {
599                 ReflectionHelper.invoke(injectMe, method, args);
600             } catch (Throwable e) {
601                 throw new MultiException(e);
602             }
603         }
604 
605     }

1183     @SuppressWarnings("unchecked")
1184     private static ScopeInfo getScopeInfo(
1185             AnnotatedElement annotatedGuy,
1186             Descriptor defaultScope,
1187             Collector collector) {
1188         AnnotatedElement topLevelElement = annotatedGuy;
1189 
1190         Annotation winnerScope = null;
1191         while (annotatedGuy != null) {
1192             Annotation current = internalGetScopeAnnotationType(
1193                     annotatedGuy,
1194                     collector);
1195             if (current != null) {
1196                 if (annotatedGuy.equals(topLevelElement)) {
1197                     // We found a winner, no matter the inherited state
1198                     winnerScope = current;
1199                     break;
1200                 } else {
1201                     if (current.annotationType().isAnnotationPresent(Inherited.class)) {
1202                         winnerScope = current;
1203                         break;
1204                     }
1205 
1206                     // This non-inherited annotation wipes out all scopes above it
1207                     break;
1208                 }
1209             }
1210 
1211             if (annotatedGuy instanceof Class) {
1212                 annotatedGuy = ((Class<?>) annotatedGuy).getSuperclass();
1213             } else {
1214                 Method theMethod = (Method) annotatedGuy;
1215                 Class<?> methodClass = theMethod.getDeclaringClass();
1216 
1217                 annotatedGuy = null;
1218                 Class<?> methodSuperclass = methodClass.getSuperclass();
1219                 while (methodSuperclass != null) {
1220                     if (Factory.class.isAssignableFrom(methodSuperclass)) {
1221                         annotatedGuy = getFactoryProvideMethod(methodSuperclass);
1222                         break;
1223                     }
1224 
1225                     methodSuperclass = methodSuperclass.getSuperclass();
1226                 }
1227             }
1228         }
1229 
1230         if (winnerScope != null) {
1231             return new ScopeInfo(winnerScope, winnerScope.annotationType());
1232         }
1233 
1234 
1235         if (topLevelElement.isAnnotationPresent(Service.class)) {
1236             return new ScopeInfo(null, Singleton.class);
1237         }
1238 
1239         if (defaultScope != null && defaultScope.getScope() != null) {
1240             Class<? extends Annotation> descScope = (Class<? extends Annotation>)
1241                     loadClass(defaultScope.getScope(), defaultScope, collector);
1242             if (descScope != null) {
1243                 return new ScopeInfo(null, descScope);
1244             }
1245         }
1246 
1247         return new ScopeInfo(null, PerLookup.class);
1248 
1249     }
